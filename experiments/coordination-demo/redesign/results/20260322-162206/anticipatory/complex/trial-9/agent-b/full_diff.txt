+// FormatTable formats headers and rows as an aligned text table. Each column
+// is auto-sized to fit the widest content (including headers), with visual
+// width calculated using StripANSI to account for ANSI escape codes. A
+// separator line of dashes divides the header row from data rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths
+	widths := make([]int, len(headers))
+	for i, h := range headers {
+		widths[i] = len(StripANSI(h))
+	}
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			w := len(StripANSI(cell))
+			if w > widths[i] {
+				widths[i] = w
+			}
+		}
+	}
+
+	var sb strings.Builder
+
+	// Helper to right-pad a string to width, accounting for ANSI codes
+	padRight := func(s string, w int) string {
+		visual := len(StripANSI(s))
+		if visual >= w {
+			return s
+		}
+		return s + strings.Repeat(" ", w-visual)
+	}
+
+	// Format header row
+	for i, h := range headers {
+		if i == 0 {
+			sb.WriteString("| ")
+		} else {
+			sb.WriteString(" | ")
+		}
+		sb.WriteString(padRight(h, widths[i]) + " ")
+	}
+	sb.WriteString("|\n")
+
+	// Format separator line
+	for i := 0; i < len(headers); i++ {
+		if i == 0 {
+			sb.WriteString("|")
+		} else {
+			sb.WriteString("-|")
+		}
+		sb.WriteString(strings.Repeat("-", widths[i]+2))
+	}
+	sb.WriteString("|\n")
+
+	// Format data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			if i == 0 {
+				sb.WriteString("| ")
+			} else {
+				sb.WriteString(" | ")
+			}
+			sb.WriteString(padRight(cell, widths[i]) + " ")
+		}
+		sb.WriteString("|\n")
+	}
+
+	return sb.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name   | Age |\n|--------|-----|\n| Alice  | 30  |\n| Bob    | 25  |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"A", "B"},
+			rows:    [][]string{},
+			want:    "| A  | B |\n|----|---|\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Value"},
+			rows: [][]string{
+				{"one"},
+				{"two"},
+			},
+			want: "| Value |\n|-------|\n| one   |\n| two   |\n",
+		},
+		{
+			name:    "ansi colored content",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"\x1b[32mgreen\x1b[0m"},
+			},
+			want: "| Status |\n|--------|\n| \x1b[32mgreen\x1b[0m  |\n",
+		},
+		{
+			name:    "mismatched column counts",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "| A  | B  | C |\n|----|----|---|\n| 1  | 2  |   |\n| x  | y  | z |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"a", "longvalue"},
+			},
+			want: "| Short  | VeryLongHeader |\n|--------|----------------|\n| a      | longvalue      |\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"X"},
+			rows:    nil,
+			want:    "| X |\n|---|\n",
+		},
+		{
+			name:    "multiple rows different widths",
+			headers: []string{"First", "Last"},
+			rows: [][]string{
+				{"a", "b"},
+				{"alice", "wonderland"},
+			},
+			want: "| First  | Last       |\n|--------|------------|\n| a      | b          |\n| alice  | wonderland |\n",
+		},
+		{
+			name:    "empty cells",
+			headers: []string{"Filled", "Empty"},
+			rows: [][]string{
+				{"data", ""},
+				{"", "info"},
+			},
+			want: "| Filled  | Empty |\n|---------|-------|\n| data    |       |\n|         | info  |\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() got:\n%q\n\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
+}
