+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It counts runes (Unicode code points), not bytes.
+func VisualWidth(s string) int {
+	clean := StripANSI(s)
+	return len([]rune(clean))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward the width. If the string
+// is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-currentWidth)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"hello world", 11},
+
+		// ANSI codes should not count
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold\x1b[0m", 4},

+		// Unicode (emoji and CJK)
+		{"你好", 2},

+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)

+		// Unicode
+		{"你好", 4, "你好  "},
+		{"😀", 3, "😀  "},
