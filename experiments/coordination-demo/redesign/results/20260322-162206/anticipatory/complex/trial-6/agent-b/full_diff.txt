+// FormatTable renders headers and rows as an aligned text table.
+// Columns are auto-sized based on the widest content (header or data).
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Handles edge cases: empty rows (headers only), rows with fewer or more columns than headers.
+// Returns an empty string if headers is empty.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and rows
+	colWidths := make([]int, len(headers))
+
+	// Initialize with header widths
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Expand for row content
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			w := len(StripANSI(row[i]))
+			if w > colWidths[i] {
+				colWidths[i] = w
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Write header
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(h)
+		// Pad to column width
+		padding := colWidths[i] - len(StripANSI(h))
+		buf.WriteString(strings.Repeat(" ", padding))
+	}
+	buf.WriteString("\n")
+
+	// Write separator
+	for i, w := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", w))
+	}
+	buf.WriteString("\n")
+
+	// Write rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+
+			var col string
+			if i < len(row) {
+				col = row[i]
+			}
+
+			buf.WriteString(col)
+			// Pad to column width
+			padding := colWidths[i] - len(StripANSI(col))
+			buf.WriteString(strings.Repeat(" ", padding))
+		}
+		buf.WriteString("\n")
+	}
+
+	return strings.TrimSuffix(buf.String(), "\n")
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "empty_headers",
+			headers: []string{},
+			rows:    nil,
+			want:    "",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Age"},
+			rows:    nil,
+			want:    "Name | Age\n-----+----",
+		},
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 ",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Name"},
+			rows: [][]string{
+				{"Alice"},
+				{"Bob"},
+			},
+			want: "Name \n-----\nAlice\nBob  ",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongerContent"},
+			rows: [][]string{
+				{"X", "Y"},
+				{"A", "VeryLongDataValue"},
+			},
+			want: "Short | LongerContent    \n------+------------------\nX     | Y                \nA     | VeryLongDataValue",
+		},
+		{
+			name:    "mismatched columns fewer in row",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25", "NYC"},
+			},
+			want: "Name  | Age | City\n------+-----+-----\nAlice | 30  |     \nBob   | 25  | NYC ",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status", "Value"},
+			rows: [][]string{
+				{"\x1b[32mactive\x1b[0m", "100"},
+				{"\x1b[31minactive\x1b[0m", "50"},
+			},
+			want: "Status   | Value\n---------+------\n\x1b[32mactive\x1b[0m   | 100  \n\x1b[31minactive\x1b[0m | 50   ",
+		},
+		{
+			name:    "ANSI in header",
+			headers: []string{"\x1b[1;33mID\x1b[0m", "Name"},
+			rows: [][]string{
+				{"1", "Alice"},
+				{"2", "Bob"},
+			},
+			want: "\x1b[1;33mID\x1b[0m | Name \n---+------\n1  | Alice\n2  | Bob  ",
+		},
+		{
+			name:    "all empty strings",
+			headers: []string{"Col1", "Col2"},
+			rows: [][]string{
+				{"", ""},
+				{"", ""},
+			},
+			want: "Col1 | Col2\n-----+-----\n     |     \n     |     ",
+		},
+		{
+			name:    "row with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"X", "Y", "Z"},
+			},
+			want: "A | B\n--+--\nX | Y",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
