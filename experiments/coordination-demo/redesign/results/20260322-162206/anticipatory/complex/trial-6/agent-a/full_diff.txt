+// VisualWidth returns the visual display width of a string, excluding ANSI escape codes.
+// It correctly handles Unicode characters by counting runes.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward the width. If the string is
+// already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mred text\x1b[0m", 8}, // "red text" = 8 chars
+		{"\x1b[1;32mbold green\x1b[0m", 10}, // "bold green" = 10 chars
+		{"\x1b[33m\x1b[1mhello\x1b[0m", 5}, // "hello" = 5 chars
+
+		// Unicode strings (emoji, CJK)
+		{"こんにちは", 5}, // 5 Japanese characters
+		{"你好", 2}, // 2 Chinese characters
+		{"hello 世界", 8}, // "hello " (6) + "世界" (2) = 8
+
+		// Mixed ANSI and Unicode
+		{"\x1b[32mhello\x1b[0m world", 11}, // "hello world" = 11 chars
+		{"\x1b[31m你好\x1b[0m", 2}, // "你好" = 2 chars
+
+		// Edge cases
+		{"\x1b[0m", 0}, // Only ANSI codes
+		{"\x1b[31m\x1b[32m\x1b[0m", 0}, // Only ANSI codes
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII - needs padding
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		{"", 3, "   "},
+
+		// Plain ASCII - already at or over width
+		{"hello", 5, "hello"},
+		{"hello", 3, "hello"}, // wider than target, unchanged
+		{"hello world", 8, "hello world"}, // wider than target, unchanged
+
+		// With ANSI codes - needs padding
+		{"\x1b[31mred\x1b[0m", 6, "\x1b[31mred\x1b[0m   "}, // "red" = 3 chars, pad to 6
+		{"\x1b[1;32mbold\x1b[0m", 8, "\x1b[1;32mbold\x1b[0m    "}, // "bold" = 4 chars, pad to 8
+
+		// With ANSI codes - already at width
+		{"\x1b[31mhello\x1b[0m", 5, "\x1b[31mhello\x1b[0m"}, // already 5 chars
+
+		// Unicode - needs padding
+		{"你好", 5, "你好   "}, // 2 chars, pad to 5
+		{"こんにちは", 7, "こんにちは  "}, // 5 chars, pad to 7
+
+		// Unicode with ANSI - needs padding
+		{"\x1b[32m世界\x1b[0m", 5, "\x1b[32m世界\x1b[0m   "}, // 2 chars, pad to 5
+
+		// Unicode with ANSI - already at width
+		{"\x1b[32mhello world\x1b[0m", 11, "\x1b[32mhello world\x1b[0m"}, // already 11 chars
+
+		// Edge cases
+		{"a", 1, "a"}, // exactly at width
+		{"", 0, ""}, // target width 0
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width is correct
+		if VisualWidth(got) < tt.width && VisualWidth(got) != 0 {
+			t.Errorf("PadToWidth(%q, %d) result has visual width %d, want %d", tt.input, tt.width, VisualWidth(got), tt.width)
+		}
+	}
+}
