+// VisualWidth returns the visual display width of a string, ignoring ANSI escape codes.
+// It counts runes (Unicode characters) rather than bytes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the visual width.
+// If the string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		desc   string
+	}{
+		{"hello", 10, "hello     ", "pad plain string"},
+		{"hello", 5, "hello", "already at width"},
+		{"hello", 3, "hello", "wider than target"},
+		{"", 5, "     ", "empty string"},
+		{"\x1b[31mred\x1b[0m", 10, "\x1b[31mred\x1b[0m       ", "pad string with ANSI codes"},
+		{"\x1b[1;32mbold green\x1b[0m", 15, "\x1b[1;32mbold green\x1b[0m     ", "pad bold green with ANSI"},
+		{"你好", 4, "你好  ", "pad unicode"},
+		{"a\x1b[35mb\x1b[0mc", 5, "a\x1b[35mb\x1b[0mc  ", "pad mixed ASCII and ANSI"},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) [%s] = %q, want %q", tt.input, tt.width, tt.desc, got, tt.want)
+		}
+		// Verify that visual width of padded result matches or exceeds target
+		visualWidth := VisualWidth(got)
+		if tt.width > VisualWidth(tt.input) && visualWidth != tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, visualWidth, tt.width)
+		}
+	}
+}
