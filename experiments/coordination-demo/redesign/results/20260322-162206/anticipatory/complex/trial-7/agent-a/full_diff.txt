+// VisualWidth returns the visual display width of a string, excluding ANSI escape codes.
+// Counts runes (not bytes) to handle Unicode correctly.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward the width.
+// If the string is already at or wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-currentWidth)
+}

+		{"\x1b[1;32mhi\x1b[0m", 5, "\x1b[1;32mhi\x1b[0m   ", 5},
+		{"café", 6, "café  ", 6},

+		}
+		visualWidth := VisualWidth(got)
+		if visualWidth != tt.wantWidth {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, visualWidth, tt.wantWidth)
