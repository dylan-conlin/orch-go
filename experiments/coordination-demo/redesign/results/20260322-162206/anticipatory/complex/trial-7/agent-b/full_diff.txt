+// FormatTable formats headers and rows as an aligned text table.
+// It auto-sizes columns based on content width and uses StripANSI to
+// correctly measure ANSI-colored strings. Returns an empty string if
+// headers is empty.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	const padding = 1
+
+	// Calculate column widths (based on visual width, ignoring ANSI codes)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			width := len(StripANSI(cell))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Helper to format a cell: content + padding to fit column
+	formatCell := func(content string, colIndex int) string {
+		stripped := StripANSI(content)
+		padRight := colWidths[colIndex] - len(stripped)
+		return content + strings.Repeat(" ", padRight)
+	}
+
+	var lines []string
+	var buf strings.Builder
+
+	// Header row
+	for i := 0; i < len(headers); i++ {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(strings.Repeat(" ", padding))
+		buf.WriteString(formatCell(headers[i], i))
+		buf.WriteString(strings.Repeat(" ", padding))
+	}
+	lines = append(lines, strings.TrimRight(buf.String(), " "))
+	buf.Reset()
+
+	// Separator
+	for i := range headers {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", colWidths[i]+padding*2))
+	}
+	lines = append(lines, buf.String())
+	buf.Reset()
+
+	// Data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			buf.WriteString(strings.Repeat(" ", padding))
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(formatCell(cell, i))
+			buf.WriteString(strings.Repeat(" ", padding))
+		}
+		lines = append(lines, strings.TrimRight(buf.String(), " "))
+		buf.Reset()
+	}
+
+	return strings.Join(lines, "\n")
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{{"Alice", "30"}, {"Bob", "25"}},
+			want: ` Name   |  Age
+--------+------
+ Alice  |  30
+ Bob    |  25`,
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want: ` Name  |  Age
+-------+------`,
+		},
+		{
+			name:    "single column",
+			headers: []string{"Name"},
+			rows:    [][]string{{"Alice"}, {"Bob"}},
+			want: ` Name
+-------
+ Alice
+ Bob`,
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name"},
+			rows:    [][]string{{"\x1b[31mAlice\x1b[0m"}, {"Bob"}},
+			want: " Name\n-------\n \x1b[31mAlice\x1b[0m\n Bob",
+		},
+		{
+			name:    "mismatched columns (short row)",
+			headers: []string{"A", "B", "C"},
+			rows:    [][]string{{"x", "y"}, {"1", "2", "3"}},
+			want: ` A  |  B  |  C
+----+-----+----
+ x  |  y  |
+ 1  |  2  |  3`,
+		},
+		{
+			name:    "wide content",
+			headers: []string{"ID", "Description"},
+			rows:    [][]string{{"1", "short"}, {"2", "this is a longer description"}},
+			want: ` ID  |  Description
+-----+-------------------------------
+ 1   |  short
+ 2   |  this is a longer description`,
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{{"x"}},
+			want:    "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"A"},
+			rows:    nil,
+			want: ` A
+---`,
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() =\n%q\n\nwant\n%q", got, tt.want)
+			}
+		})
+	}
