+// VisualWidth returns the visual display width of a string in runes,
+// ignoring ANSI escape codes. This is useful for calculating padding or
+// truncation when the string may contain ANSI color/formatting codes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return utf8.RuneCountInString(stripped)
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward the visual width. If the
+// string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	stripped := StripANSI(s)
+	visualWidth := utf8.RuneCountInString(stripped)
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII: pad when needed
+		{"hello", 10, "hello     "},
+		{"hello", 5, "hello"},
+		// Plain ASCII: no padding when already at width or wider
+		{"hello", 3, "hello"},
+		{"hello world", 5, "hello world"},
+		// Empty string
+		{"", 5, "     "},
+		{"", 0, ""},
+		// ANSI codes with ASCII
+		{"\x1b[31mhi\x1b[0m", 5, "\x1b[31mhi\x1b[0m   "},
+		{"\x1b[31mhi\x1b[0m", 2, "\x1b[31mhi\x1b[0m"},
+		{"\x1b[1;32mbold\x1b[0m", 8, "\x1b[1;32mbold\x1b[0m    "},
+		// Unicode characters
+		{"你好", 5, "你好   "},
+		{"中", 3, "中  "},
+		{"😀", 3, "😀  "},
+		// Mixed ANSI and Unicode
+		{"\x1b[1;32m中文\x1b[0m", 5, "\x1b[1;32m中文\x1b[0m   "},
+		{"\x1b[31m你好\x1b[0m", 2, "\x1b[31m你好\x1b[0m"},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width is at least the target width
+		visualWidth := VisualWidth(got)
+		if visualWidth < tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want at least %d", tt.input, tt.width, visualWidth, tt.width)
+		}
+	}
+}
