+// FormatTable renders headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including header).
+// ANSI color codes in content are stripped for width calculations but preserved
+// in the output. Returns a string with headers separated from data rows by a
+// line of dashes. Returns empty string if headers is empty.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on visual width (stripping ANSI codes)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(colWidths) {
+				w := len(StripANSI(cell))
+				if w > colWidths[i] {
+					colWidths[i] = w
+				}
+			}
+		}
+	}
+
+	var result strings.Builder
+
+	// Write header row
+	for i, h := range headers {
+		result.WriteString("| ")
+		result.WriteString(padRight(h, colWidths[i]))
+		result.WriteString(" ")
+	}
+	result.WriteString("|\n")
+
+	// Write separator line
+	for i := range headers {
+		result.WriteString("+")
+		result.WriteString(strings.Repeat("-", colWidths[i]+2))
+	}
+	result.WriteString("+\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := range headers {
+			result.WriteString("| ")
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			result.WriteString(padRight(cell, colWidths[i]))
+			result.WriteString(" ")
+		}
+		result.WriteString("|\n")
+	}
+
+	return result.String()
+}
+
+// padRight pads s to the given visual width, accounting for ANSI codes.
+// If the string contains ANSI codes, they are not counted toward the visual width.
+func padRight(s string, width int) string {
+	visualWidth := len(StripANSI(s))
+	padding := width - visualWidth
+	if padding > 0 {
+		return s + strings.Repeat(" ", padding)
+	}
+	return s
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic two-column table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name  | Age |\n+-------+-----+\n| Alice | 30  |\n| Bob   | 25  |\n",
+		},
+		{
+			name:    "headers only (no rows)",
+			headers: []string{"Header1", "Header2"},
+			rows:    [][]string{},
+			want:    "| Header1 | Header2 |\n+---------+---------+\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Col"},
+			rows: [][]string{
+				{"A"},
+				{"B"},
+			},
+			want: "| Col |\n+-----+\n| A   |\n| B   |\n",
+		},
+		{
+			name:    "ANSI colored content alignment",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[32mAlice\x1b[0m", "\x1b[31mFailed\x1b[0m"},
+				{"Bob", "OK"},
+			},
+			want: "| Name  | Status |\n+-------+--------+\n| \x1b[32mAlice\x1b[0m | \x1b[31mFailed\x1b[0m |\n| Bob   | OK     |\n",
+		},
+		{
+			name:    "row shorter than headers (missing columns)",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "| A | B | C |\n+---+---+---+\n| 1 | 2 |   |\n| 3 | 4 | 5 |\n",
+		},
+		{
+			name:    "wide content (long string)",
+			headers: []string{"Short", "Content"},
+			rows: [][]string{
+				{"A", "This is a longer string"},
+				{"BB", "X"},
+			},
+			want: "| Short | Content                 |\n+-------+-------------------------+\n| A     | This is a longer string |\n| BB    | X                       |\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "three columns with varying widths",
+			headers: []string{"ID", "Name", "Status"},
+			rows: [][]string{
+				{"1", "Alice", "Active"},
+				{"2", "Bob", "Inactive"},
+				{"3", "Charlie", "Pending"},
+			},
+			want: "| ID | Name    | Status   |\n+----+---------+----------+\n| 1  | Alice   | Active   |\n| 2  | Bob     | Inactive |\n| 3  | Charlie | Pending  |\n",
+		},
+		{
+			name:    "empty row cells",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"value1", ""},
+				{"", "value2"},
+			},
+			want: "| A      | B      |\n+--------+--------+\n| value1 |        |\n|        | value2 |\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() =\n%q\nwant\n%q", got, tt.want)
+			}
+		})
+	}
