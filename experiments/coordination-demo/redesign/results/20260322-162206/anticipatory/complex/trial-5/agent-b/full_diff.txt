+// FormatTable renders headers and rows as an aligned text table with auto-sized columns.
+// Columns are sized based on the widest content in each column (including headers).
+// ANSI-colored content is handled correctly by using StripANSI for width calculation.
+// Handles edge cases: empty rows (headers only), rows with fewer/more columns than headers.
+// Output format uses " | " to separate columns and "-+-" as separator in the header line.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Determine the number of columns
+	numCols := len(headers)
+	for _, row := range rows {
+		if len(row) > numCols {
+			numCols = len(row)
+		}
+	}
+
+	// Calculate column widths using StripANSI for correct visual width
+	colWidths := make([]int, numCols)
+
+	// Initialize with header widths
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Update with row widths
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(colWidths) {
+				w := len(StripANSI(cell))
+				if w > colWidths[i] {
+					colWidths[i] = w
+				}
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Write header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(h)
+		buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(h))))
+	}
+	buf.WriteString("\n")
+
+	// Write separator line
+	for i, width := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", width))
+	}
+	buf.WriteString("\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := 0; i < numCols; i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(cell))))
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 \n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want:    "Name | Age\n-----+----\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"\x1b[32mactive\x1b[0m"},
+				{"\x1b[31minactive\x1b[0m"},
+			},
+			want: "Status  \n--------\n\x1b[32mactive\x1b[0m  \n\x1b[31minactive\x1b[0m\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+			},
+			want: "Item  \n------\napple \nbanana\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Description"},
+			rows: [][]string{
+				{"short"},
+				{"very long description here"},
+			},
+			want: "Description               \n--------------------------\nshort                     \nvery long description here\n",
+		},
+		{
+			name:    "mismatched columns - fewer in row",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \n3 | 4 | 5\n",
+		},
+		{
+			name:    "empty headers and rows",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "single row",
+			headers: []string{"X", "Y"},
+			rows: [][]string{
+				{"a", "b"},
+			},
+			want: "X | Y\n--+--\na | b\n",
+		},
+		{
+			name:    "mixed ANSI and plain text",
+			headers: []string{"Colored", "Plain"},
+			rows: [][]string{
+				{"\x1b[33myellow\x1b[0m", "text"},
+				{"normal", "\x1b[36mcyan\x1b[0m"},
+			},
+			want: "Colored | Plain\n--------+------\n\x1b[33myellow\x1b[0m  | text \nnormal  | \x1b[36mcyan\x1b[0m \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() got:\n%q\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
