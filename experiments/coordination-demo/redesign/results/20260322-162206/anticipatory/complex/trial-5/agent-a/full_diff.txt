+// VisualWidth returns the visual display width of a string, excluding ANSI escape codes.
+// Unicode characters are correctly counted as single runes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads s with spaces to reach the target visual width.
+// ANSI escape codes are preserved in the output but do not count toward the width.
+// If the string is already at or wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string

+	}{
+		// Plain ASCII
+		{"hello", 5},

+		if got != tt.want {

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII - pad shorter strings
+		{"hello", 10, "hello     "},
+		{"hello", 5, "hello"},
+		{"hello", 3, "hello"},
+		{"a", 5, "a    "},
+
+		// Empty string
+		{"", 5, "     "},
+		{"", 0, ""},
+
+		// With ANSI codes (codes preserved, don't count toward width)
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[31mred\x1b[0m", 3, "\x1b[31mred\x1b[0m"},
+		{"\x1b[1;32mbold\x1b[0m", 8, "\x1b[1;32mbold\x1b[0m    "},
+
+		// Unicode
+		{"你好", 5, "你好   "},
+		{"你好", 2, "你好"},
+		{"こんにちは", 8, "こんにちは   "},
+
+		// Combined ANSI + Unicode
+		{"\x1b[32m你好\x1b[0m", 4, "\x1b[32m你好\x1b[0m  "},
+		{"\x1b[33m😀\x1b[0m", 3, "\x1b[33m😀\x1b[0m  "},
+	}
+
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+
+		// Verify visual width is at least the target width
+		visual := VisualWidth(got)
+		if visual < tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want >= %d", tt.input, tt.width, visual, tt.width)
+		}
+	}
+}
