+// FormatTable formats headers and rows as an aligned text table.
+// Columns are auto-sized based on the widest content (including headers).
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Handles edge cases: empty rows, rows with fewer columns than headers, nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate the width of each column (max of header and all row widths)
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			w := len(StripANSI(row[i]))
+			if w > colWidths[i] {
+				colWidths[i] = w
+			}
+		}
+	}
+
+	// Build the table
+	var buf strings.Builder
+
+	// Header row
+	for i, header := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(header)
+		buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(header))))
+	}
+	buf.WriteString("\n")
+
+	// Separator
+	for i, width := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", width))
+	}
+	buf.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(cell))))
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "Name  | Age | City\n------+-----+-----\nAlice | 30  | NYC \nBob   | 25  | LA  \n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Status"},
+			rows:    [][]string{},
+			want:    "Name | Status\n-----+-------\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+			},
+			want: "Item  \n------\napple \nbanana\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[31mError\x1b[0m", "\x1b[32mOK\x1b[0m"},
+				{"Plain", "\x1b[33mWarn\x1b[0m"},
+			},
+			want: "Name  | Status\n------+-------\n\x1b[31mError\x1b[0m | \x1b[32mOK\x1b[0m    \nPlain | \x1b[33mWarn\x1b[0m  \n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \nx | y | z\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongHeader"},
+			rows: [][]string{
+				{"X", "very long content here"},
+			},
+			want: "Short | LongHeader            \n------+-----------------------\nX     | very long content here\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "mismatched row lengths",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1"},
+				{"x", "y", "z"},
+			},
+			want: "A | B\n--+--\n1 |  \nx | y\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() = %q, want %q", got, tt.want)
+			}
+		})
+	}
+}
