+// FormatBytes formats a byte count as a human-readable string with binary units.
+// Output uses binary units (B, KiB, MiB, GiB, TiB) with 1 decimal place for
+// non-byte units. Examples: "512 B", "1.5 MiB", "2.0 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	// Define binary units
+	const (
+		unit   = 1024
+		kibibyte = unit
+		mebibyte = unit * unit
+		gibibyte = unit * unit * unit
+		tebibyte = unit * unit * unit * unit
+	)
+
+	var size float64
+	var suffix string
+
+	switch {
+	case bytes < kibibyte:
+		suffix = "B"
+		size = float64(bytes)
+	case bytes < mebibyte:
+		suffix = "KiB"
+		size = float64(bytes) / kibibyte
+	case bytes < gibibyte:
+		suffix = "MiB"
+		size = float64(bytes) / mebibyte
+	case bytes < tebibyte:
+		suffix = "GiB"
+		size = float64(bytes) / gibibyte
+	default:
+		suffix = "TiB"
+		size = float64(bytes) / tebibyte
+	}
+
+	// Format: bytes have no decimal, others have 1 decimal place
+	var result string
+	if suffix == "B" {
+		result = fmt.Sprintf("%d %s", int64(size), suffix)
+	} else {
+		result = fmt.Sprintf("%.1f %s", size, suffix)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024*10 + 512, "10.5 KiB"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1048576 * 10, "10.0 MiB"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 537870912, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB boundary and values
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		// Large values
+		{5 * 1099511627776, "5.0 TiB"},
+		{1024*1024*1024*1024*10 + 1099511627776, "11.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
