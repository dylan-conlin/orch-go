+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units with "/s" suffix: "512 B/s", "1.5 MiB/s", etc.
+// Non-byte-per-sec units show 1 decimal place.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	if bytesPerSec < KiB {
+		result = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	} else if bytesPerSec < MiB {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	} else if bytesPerSec < GiB {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		// Large values
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Fractional values
+		{1280, "1.2 KiB/s"},
+		{1350, "1.3 KiB/s"},
+		{1331200, "1.3 MiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		{-1536, "-1.5 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
