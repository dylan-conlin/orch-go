+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output uses 1 decimal place for non-byte units (e.g., "1.5 MiB").
+// Negative values are prefixed with "-". Zero returns "0 B".
+// Examples: "512 B", "1.0 KiB", "1.5 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	// Define binary unit thresholds
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes >= TiB:
+		value = float64(bytes) / float64(TiB)
+		unit = "TiB"
+	case bytes >= GiB:
+		value = float64(bytes) / float64(GiB)
+		unit = "GiB"
+	case bytes >= MiB:
+		value = float64(bytes) / float64(MiB)
+		unit = "MiB"
+	case bytes >= KiB:
+		value = float64(bytes) / float64(KiB)
+		unit = "KiB"
+	default:
+		// For bytes, show as integer
+		if negative {
+			return fmt.Sprintf("-%d B", bytes)
+		}
+		return fmt.Sprintf("%d B", bytes)
+	}
+
+	// Format with 1 decimal place
+	result := fmt.Sprintf("%.1f %s", value, unit)
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Exact boundaries and KiB
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024*5 + 512, "5.5 KiB"},
+
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{5242880, "5.0 MiB"},
+
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{5368709120, "5.0 GiB"},
+
+		// TiB boundary and values
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+
+		// Edge cases with decimal precision
+		{1024 + 102, "1.1 KiB"},  // 1.0996... rounds to 1.1
+		{1024 + 51, "1.0 KiB"},   // 0.0498... rounds down to 1.0
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
