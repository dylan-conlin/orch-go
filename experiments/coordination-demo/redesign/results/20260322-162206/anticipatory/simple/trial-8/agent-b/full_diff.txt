+// FormatRate formats a transfer rate into a human-readable string using
+// binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-B/s units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	if bytesPerSec < KiB {
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else if bytesPerSec < MiB {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	} else if bytesPerSec < GiB {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{256, "256 B/s"},
+		{512, "512 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		// Fractional values
+		{1536, "1.5 KiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Additional KiB/s values
+		{2048, "2.0 KiB/s"},
+		{3072, "3.0 KiB/s"},
+		// Additional MiB/s values
+		{2097152, "2.0 MiB/s"},
+		{5242880, "5.0 MiB/s"},
+		// Additional GiB/s values
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
