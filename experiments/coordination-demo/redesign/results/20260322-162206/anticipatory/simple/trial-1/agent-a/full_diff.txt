+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "512 B", "1.5 KiB", "1.0 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	abs := bytes
+	negative := ""
+	if bytes < 0 {
+		negative = "-"
+		abs = -bytes
+	}
+
+	// Determine the appropriate unit and value
+	switch {
+	case abs < KiB:
+		return fmt.Sprintf("%s%d B", negative, abs)
+	case abs < MiB:
+		return fmt.Sprintf("%s%.1f KiB", negative, float64(abs)/KiB)
+	case abs < GiB:
+		return fmt.Sprintf("%s%.1f MiB", negative, float64(abs)/MiB)
+	case abs < TiB:
+		return fmt.Sprintf("%s%.1f GiB", negative, float64(abs)/GiB)
+	default:
+		return fmt.Sprintf("%s%.1f TiB", negative, float64(abs)/TiB)
+	}
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB and above
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		// Edge cases between units
+		{512000, "500.0 KiB"},
+		{1024000, "1000.0 KiB"},
+		{536870912, "512.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
