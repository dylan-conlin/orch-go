+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []struct {
+		name      string
+		threshold float64
+	}{
+		{"GiB/s", 1024 * 1024 * 1024},
+		{"MiB/s", 1024 * 1024},
+		{"KiB/s", 1024},
+		{"B/s", 1},
+	}
+
+	var result string
+	for _, unit := range units {
+		if bytesPerSec >= unit.threshold {
+			value := bytesPerSec / unit.threshold
+			if unit.name == "B/s" {
+				// Integer format for bytes per second
+				result = fmt.Sprintf("%d B/s", int(value))
+			} else {
+				// 1 decimal place for other units
+				result = fmt.Sprintf("%.1f %s", value, unit.name)
+			}
+			break
+		}
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary: 1 KiB/s
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB/s
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		// Exact boundary: 1 MiB/s
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB/s
+		{1572864, "1.5 MiB/s"},
+		{512 * 1024, "512.0 KiB/s"},
+		{512 * 1024 * 1024, "512.0 MiB/s"},
+		// Exact boundary: 1 GiB/s
+		{1073741824, "1.0 GiB/s"},
+		// Large values (GiB/s)
+		{512 * 1024 * 1024 * 1024, "512.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
