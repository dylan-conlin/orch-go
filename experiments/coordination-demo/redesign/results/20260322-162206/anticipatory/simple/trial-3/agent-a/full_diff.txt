+// FormatBytes formats a byte count as a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units display 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	// Define units
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+		TiB = 1024 * 1024 * 1024 * 1024
+	)
+
+	var result string
+	if bytes >= TiB {
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/TiB)
+	} else if bytes >= GiB {
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/GiB)
+	} else if bytes >= MiB {
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/MiB)
+	} else if bytes >= KiB {
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/KiB)
+	} else {
+		result = fmt.Sprintf("%d B", bytes)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+		{1649267441664, "1.5 TiB"},
+		// Larger values
+		{2048, "2.0 KiB"},
+		{2097152, "2.0 MiB"},
+		{2147483648, "2.0 GiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
