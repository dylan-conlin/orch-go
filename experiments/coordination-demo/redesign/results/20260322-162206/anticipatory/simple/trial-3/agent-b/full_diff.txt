+// FormatRate formats a transfer rate (bytes per second) as a human-readable
+// string with binary units: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+// Zero returns "0 B/s", and negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	absRate := bytesPerSec
+	if isNegative {
+		absRate = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case absRate < KiB:
+		result = fmt.Sprintf("%.0f B/s", absRate)
+	case absRate < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", absRate/KiB)
+	case absRate < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", absRate/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", absRate/GiB)
+	}
+
+	if isNegative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 * 512, "512.0 KiB/s"},
+		// MiB boundaries
+		{1048576, "1.0 MiB/s"},
+		{1048576 * 1.5, "1.5 MiB/s"},
+		{1048576 * 2, "2.0 MiB/s"},
+		{1048576 * 512, "512.0 MiB/s"},
+		// GiB
+		{1073741824, "1.0 GiB/s"},
+		{1073741824 * 1.5, "1.5 GiB/s"},
+		{1073741824 * 2, "2.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-512, "-512 B/s"},
+		// Fractional values
+		{256.5, "256 B/s"},
+		{1536.75, "1.5 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
