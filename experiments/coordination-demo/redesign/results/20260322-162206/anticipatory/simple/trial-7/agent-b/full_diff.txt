+// FormatRate formats a transfer rate in bytes per second into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	if rate == 0 {
+		return "0 B/s"
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case rate < KiB:
+		result = fmt.Sprintf("%.0f B/s", rate)
+	case rate < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", rate/KiB)
+	case rate < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", rate/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", rate/GiB)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB/s
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// Exact MiB boundary
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB/s
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Exact GiB boundary
+		{1073741824, "1.0 GiB/s"},
+		// Large GiB values
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-512, "-512 B/s"},
+		{-1536, "-1.5 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
