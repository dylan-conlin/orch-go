+// FormatBytes formats a byte count as a human-readable string.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.3 MiB", etc.
+// Uses binary units (KiB, MiB, GiB, TiB) with 1 decimal place for non-byte units.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitIndex := 0
+	value := float64(bytes)
+
+	for unitIndex < len(units)-1 && value >= 1024 {
+		value /= 1024
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// Bytes: show as integer
+		result = fmt.Sprintf("%d B", int64(value))
+	} else {
+		// Other units: show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+		{0, "0 B"},
+		{512, "512 B"},
