+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "512 B", "1.5 KiB", "2.3 MiB", "1.0 GiB", etc.
+// Shows 1 decimal place for non-byte units and handles negative values.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitSize := int64(1024)
+
+	var value float64 = float64(bytes)
+	unitIndex := 0
+
+	for value >= float64(unitSize) && unitIndex < len(units)-1 {
+		value /= float64(unitSize)
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes, show as integer
+		result = fmt.Sprintf("%d %s", int64(value), units[unitIndex])
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries and transitions
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// GiB
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Large values
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		// Between boundaries
+		{2560, "2.5 KiB"},
+		{3072, "3.0 KiB"},
+		{5242880, "5.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
