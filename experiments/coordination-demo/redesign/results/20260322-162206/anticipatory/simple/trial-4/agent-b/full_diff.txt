+// FormatRate formats a transfer rate into a human-readable string using binary units.
+// Output style: "512 B/s", "1.5 KiB/s", "2.3 MiB/s", "1.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	rate := bytesPerSec
+	if rate < 0 {
+		negative = true
+		rate = -rate
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case rate < KiB:
+		// For B/s, show as integer
+		result = fmt.Sprintf("%.0f B/s", rate)
+	case rate < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", rate/KiB)
+	case rate < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", rate/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", rate/GiB)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s boundary and values
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024*10 + 512, "10.5 KiB/s"},
+		// MiB/s boundary and values
+		{1048576, "1.0 MiB/s"},
+		{1048576 + 524288, "1.5 MiB/s"},
+		{1048576 * 2, "2.0 MiB/s"},
+		{1048576 * 100, "100.0 MiB/s"},
+		// GiB/s boundary and values
+		{1073741824, "1.0 GiB/s"},
+		{1073741824 * 2, "2.0 GiB/s"},
+		{1073741824 * 1.5, "1.5 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
