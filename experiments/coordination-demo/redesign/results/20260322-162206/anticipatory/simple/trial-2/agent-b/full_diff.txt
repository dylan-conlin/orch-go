+// FormatRate formats a transfer rate as a human-readable string with binary units.
+// Output style: "512 B/s", "1.0 KiB/s", "1.5 MiB/s", "1.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	rate := bytesPerSec
+
+	// Find the right unit
+	for i := 0; i < len(units)-1; i++ {
+		if rate < 1024 {
+			// Format based on unit
+			var result string
+			if i == 0 { // B/s - show as integer
+				result = fmt.Sprintf("%d %s", int(rate), units[i])
+			} else { // KiB/s, MiB/s, GiB/s - show with 1 decimal place
+				result = fmt.Sprintf("%.1f %s", rate, units[i])
+			}
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+		rate /= 1024
+	}
+
+	// GiB/s or larger
+	result := fmt.Sprintf("%.1f %s", rate, units[len(units)-1])
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB boundary
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048575, "1024.0 KiB/s"},
+		// MiB boundary
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{1073741823, "1024.0 MiB/s"},
+		// GiB boundary
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		{-512, "-512 B/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
