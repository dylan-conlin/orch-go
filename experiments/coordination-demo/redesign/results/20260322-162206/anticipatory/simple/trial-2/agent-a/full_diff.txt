+// FormatBytes formats a byte count as a human-readable string.
+// Uses binary units: B, KiB, MiB, GiB, TiB with 1 decimal place for non-byte units.
+// Examples: "512 B", "1.0 KiB", "1.5 MiB", "-1.0 KiB"
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	const unitSize = 1024
+	value := float64(bytes)
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for value >= unitSize && unitIndex < len(units)-1 {
+		value /= unitSize
+		unitIndex++
+	}
+
+	// Format the output
+	var result string
+	if unitIndex == 0 {
+		// For bytes, no decimal places
+		result = fmt.Sprintf("%d %s", int64(value), units[unitIndex])
+	} else {
+		// For other units, 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		// MiB boundary
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		// TiB boundary
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Large value
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		// Between boundaries
+		{2000, "2.0 KiB"},
+		{3000000, "2.9 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
