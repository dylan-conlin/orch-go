+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string with binary units: B/s, KiB/s, MiB/s, GiB/s.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const unit = 1024.0
+	if bytesPerSec < unit {
+		if negative {
+			return fmt.Sprintf("-%.0f B/s", bytesPerSec)
+		}
+		return fmt.Sprintf("%.0f B/s", bytesPerSec)
+	}
+
+	result := ""
+	if bytesPerSec < unit*unit {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/unit)
+	} else if bytesPerSec < unit*unit*unit {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/(unit*unit))
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/(unit*unit*unit))
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary values
+		{1024, "1.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		// Fractional values
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2621440, "2.5 MiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-512, "-512 B/s"},
+		// Large values
+		{5368709120, "5.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
