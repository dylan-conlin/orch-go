+// FormatRate formats a transfer rate in bytes per second as a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const unit = 1024.0
+	if bytesPerSec < unit {
+		result := fmt.Sprintf("%.0f B/s", bytesPerSec)
+		if negative {
+			return "-" + result
+		}
+		return result
+	}
+
+	units := []string{"KiB/s", "MiB/s", "GiB/s", "TiB/s"}
+	exp := 0
+	n := bytesPerSec / unit
+	for n >= unit && exp < len(units)-1 {
+		n /= unit
+		exp++
+	}
+
+	result := fmt.Sprintf("%.1f %s", n, units[exp])
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary at 1 KiB/s
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB/s
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+		// Exact boundary at 1 MiB/s
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB/s
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Exact boundary at 1 GiB/s
+		{1073741824, "1.0 GiB/s"},
+		// Fractional GiB/s
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Large values
+		{5368709120, "5.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
