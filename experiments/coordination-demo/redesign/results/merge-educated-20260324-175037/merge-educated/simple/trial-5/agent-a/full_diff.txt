+// FormatBytes formats a byte count as a human-readable string.
+// Output uses binary units: "512 B", "1.5 KiB", "2.3 MiB", etc.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var result string
+	switch {
+	case bytes >= TiB:
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/float64(TiB))
+	case bytes >= GiB:
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/float64(GiB))
+	case bytes >= MiB:
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/float64(MiB))
+	case bytes >= KiB:
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/float64(KiB))
+	default:
+		result = fmt.Sprintf("%d B", bytes)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (< 1024)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		// Large values
+		{5368709120, "5.0 GiB"},
+		{10995116277760, "10.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
