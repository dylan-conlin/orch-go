+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte units (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	var value float64
+	var unit string
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	switch {
+	case bytesPerSec >= GiB:
+		value = bytesPerSec / GiB
+		unit = "GiB/s"
+	case bytesPerSec >= MiB:
+		value = bytesPerSec / MiB
+		unit = "MiB/s"
+	case bytesPerSec >= KiB:
+		value = bytesPerSec / KiB
+		unit = "KiB/s"
+	default:
+		// For bytes, show as integer without decimal
+		if negative {
+			return fmt.Sprintf("-%.0f B/s", bytesPerSec)
+		}
+		return fmt.Sprintf("%.0f B/s", bytesPerSec)
+	}
+
+	if negative {
+		return fmt.Sprintf("-%.1f %s", value, unit)
+	}
+	return fmt.Sprintf("%.1f %s", value, unit)
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+
+		// Fractional values
+		{1536, "1.5 KiB/s"},              // 1.5 KiB
+		{2048, "2.0 KiB/s"},              // 2 KiB
+		{1572864, "1.5 MiB/s"},           // 1.5 MiB
+		{2097152, "2.0 MiB/s"},           // 2 MiB
+		{1610612736, "1.5 GiB/s"},        // 1.5 GiB
+		{2147483648, "2.0 GiB/s"},        // 2 GiB
+
+		// Values between boundaries
+		{2560, "2.5 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		{102400, "100.0 KiB/s"},
+		{10485760, "10.0 MiB/s"},
+		{104857600, "100.0 MiB/s"},
+
+		// Large values (GiB/s)
+		{5368709120, "5.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1572864, "-1.5 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
