+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.0 MiB", "1.5 GiB", "3.2 TiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const unit = 1024
+	if bytes < unit {
+		if negative {
+			return fmt.Sprintf("-%d B", bytes)
+		}
+		return fmt.Sprintf("%d B", bytes)
+	}
+
+	div, exp := int64(unit), 0
+	for n := bytes / unit; n >= unit; n /= unit {
+		div *= unit
+		exp++
+	}
+
+	units := []string{"KiB", "MiB", "GiB", "TiB"}
+	result := fmt.Sprintf("%.1f %s", float64(bytes)/float64(div), units[exp])
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+
+		// Small values (bytes)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024*10 + 512, "10.5 KiB"},
+
+		// MiB boundaries
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{1048576 * 2, "2.0 MiB"},
+		{1048576*100 + 524288, "100.5 MiB"},
+
+		// GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{1073741824 * 5, "5.0 GiB"},
+
+		// TiB boundaries
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		{1099511627776 * 3, "3.0 TiB"},
+
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},

+
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
