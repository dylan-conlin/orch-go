+// FormatRate formats a transfer rate in bytes per second into a human-readable
+// string using binary units (B/s, KiB/s, MiB/s, GiB/s).
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	if bytesPerSec < 0 {
+		return "-" + FormatRate(-bytesPerSec)
+	}
+
+	const unit = 1024.0
+	if bytesPerSec < unit {
+		return fmt.Sprintf("%.0f B/s", bytesPerSec)
+	}
+
+	if bytesPerSec < unit*unit {
+		return fmt.Sprintf("%.1f KiB/s", bytesPerSec/unit)
+	}
+
+	if bytesPerSec < unit*unit*unit {
+		return fmt.Sprintf("%.1f MiB/s", bytesPerSec/(unit*unit))
+	}
+
+	return fmt.Sprintf("%.1f GiB/s", bytesPerSec/(unit*unit*unit))
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Fractional values
+		{1536.5, "1.5 KiB/s"},
+		{2621440, "2.5 MiB/s"},
+		// Large values
+		{5368709120, "5.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
