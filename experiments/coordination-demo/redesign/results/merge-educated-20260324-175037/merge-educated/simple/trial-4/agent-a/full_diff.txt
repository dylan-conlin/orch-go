+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const unit = 1024
+	if bytes < unit {
+		if negative {
+			return fmt.Sprintf("-%d B", bytes)
+		}
+		return fmt.Sprintf("%d B", bytes)
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	exp := 0
+	n := float64(bytes)
+	for n >= unit && exp < len(units)-1 {
+		n /= unit
+		exp++
+	}
+
+	result := fmt.Sprintf("%.1f %s", n, units[exp])
+	if negative {
+		return "-" + result
+	}
+	return result
+}
