+// FormatTable formats headers and rows as an aligned text table with ANSI-aware
+// column sizing. Columns are separated by pipes (|) with padding. The header is
+// separated from data rows with a line of dashes.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on visual width (ANSI-aware)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(row) && i < len(headers); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var result strings.Builder
+
+	// Format header row
+	result.WriteString("| ")
+	for i, h := range headers {
+		stripped := StripANSI(h)
+		padding := colWidths[i] - len(stripped)
+		result.WriteString(h)
+		result.WriteString(strings.Repeat(" ", padding))
+		result.WriteString(" | ")
+	}
+	result.WriteString("\n")
+
+	// Header separator
+	result.WriteString("|")
+	for _, w := range colWidths {
+		result.WriteString(strings.Repeat("-", w+2))
+		result.WriteString("|")
+	}
+	result.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		result.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			stripped := StripANSI(cell)
+			padding := colWidths[i] - len(stripped)
+			result.WriteString(cell)
+			result.WriteString(strings.Repeat(" ", padding))
+			result.WriteString(" | ")
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}
