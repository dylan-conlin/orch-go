+// FormatTable renders headers and rows as an aligned text table. Columns are
+// auto-sized based on the widest content (including headers), with ANSI codes
+// correctly handled via StripANSI for width calculation. The header is separated
+// from data rows with a line of dashes.
+//
+// Rows with fewer columns than headers are padded with empty strings. Rows with
+// more columns than headers have extra columns ignored.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths
+	widths := make([]int, len(headers))
+	for i, h := range headers {
+		widths[i] = len([]rune(StripANSI(h)))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			w := len([]rune(StripANSI(row[i])))
+			if w > widths[i] {
+				widths[i] = w
+			}
+		}
+	}
+
+	var b strings.Builder
+
+	// Build header row
+	b.WriteString("|")
+	for i, h := range headers {
+		stripped := StripANSI(h)
+		padding := widths[i] - len([]rune(stripped))
+		b.WriteString(" ")
+		b.WriteString(h)
+		b.WriteString(strings.Repeat(" ", padding))
+		b.WriteString(" |")
+	}
+	b.WriteString("\n")
+
+	// Build separator row
+	b.WriteString("|")
+	for _, w := range widths {
+		b.WriteString(strings.Repeat("-", w+2))
+		b.WriteString("|")
+	}
+	b.WriteString("\n")
+
+	// Build data rows
+	for _, row := range rows {
+		b.WriteString("|")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			stripped := StripANSI(cell)
+			padding := widths[i] - len([]rune(stripped))
+			b.WriteString(" ")
+			b.WriteString(cell)
+			b.WriteString(strings.Repeat(" ", padding))
+			b.WriteString(" |")
+		}
+		b.WriteString("\n")
+	}
+
+	return b.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "| Name  | Age | City |\n" +
+				"|-------|-----|------|\n" +
+				"| Alice | 30  | NYC  |\n" +
+				"| Bob   | 25  | LA   |\n",
+		},
+		{
+			name:    "empty table (headers only)",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want: "| Col1 | Col2 |\n" +
+				"|------|------|\n",
+		},
+		{
+			name:    "no headers",
+			headers: []string{},
+			rows:    [][]string{{"data"}},
+			want:    "",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status", "Value"},
+			rows: [][]string{
+				{"\x1b[32mOK\x1b[0m", "100"},
+				{"\x1b[31mERROR\x1b[0m", "0"},
+			},
+			want: "| Status | Value |\n" +
+				"|--------|-------|\n" +
+				"| \x1b[32mOK\x1b[0m     | 100   |\n" +
+				"| \x1b[31mERROR\x1b[0m  | 0     |\n",
+		},
+		{
+			name:    "rows with fewer columns",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3"},
+				{"4", "5", "6"},
+			},
+			want: "| A | B | C |\n" +
+				"|---|---|---|\n" +
+				"| 1 | 2 |   |\n" +
+				"| 3 |   |   |\n" +
+				"| 4 | 5 | 6 |\n",
+		},
+		{
+			name:    "rows with more columns (extras ignored)",
+			headers: []string{"First", "Second"},
+			rows: [][]string{
+				{"A", "B", "C", "D"},
+			},
+			want: "| First | Second |\n" +
+				"|-------|--------|\n" +
+				"| A     | B      |\n",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"Apple"},
+				{"Banana"},
+			},
+			want: "| Item   |\n" +
+				"|--------|\n" +
+				"| Apple  |\n" +
+				"| Banana |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "Very Long Header Name"},
+			rows: [][]string{
+				{"X", "This is a very long piece of text"},
+			},
+			want: "| Short | Very Long Header Name             |\n" +
+				"|-------|-----------------------------------|\n" +
+				"| X     | This is a very long piece of text |\n",
+		},
+		{
+			name:    "unicode content",
+			headers: []string{"Name", "Symbol"},
+			rows: [][]string{
+				{"Check", "✓"},
+				{"Cross", "✗"},
+				{"Heart", "❤"},
+			},
+			want: "| Name  | Symbol |\n" +
+				"|-------|--------|\n" +
+				"| Check | ✓      |\n" +
+				"| Cross | ✗      |\n" +
+				"| Heart | ❤      |\n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"A", "B"},
+			rows:    nil,
+			want: "| A | B |\n" +
+				"|---|---|\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch\nGot:\n%s\nWant:\n%s", got, tt.want)
+			}
+		})
+	}
