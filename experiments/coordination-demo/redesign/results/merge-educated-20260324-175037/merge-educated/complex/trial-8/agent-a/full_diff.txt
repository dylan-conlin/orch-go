+// VisualWidth returns the visual display width of a string, ignoring ANSI escape codes.
+// It correctly handles Unicode characters by counting runes instead of bytes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but don't count toward the width calculation.
+// If the string's visual width is already equal to or greater than the target width,
+// it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"test string", 11},
+
+		// Strings with ANSI codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[31mred\x1b[0m and \x1b[32mgreen\x1b[0m", 13}, // "red and green"
+

+		{"你好", 2},      // CJK characters (2 runes)

+		{"test🎉ing", 8}, // Mixed with emoji
+
+		// Edge cases
+		{"\x1b[31m\x1b[0m", 0}, // Only ANSI codes
+		{"   ", 3},              // Only spaces
+	}

+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		name  string
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII
+		{"shorter than width", "hello", 10, "hello     "},
+		{"exact width", "hello", 5, "hello"},
+		{"longer than width", "hello world", 8, "hello world"},
+		{"empty string", "", 5, "     "},
+
+		// Strings with ANSI codes
+		{"ANSI shorter", "\x1b[31mred\x1b[0m", 10, "\x1b[31mred\x1b[0m       "},
+		{"ANSI exact", "\x1b[31mred\x1b[0m", 3, "\x1b[31mred\x1b[0m"},
+		{"ANSI longer", "\x1b[31mhello world\x1b[0m", 5, "\x1b[31mhello world\x1b[0m"},
+
+		// Unicode strings
+		{"unicode shorter", "你好", 5, "你好   "},
+		{"unicode exact", "你好", 2, "你好"},
+		{"unicode longer", "你好世界", 3, "你好世界"},
+
+		// Edge cases
+		{"zero width", "test", 0, "test"},
+		{"negative width", "test", -5, "test"},
+		{"only ANSI codes", "\x1b[31m\x1b[0m", 5, "\x1b[31m\x1b[0m     "},
+	}
+
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) [%s] = %q, want %q",
+				tt.input, tt.width, tt.name, got, tt.want)
+		}
+
+		// Verify visual width is correct (unless input was already longer)
+		gotWidth := VisualWidth(got)
+		expectedWidth := tt.width
+		if VisualWidth(tt.input) > tt.width {
+			expectedWidth = VisualWidth(tt.input)
+		}
+		if gotWidth != expectedWidth {
+			t.Errorf("PadToWidth(%q, %d) [%s] visual width = %d, want %d",
+				tt.input, tt.width, tt.name, gotWidth, expectedWidth)
+		}
+	}
+}
