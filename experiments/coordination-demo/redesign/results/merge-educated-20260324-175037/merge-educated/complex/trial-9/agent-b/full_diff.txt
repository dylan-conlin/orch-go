+// FormatTable renders headers and rows as an aligned text table.
+// Column widths are auto-sized based on the widest content in each column
+// (including the header). ANSI color codes are handled correctly using StripANSI
+// for width calculations. The header is separated from data rows with a line of dashes.
+//
+// Edge cases:
+// - Empty rows (headers only): renders just the header with separator
+// - Rows with fewer columns than headers: missing cells are treated as empty
+// - Rows with more columns than headers: extra columns are ignored
+// - Nil rows: treated as an empty table
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	if rows == nil {
+		rows = [][]string{}
+	}
+
+	// Calculate column widths based on widest content in each column
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len([]rune(StripANSI(h)))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len([]rune(StripANSI(row[i])))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Build output
+	var b strings.Builder
+
+	// Render header
+	for i, h := range headers {
+		if i > 0 {
+			b.WriteString("  ")
+		}
+		stripped := StripANSI(h)
+		padding := colWidths[i] - len([]rune(stripped))
+		b.WriteString(h)
+		b.WriteString(strings.Repeat(" ", padding))
+	}
+	b.WriteString("\n")
+
+	// Render separator
+	for i, width := range colWidths {
+		if i > 0 {
+			b.WriteString("  ")
+		}
+		b.WriteString(strings.Repeat("-", width))
+	}
+	b.WriteString("\n")
+
+	// Render rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				b.WriteString("  ")
+			}
+
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+
+			stripped := StripANSI(cell)
+			padding := colWidths[i] - len([]rune(stripped))
+			b.WriteString(cell)
+			b.WriteString(strings.Repeat(" ", padding))
+		}
+		b.WriteString("\n")
+	}
+
+	return b.String()
+}

+func TestFormatTable(t *testing.T) {
+	t.Run("basic table", func(t *testing.T) {
+		headers := []string{"Name", "Age", "City"}
+		rows := [][]string{
+			{"Alice", "30", "NYC"},
+			{"Bob", "25", "SF"},
+			{"Charlie", "35", "LA"},
+		}
+		got := FormatTable(headers, rows)
+		want := "Name     Age  City\n" +
+			"-------  ---  ----\n" +
+			"Alice    30   NYC \n" +
+			"Bob      25   SF  \n" +
+			"Charlie  35   LA  \n"

+			t.Errorf("FormatTable() =\n%q\n\nwant:\n%q", got, want)
+		}
+	})
