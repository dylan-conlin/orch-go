+// FormatTable formats headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including ANSI codes).
+// Returns a formatted string with headers separated from rows by dashes.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths (visual width, ignoring ANSI codes)
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	// Update widths based on row content
+	for _, row := range rows {
+		for i := 0; i < len(row) && i < len(headers); i++ {
+			visualWidth := len(StripANSI(row[i]))
+			if visualWidth > colWidths[i] {
+				colWidths[i] = visualWidth
+			}
+		}
+	}
+
+	var result strings.Builder
+
+	// Build header row
+	result.WriteString("| ")
+	for i, header := range headers {
+		stripped := StripANSI(header)
+		padding := colWidths[i] - len(stripped)
+		result.WriteString(header)
+		result.WriteString(strings.Repeat(" ", padding))
+		result.WriteString(" | ")
+	}
+	result.WriteString("\n")
+
+	// Build separator row
+	result.WriteString("|")
+	for _, width := range colWidths {
+		result.WriteString(strings.Repeat("-", width+2))
+		result.WriteString("|")
+	}
+	result.WriteString("\n")
+
+	// Build data rows
+	for _, row := range rows {
+		result.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			stripped := StripANSI(cell)
+			padding := colWidths[i] - len(stripped)
+			result.WriteString(cell)
+			result.WriteString(strings.Repeat(" ", padding))
+			result.WriteString(" | ")
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}

+func TestFormatTable(t *testing.T) {

+		// Check that output contains expected structure
+		if !containsSubstring(got, "| Name") && !containsSubstring(got, "| Status") && !containsSubstring(got, "| Time") {

+// Helper functions for testing

+func containsLine(s, line string) bool {
+	for _, l := range splitLines(s) {
+		if l == line {
+			return true
+		}
+	}
+	return false
+}
+
+func containsSubstring(s, substr string) bool {
+	return len(s) > 0 && len(substr) > 0 && splitContains(s, substr)
+}
+
+func splitContains(s, substr string) bool {
+	// Simple substring check
+	for i := 0; i <= len(s)-len(substr); i++ {
+		if s[i:i+len(substr)] == substr {
+			return true
+		}
+	}
+	return false
+}
+
+func splitLines(s string) []string {
+	lines := []string{}
+	current := ""
+	for _, ch := range s {
+		if ch == '\n' {
+			lines = append(lines, current)
+			current = ""
+		} else {
+			current += string(ch)
+		}
+	}
+	if current != "" {
+		lines = append(lines, current)
+	}
+	return lines
+}
