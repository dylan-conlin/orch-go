+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. The width is calculated by counting Unicode runes in the
+// string after stripping ANSI codes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is
+// already wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := strings.Repeat(" ", width-currentWidth)
+	return s + padding
+}

+		{"pad with ANSI codes", "\x1b[31mred\x1b[0m", 6, "\x1b[31mred\x1b[0m   ", 6},
+		{"pad Unicode", "你", 4, "你   ", 4},
+		{"empty string", "", 3, "   ", 3},
+		{"zero width", "test", 0, "test", 4},
+		{"pad mixed ANSI and Unicode", "\x1b[32m世\x1b[0m", 5, "\x1b[32m世\x1b[0m    ", 5},
+	}
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := PadToWidth(tt.input, tt.width)
+			if got != tt.want {
+				t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+			}
+			gotWidth := VisualWidth(got)
+			if gotWidth != tt.wantW {
+				t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotWidth, tt.wantW)
+			}
+		})
+	}
