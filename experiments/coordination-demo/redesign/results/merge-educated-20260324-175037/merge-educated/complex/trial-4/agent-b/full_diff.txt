+// FormatTable renders headers and rows as an aligned text table.
+// Column widths are auto-sized based on the widest content (including headers).
+// ANSI escape codes are handled correctly via StripANSI for width calculation.
+// Edge cases: empty rows (headers only), rows with fewer columns than headers, nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths (using rune count for proper Unicode support)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len([]rune(StripANSI(h)))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(row) && i < len(headers); i++ {
+			width := len([]rune(StripANSI(row[i])))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var b strings.Builder
+
+	// Render header
+	for i, h := range headers {
+		stripped := StripANSI(h)
+		padding := colWidths[i] - len([]rune(stripped))
+		b.WriteString(h)
+		if padding > 0 {
+			b.WriteString(strings.Repeat(" ", padding))
+		}
+		if i < len(headers)-1 {
+			b.WriteString("  ")
+		}
+	}
+	b.WriteString("\n")
+
+	// Render separator
+	for i, width := range colWidths {
+		b.WriteString(strings.Repeat("-", width))
+		if i < len(colWidths)-1 {
+			b.WriteString("  ")
+		}
+	}
+	b.WriteString("\n")
+
+	// Render rows
+	for _, row := range rows {
+		if row == nil {
+			continue
+		}
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			stripped := StripANSI(cell)
+			padding := colWidths[i] - len([]rune(stripped))
+			b.WriteString(cell)
+			if padding > 0 {
+				b.WriteString(strings.Repeat(" ", padding))
+			}
+			if i < len(headers)-1 {
+				b.WriteString("  ")
+			}
+		}
+		b.WriteString("\n")
+	}
+
+	return b.String()
+}
