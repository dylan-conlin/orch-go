+// FormatTable renders headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including header).
+// ANSI color codes are preserved but don't count toward column width.
+// Rows with fewer columns than headers are padded with empty cells.
+// Rows with more columns than headers have extra columns ignored.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var result strings.Builder
+
+	// Render header
+	for i, h := range headers {
+		if i > 0 {
+			result.WriteString(" | ")
+		}
+		result.WriteString(h)
+		padding := colWidths[i] - len(StripANSI(h))
+		result.WriteString(strings.Repeat(" ", padding))
+	}
+	result.WriteString("\n")
+
+	// Separator line
+	for i := range headers {
+		if i > 0 {
+			result.WriteString("-+-")
+		}
+		result.WriteString(strings.Repeat("-", colWidths[i]))
+	}
+	result.WriteString("\n")
+
+	// Render rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				result.WriteString(" | ")
+			}
+
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+
+			result.WriteString(cell)
+			padding := colWidths[i] - len(StripANSI(cell))
+			result.WriteString(strings.Repeat(" ", padding))
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "Name  | Age | City\n" +
+				"------+-----+-----\n" +
+				"Alice | 30  | NYC \n" +
+				"Bob   | 25  | LA  \n",
+		},
+		{
+			name:    "empty table (headers only)",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want:    "Name | Age\n" +
+				"-----+----\n",
+		},
+		{
+			name:    "ansi colored content",
+			headers: []string{"Status", "Message"},
+			rows: [][]string{
+				{"\x1b[32mOK\x1b[0m", "Success"},
+				{"\x1b[31mERROR\x1b[0m", "Failed"},
+			},
+			want: "Status | Message\n" +
+				"-------+--------\n" +
+				"\x1b[32mOK\x1b[0m     | Success\n" +
+				"\x1b[31mERROR\x1b[0m  | Failed \n",
+		},
+		{
+			name:    "mismatched columns (fewer than headers)",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows: [][]string{
+				{"A", "B", "C"},
+				{"D", "E"},
+				{"F"},
+			},
+			want: "Col1 | Col2 | Col3\n" +
+				"-----+------+-----\n" +
+				"A    | B    | C   \n" +
+				"D    | E    |     \n" +
+				"F    |      |     \n",
+		},
+		{
+			name:    "mismatched columns (more than headers)",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3", "4"},
+				{"5", "6"},
+			},
+			want: "A | B\n" +
+				"--+--\n" +
+				"1 | 2\n" +
+				"5 | 6\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"Apple"},
+				{"Banana"},
+				{"Cherry"},
+			},
+			want: "Item  \n" +
+				"------\n" +
+				"Apple \n" +
+				"Banana\n" +
+				"Cherry\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"ID", "Description"},
+			rows: [][]string{
+				{"1", "This is a very long description that spans many characters"},
+				{"2", "Short"},
+			},
+			want: "ID | Description                                               \n" +
+				"---+-----------------------------------------------------------\n" +
+				"1  | This is a very long description that spans many characters\n" +
+				"2  | Short                                                     \n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{{"A", "B"}},
+			want:    "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Col1", "Col2"},
+			rows:    nil,
+			want:    "Col1 | Col2\n" +
+				"-----+-----\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() = \n%q\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
