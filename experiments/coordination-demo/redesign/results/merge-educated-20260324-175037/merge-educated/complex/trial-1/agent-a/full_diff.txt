+// VisualWidth returns the visual display width of a string, ignoring ANSI escape codes.
+// This is useful for alignment and padding operations where colored text needs to line up.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but don't count toward width calculation.
+// If the string's visual width is already >= width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-visualWidth)
+}

+func TestPadToWidth(t *testing.T) {

+		{"pad ANSI colored", "\x1b[31mred\x1b[0m", 6, "\x1b[31mred\x1b[0m   ", 6},
+		{"ANSI exact width", "\x1b[31mhello\x1b[0m", 5, "\x1b[31mhello\x1b[0m", 5},
+		{"ANSI already wider", "\x1b[31mhello world\x1b[0m", 5, "\x1b[31mhello world\x1b[0m", 11},
+		{"pad Unicode", "café", 6, "café  ", 6},
+		{"pad CJK", "日本", 5, "日本   ", 5},
+		{"pad emoji", "👋", 3, "👋  ", 3},
+		{"zero width", "text", 0, "text", 4},
+		{"negative width", "text", -1, "text", 4},
+		{"mixed ANSI and Unicode", "\x1b[32mcafé\x1b[0m", 7, "\x1b[32mcafé\x1b[0m   ", 7},

+			}
+			gotVW := VisualWidth(got)
+			if gotVW != tt.wantVW {
+				t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotVW, tt.wantVW)
+			}
+		})
+	}
