+// FormatTable renders headers and rows as an aligned text table.
+// Columns are auto-sized based on the widest content (ANSI codes ignored).
+// A separator line of dashes appears between the header and data rows.
+//
+// Rows with fewer columns than headers are right-padded with empty strings.
+// Rows with more columns than headers have extra columns truncated.
+// Empty rows (nil or zero-length) are skipped.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on visual width (strip ANSI codes)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len([]rune(StripANSI(h)))
+	}
+
+	// Update column widths based on row data
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			cellWidth := len([]rune(StripANSI(row[i])))
+			if cellWidth > colWidths[i] {
+				colWidths[i] = cellWidth
+			}
+		}
+	}
+
+	// Debug output
+	fmt.Printf("DEBUG colWidths: %v\n", colWidths)
+
+	var b strings.Builder
+
+	// Format header row
+	for i, h := range headers {
+		if i > 0 {
+			b.WriteString(" | ")
+		}
+		b.WriteString(h)
+		// Don't pad the last column
+		if i < len(headers)-1 {
+			stripped := StripANSI(h)
+			visualWidth := len([]rune(stripped))
+			padding := colWidths[i] - visualWidth
+			b.WriteString(strings.Repeat(" ", padding))
+		}
+	}
+	b.WriteString("\n")
+
+	// Separator line
+	for i, width := range colWidths {
+		if i > 0 {
+			b.WriteString("+")
+		}
+		// Add 1 to width for spacing (except last column)
+		dashCount := width
+		if i < len(headers)-1 {
+			dashCount = width + 1
+		}
+		b.WriteString(strings.Repeat("-", dashCount))
+	}
+	b.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		if len(row) == 0 {
+			continue // Skip empty rows
+		}
+
+		// Determine the last non-empty column for this row
+		lastCol := len(row) - 1
+		if lastCol >= len(headers) {
+			lastCol = len(headers) - 1
+		}
+
+		for i := 0; i <= lastCol; i++ {
+			if i > 0 {
+				b.WriteString(" | ")
+			}
+
+			cell := row[i]
+			b.WriteString(cell)
+
+			// Don't pad the last column in the row
+			if i < lastCol {
+				stripped := StripANSI(cell)
+				visualWidth := len([]rune(stripped))
+				padding := colWidths[i] - visualWidth
+				b.WriteString(strings.Repeat(" ", padding))
+			}
+		}
+		b.WriteString("\n")
+	}
+
+	return b.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "Name  | Age | City\n" +
+				"------+-----+-----\n" +
+				"Alice | 30  | NYC\n" +
+				"Bob   | 25  | LA\n",
+		},
+		{
+			name:    "headers only (no rows)",
+			headers: []string{"ID", "Status"},
+			rows:    [][]string{},
+			want: "ID | Status\n" +
+				"---+-------\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[32mAlice\x1b[0m", "\x1b[31mFailed\x1b[0m"},
+				{"Bob", "\x1b[32mPassed\x1b[0m"},
+			},
+			want: "Name  | Status\n" +
+				"------+-------\n" +
+				"\x1b[32mAlice\x1b[0m | \x1b[31mFailed\x1b[0m\n" +
+				"Bob   | \x1b[32mPassed\x1b[0m\n",
+		},
+		{
+			name:    "mismatched columns - fewer than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "A | B | C\n" +
+				"--+---+--\n" +
+				"1 | 2 |\n" +
+				"3 | 4 | 5\n",
+		},
+		{
+			name:    "mismatched columns - more than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3", "4"},
+				{"5", "6"},
+			},
+			want: "A | B\n" +
+				"--+--\n" +
+				"1 | 2\n" +
+				"5 | 6\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"ID"},
+			rows: [][]string{
+				{"123"},
+				{"456"},
+			},
+			want: "ID\n" +
+				"---\n" +
+				"123\n" +
+				"456\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeaderName"},
+			rows: [][]string{
+				{"A", "Small"},
+				{"B", "ExtremelyLongContentHere"},
+			},
+			want: "Short | VeryLongHeaderName\n" +
+				"------+------------------------\n" +
+				"A     | Small\n" +
+				"B     | ExtremelyLongContentHere\n",
+		},
+		{
+			name:    "empty row handling",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2"},
+				{},
+				{"3", "4"},
+			},
+			want: "A | B\n" +
+				"--+--\n" +
+				"1 | 2\n" +
+				"3 | 4\n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"X"},
+			rows:    nil,
+			want: "X\n" +
+				"-\n",
+		},
+		{
+			name:    "unicode content",
+			headers: []string{"Name", "Symbol"},
+			rows: [][]string{
+				{"Test", "✓"},
+				{"Fail", "✗"},
+			},
+			want: "Name | Symbol\n" +
+				"-----+-------\n" +
+				"Test | ✓\n" +
+				"Fail | ✗\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows: [][]string{
+				{"data"},
+			},
+			want: "",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch\nGot:\n%s\nWant:\n%s", got, tt.want)
+			}
+		})
+	}
