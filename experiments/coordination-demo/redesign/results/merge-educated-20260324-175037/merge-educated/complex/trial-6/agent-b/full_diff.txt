+// FormatTable renders headers and rows as an aligned text table with pipe
+// separators. Column widths are calculated based on the widest content in
+// each column (including headers), with ANSI escape codes correctly handled
+// via StripANSI. A separator line of dashes appears between the header and
+// data rows.
+//
+// Edge cases:
+//   - If rows is empty or nil, only the header is rendered
+//   - Rows with fewer columns than headers are padded with empty cells
+//   - Rows with more columns than headers ignore the extra columns
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on header and row content
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len([]rune(StripANSI(header)))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len([]rune(StripANSI(row[i])))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var out strings.Builder
+
+	// Render header
+	out.WriteString("| ")
+	for i, header := range headers {
+		stripped := StripANSI(header)
+		padding := colWidths[i] - len([]rune(stripped))
+		out.WriteString(header)
+		out.WriteString(strings.Repeat(" ", padding))
+		out.WriteString(" | ")
+	}
+	out.WriteString("\n")
+
+	// Render separator line
+	out.WriteString("|")
+	for _, width := range colWidths {
+		out.WriteString(strings.Repeat("-", width+2))
+		out.WriteString("|")
+	}
+	out.WriteString("\n")
+
+	// Render data rows
+	for _, row := range rows {
+		out.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			stripped := StripANSI(cell)
+			padding := colWidths[i] - len([]rune(stripped))
+			out.WriteString(cell)
+			out.WriteString(strings.Repeat(" ", padding))
+			out.WriteString(" | ")
+		}
+		out.WriteString("\n")
+	}
+
+	return out.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "| Name  | Age | City | \n" +
+				"|-------|-----|------|\n" +
+				"| Alice | 30  | NYC  | \n" +
+				"| Bob   | 25  | LA   | \n",
+		},
+		{
+			name:    "empty table (headers only)",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want: "| Col1 | Col2 | \n" +
+				"|------|------|\n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Header"},
+			rows:    nil,
+			want: "| Header | \n" +
+				"|--------|\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status", "Message"},
+			rows: [][]string{
+				{"\x1b[32mOK\x1b[0m", "Success"},
+				{"\x1b[31mERROR\x1b[0m", "Failed"},
+			},
+			want: "| Status | Message | \n" +
+				"|--------|---------|" +
+				"\n| \x1b[32mOK\x1b[0m     | Success | \n" +
+				"| \x1b[31mERROR\x1b[0m  | Failed  | \n",
+		},
+		{
+			name:    "mismatched columns (fewer than headers)",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3"},
+			},
+			want: "| A | B | C | \n" +
+				"|---|---|---|\n" +
+				"| 1 | 2 |   | \n" +
+				"| 3 |   |   | \n",
+		},
+		{
+			name:    "mismatched columns (more than headers)",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3", "4"},
+			},
+			want: "| A | B | \n" +
+				"|---|---|\n" +
+				"| 1 | 2 | \n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"Apple"},
+				{"Banana"},
+			},
+			want: "| Item   | \n" +
+				"|--------|\n" +
+				"| Apple  | \n" +
+				"| Banana | \n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"ID", "Description"},
+			rows: [][]string{
+				{"1", "This is a very long description that should determine column width"},
+			},
+			want: "| ID | Description                                                        | \n" +
+				"|----|--------------------------------------------------------------------|" +
+				"\n| 1  | This is a very long description that should determine column width | \n",
+		},
+		{
+			name:    "no headers",
+			headers: []string{},
+			rows: [][]string{
+				{"data"},
+			},
+			want: "",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch\nGot:\n%s\nWant:\n%s", got, tt.want)
+			}
+		})
+	}
