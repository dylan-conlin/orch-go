+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It counts runes (not bytes) to correctly handle Unicode.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is
+// already wider than or equal to width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualLen := VisualWidth(s)
+	if visualLen >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-visualLen)
+}

+		{"Unicode pad", "你好", 5, "你好   ", 5},
+		{"emoji pad", "👋", 3, "👋  ", 3},
+		{"ANSI Unicode pad", "\x1b[31m世界\x1b[0m", 6, "\x1b[31m世界\x1b[0m    ", 6},
+		{"zero width", "text", 0, "text", 4},
+		{"width 1 longer string", "hello", 1, "hello", 5},

+				t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotVW, tt.wantVW)
