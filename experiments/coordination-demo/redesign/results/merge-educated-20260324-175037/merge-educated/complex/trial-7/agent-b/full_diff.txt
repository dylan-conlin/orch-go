+// FormatTable renders headers and rows as an aligned text table.
+// Column widths are automatically sized based on the widest content in each column.
+// ANSI color codes are preserved in the output but do not affect column width calculations.
+// Rows with fewer columns than headers are padded; rows with more columns than headers
+// have their extra columns ignored.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on visual width (ignoring ANSI codes)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			visualWidth := len(StripANSI(row[i]))
+			if visualWidth > colWidths[i] {
+				colWidths[i] = visualWidth
+			}
+		}
+	}
+
+	var result strings.Builder
+
+	// Format header row
+	result.WriteString("|")
+	for i, h := range headers {
+		result.WriteString(" ")
+		stripped := StripANSI(h)
+		padding := colWidths[i] - len(stripped)
+		result.WriteString(h)
+		result.WriteString(strings.Repeat(" ", padding))
+		result.WriteString(" |")
+	}
+	result.WriteString("\n")
+
+	// Separator line
+	result.WriteString("|")
+	for _, width := range colWidths {
+		result.WriteString("-")
+		result.WriteString(strings.Repeat("-", width))
+		result.WriteString("-|")
+	}
+	result.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		result.WriteString("|")
+		for i := 0; i < len(headers); i++ {
+			result.WriteString(" ")
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			stripped := StripANSI(cell)
+			padding := colWidths[i] - len(stripped)
+			result.WriteString(cell)
+			result.WriteString(strings.Repeat(" ", padding))
+			result.WriteString(" |")
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "| Name  | Age | City |\n" +
+				"|-------|-----|------|\n" +
+				"| Alice | 30  | NYC  |\n" +
+				"| Bob   | 25  | LA   |\n",
+		},
+		{
+			name:    "empty table (headers only)",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want: "| Name | Age |\n" +
+				"|------|-----|\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"active"},
+				{"inactive"},
+			},
+			want: "| Status   |\n" +
+				"|----------|\n" +
+				"| active   |\n" +
+				"| inactive |\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[32mAlice\x1b[0m", "\x1b[1mactive\x1b[0m"},
+				{"Bob", "\x1b[31minactive\x1b[0m"},
+			},
+			want: "| Name  | Status   |\n" +
+				"|-------|----------|\n" +
+				"| \x1b[32mAlice\x1b[0m | \x1b[1mactive\x1b[0m   |\n" +
+				"| Bob   | \x1b[31minactive\x1b[0m |\n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob"},
+			},
+			want: "| Name  | Age | City |\n" +
+				"|-------|-----|------|\n" +
+				"| Alice | 30  | NYC  |\n" +
+				"| Bob   |     |      |\n",
+		},
+		{
+			name:    "row with more columns than headers (extra ignored)",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25", "LA", "extra"},
+			},
+			want: "| Name  | Age |\n" +
+				"|-------|-----|\n" +
+				"| Alice | 30  |\n" +
+				"| Bob   | 25  |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"ID", "Description"},
+			rows: [][]string{
+				{"1", "This is a very long description that spans many characters"},
+				{"2", "Short"},
+			},
+			want: "| ID | Description                                                |\n" +
+				"|----|------------------------------------------------------------|" + "\n" +
+				"| 1  | This is a very long description that spans many characters |\n" +
+				"| 2  | Short                                                      |\n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Name", "Age"},
+			rows:    nil,
+			want: "| Name | Age |\n" +
+				"|------|-----|\n",
+		},
+		{
+			name:    "empty headers returns empty string",
+			headers: []string{},
+			rows: [][]string{
+				{"data"},
+			},
+			want: "",
+		},
+		{
+			name:    "empty cells",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "", "NYC"},
+				{"", "25", ""},
+			},
+			want: "| Name  | Age | City |\n" +
+				"|-------|-----|------|\n" +
+				"| Alice |     | NYC  |\n" +
+				"|       | 25  |      |\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch:\nGot:\n%s\nWant:\n%s", got, tt.want)
+			}
+		})
+	}
