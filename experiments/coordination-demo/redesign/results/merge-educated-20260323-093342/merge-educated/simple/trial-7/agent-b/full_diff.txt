+// FormatRate formats a transfer rate (bytes per second) into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	var result string
+	if bytesPerSec < 1024 {
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else if bytesPerSec < 1024*1024 {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/1024)
+	} else if bytesPerSec < 1024*1024*1024 {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/(1024*1024))
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/(1024*1024*1024))
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+	return result
+}
