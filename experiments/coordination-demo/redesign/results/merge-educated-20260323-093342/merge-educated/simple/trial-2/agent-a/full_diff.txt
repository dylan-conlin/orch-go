+// FormatBytes formats a byte count as a human-readable string.
+// Output style: "0 B", "512 B", "1.5 MiB", "2.0 GiB".
+// Uses binary units (1024): B, KiB, MiB, GiB, TiB.
+// Non-byte units show 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	abs := bytes
+	if negative {
+		abs = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitDivisors := []int64{1, 1024, 1024 * 1024, 1024 * 1024 * 1024, 1024 * 1024 * 1024 * 1024}
+
+	// For bytes (unit 0), use the unit directly if value < 1024
+	if abs < 1024 {
+		result := fmt.Sprintf("%d %s", abs, units[0])
+		if negative {
+			result = "-" + result
+		}
+		return result
+	}
+
+	// For larger units, find the largest unit where abs >= divisor
+	for i := len(unitDivisors) - 1; i >= 1; i-- {
+		if abs >= unitDivisors[i] {
+			value := float64(abs) / float64(unitDivisors[i])
+			result := fmt.Sprintf("%.1f %s", value, units[i])
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	// Fallback to KiB for values between 1024 and 1 MiB
+	value := float64(abs) / float64(unitDivisors[1])
+	result := fmt.Sprintf("%.1f %s", value, units[1])
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries (KiB)
+		{1024, "1.0 KiB"},
+		{2048, "2.0 KiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{1024 + 512, "1.5 KiB"},
+		// MiB boundary
+		{1024 * 1024, "1.0 MiB"},
+		{1024 * 1024 * 2, "2.0 MiB"},
+		{1024*1024 + 512*1024, "1.5 MiB"},
+		// GiB boundary
+		{1024 * 1024 * 1024, "1.0 GiB"},
+		{1024 * 1024 * 1024 * 2, "2.0 GiB"},
+		{1024*1024*1024 + 512*1024*1024, "1.5 GiB"},
+		// TiB boundary
+		{1024 * 1024 * 1024 * 1024, "1.0 TiB"},
+		{1024 * 1024 * 1024 * 1024 * 2, "2.0 TiB"},
+		{1024*1024*1024*1024 + 512*1024*1024*1024, "1.5 TiB"},
+		// Large values
+		{1024 * 1024 * 1024 * 1024 * 512, "512.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1024 * 1024, "-1.0 MiB"},
+		{-(1024*1024 + 512*1024), "-1.5 MiB"},
+		{-1024 * 1024 * 1024, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
