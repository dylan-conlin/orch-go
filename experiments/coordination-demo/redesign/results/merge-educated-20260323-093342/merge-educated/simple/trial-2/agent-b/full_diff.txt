+// FormatRate formats a transfer rate in bytes/sec as human-readable string.
+// Output style: "512 B/s", "1.5 KiB/s", "2.3 MiB/s", "1.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []struct {
+		name    string
+		divisor float64
+	}{
+		{"GiB/s", 1024 * 1024 * 1024},
+		{"MiB/s", 1024 * 1024},
+		{"KiB/s", 1024},
+		{"B/s", 1},
+	}
+
+	for _, unit := range units {
+		if bytesPerSec >= unit.divisor {
+			value := bytesPerSec / unit.divisor
+			var result string
+			if unit.divisor == 1 {
+				// For B/s, no decimal places
+				result = fmt.Sprintf("%d %s", int64(value), unit.name)
+			} else {
+				// For other units, 1 decimal place
+				result = fmt.Sprintf("%.1f %s", value, unit.name)
+			}
+			if isNegative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	// Fallback (shouldn't reach here, but handle gracefully)
+	result := fmt.Sprintf("%d B/s", int64(bytesPerSec))
+	if isNegative {
+		result = "-" + result
+	}
+	return result
+}

+	tests := []struct {
+		input float64
+		want  string

+		// Zero and small values (B/s range)

+		{1, "1 B/s"},
+		{999, "999 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},

+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Large values
