+// FormatRate formats a transfer rate into a human-readable string with binary units.
+// Output style: "512 B/s", "1.5 KiB/s", "2.0 MiB/s", "0.5 GiB/s".
+// Shows 1 decimal place for non-byte units. Handles negative values by prefixing with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	bytesPerSec = abs(bytesPerSec)
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	divisors := []float64{1, 1024, 1024 * 1024, 1024 * 1024 * 1024}
+
+	// Find the appropriate unit
+	// For GiB (largest), use >= 0.5; for smaller units use >= 1.0; for B/s, always use.
+	lastLargeUnit := len(divisors) - 1
+	for i := lastLargeUnit; i >= 0; i-- {
+		value := bytesPerSec / divisors[i]
+		threshold := 1.0
+		if i == lastLargeUnit && lastLargeUnit > 0 {
+			// Use 0.5 threshold for the largest unit (GiB)
+			threshold = 0.5
+		}
+		if value >= threshold {
+			// For B/s, show as integer. For other units, show 1 decimal place.
+			var result string
+			if i == 0 { // B/s
+				result = fmt.Sprintf("%d %s", int64(value), units[i])
+			} else {
+				result = fmt.Sprintf("%.1f %s", value, units[i])
+			}
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	// Fallback (show as B/s)
+	result := fmt.Sprintf("%d B/s", int64(bytesPerSec))
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+// abs returns the absolute value of a float64.
+func abs(x float64) float64 {
+	if x < 0 {
+		return -x
+	}
+	return x
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1024 * 1024, "1.0 MiB/s"},
+		{1024 * 1024 * 1024, "1.0 GiB/s"},
+
+		// Fractional values
+		{1536, "1.5 KiB/s"},
+		{1.5 * 1024 * 1024, "1.5 MiB/s"},
+		{2.5 * 1024 * 1024 * 1024, "2.5 GiB/s"},
+
+		// Values between boundaries
+		{2048, "2.0 KiB/s"},
+		{2 * 1024 * 1024, "2.0 MiB/s"},
+		{500 * 1024, "500.0 KiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1.5 * 1024, "-1.5 KiB/s"},
+		{-1024 * 1024, "-1.0 MiB/s"},
+
+		// Large values
+		{10 * 1024 * 1024 * 1024, "10.0 GiB/s"},
+		{0.5 * 1024 * 1024 * 1024, "0.5 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
