+// FormatRate formats a transfer rate (bytes per second) into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	value := bytesPerSec
+	if negative {
+		value = -value
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	threshold := 1024.0
+
+	for i, unit := range units {
+		if i == len(units)-1 || value < threshold {
+			// For B/s, show as integer. For others, show 1 decimal place.
+			if i == 0 {
+				result := fmt.Sprintf("%d %s", int(value), unit)
+				if negative {
+					result = "-" + result
+				}
+				return result
+			}
+			result := fmt.Sprintf("%.1f %s", value, unit)
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+		value /= threshold
+	}
+
+	// Should never reach here, but handle gracefully
+	result := fmt.Sprintf("%.1f %s", value, units[len(units)-1])
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Large values
+		{1024 * 1024 * 1024 * 10, "10.0 GiB/s"},
+		{1024 * 1024 * 1024 * 100, "100.0 GiB/s"},
+		// Fractional values
+		{2560, "2.5 KiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{2621440, "2.5 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		{100 * 1024 * 1024, "100.0 MiB/s"},
+		// Edge cases around boundaries
+		{1023, "1023 B/s"},
+		{1025, "1.0 KiB/s"},
+		{1047552, "1023.0 KiB/s"},
+		{1049600, "1.0 MiB/s"},
+		// Negative fractional
+		{-1536, "-1.5 KiB/s"},
+		{-1572864, "-1.5 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
