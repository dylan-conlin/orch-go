+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "512 B", "1.0 KiB", "1.5 MiB", "2.3 GiB", "1.1 TiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	b := bytes
+	if negative {
+		b = -b
+	}
+
+	var result string
+	if b < KiB {
+		result = fmt.Sprintf("%d B", b)
+	} else if b < MiB {
+		result = fmt.Sprintf("%.1f KiB", float64(b)/float64(KiB))
+	} else if b < GiB {
+		result = fmt.Sprintf("%.1f MiB", float64(b)/float64(MiB))
+	} else if b < TiB {
+		result = fmt.Sprintf("%.1f GiB", float64(b)/float64(GiB))
+	} else {
+		result = fmt.Sprintf("%.1f TiB", float64(b)/float64(TiB))
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1638, "1.6 KiB"},
+		// Exact MiB boundary
+		{1048576, "1.0 MiB"},
+		// Between MiB boundaries
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Exact GiB boundary
+		{1073741824, "1.0 GiB"},
+		// Between GiB boundaries
+		{1610612736, "1.5 GiB"},
+		// Exact TiB boundary
+		{1099511627776, "1.0 TiB"},
+		// Between TiB boundaries
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		// Large values
+		{10737418240, "10.0 GiB"},
+		{10995116277760, "10.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
