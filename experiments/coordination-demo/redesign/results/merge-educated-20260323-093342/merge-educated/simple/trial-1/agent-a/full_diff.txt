+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Zero returns "0 B". Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	absBytes := bytes
+	if absBytes < 0 {
+		absBytes = -absBytes
+	}
+
+	var value float64
+	var unit string
+
+	switch {
+	case absBytes < KiB:
+		return fmt.Sprintf("%d B", bytes)
+	case absBytes < MiB:
+		value = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	case absBytes < GiB:
+		value = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes < TiB:
+		value = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	if bytes < 0 {
+		return fmt.Sprintf("-%.1f %s", value, unit)
+	}
+	return fmt.Sprintf("%.1f %s", value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small byte values
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+		// Larger KiB values
+		{10240, "10.0 KiB"},
+		{102400, "100.0 KiB"},
+		// Exact MiB boundary
+		{1048576, "1.0 MiB"},
+		// Between MiB values
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Larger MiB values
+		{10485760, "10.0 MiB"},
+		{104857600, "100.0 MiB"},
+		// Exact GiB boundary
+		{1073741824, "1.0 GiB"},
+		// Between GiB values
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Exact TiB boundary
+		{1099511627776, "1.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		// Large negative
+		{-10995116277760, "-10.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
