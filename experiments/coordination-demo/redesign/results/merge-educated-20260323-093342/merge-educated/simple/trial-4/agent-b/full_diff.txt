+// FormatRate formats a transfer rate into a human-readable string using binary
+// units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s. Values are shown with
+// 1 decimal place for non-byte-per-second units. Negative values are prefixed
+// with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	value := bytesPerSec
+
+	for i := 0; i < len(units)-1; i++ {
+		if value < 1024 {
+			if i == 0 {
+				// For bytes/sec, show as integer
+				result := fmt.Sprintf("%.0f %s", value, units[i])
+				if negative {
+					result = "-" + result
+				}
+				return result
+			}
+			// For other units, show with 1 decimal place
+			result := fmt.Sprintf("%.1f %s", value, units[i])
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+		value /= 1024
+	}
+
+	// GiB/s - show with 1 decimal place
+	result := fmt.Sprintf("%.1f %s", value, units[len(units)-1])
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s boundary and values
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 * 512, "512.0 KiB/s"},
+		{1024*1024 - 1, "1024.0 KiB/s"},
+		// MiB/s boundary and values
+		{1024 * 1024, "1.0 MiB/s"},
+		{1024*1024 + 1024*256, "1.2 MiB/s"},
+		{1024*1024*2 + 1024*512, "2.5 MiB/s"},
+		{1024*1024*512, "512.0 MiB/s"},
+		{1024*1024*1024 - 1, "1024.0 MiB/s"},
+		// GiB/s boundary and values
+		{1024 * 1024 * 1024, "1.0 GiB/s"},
+		{1024*1024*1024 + 1024*1024*256, "1.2 GiB/s"},
+		{1024*1024*1024*2 + 1024*1024*512, "2.5 GiB/s"},
+		{1024 * 1024 * 1024 * 10, "10.0 GiB/s"},
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1024 * 1024 * 1024, "-1.0 GiB/s"},
+		// Fractional boundary cases
+		{1024 * 1.5, "1.5 KiB/s"},
+		{1024 * 1024 * 1.5, "1.5 MiB/s"},
+		{1024 * 1024 * 1024 * 1.5, "1.5 GiB/s"},
+		{1024 * 0.5, "512 B/s"}, // Still under 1 KiB, so shown as B/s without decimals
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
