+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", "1.0 GiB", etc.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	sizes := []int64{1, 1024, 1024 * 1024, 1024 * 1024 * 1024, 1024 * 1024 * 1024 * 1024}
+
+	// For bytes unit, always show integer
+	if bytes < 1024 {
+		result := fmt.Sprintf("%d B", bytes)
+		if negative {
+			return "-" + result
+		}
+		return result
+	}
+
+	// For other units, show with 1 decimal place
+	for i := 1; i < len(units); i++ {
+		size := sizes[i]
+		nextSize := int64(0)
+		if i < len(sizes)-1 {
+			nextSize = sizes[i+1]
+		} else {
+			nextSize = size * 1024 // For TiB, assume we can go higher
+		}
+
+		if bytes < nextSize {
+			value := float64(bytes) / float64(size)
+			result := fmt.Sprintf("%.1f %s", value, units[i])
+			if negative {
+				return "-" + result
+			}
+			return result
+		}
+	}
+
+	// Fallback for TiB
+	value := float64(bytes) / float64(sizes[4])
+	result := fmt.Sprintf("%.1f TiB", value)
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+		// TiB boundary
