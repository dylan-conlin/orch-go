+// FormatRate formats a byte rate (bytes per second) as a human-readable string
+// using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var formatted string
+
+	switch {
+	case bytesPerSec < KiB:
+		formatted = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	case bytesPerSec < MiB:
+		formatted = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	case bytesPerSec < GiB:
+		formatted = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	default:
+		formatted = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if negative {
+		return "-" + formatted
+	}
+	return formatted
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary: 1 KiB/s
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB/s
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// Boundary: 1 MiB/s
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB/s
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Boundary: 1 GiB/s
+		{1073741824, "1.0 GiB/s"},
+		// Large GiB/s
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%.0f) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
