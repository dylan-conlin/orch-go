+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string with binary units: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	abs := bytesPerSec
+	if negative {
+		abs = -abs
+	}
+
+	const (
+		B   = 1.0
+		KiB = 1024.0
+		MiB = 1024.0 * KiB
+		GiB = 1024.0 * MiB
+	)
+
+	var result string
+	switch {
+	case abs < KiB:
+		result = fmt.Sprintf("%.0f B/s", abs)
+	case abs < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", abs/KiB)
+	case abs < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", abs/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", abs/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 * 1024, "1.0 MiB/s"},
+		{1.5 * 1024 * 1024, "1.5 MiB/s"},
+		{10 * 1024 * 1024, "10.0 MiB/s"},
+		// GiB/s
+		{1024 * 1024 * 1024, "1.0 GiB/s"},
+		{1.5 * 1024 * 1024 * 1024, "1.5 GiB/s"},
+		{10 * 1024 * 1024 * 1024, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1024 * 1024 * 1024, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
