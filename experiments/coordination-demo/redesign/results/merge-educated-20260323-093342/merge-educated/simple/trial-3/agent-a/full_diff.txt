+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output examples: "512 B", "1.0 KiB", "1.5 MiB", "2.3 GiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		B  = 1
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+		TiB = 1024 * 1024 * 1024 * 1024
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes < KiB:
+		if negative {
+			return fmt.Sprintf("-%d B", bytes)
+		}
+		return fmt.Sprintf("%d B", bytes)
+	case bytes < MiB:
+		value = float64(bytes) / float64(KiB)
+		unit = "KiB"
+	case bytes < GiB:
+		value = float64(bytes) / float64(MiB)
+		unit = "MiB"
+	case bytes < TiB:
+		value = float64(bytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(bytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	if negative {
+		return fmt.Sprintf("-%.1f %s", value, unit)
+	}
+	return fmt.Sprintf("%.1f %s", value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small byte values
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact KiB boundaries
+		{1024, "1.0 KiB"},
+		{2048, "2.0 KiB"},
+		// Fractional KiB
+		{1536, "1.5 KiB"},
+		{1331, "1.3 KiB"},
+		// Exact MiB boundaries
+		{1048576, "1.0 MiB"},
+		{2097152, "2.0 MiB"},
+		// Fractional MiB
+		{1572864, "1.5 MiB"},
+		// Exact GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Fractional GiB
+		{1610612736, "1.5 GiB"},
+		// TiB
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
