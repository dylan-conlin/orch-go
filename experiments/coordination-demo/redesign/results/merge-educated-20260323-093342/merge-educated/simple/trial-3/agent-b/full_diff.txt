+// FormatRate formats a transfer rate into a human-readable string using binary
+// units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units.
+// Handles zero (returns "0 B/s") and negative values (prefixed with "-").
+func FormatRate(bytesPerSec float64) string {
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var prefix string
+	if negative {
+		prefix = "-"
+	}
+
+	if rate == 0 {
+		return "0 B/s"
+	}
+
+	if rate < KiB {
+		return fmt.Sprintf("%s%d B/s", prefix, int64(rate))
+	}
+
+	if rate < MiB {
+		return fmt.Sprintf("%s%.1f KiB/s", prefix, rate/KiB)
+	}
+
+	if rate < GiB {
+		return fmt.Sprintf("%s%.1f MiB/s", prefix, rate/MiB)
+	}
+
+	return fmt.Sprintf("%s%.1f GiB/s", prefix, rate/GiB)
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries and fractional values
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+
+		// MiB boundary
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{3145728, "3.0 MiB/s"},
+
+		// GiB boundary and large values
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+
+		// Edge cases with fractional bytes per second
+		{100.5, "100 B/s"},
+		{1024.5, "1.0 KiB/s"},
+		{1536.7, "1.5 KiB/s"},
+		{1048576.2, "1.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
