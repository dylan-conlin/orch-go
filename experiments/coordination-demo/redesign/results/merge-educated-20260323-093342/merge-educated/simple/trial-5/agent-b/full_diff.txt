+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units. Output style: "512 B/s", "1.5 KiB/s", "1.0 MiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	threshold := 1024.0
+
+	for i, unit := range units {
+		if bytesPerSec < threshold || i == len(units)-1 {
+			var result string
+			if i == 0 {
+				// For B/s, show as integer
+				result = fmt.Sprintf("%d %s", int64(bytesPerSec), unit)
+			} else {
+				// For larger units, show 1 decimal place
+				result = fmt.Sprintf("%.1f %s", bytesPerSec, unit)
+			}
+			if isNegative {
+				result = "-" + result
+			}
+			return result
+		}
+		bytesPerSec /= threshold
+	}
+
+	return "0 B/s"
+}
