+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Examples: "0 B", "512 B", "1.5 KiB", "1.0 MiB".
+func FormatBytes(bytes int64) string {
+	const (
+		unit = 1024
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	value := float64(bytes)
+
+	for i, u := range units {
+		if i == 0 {
+			// For bytes, show as integer
+			if value < unit {
+				if negative {
+					return fmt.Sprintf("-%d %s", int64(value), u)
+				}
+				return fmt.Sprintf("%d %s", int64(value), u)
+			}
+		} else {
+			// For larger units, show 1 decimal place
+			if value < unit {
+				if negative {
+					return fmt.Sprintf("-%.1f %s", value, u)
+				}
+				return fmt.Sprintf("%.1f %s", value, u)
+			}
+		}
+		value /= unit
+	}
+
+	// For values >= 1024 TiB, return as TiB
+	if negative {
+		return fmt.Sprintf("-%.1f TiB", value*unit)
+	}
+	return fmt.Sprintf("%.1f TiB", value*unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		// MiB boundaries
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{10485760, "10.0 MiB"},
+		// GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB boundaries
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-512, "-512 B"},
+		// Large values
+		{1099511627776 * 10, "10.0 TiB"},
+		{1099511627776 * 100, "100.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
