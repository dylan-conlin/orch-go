+// FormatTable renders headers and rows as an aligned text table with
+// auto-sized columns. Column widths are calculated using visual width
+// (ANSI codes excluded). Headers are separated from data by a dash line.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and all rows
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Check all rows to find max width for each column
+	for _, row := range rows {
+		if row == nil {
+			continue
+		}
+		for i, cell := range row {
+			if i < len(headers) {
+				width := len(StripANSI(cell))
+				if width > colWidths[i] {
+					colWidths[i] = width
+				}
+			}
+		}
+	}
+
+	// Build header row
+	var result strings.Builder
+	result.WriteString("|")
+	for i, h := range headers {
+		result.WriteString(" ")
+		result.WriteString(padString(h, colWidths[i]))
+		result.WriteString(" ")
+		result.WriteString("|")
+	}
+	result.WriteString("\n")
+
+	// Build separator line
+	result.WriteString("|")
+	for _, w := range colWidths {
+		result.WriteString(strings.Repeat("-", w+2))
+		result.WriteString("|")
+	}
+	result.WriteString("\n")
+
+	// Build data rows
+	for _, row := range rows {
+		if row == nil {
+			continue
+		}
+		result.WriteString("|")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			result.WriteString(" ")
+			result.WriteString(padString(cell, colWidths[i]))
+			result.WriteString(" ")
+			result.WriteString("|")
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}
+
+// padString pads s to the target width using spaces. ANSI codes are
+// preserved but don't count toward the visual width.
+func padString(s string, width int) string {
+	visualLen := len(StripANSI(s))
+	if visualLen >= width {
+		return s
+	}
+	padding := width - visualLen
+	return s + strings.Repeat(" ", padding)
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "New York"},
+				{"Bob", "25", "LA"},
+				{"Charlie", "35", "Chicago"},
+			},
+			want: "| Name    | Age | City     |\n|---------|-----|----------|\n| Alice   | 30  | New York |\n| Bob     | 25  | LA       |\n| Charlie | 35  | Chicago  |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "| Col1 | Col2 |\n|------|------|\n",
+		},
+		{
+			name:    "headers only with nil rows",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{nil, nil},
+			want:    "| Col1 | Col2 |\n|------|------|\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status", "Value"},
+			rows: [][]string{
+				{"\x1b[32mPASS\x1b[0m", "100"},
+				{"\x1b[31mFAIL\x1b[0m", "50"},
+			},
+			want: "| Status | Value |\n|--------|-------|\n| \x1b[32mPASS\x1b[0m   | 100   |\n| \x1b[31mFAIL\x1b[0m   | 50    |\n",
+		},
+		{
+			name:    "mismatched column counts",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5", "6"},
+				{"7"},
+			},
+			want: "| A | B | C |\n|---|---|---|\n| 1 | 2 |   |\n| 3 | 4 | 5 |\n| 7 |   |   |\n",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"ID"},
+			rows: [][]string{
+				{"12345"},
+				{"67890"},
+			},
+			want: "| ID    |\n|-------|\n| 12345 |\n| 67890 |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "Long"},
+			rows: [][]string{
+				{"a", "This is a very long string that should determine column width"},
+				{"bb", "short"},
+			},
+			want: "| Short | Long                                                          |\n|-------|---------------------------------------------------------------|\n| a     | This is a very long string that should determine column width |\n| bb    | short                                                         |\n",
+		},
+		{
+			name:    "empty table",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() = %q, want %q", got, tt.want)
+			}
+		})
+	}
+}
