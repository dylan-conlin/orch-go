+// VisualWidth returns the visual display width of a string, ignoring ANSI escape codes.
+// Unicode characters (including emoji and CJK) are counted as single runes.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is already
+// wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"hello world", 11},
+		{"", 0},
+		{"a", 1},
+
+		// ANSI color codes (should be stripped)
+		{"\x1b[31mred text\x1b[0m", 8},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m", 6},
+
+		// Unicode: CJK characters (each rune = 1 width in this implementation)
+		{"你好", 2},
+		{"中文", 2},
+		{"hello中文world", 12},
+
+		// Emoji (counted as single rune)
+		{"hello 👋", 7},
+		{"hello👋", 6},
+		{"🎉🎊", 2},
+
+		// Mixed ANSI + Unicode
+		{"\x1b[36m你好\x1b[0m", 2},
+
+		// Edge cases
+		{"   ", 3},
+		{"\x1b[0m", 0}, // Only ANSI codes
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain strings - pad to larger width
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		{"", 3, "   "},
+
+		// Exact width - no padding needed
+		{"exact", 5, "exact"},
+		{"hello", 5, "hello"},
+
+		// Already wider - return unchanged
+		{"hello world", 5, "hello world"},
+		{"test", 2, "test"},
+
+		// ANSI codes - preserved, don't count toward width
+		{"\x1b[31mred\x1b[0m", 6, "\x1b[31mred\x1b[0m   "},
+		{"\x1b[1;32mbold\x1b[0m", 8, "\x1b[1;32mbold\x1b[0m    "},
+
+		// Unicode strings
+		{"你好", 4, "你好  "},
+		{"中", 3, "中  "},
+		{"hello👋", 10, "hello👋    "},
+
+		// ANSI + Unicode
+		{"\x1b[33m你好\x1b[0m", 5, "\x1b[33m你好\x1b[0m   "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+
+		// Verify visual width of result is correct (unless already wider)
+		if VisualWidth(tt.input) <= tt.width {
+			resultWidth := VisualWidth(got)
+			if resultWidth != tt.width {
+				t.Errorf("PadToWidth(%q, %d) result has visual width %d, want %d", tt.input, tt.width, resultWidth, tt.width)
+			}
+		}
+	}
+}
