+// VisualWidth returns the visual display width of a string, excluding ANSI
+// escape codes. It correctly handles Unicode characters (runes).
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward width. If the string is
+// already wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+		// ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow text here\x1b[0m", 16},
+		// Unicode (runes)
+		{"café", 4},
+		{"你好", 2},
+		{"👍", 1},
+		// ANSI + Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1;32m👍test\x1b[0m", 5},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII padding
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		{"", 5, "     "},
+		// Already at width
+		{"hello", 5, "hello"},
+		// Already wider than target
+		{"hello world", 5, "hello world"},
+		// ANSI codes with padding
+		{"\x1b[31mred\x1b[0m", 6, "\x1b[31mred\x1b[0m   "},
+		{"\x1b[1;32mbold\x1b[0m", 8, "\x1b[1;32mbold\x1b[0m    "},
+		// Unicode with padding
+		{"café", 6, "café  "},
+		{"你好", 4, "你好  "},
+		// Unicode + ANSI
+		{"\x1b[31m你好\x1b[0m", 4, "\x1b[31m你好\x1b[0m  "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+	}
+}
