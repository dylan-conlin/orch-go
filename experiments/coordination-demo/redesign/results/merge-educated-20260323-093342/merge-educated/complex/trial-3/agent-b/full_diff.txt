+// FormatTable formats headers and rows as an aligned text table.
+// Column widths are auto-sized based on the widest content in each column.
+// Uses StripANSI to correctly calculate widths of ANSI-colored content.
+// Handles edge cases: empty rows (headers only), rows with fewer columns than headers, nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths (max of header width and any row width)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(headers) {
+				w := len(StripANSI(cell))
+				if w > colWidths[i] {
+					colWidths[i] = w
+				}
+			}
+		}
+	}
+
+	// Helper to pad a string to a target visual width while preserving ANSI codes
+	padCell := func(s string, width int) string {
+		visualWidth := len(StripANSI(s))
+		if visualWidth >= width {
+			return s
+		}
+		return s + strings.Repeat(" ", width-visualWidth)
+	}
+
+	// Format the table
+	var buf strings.Builder
+
+	// Header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(padCell(h, colWidths[i]))
+	}
+	buf.WriteString("\n")
+
+	// Separator line
+	for i, w := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", w))
+	}
+	buf.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(padCell(cell, colWidths[i]))
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+		headers []string
+		rows    [][]string
+		want    string
+	}{

+		{
+			name:    "headers only (empty rows)",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows:    [][]string{},

+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{

+				{"banana"},
+			},
+			want: "Item  \n------\napple \nbanana\n",
+		},
+		{
+			name:    "rows with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{

+				{"x", "y", "z"},

+			name:    "wide content",

+			want: "Name   | Status\n-------+-------\n\x1b[32mActive\x1b[0m | \x1b[31mRed\x1b[0m   \nPlain  | Text  \n",
