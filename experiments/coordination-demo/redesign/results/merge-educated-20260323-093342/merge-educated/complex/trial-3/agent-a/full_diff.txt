+// VisualWidth returns the visual display width of a string, ignoring ANSI escape codes.
+// It counts runes, not bytes, to correctly handle multi-byte characters like emoji and CJK.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward the width.
+// If the string is already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"hello world", 11},
+		{"", 0},
+		{"a", 1},
+		// ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow text here\x1b[0m", 16},
+		// Unicode: CJK characters
+		{"你好", 2},
+		{"hello你好", 7},
+		// Emoji
+		{"👋", 1},
+		{"hello 👋 world", 13},
+		// Mixed ANSI and Unicode
+		{"\x1b[36m你好\x1b[0m", 2},
+		{"\x1b[1;31m👋\x1b[0m", 1},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII padding
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		// Already at width
+		{"hello", 5, "hello"},
+		// Already wider than width
+		{"hello world", 5, "hello world"},
+		// Empty string
+		{"", 5, "     "},
+		// ANSI codes preserved
+		{"\x1b[31mred\x1b[0m", 6, "\x1b[31mred\x1b[0m   "},
+		{"\x1b[1;32mbold green\x1b[0m", 12, "\x1b[1;32mbold green\x1b[0m  "},
+		// Unicode (CJK)
+		{"你好", 4, "你好  "},
+		{"你", 3, "你  "},
+		// Emoji
+		{"👋", 3, "👋  "},
+		{"hello 👋", 10, "hello 👋   "},
+		// Mixed ANSI and Unicode
+		{"\x1b[36m你好\x1b[0m", 4, "\x1b[36m你好\x1b[0m  "},
+		{"\x1b[1;31m👋\x1b[0m", 3, "\x1b[1;31m👋\x1b[0m  "},
+		// Width of 0
+		{"hello", 0, "hello"},
+		// Negative width (treated as 0, string returned unchanged)
+		{"test", -5, "test"},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width of result is at least the requested width
+		gotWidth := VisualWidth(got)
+		if gotWidth < tt.width && gotWidth < VisualWidth(tt.input) {
+			t.Errorf("PadToWidth(%q, %d) result width %d < target width %d", tt.input, tt.width, gotWidth, tt.width)
+		}
+	}
+}
