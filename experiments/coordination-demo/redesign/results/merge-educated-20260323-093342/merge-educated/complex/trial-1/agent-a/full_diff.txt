+// VisualWidth returns the visual display width of a string, counting runes
+// and ignoring ANSI escape codes. Useful for alignment in terminals.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward the visual width.
+// If the string is already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Empty string padding
+		{"", 0, ""},
+		{"", 5, "     "},
+		// Plain ASCII padding
+		{"hi", 5, "hi   "},
+		{"hello", 5, "hello"},
+		{"hello", 3, "hello"}, // Already wider, no change
+		// ANSI codes preserved, spaces added correctly
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "}, // 3 chars + 2 spaces
+		{"\x1b[1;32mbold\x1b[0m", 6, "\x1b[1;32mbold\x1b[0m  "}, // 4 chars + 2 spaces
+		// Unicode with ANSI codes
+		{"\x1b[31m你好\x1b[0m", 4, "\x1b[31m你好\x1b[0m  "}, // 2 chars + 2 spaces
+		// Already at target width
+		{"hello", 5, "hello"},
+		// Wider than target width
+		{"hello world", 5, "hello world"}, // No padding
+		{"hello", 2, "hello"}, // No padding
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width matches or exceeds target (if not already wider)
+		visualWidth := VisualWidth(got)
+		if VisualWidth(tt.input) < tt.width && visualWidth != tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, visualWidth, tt.width)
+		}
+	}
+}
