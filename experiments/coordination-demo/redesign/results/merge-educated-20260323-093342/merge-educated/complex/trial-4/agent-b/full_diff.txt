+// FormatTable renders headers and rows as an aligned text table.
+// It auto-sizes each column based on the widest content in that column
+// (including the header). The header is separated from data rows with a
+// line of dashes. ANSI color codes are preserved in the output but do not
+// affect column width calculations. Handles edge cases: empty rows, rows
+// with fewer columns than headers, and nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and rows
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Build the table
+	var buf strings.Builder
+	padding := 1
+
+	// Format header row
+	buf.WriteString("| ")
+	for i, header := range headers {
+		width := colWidths[i]
+		paddedHeader := header + strings.Repeat(" ", width-len(StripANSI(header)))
+		buf.WriteString(paddedHeader)
+		if i < len(headers)-1 {
+			buf.WriteString(" | ")
+		} else {
+			buf.WriteString(" |")
+		}
+	}
+	buf.WriteString("\n")
+
+	// Format separator line
+	buf.WriteString("|")
+	for i := range headers {
+		width := colWidths[i]
+		buf.WriteString(strings.Repeat("-", width+2*padding))
+		buf.WriteString("|")
+	}
+	buf.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		buf.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			width := colWidths[i]
+			visualWidth := len(StripANSI(cell))
+			paddedCell := cell + strings.Repeat(" ", width-visualWidth)
+			buf.WriteString(paddedCell)
+			if i < len(headers)-1 {
+				buf.WriteString(" | ")
+			} else {
+				buf.WriteString(" |")
+			}
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}
