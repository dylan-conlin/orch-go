+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It correctly handles Unicode by counting runes, not bytes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the width. If the
+// string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+		// Strings with ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m text", 11},
+		// Unicode strings (emoji, CJK)
+		{"😀", 1},
+		{"你好", 2},
+		{"こんにちは", 5},
+		{"hello 😀 world", 13},
+		{"\x1b[31m你好\x1b[0m", 2},
+		// Complex mixed content
+		{"\x1b[1;32m你好\x1b[0m \x1b[33m🌍\x1b[0m", 4},
+		// Edge cases
+		{"   ", 3},
+		{"\t", 1},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		wantLen int // visual width of result
+	}{
+		// Plain ASCII strings - padding needed
+		{"hello", 10, "hello     ", 10},
+		{"hi", 5, "hi   ", 5},
+		{"", 3, "   ", 3},
+		// Plain ASCII strings - no padding needed
+		{"hello", 5, "hello", 5},
+		{"hello", 3, "hello", 5},
+		{"hello world", 5, "hello world", 11},
+		// ANSI codes preserved, padding based on visual width
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  ", 5},
+		{"\x1b[31mred\x1b[0m", 3, "\x1b[31mred\x1b[0m", 3},
+		{"\x1b[31mred\x1b[0m", 2, "\x1b[31mred\x1b[0m", 3},
+		// Unicode with padding
+		{"你好", 4, "你好  ", 4},
+		{"😀", 3, "😀  ", 3},
+		// Mixed ANSI + Unicode
+		{"\x1b[32m你好\x1b[0m", 5, "\x1b[32m你好\x1b[0m   ", 5},
+		{"\x1b[32m你好\x1b[0m", 2, "\x1b[32m你好\x1b[0m", 2},
+		// Already at target width
+		{"test", 4, "test", 4},
+		// Wider than target width
+		{"toolong", 3, "toolong", 7},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		visualWidth := VisualWidth(got)
+		if visualWidth < tt.wantLen {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want >= %d", tt.input, tt.width, visualWidth, tt.wantLen)
+		}
+	}
+}
