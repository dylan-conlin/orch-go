+// VisualWidth returns the visual display width of a string, correctly handling
+// ANSI escape codes (which are stripped before counting) and multi-byte Unicode
+// characters. The result counts runes, not bytes.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the width calculation.
+// If the string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := strings.Repeat(" ", width-currentWidth)
+	return s + padding
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"test string", 11},
+		// ANSI color codes (should be stripped)
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m and \x1b[34mblue\x1b[0m", 15},
+		// Unicode strings (CJK characters)
+		{"你好", 2},
+		{"こんにちは", 5},
+		{"한글", 2},
+		// Emoji
+		{"😀", 1},
+		{"😀😁", 2},
+		// Mixed content
+		{"hello 世界", 8},
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[32m😀\x1b[0m", 1},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		wantWd int
+	}{
+		// Plain ASCII - needs padding
+		{"hello", 10, "hello     ", 10},
+		// Already at width
+		{"hello", 5, "hello", 5},
+		// Already wider than target
+		{"hello", 3, "hello", 5},
+		// Empty string
+		{"", 5, "     ", 5},
+		// With ANSI codes - padding applied, codes preserved
+		{"\x1b[31mred\x1b[0m", 6, "\x1b[31mred\x1b[0m   ", 6},
+		{"\x1b[1;32mbold green\x1b[0m", 15, "\x1b[1;32mbold green\x1b[0m     ", 15},
+		// Unicode - padding to reach target visual width
+		{"你好", 5, "你好   ", 5},
+		{"😀", 3, "😀  ", 3},
+		// Mixed ANSI and Unicode
+		{"\x1b[31m你好\x1b[0m", 5, "\x1b[31m你好\x1b[0m   ", 5},
+		// No padding needed
+		{"test", 4, "test", 4},
+		// Single character
+		{"a", 3, "a  ", 3},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width matches target (or exceeds if already wide)
+		gotWd := VisualWidth(got)
+		if gotWd != tt.wantWd {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotWd, tt.wantWd)
+		}
+	}
+}
