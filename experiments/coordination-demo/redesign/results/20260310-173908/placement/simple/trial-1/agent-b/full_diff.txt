+// FormatRate formats a transfer rate as a human-readable string.
+// Output uses binary units with "/s" suffix: "512 B/s", "1.5 KiB/s", "1.0 MiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	if bytesPerSec < KiB {
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else if bytesPerSec < MiB {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	} else if bytesPerSec < GiB {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+
+	return result
+}
