+// FormatBytes formats a byte count as a human-readable string using binary units
+// (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	units := []struct {
+		size int64
+		name string
+	}{
+		{TiB, "TiB"},
+		{GiB, "GiB"},
+		{MiB, "MiB"},
+		{KiB, "KiB"},
+	}
+
+	for _, u := range units {
+		if bytes >= u.size {
+			value := float64(bytes) / float64(u.size)
+			result := fmt.Sprintf("%.1f %s", value, u.name)
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	// Bytes (no decimal)
+	result := fmt.Sprintf("%d B", bytes)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1024 * 1024, "1.0 MiB"},
+		{1024 * 1024 * 1024, "1.0 GiB"},
+		{1024 * 1024 * 1024 * 1024, "1.0 TiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{1024*1024 + 512*1024, "1.5 MiB"},
+		{1024*1024*1024 + 512*1024*1024, "1.5 GiB"},
+		// Large values
+		{2 * 1024 * 1024 * 1024, "2.0 GiB"},
+		{5 * 1024 * 1024 * 1024 * 1024, "5.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
