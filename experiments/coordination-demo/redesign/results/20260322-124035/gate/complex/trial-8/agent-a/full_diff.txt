+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It correctly handles Unicode by counting runes, not bytes.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads s with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the visual width.
+// If the string is already at or wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"hello world", 11},
+		{"", 0},
+		{" ", 1},
+		{"123", 3},
+		// Strings with ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[31mh\x1b[32me\x1b[34ml\x1b[35ml\x1b[36mo\x1b[0m", 5},
+		// Unicode strings (CJK characters)
+		{"你好", 2},
+		{"你好世界", 4},
+		// Emoji
+		{"👋", 1},
+		{"👋 hello", 7},
+		// Mixed: Unicode with ANSI
+		{"\x1b[31m你好\x1b[0m", 2},
+		// Edge case: only ANSI codes
+		{"\x1b[31m\x1b[0m", 0},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII padding
+		{"hello", 10, "hello     "},
+		{"hello", 5, "hello"},
+		{"hello", 3, "hello"}, // already wider, no change
+		{"", 5, "     "},
+		// ANSI-colored content with padding
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[31mred\x1b[0m", 3, "\x1b[31mred\x1b[0m"},
+		{"\x1b[1;32mbold green\x1b[0m", 15, "\x1b[1;32mbold green\x1b[0m     "},
+		// Unicode strings (rune-based, not terminal-wide)
+		{"你好", 4, "你好  "},
+		{"你好", 5, "你好   "},
+		{"你好", 2, "你好"},
+		// Edge cases
+		{"hi", 0, "hi"},
+		{" ", 3, "   "},
+		// ANSI with exact width
+		{"\x1b[31mhi\x1b[0m", 2, "\x1b[31mhi\x1b[0m"},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+	}
+}
