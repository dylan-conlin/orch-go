+// VisualWidth returns the visual display width of a string, ignoring ANSI escape codes.
+// It correctly counts Unicode runes, not bytes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward the width.
+// If the string is already at or wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"hello world", 11},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[0m", 0},
+
+		// Unicode strings (CJK, emoji)
+		{"你好", 2},
+		{"😀", 1},
+		{"hello😀world", 11},
+		{"🎉test", 5},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII strings
+		{"hello", 10, "hello     "},
+		{"hello", 5, "hello"},
+		{"hello", 3, "hello"},  // Already wider than target
+
+		// Empty string
+		{"", 5, "     "},
+		{"", 0, ""},
+
+		// Strings with ANSI codes
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[1;32mbold green\x1b[0m", 12, "\x1b[1;32mbold green\x1b[0m  "},
+
+		// Already at target width
+		{"exact", 5, "exact"},
+
+		// Edge case: width 0
+		{"hello", 0, "hello"},
+
+		// Unicode strings
+		{"你好", 4, "你好  "},
+		{"😀", 3, "😀  "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+	}
+}
