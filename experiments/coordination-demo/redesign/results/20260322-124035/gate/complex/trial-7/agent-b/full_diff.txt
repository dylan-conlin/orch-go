+// FormatTable renders headers and rows as an aligned text table. Columns are
+// auto-sized based on the widest content (including headers). ANSI-colored
+// content is handled correctly using StripANSI to calculate true visual widths.
+// Headers are separated from data rows by a line of dashes. Handles edge cases:
+// empty tables (headers only), rows with fewer columns than headers, and nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and all rows
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cellContent string
+			if i < len(row) {
+				cellContent = row[i]
+			}
+			visualWidth := len(StripANSI(cellContent))
+			if visualWidth > colWidths[i] {
+				colWidths[i] = visualWidth
+			}
+		}
+	}
+
+	// Helper to pad a cell to width, preserving ANSI codes
+	padCell := func(s string, width int) string {
+		visualWidth := len(StripANSI(s))
+		padding := width - visualWidth
+		if padding <= 0 {
+			return s
+		}
+		return s + strings.Repeat(" ", padding)
+	}
+
+	var buf strings.Builder
+
+	// Write header row
+	for i, h := range headers {
+		buf.WriteString("| ")
+		buf.WriteString(padCell(h, colWidths[i]))
+		buf.WriteString(" ")
+	}
+	buf.WriteString("|\n")
+
+	// Write separator
+	for i := range headers {
+		buf.WriteString("|")
+		buf.WriteString(strings.Repeat("-", colWidths[i]+2))
+	}
+	buf.WriteString("|\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			buf.WriteString("| ")
+			var cellContent string
+			if i < len(row) {
+				cellContent = row[i]
+			}
+			buf.WriteString(padCell(cellContent, colWidths[i]))
+			buf.WriteString(" ")
+		}
+		buf.WriteString("|\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name  | Age |\n|-------|-----|\n| Alice | 30  |\n| Bob   | 25  |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "| Col1 | Col2 |\n|------|------|\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"Running"},
+				{"Stopped"},
+			},
+			want: "| Status  |\n|---------|\n| Running |\n| Stopped |\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Item", "Color"},
+			rows: [][]string{
+				{"\x1b[31mError\x1b[0m", "\x1b[31mRed\x1b[0m"},
+				{"Normal", "\x1b[32mGreen\x1b[0m"},
+			},
+			want: "| Item   | Color |\n|--------|-------|\n| \x1b[31mError\x1b[0m  | \x1b[31mRed\x1b[0m   |\n| Normal | \x1b[32mGreen\x1b[0m |\n",
+		},
+		{
+			name:    "rows with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "| A | B | C |\n|---|---|---|\n| 1 | 2 |   |\n| 3 | 4 | 5 |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeaderName"},
+			rows: [][]string{
+				{"X", "A very long value indeed"},
+				{"Y", "B"},
+			},
+			want: "| Short | VeryLongHeaderName       |\n|-------|--------------------------|\n| X     | A very long value indeed |\n| Y     | B                        |\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"A", "B"},
+			rows:    nil,
+			want:    "| A | B |\n|---|---|\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch\ngot:\n%q\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
+}
