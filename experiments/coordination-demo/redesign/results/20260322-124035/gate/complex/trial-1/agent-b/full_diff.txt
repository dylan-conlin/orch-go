+// FormatTable renders headers and rows as an aligned text table with pipe separators.
+// It auto-sizes columns based on the widest content (including headers), separates
+// the header from data rows with dashes, and correctly handles ANSI-colored content
+// by using StripANSI for width calculation. Rows with fewer columns than headers are
+// padded; rows with more columns are truncated to match the header count.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Handle nil or empty rows
+	if rows == nil {
+		rows = [][]string{}
+	}
+
+	// Calculate column widths based on headers and content
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			width := len(StripANSI(content))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Build output with pipe separators and spaces
+	var buf strings.Builder
+
+	// Header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		padding := colWidths[i] - len(StripANSI(h))
+		buf.WriteString(h)
+		buf.WriteString(strings.Repeat(" ", padding))
+	}
+	buf.WriteString("\n")
+
+	// Separator line with dashes
+	for i, width := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", width))
+	}
+	buf.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		for i, width := range colWidths {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			padding := width - len(StripANSI(content))
+			buf.WriteString(content)
+			buf.WriteString(strings.Repeat(" ", padding))
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"Alice", "Done"},
+				{"Bob", "Pending"},
+			},
+			want: "Name  | Status \n------+--------\nAlice | Done   \nBob   | Pending\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"ID", "Value"},
+			rows:    [][]string{},
+			want:    "ID | Value\n---+------\n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Col1", "Col2"},
+			rows:    nil,
+			want:    "Col1 | Col2\n-----+-----\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Task"},
+			rows: [][]string{
+				{"Write tests"},
+				{"Fix bug"},
+			},
+			want: "Task       \n-----------\nWrite tests\nFix bug    \n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m", "Active"},
+				{"Normal", "\x1b[32mGreen\x1b[0m"},
+			},
+			want: "Name   | Status\n-------+-------\n\x1b[31mRed\x1b[0m    | Active\nNormal | \x1b[32mGreen\x1b[0m \n",
+		},
+		{
+			name:    "row fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \n3 | 4 | 5\n",
+		},
+		{
+			name:    "row more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+			},
+			want: "A | B\n--+--\n1 | 2\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeaderName"},
+			rows: [][]string{
+				{"A", "B"},
+				{"X", "VeryLongContentValue"},
+			},
+			want: "Short | VeryLongHeaderName  \n------+---------------------\nA     | B                   \nX     | VeryLongContentValue\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "mixed content types",
+			headers: []string{"ID", "Name", "Count"},
+			rows: [][]string{
+				{"1", "Alice", "42"},
+				{"2", "Bob", "7"},
+				{"10", "Charlie", "999"},
+			},
+			want: "ID | Name    | Count\n---+---------+------\n1  | Alice   | 42   \n2  | Bob     | 7    \n10 | Charlie | 999  \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
