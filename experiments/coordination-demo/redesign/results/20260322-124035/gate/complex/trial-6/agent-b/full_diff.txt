+// FormatTable renders headers and rows as an aligned text table with auto-sized
+// columns. Column widths are calculated using StripANSI to handle ANSI-colored
+// content correctly. The header is separated from data rows by a line of dashes.
+//
+// Handles edge cases: empty rows (headers only), rows with fewer columns than
+// headers, and nil rows. Rows with more columns than headers are truncated.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and all rows
+	widths := make([]int, len(headers))
+	for i, h := range headers {
+		widths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			w := len(StripANSI(row[i]))
+			if w > widths[i] {
+				widths[i] = w
+			}
+		}
+	}
+
+	// Format header row
+	var buf strings.Builder
+	buf.WriteString("| ")
+	for i, h := range headers {
+		buf.WriteString(h)
+		buf.WriteString(strings.Repeat(" ", widths[i]-len(StripANSI(h))))
+		if i < len(headers)-1 {
+			buf.WriteString(" | ")
+		}
+	}
+	buf.WriteString(" |\n")
+
+	// Format separator line
+	buf.WriteString("|")
+	for i, w := range widths {
+		buf.WriteString(strings.Repeat("-", w+2))
+		if i < len(widths)-1 {
+			buf.WriteString("|")
+		}
+	}
+	buf.WriteString("|\n")
+
+	// Format data rows
+	for _, row := range rows {
+		buf.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", widths[i]-len(StripANSI(cell))))
+			if i < len(headers)-1 {
+				buf.WriteString(" | ")
+			}
+		}
+		buf.WriteString(" |\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"Alice", "active"},
+				{"Bob", "idle"},
+			},
+			want: "| Name  | Status |\n|-------|--------|\n| Alice | active |\n| Bob   | idle   |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "| Col1 | Col2 |\n|------|------|\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+			},
+			want: "| Item   |\n|--------|\n| apple  |\n| banana |\n",
+		},
+		{
+			name:    "ansi colored content",
+			headers: []string{"Color"},
+			rows: [][]string{
+				{"\x1b[31mred\x1b[0m"},
+				{"\x1b[32mgreen\x1b[0m"},
+			},
+			want: "| Color |\n|-------|\n| \x1b[31mred\x1b[0m   |\n| \x1b[32mgreen\x1b[0m |\n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "| A | B | C |\n|---|---|---|\n| 1 | 2 |   |\n| x | y | z |\n",
+		},
+		{
+			name:    "row with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+			},
+			want: "| A | B |\n|---|---|\n| 1 | 2 |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongerHeader"},
+			rows: [][]string{
+				{"x", "verylongcontent"},
+			},
+			want: "| Short | LongerHeader    |\n|-------|-----------------|\n| x     | verylongcontent |\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows: [][]string{
+				{"a", "b"},
+			},
+			want: "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Col"},
+			rows:    nil,
+			want:    "| Col |\n|-----|\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
