+// FormatTable renders headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content in that column
+// (including headers). ANSI color codes are preserved but do not count toward
+// column width. Rows with fewer columns than headers are padded with empty cells.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate max width for each column
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Check all rows to update column widths
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			w := len(StripANSI(row[i]))
+			if w > colWidths[i] {
+				colWidths[i] = w
+			}
+		}
+	}
+
+	var b strings.Builder
+
+	// Header row
+	for i, h := range headers {
+		if i > 0 {
+			b.WriteString(" | ")
+		}
+		padding := colWidths[i] - len(StripANSI(h))
+		b.WriteString(h)
+		b.WriteString(strings.Repeat(" ", padding))
+	}
+	b.WriteString("\n")
+
+	// Separator line
+	for i := 0; i < len(headers); i++ {
+		if i > 0 {
+			b.WriteString("-+-")
+		}
+		b.WriteString(strings.Repeat("-", colWidths[i]))
+	}
+	b.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				b.WriteString(" | ")
+			}
+			if i < len(row) {
+				padding := colWidths[i] - len(StripANSI(row[i]))
+				b.WriteString(row[i])
+				b.WriteString(strings.Repeat(" ", padding))
+			} else {
+				// Pad with spaces if row is shorter than headers
+				b.WriteString(strings.Repeat(" ", colWidths[i]))
+			}
+		}
+		b.WriteString("\n")
+	}
+
+	return b.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 \n",
+		},
+		{
+			name:    "headers only (empty table)",
+			headers: []string{"Column1", "Column2"},
+			rows:    [][]string{},
+			want:    "Column1 | Column2\n--------+--------\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status", "Message"},
+			rows: [][]string{
+				{"\x1b[32mok\x1b[0m", "task done"},
+				{"\x1b[31merror\x1b[0m", "failed"},
+			},
+			want: "Status | Message  \n-------+----------\n\x1b[32mok\x1b[0m     | task done\n\x1b[31merror\x1b[0m  | failed   \n",
+		},
+		{
+			name:    "mismatched columns (rows with fewer cols)",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \nx | y | z\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+				{"cherry"},
+			},
+			want: "Item  \n------\napple \nbanana\ncherry\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"x", "a"},
+				{"abc", "this is a very long string"},
+			},
+			want: "Short | VeryLongHeader            \n------+---------------------------\nx     | a                         \nabc   | this is a very long string\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows: [][]string{
+				{"a", "b"},
+			},
+			want: "",
+		},
+		{
+			name:    "row with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+			},
+			want: "A | B\n--+--\n1 | 2\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v)\n= %q\nwant %q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
