+// FormatTable renders headers and rows as an aligned text table. Each column
+// is auto-sized based on the widest content (including headers). ANSI-colored
+// content is handled correctly — visual widths are calculated after stripping
+// ANSI codes. Columns are separated by " | " and headers are separated from
+// data rows with a line of dashes.
+//
+// Edge cases:
+// - nil or empty rows: renders headers only with separator
+// - rows with fewer columns: pads with empty strings
+// - rows with more columns than headers: extra columns are ignored
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate max width for each column based on header and row content.
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	// Update widths based on row content.
+	for _, row := range rows {
+		for i := 0; i < len(colWidths) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Build output with header, separator, and rows.
+	var buf strings.Builder
+
+	// Write header row, padding each column to its max width.
+	for i, header := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		visualWidth := len(StripANSI(header))
+		padAmount := colWidths[i] - visualWidth
+		buf.WriteString(header)
+		buf.WriteString(strings.Repeat(" ", padAmount))
+	}
+	buf.WriteString("\n")
+
+	// Write separator row.
+	for i := range headers {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", colWidths[i]))
+	}
+	buf.WriteString("\n")
+
+	// Write data rows, padding each cell to its column width.
+	for _, row := range rows {
+		for i := range headers {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			visualWidth := len(StripANSI(cell))
+			padAmount := colWidths[i] - visualWidth
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", padAmount))
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table with multiple rows",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"Alice", "active"},
+				{"Bob", "inactive"},
+			},
+			want: "Name  | Status  \n------+---------\nAlice | active  \nBob   | inactive\n",
+		},
+		{
+			name:    "headers only (no rows)",
+			headers: []string{"ID", "Value"},
+			rows:    nil,
+			want:    "ID | Value\n---+------\n",
+		},
+		{
+			name:    "empty rows slice",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "Col1 | Col2\n-----+-----\n",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"pending"},
+				{"complete"},
+			},
+			want: "Status  \n--------\npending \ncomplete\n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"val1", "val2", "val3"},
+				{"short", "x"},
+			},
+			want: "A     | B    | C   \n------+------+-----\nval1  | val2 | val3\nshort | x    |     \n",
+		},
+		{
+			name:    "row with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"val1", "val2", "extra_ignored"},
+			},
+			want: "A    | B   \n-----+-----\nval1 | val2\n",
+		},
+		{
+			name:    "ANSI colored content alignment",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[31mAlice\x1b[0m", "\x1b[32mactive\x1b[0m"},
+				{"Bob", "inactive"},
+			},
+			want: "Name  | Status  \n------+---------\n\x1b[31mAlice\x1b[0m | \x1b[32mactive\x1b[0m  \nBob   | inactive\n",
+		},
+		{
+			name:    "wide content alignment",
+			headers: []string{"Short", "Very Long Header"},
+			rows: [][]string{
+				{"a", "b"},
+				{"this is a longer value", "x"},
+			},
+			want: "Short                  | Very Long Header\n-----------------------+-----------------\na                      | b               \nthis is a longer value | x               \n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{{"a", "b"}},
+			want:    "",
+		},
+		{
+			name:    "all empty cells",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"", ""},
+				{"", ""},
+			},
+			want: "A | B\n--+--\n  |  \n  |  \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
