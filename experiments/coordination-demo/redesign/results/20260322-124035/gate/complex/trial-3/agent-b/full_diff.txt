+// FormatTable renders headers and rows as an aligned text table.
+// Column widths are auto-sized based on the widest content in each column
+// (including headers). ANSI color codes are preserved but don't count toward
+// visual width. Handles edge cases: empty rows (headers only), rows with fewer
+// columns than headers, and nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths using visual width (ANSI-stripped)
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = visualWidth(header)
+	}
+
+	// Check all rows to find max width for each column
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(colWidths) {
+				w := visualWidth(cell)
+				if w > colWidths[i] {
+					colWidths[i] = w
+				}
+			}
+		}
+	}
+
+	// Build output
+	var result strings.Builder
+
+	// Format headers
+	for i, header := range headers {
+		result.WriteString("| ")
+		result.WriteString(padToWidth(header, colWidths[i]))
+		result.WriteString(" ")
+	}
+	result.WriteString("|\n")
+
+	// Separator line
+	for i := range headers {
+		result.WriteString("+")
+		result.WriteString(strings.Repeat("-", colWidths[i]+2))
+	}
+	result.WriteString("+\n")
+
+	// Format rows
+	for _, row := range rows {
+		for i := range headers {
+			result.WriteString("| ")
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			result.WriteString(padToWidth(cell, colWidths[i]))
+			result.WriteString(" ")
+		}
+		result.WriteString("|\n")
+	}
+
+	return result.String()
+}
+
+// visualWidth returns the visual display width of a string, ignoring ANSI codes.
+func visualWidth(s string) int {
+	return len(StripANSI(s))
+}
+
+// padToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width.
+func padToWidth(s string, width int) string {
+	stripped := StripANSI(s)
+	padding := width - len(stripped)
+	if padding <= 0 {
+		return s
+	}
+	return s + strings.Repeat(" ", padding)
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name  | Age |\n+-------+-----+\n| Alice | 30  |\n| Bob   | 25  |\n",
+		},
+		{
+			name:    "headers only (empty rows)",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows:    [][]string{},
+			want:    "| Col1 | Col2 | Col3 |\n+------+------+------+\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"Running"},
+				{"Stopped"},
+			},
+			want: "| Status  |\n+---------+\n| Running |\n| Stopped |\n",
+		},
+		{
+			name:    "mismatched column counts - row shorter than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"a1", "b1"},
+				{"a2", "b2", "c2"},
+			},
+			want: "| A  | B  | C  |\n+----+----+----+\n| a1 | b1 |    |\n| a2 | b2 | c2 |\n",
+		},
+		{
+			name:    "wide content with long strings",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"x", "this is a very long cell content"},
+				{"longer", "y"},
+			},
+			want: "| Short  | VeryLongHeader                   |\n+--------+----------------------------------+\n| x      | this is a very long cell content |\n| longer | y                                |\n",
+		},
+		{
+			name:    "ANSI colored content alignment",
+			headers: []string{"Plain", "Colored"},
+			rows: [][]string{
+				{"abc", "\x1b[31mred\x1b[0m"},
+				{"\x1b[32mgre\x1b[0m", "longer"},
+			},
+			want: "| Plain | Colored |\n+-------+---------+\n| abc   | \x1b[31mred\x1b[0m     |\n| \x1b[32mgre\x1b[0m   | longer  |\n",
+		},
+		{
+			name:    "empty strings in cells",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"", "val"},
+				{"x", ""},
+			},
+			want: "| A | B   |\n+---+-----+\n|   | val |\n| x |     |\n",
+		},
+		{
+			name:    "single row",
+			headers: []string{"ID", "Value"},
+			rows: [][]string{
+				{"1", "test"},
+			},
+			want: "| ID | Value |\n+----+-------+\n| 1  | test  |\n",
+		},
+		{
+			name:    "nil rows (treated as empty)",
+			headers: []string{"Header"},
+			rows:    nil,
+			want:    "| Header |\n+--------+\n",
+		},
+		{
+			name:    "empty headers returns empty string",
+			headers: []string{},
+			rows: [][]string{
+				{"a", "b"},
+			},
+			want: "",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
