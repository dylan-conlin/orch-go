+// FormatTable renders headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including headers).
+// Headers are separated from rows with a line of dashes.
+// ANSI color codes are handled correctly using StripANSI for width calculation.
+// Handles edge cases: empty rows, rows with fewer columns than headers, and nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and all rows
+	widths := make([]int, len(headers))
+	for i, h := range headers {
+		widths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			w := len(StripANSI(row[i]))
+			if w > widths[i] {
+				widths[i] = w
+			}
+		}
+	}
+
+	// Build the table
+	var buf strings.Builder
+	const colPadding = 1
+	const separator = " | "
+
+	// Write header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(separator)
+		}
+		buf.WriteString(h)
+		// Pad to column width
+		visualWidth := len(StripANSI(h))
+		padding := widths[i] - visualWidth
+		buf.WriteString(strings.Repeat(" ", padding))
+	}
+	buf.WriteString("\n")
+
+	// Write separator line
+	for i := range headers {
+		if i > 0 {
+			buf.WriteString(" + ")
+		}
+		buf.WriteString(strings.Repeat("-", widths[i]))
+	}
+	buf.WriteString("\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := range headers {
+			if i > 0 {
+				buf.WriteString(separator)
+			}
+			if i < len(row) {
+				cell := row[i]
+				buf.WriteString(cell)
+				// Pad to column width
+				visualWidth := len(StripANSI(cell))
+				padding := widths[i] - visualWidth
+				buf.WriteString(strings.Repeat(" ", padding))
+			} else {
+				// Missing cell in row - pad with spaces
+				buf.WriteString(strings.Repeat(" ", widths[i]))
+			}
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Status", "Count"},
+			rows: [][]string{
+				{"Alice", "active", "42"},
+				{"Bob", "idle", "7"},
+			},
+			want: "Name  | Status | Count\n----- + ------ + -----\nAlice | active | 42   \nBob   | idle   | 7    \n",
+		},
+		{
+			name:    "headers only (no rows)",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "Col1 | Col2\n---- + ----\n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"A", "B"},
+			rows:    nil,
+			want:    "A | B\n- + -\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Value"},
+			rows: [][]string{
+				{"test"},
+				{"longer value"},
+			},
+			want: "Value       \n------------\ntest        \nlonger value\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"a", "b"},
+				{"short", "this is a very long string"},
+			},
+			want: "Short | VeryLongHeader            \n----- + --------------------------\na     | b                         \nshort | this is a very long string\n",
+		},
+		{
+			name:    "rows with fewer columns than headers",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows: [][]string{
+				{"a", "b"},
+				{"x", "y", "z"},
+			},
+			want: "Col1 | Col2 | Col3\n---- + ---- + ----\na    | b    |     \nx    | y    | z   \n",
+		},
+		{
+			name:    "rows with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+			},
+			want: "A | B\n- + -\n1 | 2\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Color"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m", "\x1b[32mGreen\x1b[0m"},
+				{"Plain", "\x1b[1;33mBold Yellow\x1b[0m"},
+			},
+			want: "Name  | Color      \n----- + -----------\n\x1b[31mRed\x1b[0m   | \x1b[32mGreen\x1b[0m      \nPlain | \x1b[1;33mBold Yellow\x1b[0m\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) = %q, want %q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
