+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. Unicode characters (including emoji) are counted as single
+// runes. This is useful for aligning text with colored or formatted output.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward the visual width. If the
+// string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+		// Strings with ANSI color codes
+		{"\x1b[31mred text\x1b[0m", 8},           // "red text"
+		{"\x1b[1;32mbold green\x1b[0m", 10},      // "bold green"
+		{"\x1b[33myellow\x1b[0m", 6},             // "yellow"
+		// Unicode: CJK characters
+		{"hello世界", 7},                         // 5 ascii + 2 chinese chars
+		{"世界", 2},                               // 2 chinese chars
+		{"你好世界", 4},                         // 4 chinese chars
+		// Unicode: emoji
+		{"👋", 1},                                 // emoji is one rune
+		{"hello👋world", 11},                    // 5 + 1 + 5
+		// Mixed ANSI and Unicode
+		{"\x1b[32m你好\x1b[0m", 2},               // green + chinese + reset, visual width 2
+		{"\x1b[31m👋\x1b[0m hello", 7},          // red emoji reset hello
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII: padding needed
+		{"hi", 5, "hi   "},
+		{"hello", 10, "hello     "},
+		{"", 3, "   "},
+		{"a", 1, "a"},
+		// Plain ASCII: already at width
+		{"hello", 5, "hello"},
+		{"exact", 5, "exact"},
+		// Plain ASCII: wider than target (no change)
+		{"hello", 3, "hello"},
+		{"hello world", 5, "hello world"},
+		// ANSI codes: should preserve codes but pad visually
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[1;32mbold\x1b[0m", 6, "\x1b[1;32mbold\x1b[0m  "},
+		// Unicode: padding with unicode content
+		{"你好", 4, "你好  "},
+		{"世界", 3, "世界 "},
+		// ANSI with Unicode
+		{"\x1b[33m你\x1b[0m", 3, "\x1b[33m你\x1b[0m  "},
+		// Mixed: long string with ANSI and emoji
+		{"\x1b[35m👋hello\x1b[0m", 10, "\x1b[35m👋hello\x1b[0m    "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width matches or exceeds target
+		gotWidth := VisualWidth(got)
+		if gotWidth < tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want >= %d", tt.input, tt.width, gotWidth, tt.width)
+		}
+	}
+}
