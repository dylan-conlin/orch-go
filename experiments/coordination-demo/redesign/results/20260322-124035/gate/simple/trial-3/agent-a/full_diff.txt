+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units are shown with 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	const (
+		B  = 1
+		KB = 1024 * B
+		MB = 1024 * KB
+		GB = 1024 * MB
+		TB = 1024 * GB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	var value float64
+	var unit string
+
+	prefix := ""
+	if negative {
+		prefix = "-"
+	}
+
+	switch {
+	case bytes < KB:
+		return fmt.Sprintf("%s%d B", prefix, bytes)
+	case bytes < MB:
+		value = float64(bytes) / float64(KB)
+		unit = "KiB"
+	case bytes < GB:
+		value = float64(bytes) / float64(MB)
+		unit = "MiB"
+	case bytes < TB:
+		value = float64(bytes) / float64(GB)
+		unit = "GiB"
+	default:
+		value = float64(bytes) / float64(TB)
+		unit = "TiB"
+	}
+
+	return fmt.Sprintf("%s%.1f %s", prefix, value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Exact boundaries and transitions
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+
+		// Values between boundaries
+		{2560, "2.5 KiB"},
+		{5242880, "5.0 MiB"},
+		{10485760, "10.0 MiB"},
+		{1073741824 * 10, "10.0 GiB"},
+
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1099511627776, "-1.0 TiB"},
+		{-512, "-512 B"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
