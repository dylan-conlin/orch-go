+// FormatRate formats a transfer rate (bytes per second) into a human-readable string.
+// Output style: "512 B/s", "1.5 MiB/s", etc.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	const divisor = 1024.0
+
+	unitIndex := 0
+	value := bytesPerSec
+
+	for unitIndex < len(units)-1 && value >= divisor {
+		value /= divisor
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes per second, show as integer
+		result = fmt.Sprintf("%d %s", int64(value), units[unitIndex])
+	} else {
+		// For larger units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{256, "256 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+
+		// Fractional values
+		{1536, "1.5 KiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1610612736, "1.5 GiB/s"},
+
+		// Other intermediate values
+		{2048, "2.0 KiB/s"},
+		{2097152, "2.0 MiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+
+		// Large values (beyond GiB)
+		{10737418240, "10.0 GiB/s"},  // 10 GiB/s
+		{1099511627776, "1024.0 GiB/s"}, // 1 TiB/s
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
