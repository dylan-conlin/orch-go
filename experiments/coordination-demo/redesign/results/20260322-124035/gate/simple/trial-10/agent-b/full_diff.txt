+// FormatRate formats a transfer rate (bytes/sec) as a human-readable string.
+// Output style: "512 B/s", "1.5 MiB/s", "2.1 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	const divisor = 1024.0
+
+	unitIdx := 0
+	for unitIdx < len(units)-1 && rate >= divisor {
+		rate /= divisor
+		unitIdx++
+	}
+
+	// Format: no decimals for B/s, 1 decimal for others
+	var result string
+	if unitIdx == 0 {
+		// B/s: show as integer
+		result = fmt.Sprintf("%.0f %s", rate, units[unitIdx])
+	} else {
+		// KiB/s, MiB/s, GiB/s: show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", rate, units[unitIdx])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary (1024 B = 1 KiB)
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB values
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// Exact MiB boundary (1048576 B = 1 MiB)
+		{1048576, "1.0 MiB/s"},
+		{1048576 * 1.5, "1.5 MiB/s"},
+		// GiB values
+		{1048576 * 1024, "1.0 GiB/s"},
+		{1048576 * 1024 * 2.5, "2.5 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1048576 * 1024 * 3.7, "-3.7 GiB/s"},
+		// Fractional inputs that round to integers
+		{512.5, "512 B/s"},
+		{1024.5, "1.0 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
