+// FormatBytes formats byte counts into human-readable strings using binary units.
+// Examples: "512 B", "1.5 KiB", "2.0 MiB"
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitSize := int64(1024)
+
+	value := float64(bytes)
+	unitIndex := 0
+
+	for unitIndex < len(units)-1 && value >= float64(unitSize) {
+		value /= float64(unitSize)
+		unitIndex++
+	}
+
+	prefix := ""
+	if negative {
+		prefix = "-"
+	}
+
+	if unitIndex == 0 { // Bytes - no decimal
+		return fmt.Sprintf("%s%d B", prefix, int64(value))
+	}
+	return fmt.Sprintf("%s%.1f %s", prefix, value, units[unitIndex])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (no units)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		// Exact MiB boundary
+		{1048576, "1.0 MiB"},
+		// Between MiB boundaries
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Exact GiB boundary
+		{1073741824, "1.0 GiB"},
+		// Between GiB boundaries
+		{1610612736, "1.5 GiB"},
+		// Exact TiB boundary
+		{1099511627776, "1.0 TiB"},
+		// Large TiB value
+		{2199023255552, "2.0 TiB"},
+		{5497558138880, "5.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
