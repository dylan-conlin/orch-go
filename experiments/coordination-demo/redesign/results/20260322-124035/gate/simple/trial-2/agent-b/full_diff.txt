+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units with "/s" suffix: "B/s", "KiB/s", "MiB/s", "GiB/s".
+// Non-byte-per-second rates are shown with 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	absRate := bytesPerSec
+	if isNegative {
+		absRate = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024.0
+		MiB = 1024.0 * 1024.0
+		GiB = 1024.0 * 1024.0 * 1024.0
+	)
+
+	var (
+		value float64
+		unit  string
+	)
+
+	switch {
+	case absRate >= GiB:
+		value = absRate / GiB
+		unit = "GiB/s"
+	case absRate >= MiB:
+		value = absRate / MiB
+		unit = "MiB/s"
+	case absRate >= KiB:
+		value = absRate / KiB
+		unit = "KiB/s"
+	default:
+		// Less than 1 KiB/s, show as bytes per second
+		value = absRate
+		unit = "B/s"
+	}
+
+	var result string
+	if unit == "B/s" {
+		// Show as integer for byte rates
+		result = fmt.Sprintf("%d %s", int(value), unit)
+	} else {
+		// Show 1 decimal place for other units
+		result = fmt.Sprintf("%.1f %s", value, unit)
+	}
+
+	if isNegative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary (1 KiB)
+		{1024, "1.0 KiB/s"},
+		// Between boundaries
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 + 512, "1.5 KiB/s"},
+		// Exact boundary (1 MiB)
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB
+		{1048576 + 524288, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// GiB rates
+		{1073741824, "1.0 GiB/s"},
+		{1073741824 + 536870912, "1.5 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-512, "-512 B/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Edge cases
+		{0.5, "0 B/s"}, // Fractional bytes rounds to 0
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
