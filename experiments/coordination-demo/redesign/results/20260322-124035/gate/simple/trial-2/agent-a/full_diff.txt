+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output uses 1 decimal place for non-byte units: "0 B", "1.5 KiB", "2.3 MiB", etc.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	const (
+		Ki = 1024
+		Mi = 1024 * 1024
+		Gi = 1024 * 1024 * 1024
+		Ti = 1024 * 1024 * 1024 * 1024
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes >= Ti:
+		value = float64(bytes) / float64(Ti)
+		unit = "TiB"
+	case bytes >= Gi:
+		value = float64(bytes) / float64(Gi)
+		unit = "GiB"
+	case bytes >= Mi:
+		value = float64(bytes) / float64(Mi)
+		unit = "MiB"
+	case bytes >= Ki:
+		value = float64(bytes) / float64(Ki)
+		unit = "KiB"
+	default:
+		unit = "B"
+		if negative {
+			return fmt.Sprintf("-%d %s", bytes, unit)
+		}
+		return fmt.Sprintf("%d %s", bytes, unit)
+	}
+
+	result := fmt.Sprintf("%.1f %s", value, unit)
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		{0, "0 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048575, "1024.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741823, "1024.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 * 2, "2.0 TiB"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
