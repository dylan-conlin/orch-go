+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte units are shown with 1 decimal place.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kib = 1024
+		mib = kib * 1024
+		gib = mib * 1024
+	)
+
+	var result string
+	if bytesPerSec < kib {
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else if bytesPerSec < mib {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kib)
+	} else if bytesPerSec < gib {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/mib)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/gib)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		// Fractional values
+		{1536, "1.5 KiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Between boundaries
+		{2048, "2.0 KiB/s"},
+		{3072, "3.0 KiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{3145728, "3.0 MiB/s"},
+		// Large values
+		{10737418240, "10.0 GiB/s"},
+		{1099511627776, "1024.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
