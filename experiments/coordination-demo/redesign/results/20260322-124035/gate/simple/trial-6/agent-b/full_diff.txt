+// FormatRate formats a transfer rate in bytes per second as a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units (e.g., "1.5 MiB/s").
+// Returns "0 B/s" for zero. Prefixes negative values with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	divisor := 1024.0
+	unitIndex := 0
+
+	for unitIndex < len(units)-1 && bytesPerSec >= divisor {
+		bytesPerSec /= divisor
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes/sec, no decimal
+		result = fmt.Sprintf("%.0f %s", bytesPerSec, units[unitIndex])
+	} else {
+		// For other units, 1 decimal place
+		result = fmt.Sprintf("%.1f %s", bytesPerSec, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+		{512, "512 B/s"},
