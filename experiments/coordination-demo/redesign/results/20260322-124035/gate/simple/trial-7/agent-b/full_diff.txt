+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units. Output style: "512 B/s", "1.5 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	threshold := 1024.0
+
+	for i, unit := range units {
+		if bytesPerSec < threshold || i == len(units)-1 {
+			var formatted string
+			if i == 0 {
+				formatted = fmt.Sprintf("%.0f %s", bytesPerSec, unit)
+			} else {
+				formatted = fmt.Sprintf("%.1f %s", bytesPerSec, unit)
+			}
+			if negative {
+				return "-" + formatted
+			}
+			return formatted
+		}
+		bytesPerSec /= threshold
+	}
+	return ""
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Large values
+		{10737418240, "10.0 GiB/s"},
+		{1099511627776, "1024.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Fractional values
+		{2560, "2.5 KiB/s"},
+		{5242880, "5.0 MiB/s"},
+		{7864320, "7.5 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
