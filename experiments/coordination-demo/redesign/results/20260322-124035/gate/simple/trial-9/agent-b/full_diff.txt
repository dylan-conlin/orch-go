+// FormatRate formats a transfer rate (bytes per second) into a human-readable string.
+// Uses binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place for non-byte-per-sec units.
+// Handles zero as "0 B/s" and negative values with "-" prefix.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		ki = 1024.0
+		mi = ki * 1024
+		gi = mi * 1024
+	)
+
+	var result string
+	switch {
+	case bytesPerSec < ki:
+		// For bytes per second, show as integer
+		result = fmt.Sprintf("%d B/s", int64(bytesPerSec))
+	case bytesPerSec < mi:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/ki)
+	case bytesPerSec < gi:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/mi)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/gi)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s boundary
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 * 100, "100.0 KiB/s"},
+		{1024*1024 - 1, "1024.0 KiB/s"},
+		// MiB/s boundary
+		{1024 * 1024, "1.0 MiB/s"},
+		{1024 * 1024 * 1.5, "1.5 MiB/s"},
+		{1024 * 1024 * 10, "10.0 MiB/s"},
+		{1024*1024*1024 - 1, "1024.0 MiB/s"},
+		// GiB/s boundary and large values
+		{1024 * 1024 * 1024, "1.0 GiB/s"},
+		{1024 * 1024 * 1024 * 2.5, "2.5 GiB/s"},
+		{1024 * 1024 * 1024 * 100, "100.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1024 * 1024, "-1.0 MiB/s"},
+		{-1024 * 1024 * 1024, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%.1f) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
