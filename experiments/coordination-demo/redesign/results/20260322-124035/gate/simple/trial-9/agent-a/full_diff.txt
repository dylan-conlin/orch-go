+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "512 B", "1.5 KiB", "2.3 MiB", "1.0 GiB", etc.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case absBytes >= TiB:
+		value = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	case absBytes >= GiB:
+		value = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	case absBytes >= MiB:
+		value = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes >= KiB:
+		value = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	default:
+		// For bytes, show no decimal
+		if negative {
+			return fmt.Sprintf("-%d B", absBytes)
+		}
+		return fmt.Sprintf("%d B", absBytes)
+	}
+
+	if negative {
+		return fmt.Sprintf("-%.1f %s", value, unit)
+	}
+	return fmt.Sprintf("%.1f %s", value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		{0, "0 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
