+// FormatBytes formats a byte count into a human-readable string.
+// Output uses binary units (B, KiB, MiB, GiB, TiB) with 1 decimal place
+// for non-byte units. Negative values are prefixed with "-".
+// Examples: "0 B", "512 B", "1.5 KiB", "1.0 MiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	absBytes := bytes
+	negative := false
+	if bytes < 0 {
+		negative = true
+		absBytes = -bytes
+	}
+
+	const (
+		unit = 1024
+		kiB  = unit
+		miB  = unit * unit
+		giB  = unit * unit * unit
+		tiB  = unit * unit * unit * unit
+	)
+
+	var result string
+
+	if absBytes < kiB {
+		result = fmt.Sprintf("%d B", absBytes)
+	} else if absBytes < miB {
+		val := float64(absBytes) / float64(kiB)
+		result = fmt.Sprintf("%.1f KiB", val)
+	} else if absBytes < giB {
+		val := float64(absBytes) / float64(miB)
+		result = fmt.Sprintf("%.1f MiB", val)
+	} else if absBytes < tiB {
+		val := float64(absBytes) / float64(giB)
+		result = fmt.Sprintf("%.1f GiB", val)
+	} else {
+		val := float64(absBytes) / float64(tiB)
+		result = fmt.Sprintf("%.1f TiB", val)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero case
+		{0, "0 B"},
+		// Bytes range
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		// TiB boundary and values
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
