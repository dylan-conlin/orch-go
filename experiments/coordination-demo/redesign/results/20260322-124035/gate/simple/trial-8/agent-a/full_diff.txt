+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units display 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	const (
+		B  = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	var value float64
+	var unit string
+
+	switch {
+	case absBytes < KiB:
+		return fmt.Sprintf("%d B", bytes)
+	case absBytes < MiB:
+		value = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	case absBytes < GiB:
+		value = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes < TiB:
+		value = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	if negative {
+		return fmt.Sprintf("-%.1f %s", value, unit)
+	}
+	return fmt.Sprintf("%.1f %s", value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 + 512, "1.5 KiB"},
+		// Between KiB and MiB
+		{512 * 1024, "512.0 KiB"},
+		{768 * 1024, "768.0 KiB"},
+		{1023 * 1024, "1023.0 KiB"},
+		// MiB boundary
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{2 * 1048576, "2.0 MiB"},
+		// Between MiB and GiB
+		{512 * 1048576, "512.0 MiB"},
+		{768 * 1048576, "768.0 MiB"},
+		{1023 * 1048576, "1023.0 MiB"},
+		// GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{2 * 1073741824, "2.0 GiB"},
+		// Between GiB and TiB
+		{512 * 1073741824, "512.0 GiB"},
+		{768 * 1073741824, "768.0 GiB"},
+		// TiB boundary
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		{10 * 1099511627776, "10.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		// Fractional values
+		{1536, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
