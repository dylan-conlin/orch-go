+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units display 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	const binary = 1024.0
+
+	value := bytesPerSec
+	unitIndex := 0
+
+	for unitIndex < len(units)-1 && value >= binary {
+		value /= binary
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// Bytes/sec: show as integer
+		result = fmt.Sprintf("%.0f %s", value, units[unitIndex])
+	} else {
+		// KiB/s, MiB/s, GiB/s: show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes/sec, shown as integers)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// KiB/s
+		{512 * 1024, "512.0 KiB/s"},
+		{1.5 * 1024 * 1024, "1.5 MiB/s"},
+		// MiB/s boundary and values
+		{1048576, "1.0 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{3145728, "3.0 MiB/s"},
+		// Fractional MiB/s
+		{1572864, "1.5 MiB/s"},
+		{2621440, "2.5 MiB/s"},
+		// GiB/s boundary and values
+		{1073741824, "1.0 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Large GiB/s
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Edge case: very small value
+		{0.5, "0 B/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
