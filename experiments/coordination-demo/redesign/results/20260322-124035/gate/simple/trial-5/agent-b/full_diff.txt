+// FormatRate formats a transfer rate (bytes per second) into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte rates show 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	// Check units from largest to smallest
+	if bytesPerSec >= 1024*1024*1024 {
+		val := bytesPerSec / (1024 * 1024 * 1024)
+		if negative {
+			return fmt.Sprintf("-%.1f GiB/s", val)
+		}
+		return fmt.Sprintf("%.1f GiB/s", val)
+	}
+	if bytesPerSec >= 1024*1024 {
+		val := bytesPerSec / (1024 * 1024)
+		if negative {
+			return fmt.Sprintf("-%.1f MiB/s", val)
+		}
+		return fmt.Sprintf("%.1f MiB/s", val)
+	}
+	if bytesPerSec >= 1024 {
+		val := bytesPerSec / 1024
+		if negative {
+			return fmt.Sprintf("-%.1f KiB/s", val)
+		}
+		return fmt.Sprintf("%.1f KiB/s", val)
+	}
+
+	// B/s - no decimal places
+	if negative {
+		return fmt.Sprintf("-%.0f B/s", bytesPerSec)
+	}
+	return fmt.Sprintf("%.0f B/s", bytesPerSec)
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary at 1 KiB
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// Boundary at 1 MiB
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Boundary at 1 GiB
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Large values
+		{5368709120, "5.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
