+// FormatBytes formats a byte count into human-readable binary units.
+// Output style: "512 B", "1.5 KiB", "1.0 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	sign := ""
+	if bytes < 0 {
+		sign = "-"
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	value := float64(bytes)
+
+	for i := 0; i < len(units)-1; i++ {
+		if value < 1024 {
+			if i == 0 {
+				return fmt.Sprintf("%s%d %s", sign, int64(value), units[i])
+			}
+			return fmt.Sprintf("%s%.1f %s", sign, value, units[i])
+		}
+		value /= 1024
+	}
+
+	return fmt.Sprintf("%s%.1f %s", sign, value, units[len(units)-1])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
