+// FormatRate formats a transfer rate (bytes per second) as a human-readable
+// string using binary units. Output style: "512 B/s", "1.5 KiB/s", "2.0 MiB/s".
+func FormatRate(bytesPerSec float64) string {
+	// Handle negative values
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	// Define binary unit thresholds
+	const (
+		KiB = 1024.0
+		MiB = 1024.0 * 1024.0
+		GiB = 1024.0 * 1024.0 * 1024.0
+	)
+
+	var result string
+	if rate >= GiB {
+		result = fmt.Sprintf("%.1f GiB/s", rate/GiB)
+	} else if rate >= MiB {
+		result = fmt.Sprintf("%.1f MiB/s", rate/MiB)
+	} else if rate >= KiB {
+		result = fmt.Sprintf("%.1f KiB/s", rate/KiB)
+	} else {
+		result = fmt.Sprintf("%d B/s", int(rate))
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Large values
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
