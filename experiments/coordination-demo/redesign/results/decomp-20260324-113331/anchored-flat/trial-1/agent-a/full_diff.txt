+// FormatBytes formats a byte count into a human-readable string.
+// Uses binary units (B, KiB, MiB, GiB, TiB) with 1 decimal place for
+// non-byte values. Examples: "0 B", "512 B", "1.5 MiB", "-1.0 KiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		unit = 1024
+	)
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	value := float64(bytes)
+
+	for i := 0; i < len(units)-1; i++ {
+		if value < unit {
+			if i == 0 {
+				// For bytes, show no decimal places
+				if negative {
+					return fmt.Sprintf("-%d %s", int64(value), units[i])
+				}
+				return fmt.Sprintf("%d %s", int64(value), units[i])
+			}
+			// For other units, show 1 decimal place
+			if negative {
+				return fmt.Sprintf("-%.1f %s", value, units[i])
+			}
+			return fmt.Sprintf("%.1f %s", value, units[i])
+		}
+		value /= unit
+	}
+
+	// TiB or larger
+	if negative {
+		return fmt.Sprintf("-%.1f %s", value, units[len(units)-1])
+	}
+	return fmt.Sprintf("%.1f %s", value, units[len(units)-1])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+
+		// Small bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+
+		// Large values
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		{5497558138880, "5.0 TiB"},
+
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
