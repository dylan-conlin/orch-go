+// FormatBytes formats a byte count as a human-readable string.
+// Output style: "0 B", "512 B", "1.5 KiB", "1.0 MiB", "2.3 GiB".
+// Uses binary units: B, KiB, MiB, GiB, TiB with 1 decimal place for non-byte units.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	size := float64(bytes)
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for size >= 1024 && unitIndex < len(units)-1 {
+		size /= 1024
+		unitIndex++
+	}
+
+	// Format the output
+	var result string
+	if unitIndex == 0 {
+		// For bytes, no decimal places
+		result = fmt.Sprintf("%d B", int64(size))
+	} else {
+		// For other units, 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+		{1649267441664, "1.5 TiB"},
+		// Various values
+		{512 * 1024, "512.0 KiB"},
+		{2 * 1024 * 1024, "2.0 MiB"},
+		{2560 * 1024, "2.5 MiB"},
+		{512 * 1024 * 1024, "512.0 MiB"},
+		{10 * 1024 * 1024 * 1024, "10.0 GiB"},
+		{2 * 1024 * 1024 * 1024 * 1024, "2.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
