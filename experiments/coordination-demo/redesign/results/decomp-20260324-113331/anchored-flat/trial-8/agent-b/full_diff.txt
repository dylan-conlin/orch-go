+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Output style: "512 B/s", "1.5 KiB/s", "1.0 MiB/s", "0.5 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	isNegative := bytesPerSec < 0
+	abs := bytesPerSec
+	if isNegative {
+		abs = -abs
+	}
+
+	// Binary units (1024-based)
+	const (
+		kib = 1024
+		mib = 1024 * 1024
+		gib = 1024 * 1024 * 1024
+	)
+
+	var result string
+	switch {
+	case abs >= gib:
+		result = fmt.Sprintf("%.1f GiB/s", abs/gib)
+	case abs >= mib:
+		result = fmt.Sprintf("%.1f MiB/s", abs/mib)
+	case abs >= kib:
+		result = fmt.Sprintf("%.1f KiB/s", abs/kib)
+	default:
+		result = fmt.Sprintf("%.0f B/s", abs)
+	}
+
+	if isNegative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		{1048575, "1024.0 KiB/s"},
+		// MiB boundaries
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		{1073741823, "1024.0 MiB/s"},
+		// GiB range
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Fractional values
+		{512.5, "512 B/s"},
+		{1024.5, "1.0 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%.1f) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
