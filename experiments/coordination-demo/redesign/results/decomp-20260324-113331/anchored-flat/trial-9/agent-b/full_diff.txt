+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place
+// for non-byte-per-second values.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	// Define binary units
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+
+	switch {
+	case rate < KiB:
+		result = fmt.Sprintf("%.0f B/s", rate)
+	case rate < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", rate/KiB)
+	case rate < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", rate/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", rate/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero case
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary: 1 KiB/s
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// Larger values in KiB/s range
+		{102400, "100.0 KiB/s"},
+		{512000, "500.0 KiB/s"},
+		// Boundary: 1 MiB/s
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Larger values in MiB/s range
+		{104857600, "100.0 MiB/s"},
+		{524288000, "500.0 MiB/s"},
+		// Boundary: 1 GiB/s
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Large values in GiB/s range
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},

+		{1536.5, "1.5 KiB/s"},
