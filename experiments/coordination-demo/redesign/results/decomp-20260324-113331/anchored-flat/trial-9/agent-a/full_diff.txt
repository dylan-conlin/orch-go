+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "1.0 MiB", "2.3 GiB", "1.1 TiB".
+// Non-byte units show 1 decimal place. Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	sign := ""
+	if negative {
+		sign = "-"
+	}
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes < KiB:
+		return fmt.Sprintf("%s%d B", sign, bytes)
+	case bytes < MiB:
+		value = float64(bytes) / float64(KiB)
+		unit = "KiB"
+	case bytes < GiB:
+		value = float64(bytes) / float64(MiB)
+		unit = "MiB"
+	case bytes < TiB:
+		value = float64(bytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(bytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	return fmt.Sprintf("%s%.1f %s", sign, value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{1024 + 512, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Large values
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1536, "-1.5 KiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
