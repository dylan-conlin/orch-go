+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Output style: "512 B/s", "1.5 MiB/s", "10.2 GiB/s".
+// Uses binary units (KiB, MiB, GiB) with 1 decimal place for non-byte-per-sec rates.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	sign := ""
+	if bytesPerSec < 0 {
+		sign = "-"
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	if bytesPerSec < KiB {
+		return fmt.Sprintf("%s%.0f B/s", sign, bytesPerSec)
+	}
+	if bytesPerSec < MiB {
+		return fmt.Sprintf("%s%.1f KiB/s", sign, bytesPerSec/KiB)
+	}
+	if bytesPerSec < GiB {
+		return fmt.Sprintf("%s%.1f MiB/s", sign, bytesPerSec/MiB)
+	}
+	return fmt.Sprintf("%s%.1f GiB/s", sign, bytesPerSec/GiB)
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		{0, "0 B/s"},
+		{256, "256 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1048576 * 1.5, "1.5 MiB/s"},
+		{1048576 * 2.5, "2.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1073741824 * 3.7, "3.7 GiB/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
