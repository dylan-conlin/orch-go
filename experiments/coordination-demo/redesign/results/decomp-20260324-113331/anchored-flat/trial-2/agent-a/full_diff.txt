+// FormatBytes formats byte counts into human-readable strings using binary units
+// (B, KiB, MiB, GiB, TiB). Non-byte values are displayed with 1 decimal place.
+// Negative values are prefixed with "-".
+// Examples: "0 B", "512 B", "1.5 KiB", "1.0 MiB", "-1.0 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	neg := bytes < 0
+	if neg {
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	size := float64(bytes)
+	unitIdx := 0
+
+	for size >= 1024 && unitIdx < len(units)-1 {
+		size /= 1024
+		unitIdx++
+	}
+
+	var result string
+	if unitIdx == 0 {
+		result = fmt.Sprintf("%d B", int64(size))
+	} else {
+		result = fmt.Sprintf("%.1f %s", size, units[unitIdx])
+	}
+
+	if neg {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{10737418240, "10.0 GiB"},
+		// TiB boundary and values
+		{1099511627776, "1.0 TiB"},
+		{1648721223680, "1.5 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		// Edge cases
+		{1024*1024 - 1, "1024.0 KiB"},
+		{1024*1024*1024 - 1, "1024.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
