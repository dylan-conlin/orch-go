+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.0 MiB", "1.2 GiB".
+// Negative values are prefixed with a minus sign.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var unit string
+	var size float64
+
+	switch {
+	case bytes < KiB:
+		result := fmt.Sprintf("%d B", bytes)
+		if negative {
+			result = "-" + result
+		}
+		return result
+	case bytes < MiB:
+		unit = "KiB"
+		size = float64(bytes) / float64(KiB)
+	case bytes < GiB:
+		unit = "MiB"
+		size = float64(bytes) / float64(MiB)
+	case bytes < TiB:
+		unit = "GiB"
+		size = float64(bytes) / float64(GiB)
+	default:
+		unit = "TiB"
+		size = float64(bytes) / float64(TiB)
+	}
+
+	result := fmt.Sprintf("%.1f %s", size, unit)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 + 512, "1.5 KiB"},
+		// MiB boundary
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB boundary
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-512, "-512 B"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
