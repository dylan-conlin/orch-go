+// FormatRate formats a transfer rate as a human-readable string using binary
+// units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Byte-per-second rates show no decimal places; larger units show 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	value := bytesPerSec
+	if negative {
+		value = -value
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	divisor := 1024.0
+
+	// Find the appropriate unit
+	unitIdx := 0
+	for unitIdx < len(units)-1 && value >= divisor {
+		value /= divisor
+		unitIdx++
+	}
+
+	// Format the result
+	var result string
+	if unitIdx == 0 {
+		// B/s - no decimal places
+		result = fmt.Sprintf("%d %s", int(value), units[unitIdx])
+	} else {
+		// Other units - 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIdx])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary at 1 KiB/s
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// Fractional KiB/s
+		{2560, "2.5 KiB/s"},
+		{3072, "3.0 KiB/s"},
+		// Boundary at 1 MiB/s
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Fractional MiB/s
+		{3145728, "3.0 MiB/s"},
+		// Boundary at 1 GiB/s
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Large values
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
