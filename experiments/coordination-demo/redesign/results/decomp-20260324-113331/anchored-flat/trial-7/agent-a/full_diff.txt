+// FormatBytes formats a byte count into a human-readable string.
+// Uses binary units: B, KiB, MiB, GiB, TiB.
+// Shows 1 decimal place for non-byte units. Handles negative values.
+// Examples: "0 B", "512 B", "1.5 KiB", "1.0 MiB", "-1.0 KiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	sign := ""
+	value := bytes
+	if bytes < 0 {
+		sign = "-"
+		value = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	switch {
+	case value < KiB:
+		return fmt.Sprintf("%s%d B", sign, value)
+	case value < MiB:
+		return fmt.Sprintf("%s%.1f KiB", sign, float64(value)/KiB)
+	case value < GiB:
+		return fmt.Sprintf("%s%.1f MiB", sign, float64(value)/MiB)
+	case value < TiB:
+		return fmt.Sprintf("%s%.1f GiB", sign, float64(value)/GiB)
+	default:
+		return fmt.Sprintf("%s%.1f TiB", sign, float64(value)/TiB)
+	}
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// KiB boundaries and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+		{1048575, "1024.0 KiB"},
+
+		// MiB boundaries and values
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1073741823, "1024.0 MiB"},
+
+		// GiB boundaries and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{10737418240, "10.0 GiB"},
+		{1099511627775, "1024.0 GiB"},
+
+		// TiB values
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{10995116277760, "10.0 TiB"},
+
+		// Negative values
+		{-1, "-1 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
