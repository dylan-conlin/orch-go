+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Output style: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s".
+// Uses binary units (KiB, MiB, GiB) with "/s" suffix and 1 decimal place for non-byte units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	sign := ""
+	if bytesPerSec < 0 {
+		sign = "-"
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	divisor := 1024.0
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for unitIndex < len(units)-1 && bytesPerSec >= divisor {
+		bytesPerSec /= divisor
+		unitIndex++
+	}
+
+	// Format: no decimal for bytes, 1 decimal for others
+	if unitIndex == 0 {
+		return fmt.Sprintf("%s%d %s", sign, int(bytesPerSec), units[unitIndex])
+	}
+	return fmt.Sprintf("%s%.1f %s", sign, bytesPerSec, units[unitIndex])
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB boundary
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		{524288, "512.0 KiB/s"},
+		// MiB boundary
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{5242880, "5.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		// GiB boundary
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Fractional values
+		{2560, "2.5 KiB/s"},
+		{3145728, "3.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
