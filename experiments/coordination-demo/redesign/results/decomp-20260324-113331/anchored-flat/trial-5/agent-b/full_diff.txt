+// FormatRate formats a transfer rate in bytes-per-second as a human-readable
+// string using binary units with "/s" suffix: "0 B/s", "1.5 MiB/s", etc.
+// Negative values are prefixed with "-". Non-byte-per-sec units show 1 decimal
+// place (e.g., "1.5 KiB/s"), while B/s is shown as an integer.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := ""
+	if bytesPerSec < 0 {
+		negative = "-"
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KB = 1024
+		MB = 1024 * KB
+		GB = 1024 * MB
+	)
+
+	if bytesPerSec >= GB {
+		return fmt.Sprintf("%s%.1f GiB/s", negative, bytesPerSec/GB)
+	}
+	if bytesPerSec >= MB {
+		return fmt.Sprintf("%s%.1f MiB/s", negative, bytesPerSec/MB)
+	}
+	if bytesPerSec >= KB {
+		return fmt.Sprintf("%s%.1f KiB/s", negative, bytesPerSec/KB)
+	}
+	return fmt.Sprintf("%s%d B/s", negative, int(bytesPerSec))
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero value
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary: exactly 1 KiB/s
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB/s
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+		// Boundary: exactly 1 MiB/s
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB/s
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Boundary: exactly 1 GiB/s
+		{1073741824, "1.0 GiB/s"},
+		// Fractional GiB/s
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-512, "-512 B/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
