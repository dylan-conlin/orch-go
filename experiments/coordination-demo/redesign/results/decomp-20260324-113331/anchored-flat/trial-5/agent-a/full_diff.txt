+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Shows 1 decimal place for non-byte units.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	var sign string
+	absBytes := bytes
+	if bytes < 0 {
+		sign = "-"
+		absBytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	var value float64 = float64(absBytes)
+	unitIndex := 0
+
+	for unitIndex < len(units)-1 && value >= 1024 {
+		value /= 1024
+		unitIndex++
+	}
+
+	if unitIndex == 0 {
+		// For bytes, show as integer
+		return fmt.Sprintf("%s%d B", sign, absBytes)
+	}
+
+	return fmt.Sprintf("%s%.1f %s", sign, value, units[unitIndex])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{1024 + 256, "1.2 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+
+		// Larger values
+		{10 * 1024, "10.0 KiB"},
+		{100 * 1024 * 1024, "100.0 MiB"},
+		{5 * 1024 * 1024 * 1024, "5.0 GiB"},
+		{2 * 1024 * 1024 * 1024 * 1024, "2.0 TiB"},
+
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+
+		// Edge cases with rounding
+		{1229, "1.2 KiB"},  // 1229 / 1024 = 1.199...
+		{1280, "1.2 KiB"},  // 1280 / 1024 = 1.25, rounds to 1.2
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
