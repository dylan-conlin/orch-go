+// FormatBytes formats a byte count as a human-readable string.
+// Uses binary units: B, KiB, MiB, GiB, TiB.
+// Non-byte values show 1 decimal place (e.g., "1.5 MiB").
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	const unit = 1024
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+
+	value := float64(bytes)
+	unitIndex := 0
+
+	for value >= unit && unitIndex < len(units)-1 {
+		value /= unit
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// Bytes: show as integer
+		result = fmt.Sprintf("%d B", int64(value))
+	} else {
+		// Other units: show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+		{1649267441664, "1.5 TiB"},
+		// Larger values
+		{5242880, "5.0 MiB"},
+		{10737418240, "10.0 GiB"},
+		// Rounding tests
+		{2560, "2.5 KiB"},
+		{2621440, "2.5 MiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+		{-1536, "-1.5 KiB"},
+		// Edge case: single byte negative
+		{-1, "-1 B"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
