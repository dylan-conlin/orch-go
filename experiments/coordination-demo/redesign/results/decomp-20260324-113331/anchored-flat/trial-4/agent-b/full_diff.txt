+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Output style: "0 B/s", "512 B/s", "1.5 KiB/s", "1.0 MiB/s", "2.3 GiB/s".
+// Uses binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place for non-byte units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	if rate >= GiB {
+		result = fmt.Sprintf("%.1f GiB/s", rate/GiB)
+	} else if rate >= MiB {
+		result = fmt.Sprintf("%.1f MiB/s", rate/MiB)
+	} else if rate >= KiB {
+		result = fmt.Sprintf("%.1f KiB/s", rate/KiB)
+	} else {
+		result = fmt.Sprintf("%.0f B/s", rate)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact and near boundaries (KiB)
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 * 10, "10.0 KiB/s"},
+		{1024 * 512, "512.0 KiB/s"},
+		{1024*1024 - 1, "1024.0 KiB/s"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB/s"},
+		{1048576 * 1.5, "1.5 MiB/s"},
+		{1048576 * 2, "2.0 MiB/s"},
+		{1048576 * 10, "10.0 MiB/s"},
+		{1048576 * 512, "512.0 MiB/s"},
+		{1048576*1024 - 1, "1024.0 MiB/s"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB/s"},
+		{1073741824 * 1.5, "1.5 GiB/s"},
+		{1073741824 * 2, "2.0 GiB/s"},
+		{1073741824 * 10, "10.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-512, "-512 B/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
