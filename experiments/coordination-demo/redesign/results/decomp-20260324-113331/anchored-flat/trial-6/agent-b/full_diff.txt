+// FormatRate formats a transfer rate in bytes per second as a human-readable
+// string using binary units with "/s" suffix.
+// Output style: "0 B/s", "512 B/s", "1.5 KiB/s", "10.2 MiB/s", "2.1 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	// Handle zero case
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	isNegative := bytesPerSec < 0
+	value := bytesPerSec
+	if isNegative {
+		value = -value
+	}
+
+	// Define binary units
+	const (
+		kiB = 1024
+		miB = 1024 * 1024
+		giB = 1024 * 1024 * 1024
+	)
+
+	var result string
+
+	// Choose appropriate unit
+	if value >= giB {
+		result = fmt.Sprintf("%.1f GiB/s", value/giB)
+	} else if value >= miB {
+		result = fmt.Sprintf("%.1f MiB/s", value/miB)
+	} else if value >= kiB {
+		result = fmt.Sprintf("%.1f KiB/s", value/kiB)
+	} else {
+		result = fmt.Sprintf("%.0f B/s", value)
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero case
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s boundaries and values
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{512 * 1024, "512.0 KiB/s"},
+		// MiB/s boundaries and values
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		// GiB/s values
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
