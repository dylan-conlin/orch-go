+// FormatRate formats a transfer rate in bytes per second into a human-readable
+// string using binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place
+// for non-byte-per-sec units. Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	// Define units and threshold
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	threshold := 1024.0
+
+	// Find the appropriate unit
+	unitIndex := 0
+	for unitIndex < len(units)-1 && bytesPerSec >= threshold {
+		bytesPerSec /= threshold
+		unitIndex++
+	}
+
+	// Format output: no decimal for B/s, 1 decimal for others
+	var result string
+	if unitIndex == 0 {
+		result = fmt.Sprintf("%.0f %s", bytesPerSec, units[unitIndex])
+	} else {
+		result = fmt.Sprintf("%.1f %s", bytesPerSec, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries and KiB/s
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+
+		// MiB/s
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+
+		// GiB/s
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
