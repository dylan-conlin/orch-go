+// FormatBytes formats a byte count into a human-readable string using binary
+// units. Output style: "0 B", "512 B", "1.5 KiB", "2.0 MiB", "1.0 GiB", "1.0 TiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	neg := false
+	if bytes < 0 {
+		neg = true
+		bytes = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+		TiB = 1024 * 1024 * 1024 * 1024
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes < KiB:
+		sign := ""
+		if neg {
+			sign = "-"
+		}
+		return fmt.Sprintf("%s%d B", sign, bytes)
+	case bytes < MiB:
+		value = float64(bytes) / KiB
+		unit = "KiB"
+	case bytes < GiB:
+		value = float64(bytes) / MiB
+		unit = "MiB"
+	case bytes < TiB:
+		value = float64(bytes) / GiB
+		unit = "GiB"
+	default:
+		value = float64(bytes) / TiB
+		unit = "TiB"
+	}
+
+	sign := ""
+	if neg {
+		sign = "-"
+	}
+	return fmt.Sprintf("%s%.1f %s", sign, value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1610612736, "-1.5 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		// Large values
+		{1099511627776 * 10, "10.0 TiB"},
+		{1099511627776 * 100, "100.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
