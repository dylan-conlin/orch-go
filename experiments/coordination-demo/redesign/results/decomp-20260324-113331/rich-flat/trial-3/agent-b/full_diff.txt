+// FormatRate formats a transfer rate as a human-readable string.
+// Output uses binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place
+// for non-byte-per-sec units. Negative rates are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB"}
+	divisor := 1024.0
+
+	// Find the appropriate unit
+	unitIdx := 0
+	for unitIdx < len(units)-1 && bytesPerSec >= divisor {
+		bytesPerSec /= divisor
+		unitIdx++
+	}
+
+	// Format the output
+	var result string
+	if unitIdx == 0 {
+		// B/s - no decimal places
+		result = fmt.Sprintf("%.0f %s/s", bytesPerSec, units[unitIdx])
+	} else {
+		// KiB/s, MiB/s, GiB/s - 1 decimal place
+		result = fmt.Sprintf("%.1f %s/s", bytesPerSec, units[unitIdx])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		// Large values
+		{5368709120, "5.0 GiB/s"},
+		// Fractional values
+		{2560, "2.5 KiB/s"},
+		{5242880, "5.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
