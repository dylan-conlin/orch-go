+// FormatBytes formats byte counts into human-readable strings using binary units.
+// Output uses binary units (B, KiB, MiB, GiB, TiB) with 1 decimal place for
+// non-byte units. Examples: "512 B", "1.5 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	isNegative := bytes < 0
+	b := bytes
+	if isNegative {
+		b = -b
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitSize := int64(1024)
+
+	size := float64(b)
+	unitIndex := 0
+
+	for unitIndex < len(units)-1 && size >= float64(unitSize) {
+		size /= float64(unitSize)
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// Bytes - no decimal
+		result = fmt.Sprintf("%d B", int64(size))
+	} else {
+		// Other units - 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[unitIndex])
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{1649267441664, "1.5 TiB"},
+		// Large values
+		{10737418240, "10.0 GiB"},
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
