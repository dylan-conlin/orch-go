+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "512 B", "1.0 KiB", "1.5 MiB", "1.0 GiB", "2.3 TiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes >= TiB:
+		value = float64(bytes) / float64(TiB)
+		unit = "TiB"
+	case bytes >= GiB:
+		value = float64(bytes) / float64(GiB)
+		unit = "GiB"
+	case bytes >= MiB:
+		value = float64(bytes) / float64(MiB)
+		unit = "MiB"
+	case bytes >= KiB:
+		value = float64(bytes) / float64(KiB)
+		unit = "KiB"
+	default:
+		// For bytes, show as integer
+		if negative {
+			return fmt.Sprintf("-%d B", bytes)
+		}
+		return fmt.Sprintf("%d B", bytes)
+	}
+
+	result := fmt.Sprintf("%.1f %s", value, unit)
+	if negative {
+		return "-" + result
+	}
+	return result
+}
