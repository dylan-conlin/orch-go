+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units are shown with 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	neg := ""
+	if bytesPerSec < 0 {
+		neg = "-"
+		bytesPerSec = -bytesPerSec
+	}
+
+	// Define units in ascending order
+	const (
+		kiB = 1024
+		miB = 1024 * 1024
+		giB = 1024 * 1024 * 1024
+	)
+
+	// Choose the appropriate unit
+	if bytesPerSec < kiB {
+		return fmt.Sprintf("%s%d B/s", neg, int(bytesPerSec))
+	}
+	if bytesPerSec < miB {
+		return fmt.Sprintf("%s%.1f KiB/s", neg, bytesPerSec/kiB)
+	}
+	if bytesPerSec < giB {
+		return fmt.Sprintf("%s%.1f MiB/s", neg, bytesPerSec/miB)
+	}
+	return fmt.Sprintf("%s%.1f GiB/s", neg, bytesPerSec/giB)
+}
