+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units are shown with 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	sign := ""
+	rate := bytesPerSec
+	if rate < 0 {
+		sign = "-"
+		rate = -rate
+	}
+
+	if rate == 0 {
+		return "0 B/s"
+	}
+
+	const (
+		kiB = 1024
+		miB = 1024 * 1024
+		giB = 1024 * 1024 * 1024
+	)
+
+	if rate < kiB {
+		return fmt.Sprintf("%s%d B/s", sign, int(rate))
+	}
+	if rate < miB {
+		return fmt.Sprintf("%s%.1f KiB/s", sign, rate/kiB)
+	}
+	if rate < giB {
+		return fmt.Sprintf("%s%.1f MiB/s", sign, rate/miB)
+	}
+	return fmt.Sprintf("%s%.1f GiB/s", sign, rate/giB)
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		bytesPerSec float64
+		want        string
+	}{
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1572864, "-1.5 MiB/s"},
+		{100, "100 B/s"},
+		{2048, "2.0 KiB/s"},
+		{5242880, "5.0 MiB/s"},
+		{10737418240, "10.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.bytesPerSec)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.bytesPerSec, got, tt.want)
+		}
+	}
