+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Values are shown with 1 decimal place for
+// non-byte units. Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var result string
+	switch {
+	case bytes < KiB:
+		result = fmt.Sprintf("%d B", bytes)
+	case bytes < MiB:
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/float64(KiB))
+	case bytes < GiB:
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/float64(MiB))
+	case bytes < TiB:
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/float64(GiB))
+	default:
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/float64(TiB))
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{10737418240, "10.0 GiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
