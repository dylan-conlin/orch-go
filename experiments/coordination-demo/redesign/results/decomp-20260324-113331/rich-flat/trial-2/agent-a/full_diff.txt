+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units are shown with 1 decimal place.
+// Negative values are prefixed with "-". Examples: "0 B", "1.5 KiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	const (
+		KB int64 = 1024
+		MB int64 = 1024 * KB
+		GB int64 = 1024 * MB
+		TB int64 = 1024 * GB
+	)
+
+	var result string
+	switch {
+	case bytes < KB:
+		result = fmt.Sprintf("%d B", bytes)
+	case bytes < MB:
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/float64(KB))
+	case bytes < GB:
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/float64(MB))
+	case bytes < TB:
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/float64(GB))
+	default:
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/float64(TB))
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+		{10 * 1024, "10.0 KiB"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1048576 + 512*1024, "1.5 MiB"},
+		{2 * 1048576, "2.0 MiB"},
+		{10 * 1048576, "10.0 MiB"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{2 * 1073741824, "2.0 GiB"},
+		// TiB boundary and values
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		{2 * 1099511627776, "2.0 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
