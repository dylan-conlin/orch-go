+// FormatRate formats a transfer rate in bytes per second into a human-readable
+// string using binary units (B/s, KiB/s, MiB/s, GiB/s) with one decimal place
+// for non-byte-per-second units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case bytesPerSec >= GiB:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	case bytesPerSec >= MiB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	case bytesPerSec >= KiB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	default:
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+
+		// Fractional values in KiB
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+
+		// Fractional values in MiB
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{2621440, "2.5 MiB/s"},
+
+		// Fractional values in GiB
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+
+		// Large values
+		{10737418240, "10.0 GiB/s"},
+		{107374182400, "100.0 GiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+
+		// Edge cases with decimals
+		{1126.4, "1.1 KiB/s"},
+		{2560.5, "2.5 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
