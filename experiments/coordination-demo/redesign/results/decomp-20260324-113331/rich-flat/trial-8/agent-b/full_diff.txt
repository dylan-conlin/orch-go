+// FormatRate formats a transfer rate in bytes per second into a human-readable
+// string using binary units (B/s, KiB/s, MiB/s, GiB/s).
+// Non-byte units display one decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	unitIndex := 0
+	value := bytesPerSec
+
+	// Scale down until we find an appropriate unit
+	for value >= 1024 && unitIndex < len(units)-1 {
+		value /= 1024
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes per second, show as integer
+		result = fmt.Sprintf("%.0f %s", value, units[unitIndex])
+	} else {
+		// For larger units, show one decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and near-zero values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		// Exact boundary: 1 KiB
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB values
+		{1536, "1.5 KiB/s"},
+		{1024 * 1.2, "1.2 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// Exact boundary: 1 MiB
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB values
+		{1048576 * 1.5, "1.5 MiB/s"},
+		{1048576 * 2, "2.0 MiB/s"},
+		// Exact boundary: 1 GiB
+		{1073741824, "1.0 GiB/s"},
+		// Large GiB values
+		{1073741824 * 10, "10.0 GiB/s"},
+		{1073741824 * 2.5, "2.5 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
