+// FormatBytes formats a byte count as a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Negative values are prefixed with "-". Examples: "512 B", "1.5 MiB", "3.2 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const unit = 1024
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+
+	size := float64(bytes)
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for size >= unit && unitIndex < len(units)-1 {
+		size /= unit
+		unitIndex++
+	}
+
+	// Format output
+	var result string
+	if unitIndex == 0 {
+		// Bytes: no decimal places
+		result = fmt.Sprintf("%d %s", int64(size), units[unitIndex])
+	} else {
+		// Other units: 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries and just over
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Large values
+		{5368709120, "5.0 GiB"},
+		{10995116277760, "10.0 TiB"},
+		// Values between boundaries
+		{2560, "2.5 KiB"},
+		{3145728, "3.0 MiB"},
+		{5368709120, "5.0 GiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		// Edge case: one byte less than boundary
+		{1023, "1023 B"},
+		{1048575, "1024.0 KiB"},
+		{1073741823, "1024.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
