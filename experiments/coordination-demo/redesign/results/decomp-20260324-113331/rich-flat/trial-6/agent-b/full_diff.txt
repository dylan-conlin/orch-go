+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units (B/s, KiB/s, MiB/s, GiB/s). Non-byte-per-sec units
+// are displayed with 1 decimal place. Negative rates are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	bytes := bytesPerSec
+	if isNegative {
+		bytes = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var formatted string
+	switch {
+	case bytes < KiB:
+		formatted = fmt.Sprintf("%.0f B/s", bytes)
+	case bytes < MiB:
+		formatted = fmt.Sprintf("%.1f KiB/s", bytes/KiB)
+	case bytes < GiB:
+		formatted = fmt.Sprintf("%.1f MiB/s", bytes/MiB)
+	default:
+		formatted = fmt.Sprintf("%.1f GiB/s", bytes/GiB)
+	}
+
+	if isNegative {
+		formatted = "-" + formatted
+	}
+	return formatted
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries and transitions
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Large values
+		{10737418240, "10.0 GiB/s"},
+		{107374182400, "100.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-512, "-512 B/s"},
+		// Fractional values less than 1 KiB
+		{512.5, "512 B/s"},
+		{1023.9, "1024 B/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
