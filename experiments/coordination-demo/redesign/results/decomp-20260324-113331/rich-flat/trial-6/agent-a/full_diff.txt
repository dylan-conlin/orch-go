+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units are shown with 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitSize := int64(1024)
+
+	// Handle negative values
+	negative := false
+	absBytes := bytes
+	if bytes < 0 {
+		negative = true
+		absBytes = -bytes
+	}
+
+	// Find the appropriate unit
+	unitIndex := 0
+	value := float64(absBytes)
+
+	for unitIndex < len(units)-1 && value >= float64(unitSize) {
+		value /= float64(unitSize)
+		unitIndex++
+	}
+
+	// Format the result
+	var result string
+	if unitIndex == 0 {
+		// Bytes unit - no decimal places
+		result = fmt.Sprintf("%d B", int64(value))
+	} else {
+		// Larger units - 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+	tests := []struct {
+		input int64
+		want  string

+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},

+		{1572864, "1.5 MiB"},

+		{1649267441664, "1.5 TiB"},

+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
