+// FormatBytes formats a byte count as a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB) with 1 decimal place for non-byte units.
+// Negative values are prefixed with "-". Zero returns "0 B".
+func FormatBytes(bytes int64) string {
+	const (
+		KiB = 1024
+		MiB = KiB * 1024
+		GiB = MiB * 1024
+		TiB = GiB * 1024
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	var result string
+	if bytes < KiB {
+		result = fmt.Sprintf("%d B", bytes)
+	} else if bytes < MiB {
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/float64(KiB))
+	} else if bytes < GiB {
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/float64(MiB))
+	} else if bytes < TiB {
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/float64(GiB))
+	} else {
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/float64(TiB))
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 * 10, "10.0 KiB"},
+		// MiB boundaries
+		{1048576, "1.0 MiB"},
+		{1048576 + 512*1024, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1048576 * 10, "10.0 MiB"},
+		// GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB boundaries
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
