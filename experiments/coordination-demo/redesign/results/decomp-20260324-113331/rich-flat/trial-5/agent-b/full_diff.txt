+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units.
+// Handles zero as "0 B/s" and negative values with "-" prefix.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	prefix := ""
+	if isNegative {
+		prefix = "-"
+	}
+
+	if bytesPerSec < KiB {
+		return fmt.Sprintf("%s%.0f B/s", prefix, bytesPerSec)
+	} else if bytesPerSec < MiB {
+		return fmt.Sprintf("%s%.1f KiB/s", prefix, bytesPerSec/KiB)
+	} else if bytesPerSec < GiB {
+		return fmt.Sprintf("%s%.1f MiB/s", prefix, bytesPerSec/MiB)
+	} else {
+		return fmt.Sprintf("%s%.1f GiB/s", prefix, bytesPerSec/GiB)
+	}
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Kilobytes per second (1 KiB = 1024 bytes)
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048575, "1024.0 KiB/s"},
+		// Megabytes per second (1 MiB = 1048576 bytes)
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{5242880, "5.0 MiB/s"},
+		{1073741823, "1024.0 MiB/s"},
+		// Gigabytes per second (1 GiB = 1073741824 bytes)
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		{-1536, "-1.5 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
