+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "512 B", "1.5 KiB", "2.0 MiB", "1.0 GiB", "5.0 TiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+
+	// Handle negative values
+	isNegative := bytes < 0
+	size := float64(bytes)
+	if isNegative {
+		size = -size
+	}
+
+	// Find the appropriate unit
+	unitIndex := 0
+	for unitIndex < len(units)-1 && size >= 1024 {
+		size /= 1024
+		unitIndex++
+	}
+
+	// Format based on unit
+	prefix := ""
+	if isNegative {
+		prefix = "-"
+	}
+
+	if unitIndex == 0 {
+		// Bytes - show as integer
+		return fmt.Sprintf("%s%d %s", prefix, int64(size), units[unitIndex])
+	} else {
+		// KiB and higher - show with 1 decimal place
+		return fmt.Sprintf("%s%.1f %s", prefix, size, units[unitIndex])
+	}
+}
