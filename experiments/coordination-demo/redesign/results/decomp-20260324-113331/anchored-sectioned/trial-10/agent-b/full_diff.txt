+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units. Output style: "512 B/s", "1.5 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative rates by extracting sign and working with absolute value
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	// Define units and thresholds
+	units := []struct {
+		name      string
+		threshold float64
+	}{
+		{"B", 1},
+		{"KiB", 1024},
+		{"MiB", 1024 * 1024},
+		{"GiB", 1024 * 1024 * 1024},
+	}
+
+	// Find the appropriate unit
+	var unitName string
+	var divisor float64
+	for i := len(units) - 1; i >= 0; i-- {
+		if rate >= units[i].threshold {
+			unitName = units[i].name
+			divisor = units[i].threshold
+			break
+		}
+	}
+
+	// Format with appropriate decimal places
+	formatted := rate / divisor
+	var result string
+	if unitName == "B" {
+		// No decimal places for bytes
+		result = fmt.Sprintf("%d %s/s", int(formatted), unitName)
+	} else {
+		// 1 decimal place for other units
+		result = fmt.Sprintf("%.1f %s/s", formatted, unitName)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
