+// === Size Formatting ===
+
+// FormatBytes formats a byte count as a human-readable string.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.0 MiB", "1.0 GiB", "1.0 TiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	var divisor int64 = 1
+	unitIndex := 0
+
+	absBytes := bytes
+	if bytes < 0 {
+		absBytes = -bytes
+	}
+
+	// Find the appropriate unit by dividing by 1024
+	for unitIndex < len(units)-1 && absBytes >= 1024*divisor {
+		divisor *= 1024
+		unitIndex++
+	}
+
+	// Bytes: no decimal place
+	if unitIndex == 0 {
+		if bytes < 0 {
+			return fmt.Sprintf("-%d B", absBytes)
+		}
+		return fmt.Sprintf("%d B", bytes)
+	}
+
+	// KiB and larger: 1 decimal place
+	value := float64(absBytes) / float64(divisor)
+	if bytes < 0 {
+		return fmt.Sprintf("-%.1f %s", value, units[unitIndex])
+	}
+	return fmt.Sprintf("%.1f %s", value, units[unitIndex])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1024 * 1024, "1.0 MiB"},
+		{1024 * 1024 * 1024, "1.0 GiB"},
+		{1024 * 1024 * 1024 * 1024, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"}, // 1.5 * 1024 * 1024
+		{1610612736, "1.5 GiB"}, // 1.5 * 1024 * 1024 * 1024
+		{1688849860000, "1.5 TiB"}, // 1.5 * 1024 * 1024 * 1024 * 1024
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1024 * 1024, "-1.0 MiB"},
+		{-1572864, "-1.5 MiB"},
+		// Large values
+		{1024 * 1024 * 1024 * 1024 * 1024, "1024.0 TiB"}, // 1 PiB (stays in TiB)
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
