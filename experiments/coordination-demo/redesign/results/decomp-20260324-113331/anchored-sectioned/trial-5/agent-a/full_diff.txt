+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string using
+// binary units (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	value := float64(bytes)
+
+	for _, unit := range units {
+		if value < 1024.0 {
+			var formatted string
+			if unit == "B" {
+				formatted = fmt.Sprintf("%.0f %s", value, unit)
+			} else {
+				formatted = fmt.Sprintf("%.1f %s", value, unit)
+			}
+			if negative {
+				return "-" + formatted
+			}
+			return formatted
+		}
+		value /= 1024.0
+	}
+
+	// If we've exhausted all units, format in TiB with 1 decimal
+	formatted := fmt.Sprintf("%.1f TiB", value)
+	if negative {
+		return "-" + formatted
+	}
+	return formatted
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{1024 + 512, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+
+		// Larger values
+		{2048, "2.0 KiB"},
+		{3 * 1024 * 1024, "3.0 MiB"},
+		{5 * 1024 * 1024 * 1024, "5.0 GiB"},
+		{10 * 1024 * 1024 * 1024 * 1024, "10.0 TiB"},
+
+		// Negative values
+		{-1, "-1 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+
+		// Edge cases with decimals
+		{2560, "2.5 KiB"},
+		{2621440, "2.5 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
