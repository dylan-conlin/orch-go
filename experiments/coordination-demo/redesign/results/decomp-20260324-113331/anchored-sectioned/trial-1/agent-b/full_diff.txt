+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	switch {
+	case rate < KiB:
+		result = fmt.Sprintf("%g B/s", rate)
+	case rate < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", rate/KiB)
+	case rate < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", rate/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", rate/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}
