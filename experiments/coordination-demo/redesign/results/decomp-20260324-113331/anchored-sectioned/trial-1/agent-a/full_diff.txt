+// === Size Formatting ===
+
+// FormatBytes converts a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units are shown with one decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var result string
+	switch {
+	case bytes < KiB:
+		result = fmt.Sprintf("%d B", bytes)
+	case bytes < MiB:
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/float64(KiB))
+	case bytes < GiB:
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/float64(MiB))
+	case bytes < TiB:
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/float64(GiB))
+	default:
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/float64(TiB))
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}
