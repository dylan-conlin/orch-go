+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place
+// for non-byte-per-sec units.
+func FormatRate(bytesPerSec float64) string {
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	absRate := bytesPerSec
+	if negative {
+		absRate = -bytesPerSec
+	}
+
+	var result string
+	switch {
+	case absRate < KiB:
+		// Return as B/s without decimal
+		result = fmt.Sprintf("%.0f B/s", absRate)
+	case absRate < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", absRate/KiB)
+	case absRate < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", absRate/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", absRate/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
