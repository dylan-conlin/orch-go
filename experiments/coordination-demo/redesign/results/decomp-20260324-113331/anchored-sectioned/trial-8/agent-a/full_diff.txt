+// === Size Formatting ===
+
+// FormatBytes converts a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte values show 1 decimal place.
+// Examples: "512 B", "1.5 MiB", "-1.0 KiB".
+func FormatBytes(bytes int64) string {
+	const (
+		B  = 1
+		Ki = 1024
+		Mi = Ki * 1024
+		Gi = Mi * 1024
+		Ti = Gi * 1024
+	)
+
+	abs := bytes
+	sign := ""
+	if bytes < 0 {
+		sign = "-"
+		abs = -bytes
+	}
+
+	if abs < Ki {
+		return fmt.Sprintf("%s%d B", sign, abs)
+	}
+
+	if abs < Mi {
+		return fmt.Sprintf("%s%.1f KiB", sign, float64(abs)/float64(Ki))
+	}
+
+	if abs < Gi {
+		return fmt.Sprintf("%s%.1f MiB", sign, float64(abs)/float64(Mi))
+	}
+
+	if abs < Ti {
+		return fmt.Sprintf("%s%.1f GiB", sign, float64(abs)/float64(Gi))
+	}
+
+	return fmt.Sprintf("%s%.1f TiB", sign, float64(abs)/float64(Ti))
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Kilobytes - exact and boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 + 512, "1.5 KiB"},
+
+		// Megabytes
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{10485760, "10.0 MiB"},
+
+		// Gigabytes
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+
+		// Terabytes
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+
+		// Negative values
+		{-1, "-1 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+
+		// Edge cases
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1025, "1.0 KiB"},
+	}
+

+		got := FormatBytes(tt.input)
