+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate as a human-readable string.
+// Output style: "512 B/s", "1.5 MiB/s", "1.0 GiB/s".
+// Uses binary units (KiB/s, MiB/s, GiB/s) with 1 decimal place for non-byte units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	threshold := 1024.0
+
+	// Find the appropriate unit by dividing down from bytes
+	for i, unit := range units {
+		if bytesPerSec < threshold || i == len(units)-1 {
+			var result string
+			if i == 0 {
+				// B/s: show no decimals
+				result = fmt.Sprintf("%.0f %s", bytesPerSec, unit)
+			} else {
+				// Others: show 1 decimal place
+				result = fmt.Sprintf("%.1f %s", bytesPerSec, unit)
+			}
+			if isNegative {
+				result = "-" + result
+			}
+			return result
+		}
+		bytesPerSec /= threshold
+	}
+
+	return ""
+}
