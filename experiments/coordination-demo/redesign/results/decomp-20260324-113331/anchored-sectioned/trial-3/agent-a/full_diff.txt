+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	units := []struct {
+		name string
+		size int64
+	}{
+		{"TiB", 1024 * 1024 * 1024 * 1024},
+		{"GiB", 1024 * 1024 * 1024},
+		{"MiB", 1024 * 1024},
+		{"KiB", 1024},
+		{"B", 1},
+	}
+
+	for _, unit := range units {
+		if absBytes >= unit.size {
+			if unit.name == "B" {
+				result := fmt.Sprintf("%d %s", absBytes, unit.name)
+				if negative {
+					result = "-" + result
+				}
+				return result
+			}
+			value := float64(absBytes) / float64(unit.size)
+			result := fmt.Sprintf("%.1f %s", value, unit.name)
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	return "0 B"
+}
