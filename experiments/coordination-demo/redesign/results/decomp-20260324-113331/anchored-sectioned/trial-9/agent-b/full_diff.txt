+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate (bytes per second) as a human-readable
+// string using binary units with "/s" suffix.
+// Shows 1 decimal place for non-byte-per-sec units.
+// Examples: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kiB = 1024
+		miB = 1024 * kiB
+		giB = 1024 * miB
+	)
+
+	var result string
+	switch {
+	case bytesPerSec < kiB:
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	case bytesPerSec < miB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kiB)
+	case bytesPerSec < giB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/miB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/giB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}
