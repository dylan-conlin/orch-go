+// === Size Formatting ===
+
+// FormatBytes converts a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units display with 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	sign := ""
+	if bytes < 0 {
+		sign = "-"
+		bytes = -bytes
+	}
+
+	const (
+		B  = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	switch {
+	case bytes < KiB:
+		return fmt.Sprintf("%s%d B", sign, bytes)
+	case bytes < MiB:
+		return fmt.Sprintf("%s%.1f KiB", sign, float64(bytes)/float64(KiB))
+	case bytes < GiB:
+		return fmt.Sprintf("%s%.1f MiB", sign, float64(bytes)/float64(MiB))
+	case bytes < TiB:
+		return fmt.Sprintf("%s%.1f GiB", sign, float64(bytes)/float64(GiB))
+	default:
+		return fmt.Sprintf("%s%.1f TiB", sign, float64(bytes)/float64(TiB))
+	}
+}
