+// === Rate Formatting ===

+// FormatRate formats a transfer rate as a human-readable string using binary units.
+// Output style: "0 B/s", "512 B/s", "1.5 KiB/s", "1.0 MiB/s", "2.3 GiB/s".
+// Shows 1 decimal place for non-byte-per-sec units. Handles negative values by prefixing with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	divisor := 1024.0
+
+	value := bytesPerSec
+	unitIndex := 0
+
+	for unitIndex < len(units)-1 && value >= divisor {
+		value /= divisor
+		unitIndex++
+	}
+
+	// Format output
+	var result string
+	if unitIndex == 0 {
+		// For bytes per second, show as integer
+		result = fmt.Sprintf("%d %s", int(value), units[unitIndex])
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
