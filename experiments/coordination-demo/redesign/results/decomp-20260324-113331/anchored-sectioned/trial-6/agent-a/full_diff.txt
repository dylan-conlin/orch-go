+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string with binary units.
+// Returns strings like "512 B", "1.5 KiB", "2.3 GiB" with one decimal place for
+// non-byte units. Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	var result string
+	switch {
+	case bytes < KiB:
+		result = fmt.Sprintf("%d B", bytes)
+	case bytes < MiB:
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/float64(KiB))
+	case bytes < GiB:
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/float64(MiB))
+	case bytes < TiB:
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/float64(GiB))
+	default:
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/float64(TiB))
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}
