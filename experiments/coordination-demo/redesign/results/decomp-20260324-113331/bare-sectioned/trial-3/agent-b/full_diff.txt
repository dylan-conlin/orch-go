+// === Rate Formatting ===
+
+// FormatRate formats a rate (bytes per second) as a human-readable string.
+// Output style: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s", "1.0 GiB/s", etc.
+// Shows 1 decimal place for non-byte-per-sec units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	divisor := 1024.0
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for unitIndex < len(units)-1 && bytesPerSec >= divisor {
+		bytesPerSec /= divisor
+		unitIndex++
+	}
+
+	// Format the result
+	var result string
+	if unitIndex == 0 {
+		// For bytes per second, show as integer
+		result = fmt.Sprintf("%d %s", int(bytesPerSec), units[unitIndex])
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", bytesPerSec, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
