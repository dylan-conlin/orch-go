+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte values display 1 decimal place.
+// Negative values are prefixed with "-". Zero returns "0 B".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	sign := ""
+	if bytes < 0 {
+		sign = "-"
+		bytes = -bytes
+	}
+
+	const (
+		_   = iota
+		KiB = 1 << (10 * iota)
+		MiB
+		GiB
+		TiB
+	)
+
+	switch {
+	case bytes < KiB:
+		return fmt.Sprintf("%s%d B", sign, bytes)
+	case bytes < MiB:
+		return fmt.Sprintf("%s%.1f KiB", sign, float64(bytes)/KiB)
+	case bytes < GiB:
+		return fmt.Sprintf("%s%.1f MiB", sign, float64(bytes)/MiB)
+	case bytes < TiB:
+		return fmt.Sprintf("%s%.1f GiB", sign, float64(bytes)/GiB)
+	default:
+		return fmt.Sprintf("%s%.1f TiB", sign, float64(bytes)/TiB)
+	}
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		// Exact MiB boundary
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{10485760, "10.0 MiB"},
+		// Exact GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{5368709120, "5.0 GiB"},
+		// Exact TiB boundary
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{5497558138880, "5.0 TiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{2560, "2.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-512, "-512 B"},
+		{-1536, "-1.5 KiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
