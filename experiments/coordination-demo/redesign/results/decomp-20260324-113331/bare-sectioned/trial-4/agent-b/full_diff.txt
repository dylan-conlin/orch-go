+// === Rate Formatting ===

+// FormatRate formats a transfer rate into a human-readable string using binary units.
+// Output style: "512 B/s", "1.5 MiB/s", "2.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kib = 1024
+		mib = 1024 * 1024
+		gib = 1024 * 1024 * 1024
+	)
+
+	var result string
+
+	if bytesPerSec < kib {
+		result = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	} else if bytesPerSec < mib {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kib)
+	} else if bytesPerSec < gib {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/mib)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/gib)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
