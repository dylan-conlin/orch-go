+// === Size Formatting ===
+
+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "512 B", "1.0 KiB", "1.5 MiB", "2.3 GiB", etc.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const unit = 1024
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+
+	size := float64(bytes)
+	unitIdx := 0
+
+	for size >= unit && unitIdx < len(units)-1 {
+		size /= unit
+		unitIdx++
+	}
+
+	var result string
+	if unitIdx == 0 {
+		result = fmt.Sprintf("%d %s", int64(size), units[unitIdx])
+	} else {
+		result = fmt.Sprintf("%.1f %s", size, units[unitIdx])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+		{1048576 * 2, "2.0 MiB"},
+		{1048576 * 512, "512.0 MiB"},

+		{1073741824, "1.0 GiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{1073741824 * 2, "2.0 GiB"},
+		{1073741824 * 10, "10.0 GiB"},

+		{1099511627776 * 2, "2.0 TiB"},
