+// === Rate Formatting ===

+// FormatRate formats a transfer rate in bytes per second as a human-readable string.
+// Output style: "512 B/s", "1.5 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var formatted string
+	if bytesPerSec >= GiB {
+		formatted = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	} else if bytesPerSec >= MiB {
+		formatted = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	} else if bytesPerSec >= KiB {
+		formatted = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	} else {
+		formatted = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	}
+
+	if negative {
+		return "-" + formatted
+	}
+	return formatted
+}
