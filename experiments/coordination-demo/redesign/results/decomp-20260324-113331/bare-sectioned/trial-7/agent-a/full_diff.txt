+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string using binary
+// units. Output style: "512 B", "1.0 KiB", "1.5 MiB", "1.0 GiB", "1.0 TiB".
+// Negative values are formatted with a leading "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		b   = 1
+		kib = 1024 * b
+		mib = 1024 * kib
+		gib = 1024 * mib
+		tib = 1024 * gib
+	)
+
+	var result string
+	switch {
+	case bytes < kib:
+		result = fmt.Sprintf("%d B", bytes)
+	case bytes < mib:
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/float64(kib))
+	case bytes < gib:
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/float64(mib))
+	case bytes < tib:
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/float64(gib))
+	default:
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/float64(tib))
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+	tests := []struct {
+		input int64
+		want  string

+		{512, "512 B"},

+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		// Between KiB and MiB
+		{10240, "10.0 KiB"},
+		{102400, "100.0 KiB"},
+		// Exact boundary: 1 MiB
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Between MiB and GiB
+		{10485760, "10.0 MiB"},
+		{104857600, "100.0 MiB"},
+		// Exact boundary: 1 GiB
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Between GiB and TiB
+		{10737418240, "10.0 GiB"},
+		{107374182400, "100.0 GiB"},
+		// Exact boundary: 1 TiB
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},

+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
