+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output uses binary units: B, KiB, MiB, GiB, TiB with 1 decimal place for
+// non-byte values. Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	abs := bytes
+	negative := false
+	if bytes < 0 {
+		negative = true
+		abs = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	size := float64(abs)
+	unitIdx := 0
+
+	for unitIdx < len(units)-1 && size >= 1024 {
+		size /= 1024
+		unitIdx++
+	}
+
+	var result string
+	if unitIdx == 0 {
+		// For bytes, show as integer
+		result = fmt.Sprintf("%d B", abs)
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[unitIdx])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+

+
+		// Large values
+		{10737418240, "10.0 GiB"},
+		{1099511627776 * 5, "5.0 TiB"},
+		{1099511627776 * 10, "10.0 TiB"},
+

+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
