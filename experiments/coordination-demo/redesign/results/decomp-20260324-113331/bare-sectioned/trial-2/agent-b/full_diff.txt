+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+// Zero is returned as "0 B/s". Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	threshold := 1024.0
+
+	for i := 0; i < len(units)-1; i++ {
+		if bytesPerSec < threshold {
+			if i == 0 {
+				// B/s: no decimal places
+				result := fmt.Sprintf("%d %s", int(bytesPerSec), units[i])
+				if negative {
+					result = "-" + result
+				}
+				return result
+			}
+			// KiB/s and above: 1 decimal place
+			result := fmt.Sprintf("%.1f %s", bytesPerSec, units[i])
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+		bytesPerSec /= threshold
+	}
+
+	// GiB/s: 1 decimal place
+	result := fmt.Sprintf("%.1f %s", bytesPerSec, units[len(units)-1])
+	if negative {
+		result = "-" + result
+	}
+	return result
+}
