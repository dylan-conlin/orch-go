+// === Size Formatting ===
+
+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", "1.0 GiB", "2.3 TiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var size float64
+	var unit string
+
+	switch {
+	case absBytes < KiB:
+		result := fmt.Sprintf("%d B", absBytes)
+		if negative {
+			return "-" + result
+		}
+		return result
+	case absBytes < MiB:
+		size = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	case absBytes < GiB:
+		size = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes < TiB:
+		size = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		size = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	result := fmt.Sprintf("%.1f %s", size, unit)
+	if negative {
+		return "-" + result
+	}
+	return result
+}
