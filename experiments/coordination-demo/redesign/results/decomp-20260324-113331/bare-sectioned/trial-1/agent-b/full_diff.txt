+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate into a human-readable string.
+// Uses binary units: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units.
+// Example: FormatRate(1024) returns "1.0 KiB/s"
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	value := bytesPerSec
+	if isNegative {
+		value = -value
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	if value < KiB {
+		result = fmt.Sprintf("%.0f B/s", value)
+	} else if value < MiB {
+		result = fmt.Sprintf("%.1f KiB/s", value/KiB)
+	} else if value < GiB {
+		result = fmt.Sprintf("%.1f MiB/s", value/MiB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", value/GiB)
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+
+	return result
+}
