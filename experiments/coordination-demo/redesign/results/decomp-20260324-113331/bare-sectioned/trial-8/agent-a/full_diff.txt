+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.0 MiB", "1.2 GiB", "5.3 TiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	boundaries := []int64{1, 1024, 1024 * 1024, 1024 * 1024 * 1024, 1024 * 1024 * 1024 * 1024}
+
+	var result string
+	for i := len(boundaries) - 1; i >= 0; i-- {
+		if bytes >= boundaries[i] {
+			if i == 0 {
+				// Bytes unit - no decimal
+				result = fmt.Sprintf("%d B", bytes)
+			} else {
+				// Other units - 1 decimal place
+				value := float64(bytes) / float64(boundaries[i])
+				result = fmt.Sprintf("%.1f %s", value, units[i])
+			}
+			break
+		}
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}
