+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate in bytes per second as a human-readable
+// string using binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place
+// for non-byte-per-second units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	prefix := ""
+	if bytesPerSec < 0 {
+		prefix = "-"
+		bytesPerSec = -bytesPerSec
+	}
+
+	// Binary units (1024 bytes = 1 KiB)
+	const (
+		kiB = 1024
+		miB = 1024 * kiB
+		giB = 1024 * miB
+	)
+
+	if bytesPerSec >= giB {
+		return fmt.Sprintf("%s%.1f GiB/s", prefix, bytesPerSec/giB)
+	}
+	if bytesPerSec >= miB {
+		return fmt.Sprintf("%s%.1f MiB/s", prefix, bytesPerSec/miB)
+	}
+	if bytesPerSec >= kiB {
+		return fmt.Sprintf("%s%.1f KiB/s", prefix, bytesPerSec/kiB)
+	}
+
+	// B/s - use integer format
+	return fmt.Sprintf("%s%d B/s", prefix, int(bytesPerSec))
+}
