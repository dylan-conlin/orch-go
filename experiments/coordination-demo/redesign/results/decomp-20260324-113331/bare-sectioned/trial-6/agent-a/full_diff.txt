+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", etc.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes < KiB:
+		value = float64(bytes)
+		unit = "B"
+	case bytes < MiB:
+		value = float64(bytes) / KiB
+		unit = "KiB"
+	case bytes < GiB:
+		value = float64(bytes) / MiB
+		unit = "MiB"
+	case bytes < TiB:
+		value = float64(bytes) / GiB
+		unit = "GiB"
+	default:
+		value = float64(bytes) / TiB
+		unit = "TiB"
+	}
+
+	var result string
+	if unit == "B" {
+		result = fmt.Sprintf("%d %s", int64(value), unit)
+	} else {
+		result = fmt.Sprintf("%.1f %s", value, unit)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},

+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+		{1649267441664, "1.5 TiB"},
+		// Other values
+		{2048, "2.0 KiB"},
+		{512000, "500.0 KiB"},
+		{5368709120, "5.0 GiB"},
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-512, "-512 B"},
+		{-1536, "-1.5 KiB"},
+		// Large values (beyond 1 TiB, still shown in TiB)
+		{1125899906842624, "1024.0 TiB"},
+		{1099511627776000, "1000.0 TiB"},
