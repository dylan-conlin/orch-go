+// === Rate Formatting ===
+
+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units.
+// Examples: "0 B/s", "512 B/s", "1.5 MiB/s", "2.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	// Handle zero
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	// Define unit boundaries (binary units, largest first)
+	units := []struct {
+		name     string
+		boundary float64
+	}{
+		{"GiB/s", 1024 * 1024 * 1024},
+		{"MiB/s", 1024 * 1024},
+		{"KiB/s", 1024},
+		{"B/s", 1},
+	}
+
+	// Find the appropriate unit and format
+	var result string
+	for _, unit := range units {
+		if bytesPerSec >= unit.boundary {
+			value := bytesPerSec / unit.boundary
+			if unit.boundary == 1 {
+				// For B/s, show as integer
+				result = fmt.Sprintf("%.0f %s", value, unit.name)
+			} else {
+				// For KiB/s and larger, show 1 decimal place
+				result = fmt.Sprintf("%.1f %s", value, unit.name)
+			}
+			break
+		}
+	}
+
+	// Prefix with "-" if negative
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
