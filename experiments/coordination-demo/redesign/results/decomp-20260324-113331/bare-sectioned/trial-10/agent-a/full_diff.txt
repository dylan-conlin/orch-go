+// === Size Formatting ===
+
+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", "1.0 GiB", "1.0 TiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	isNegative := bytes < 0
+	absBytes := bytes
+	if isNegative {
+		absBytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	const unitStep int64 = 1024
+
+	// Find the appropriate unit
+	var size float64 = float64(absBytes)
+	unitIndex := 0
+
+	for unitIndex = 0; unitIndex < len(units)-1; unitIndex++ {
+		if size < float64(unitStep) {
+			break
+		}
+		size /= float64(unitStep)
+	}
+
+	// Format the result
+	var result string
+	if unitIndex == 0 {
+		// Bytes - no decimal places
+		result = fmt.Sprintf("%d B", int64(size))
+	} else {
+		// Other units - 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[unitIndex])
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+
+	return result
+}

+		input int64
+		want  string

+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+
+		// Large values
+		{1099511627776 * 2, "2.0 TiB"},
+		{1099511627776 * 5, "5.0 TiB"},
+
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+

+		{1229, "1.2 KiB"},   // 1.199... rounds to 1.2
