+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte values show 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	var value float64 = float64(bytes)
+
+	for i := 0; i < len(units)-1; i++ {
+		if value < 1024 {
+			if i == 0 {
+				// Byte unit: no decimal places
+				prefix := ""
+				if negative {
+					prefix = "-"
+				}
+				return fmt.Sprintf("%s%d %s", prefix, int64(value), units[i])
+			}
+			// Other units: 1 decimal place
+			prefix := ""
+			if negative {
+				prefix = "-"
+			}
+			return fmt.Sprintf("%s%.1f %s", prefix, value, units[i])
+		}
+		value /= 1024
+	}
+
+	// TiB (largest unit)
+	prefix := ""
+	if negative {
+		prefix = "-"
+	}
+	return fmt.Sprintf("%s%.1f %s", prefix, value, units[len(units)-1])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (< 1 KiB)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries (KiB)
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+		// Values between boundaries (MiB)
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{3145728, "3.0 MiB"},
+		// Values between boundaries (GiB)
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Large values (TiB)
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
