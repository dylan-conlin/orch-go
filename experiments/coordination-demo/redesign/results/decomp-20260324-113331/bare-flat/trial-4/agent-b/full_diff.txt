+// FormatRate formats a transfer rate (bytes per second) into human-readable form.
+// Uses binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place for
+// non-byte-per-second units. Examples: "0 B/s", "512 B/s", "1.5 MiB/s".
+func FormatRate(bytesPerSec float64) string {
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	// Handle zero
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	abs := bytesPerSec
+	if negative {
+		abs = -abs
+	}
+
+	// Choose unit and format
+	var result string
+	if abs >= GiB {
+		result = fmt.Sprintf("%.1f GiB/s", abs/GiB)
+	} else if abs >= MiB {
+		result = fmt.Sprintf("%.1f MiB/s", abs/MiB)
+	} else if abs >= KiB {
+		result = fmt.Sprintf("%.1f KiB/s", abs/KiB)
+	} else {
+		result = fmt.Sprintf("%.0f B/s", abs)
+	}
+
+	// Add negative prefix if needed
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries (1024 bytes = 1 KiB)
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// KiB to MiB boundary
+		{1023 * 1024, "1023.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1536 * 1024, "1.5 MiB/s"},
+		{2 * 1024 * 1024, "2.0 MiB/s"},
+		// MiB to GiB boundary
+		{1023 * 1024 * 1024, "1023.0 MiB/s"},
+		{1024 * 1024 * 1024, "1.0 GiB/s"},
+		{1536 * 1024 * 1024, "1.5 GiB/s"},
+		// Large GiB values
+		{10 * 1024 * 1024 * 1024, "10.0 GiB/s"},
+		{100.5 * 1024 * 1024 * 1024, "100.5 GiB/s"},
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1024 * 1024 * 1024, "-1.0 GiB/s"},
+		// Fractional values
+		{1536.5, "1.5 KiB/s"},
+		{2048.75, "2.0 KiB/s"},
+		{512.5, "512 B/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
