+// FormatBytes formats a byte count as a human-readable string.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", etc.
+// Uses binary units (1024 bytes per unit): B, KiB, MiB, GiB, TiB.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	var size float64 = float64(bytes)
+
+	for i := 0; i < len(units); i++ {
+		if size < 1024.0 || i == len(units)-1 {
+			// Format the result
+			var result string
+			if i == 0 {
+				// Bytes: no decimal places
+				result = fmt.Sprintf("%d %s", int64(size), units[i])
+			} else {
+				// Other units: 1 decimal place
+				result = fmt.Sprintf("%.1f %s", size, units[i])
+			}
+
+			if negative {
+				return "-" + result
+			}
+			return result
+		}
+		size /= 1024.0
+	}
+
+	// Should never reach here
+	return ""
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small byte values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Kibibytes: exact boundaries and in-between
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+		{1024*1024 - 1, "1024.0 KiB"},
+
+		// Mebibytes
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{10485760, "10.0 MiB"},
+		{1024*1024*1024 - 1, "1024.0 MiB"},
+
+		// Gibibytes
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776 - 1, "1024.0 GiB"},
+
+		// Tebibytes
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
