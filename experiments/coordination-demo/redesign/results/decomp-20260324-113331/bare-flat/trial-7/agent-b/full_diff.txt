+// FormatRate formats a transfer rate into a human-readable string.
+// Uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-B/s units. Handles zero and negative values.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kib = 1024
+		mib = 1024 * kib
+		gib = 1024 * mib
+	)
+
+	var result string
+	if bytesPerSec < kib {
+		// For B/s, show no decimal places
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else if bytesPerSec < mib {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kib)
+	} else if bytesPerSec < gib {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/mib)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/gib)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Fractional values in different units
+		{512.5, "512 B/s"},
+		{1024.5, "1.0 KiB/s"},
+		{1024 * 1.25, "1.2 KiB/s"},
+		{1024 * 1024 * 1.75, "1.8 MiB/s"},
+		// Large values
+		{1024 * 1024 * 1024 * 10, "10.0 GiB/s"},
+		{1024 * 1024 * 1024 * 100, "100.0 GiB/s"},
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		{-1536, "-1.5 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
