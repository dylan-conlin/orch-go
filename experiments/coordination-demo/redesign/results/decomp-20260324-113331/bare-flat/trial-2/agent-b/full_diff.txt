+// FormatRate formats a transfer rate into a human-readable string.
+// Output style: "512 B/s", "1.5 KiB/s", "1.0 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	value := bytesPerSec
+
+	for i := 0; i < len(units)-1; i++ {
+		if value < 1024 {
+			var result string
+			if i == 0 {
+				// For bytes, show as integer
+				result = fmt.Sprintf("%.0f %s", value, units[i])
+			} else {
+				// For other units, show 1 decimal place
+				result = fmt.Sprintf("%.1f %s", value, units[i])
+			}
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+		value /= 1024
+	}
+
+	// GiB/s case (last unit)
+	result := fmt.Sprintf("%.1f %s", value, units[len(units)-1])
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary: 1 KiB
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+		// Boundary: 1 MiB
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		// Boundary: 1 GiB
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},

+		{2560, "2.5 KiB/s"},
