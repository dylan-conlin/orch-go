+// FormatRate formats a transfer rate in bytes per second as a human-readable string.
+// Output style: "0 B/s", "512 B/s", "1.5 MiB/s", "3.2 GiB/s".
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kiB = 1024
+		miB = 1024 * 1024
+		giB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	switch {
+	case bytesPerSec < kiB:
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	case bytesPerSec < miB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kiB)
+	case bytesPerSec < giB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/miB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/giB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary: 1 KiB
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB
+		{1536, "1.5 KiB/s"},
+		{1024 * 1.75, "1.8 KiB/s"},
+		// Larger KiB values
+		{1024 * 512, "512.0 KiB/s"},
+		// Exact boundary: 1 MiB
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB
+		{1048576 * 1.5, "1.5 MiB/s"},
+		{1048576 * 2.25, "2.2 MiB/s"},
+		// Larger MiB values
+		{1048576 * 512, "512.0 MiB/s"},
+		// Exact boundary: 1 GiB
+		{1073741824, "1.0 GiB/s"},
+		// Fractional GiB
+		{1073741824 * 2.5, "2.5 GiB/s"},
+		// Very large GiB values
+		{1073741824 * 100, "100.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		{-1536, "-1.5 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
