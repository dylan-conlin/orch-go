+// FormatRate formats a transfer rate in bytes per second into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte units are shown with 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kiB = 1024
+		miB = 1024 * kiB
+		giB = 1024 * miB
+	)
+
+	var result string
+
+	if bytesPerSec < kiB {
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else if bytesPerSec < miB {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kiB)
+	} else if bytesPerSec < giB {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/miB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/giB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary: 1 KiB
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB values
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{3072, "3.0 KiB/s"},
+		{5120, "5.0 KiB/s"},
+		// Near MiB boundary
+		{1048575, "1024.0 KiB/s"}, // Just before 1 MiB
+		// Boundary: 1 MiB
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB values
+		{1572864, "1.5 MiB/s"},     // 1.5 MiB
+		{5242880, "5.0 MiB/s"},     // 5 MiB
+		{104857600, "100.0 MiB/s"}, // 100 MiB
+		// Near GiB boundary
+		{1073741823, "1024.0 MiB/s"}, // Just before 1 GiB
+		// Boundary: 1 GiB
+		{1073741824, "1.0 GiB/s"},
+		// Fractional GiB values
+		{1610612736, "1.5 GiB/s"},   // 1.5 GiB
+		{5368709120, "5.0 GiB/s"},   // 5 GiB
+		{10737418240, "10.0 GiB/s"}, // 10 GiB
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
