+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", "1.0 GiB", "1.0 TiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	// Define units in order of size
+	units := []struct {
+		name  string
+		bytes int64
+	}{
+		{"TiB", 1024 * 1024 * 1024 * 1024},
+		{"GiB", 1024 * 1024 * 1024},
+		{"MiB", 1024 * 1024},
+		{"KiB", 1024},
+		{"B", 1},
+	}
+
+	for _, unit := range units {
+		if absBytes >= unit.bytes {
+			// For bytes, show as integer
+			if unit.name == "B" {
+				if negative {
+					return fmt.Sprintf("-%d B", absBytes)
+				}
+				return fmt.Sprintf("%d B", absBytes)
+			}
+
+			// For other units, show 1 decimal place
+			value := float64(absBytes) / float64(unit.bytes)
+			if negative {
+				return fmt.Sprintf("-%.1f %s", value, unit.name)
+			}
+			return fmt.Sprintf("%.1f %s", value, unit.name)
+		}
+	}
+
+	// Should not reach here, but fallback to bytes
+	if negative {
+		return fmt.Sprintf("-%d B", absBytes)
+	}
+	return fmt.Sprintf("%d B", absBytes)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+
+		// MiB boundary
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{10485760, "10.0 MiB"},
+
+		// GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{10737418240, "10.0 GiB"},
+
+		// TiB boundary
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+
+		// Between boundaries
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1047552, "1023.0 KiB"},
+		{1048576, "1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
