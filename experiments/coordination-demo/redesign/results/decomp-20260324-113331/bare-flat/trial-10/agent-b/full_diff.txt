+// FormatRate formats a byte rate into a human-readable string with binary units.
+// Output style: "0 B/s", "512 B/s", "1.5 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	val := bytesPerSec
+	if negative {
+		val = -val
+	}
+
+	// Binary units: 1 KiB = 1024 bytes
+	const (
+		kib = 1024
+		mib = 1024 * 1024
+		gib = 1024 * 1024 * 1024
+	)
+
+	var result string
+	if val >= gib {
+		result = fmt.Sprintf("%.1f GiB/s", val/gib)
+	} else if val >= mib {
+		result = fmt.Sprintf("%.1f MiB/s", val/mib)
+	} else if val >= kib {
+		result = fmt.Sprintf("%.1f KiB/s", val/kib)
+	} else {
+		result = fmt.Sprintf("%.0f B/s", val)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary: 1 KiB
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// Large KiB values
+		{512 * 1024, "512.0 KiB/s"},
+		// Boundary: 1 MiB
+		{1024 * 1024, "1.0 MiB/s"},
+		// Fractional MiB
+		{1.5 * 1024 * 1024, "1.5 MiB/s"},
+		{2 * 1024 * 1024, "2.0 MiB/s"},
+		// Boundary: 1 GiB
+		{1024 * 1024 * 1024, "1.0 GiB/s"},
+		// Fractional GiB
+		{1.5 * 1024 * 1024 * 1024, "1.5 GiB/s"},
+		{5 * 1024 * 1024 * 1024, "5.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1024 * 1024, "-1.0 MiB/s"},
+		{-1024 * 1024 * 1024, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
