+// FormatRate formats a transfer rate in bytes per second into a human-readable
+// string using binary units. Output style: "512 B/s", "1.5 KiB/s", "1.0 MiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kiB = 1024
+		miB = 1024 * 1024
+		giB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	if bytesPerSec < kiB {
+		result = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	} else if bytesPerSec < miB {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kiB)
+	} else if bytesPerSec < giB {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/miB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/giB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Boundary: 1024 bytes = 1 KiB
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{3072, "3.0 KiB/s"},
+		// Larger KiB values
+		{5120, "5.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		{102400, "100.0 KiB/s"},
+		// Boundary: 1048576 bytes = 1 MiB
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		// Boundary: 1073741824 bytes = 1 GiB
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Fractional values (testing decimal precision)
+		{768, "768 B/s"},  // still less than 1024
+		{1280, "1.2 KiB/s"},  // 1280 / 1024 = 1.25, formatted as 1.2
+		{1572864, "1.5 MiB/s"}, // exact .5
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
