+// FormatTable formats headers and rows as an aligned text table.
+// Each column is sized to fit the widest content in that column (including header).
+// The header is separated from data rows by a line of dashes.
+// ANSI-colored content is measured using StripANSI to ensure correct alignment.
+// Rows with fewer columns than headers are padded with empty cells.
+// Rows with more columns than headers have extra columns ignored.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate the maximum width for each column
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Write header
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		padding := colWidths[i] - len(StripANSI(h))
+		buf.WriteString(h)
+		buf.WriteString(strings.Repeat(" ", padding))
+	}
+	buf.WriteString("\n")
+
+	// Write separator line
+	for i, width := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", width))
+	}
+	buf.WriteString("\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			padding := colWidths[i] - len(StripANSI(cell))
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", padding))
+		}
+		buf.WriteString("\n")
+	}
+
+	return strings.TrimSuffix(buf.String(), "\n")
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 ",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want:    "Name | Age\n-----+----",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"active"},
+				{"inactive"},
+			},
+			want: "Status  \n--------\nactive  \ninactive",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"\x1b[32mactive\x1b[0m"},
+				{"\x1b[31minactive\x1b[0m"},
+			},
+			want: "Status  \n--------\n\x1b[32mactive\x1b[0m  \n\x1b[31minactive\x1b[0m",
+		},
+		{
+			name:    "rows with fewer columns",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \nx |   |  ",
+		},
+		{
+			name:    "rows with more columns (extra ignored)",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+				{"x", "y", "z"},
+			},
+			want: "A | B\n--+--\n1 | 2\nx | y",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeaderName"},
+			rows: [][]string{
+				{"a", "short"},
+				{"medium text", "verylongcontentvalue"},
+			},
+			want: "Short       | VeryLongHeaderName  \n------------+---------------------\na           | short               \nmedium text | verylongcontentvalue",
+		},
+		{
+			name:    "empty headers returns empty",
+			headers: []string{},
+			rows:    [][]string{{"a", "b"}},
+			want:    "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Col"},
+			rows:    nil,
+			want:    "Col\n---",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() =\n%q\n\nwant\n%q", got, tt.want)
+			}
+		})
+	}
+}
