+// VisualWidth returns the visual display width of a string, excluding ANSI
+// escape codes. It counts runes (not bytes) to handle Unicode correctly.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward width. If the string is
+// already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII padding
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		{"", 5, "     "},
+		// Already at width
+		{"hello", 5, "hello"},
+		// Already wider than width
+		{"hello", 3, "hello"},
+		{"hello world", 5, "hello world"},
+		// ANSI codes preserved
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[1;32mbold green\x1b[0m", 15, "\x1b[1;32mbold green\x1b[0m     "},
+		// Unicode
+		{"你好", 5, "你好   "},
+		{"👋", 3, "👋  "},
+		// Mixed ANSI and Unicode
+		{"\x1b[31m你好\x1b[0m", 5, "\x1b[31m你好\x1b[0m   "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width is at least the target width (or input is wider)
+		visualWidth := VisualWidth(got)
+		if visualWidth < tt.width && VisualWidth(tt.input) < tt.width {
+			t.Errorf("PadToWidth(%q, %d): visual width = %d, want at least %d", tt.input, tt.width, visualWidth, tt.width)
+		}
+	}
+}
