+// VisualWidth returns the visual display width of a string, excluding ANSI
+// escape codes. It correctly counts Unicode runes instead of bytes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads s with spaces to reach the target visual width.
+// ANSI escape codes are preserved but don't count toward the width.
+// If s is already at or wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input   string
+		width   int
+		want    string
+		wantLen int
+	}{
+		// Plain ASCII - needs padding
+		{"hello", 10, "hello     ", 10},
+		{"hi", 5, "hi   ", 5},
+
+		// Plain ASCII - no padding needed
+		{"hello", 5, "hello", 5},
+		{"hello world", 5, "hello world", 11},
+
+		// ANSI codes - padding needed
+		{"\x1b[31mred\x1b[0m", 10, "\x1b[31mred\x1b[0m       ", 10},
+		{"\x1b[1;32mbold green\x1b[0m", 15, "\x1b[1;32mbold green\x1b[0m     ", 15},
+
+		// ANSI codes - no padding needed
+		{"\x1b[31mred\x1b[0m", 3, "\x1b[31mred\x1b[0m", 3},
+
+		// Unicode
+		{"你好", 5, "你好   ", 5},
+		{"😀test", 6, "😀test ", 6},
+
+		// Empty string
+		{"", 5, "     ", 5},
+		{"", 0, "", 0},
+
+		// Already at width
+		{"exact", 5, "exact", 5},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		if VisualWidth(got) != tt.wantLen {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, VisualWidth(got), tt.wantLen)
+		}
+	}
+}
