+// FormatTable renders headers and rows as an aligned text table with
+// pipe-separated columns. Each column is auto-sized based on the widest
+// content (including header). ANSI-colored content is handled correctly
+// using StripANSI for width calculation. Handles edge cases: empty rows
+// (headers only), rows with fewer columns than headers, and nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and rows
+	colWidths := make([]int, len(headers))
+
+	// Start with header widths
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Update widths based on row content
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(colWidths) {
+				width := len(StripANSI(cell))
+				if width > colWidths[i] {
+					colWidths[i] = width
+				}
+			}
+		}
+	}
+
+	var buf strings.Builder
+	lastCol := len(headers) - 1
+
+	// Header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(h)
+		// Don't pad the last column
+		if i < lastCol {
+			buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(h))))
+		}
+	}
+	buf.WriteString("\n")
+
+	// Separator line (dashes with pipes)
+	for i := 0; i < len(headers); i++ {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", colWidths[i]))
+	}
+	buf.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(cell)
+			// Don't pad the last column
+			if i < lastCol {
+				buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(cell))))
+			}
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "headers only (no rows)",
+			headers: []string{"Name", "Status"},
+			rows:    [][]string{},
+			want:    "Name | Status\n-----+-------\n",
+		},
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"Alice", "Active"},
+				{"Bob", "Inactive"},
+			},
+			want: "Name  | Status\n------+---------\nAlice | Active\nBob   | Inactive\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+			},
+			want: "Item\n------\napple\nbanana\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "Long"},
+			rows: [][]string{
+				{"A", "This is a very long string"},
+				{"BC", "Short"},
+			},
+			want: "Short | Long\n------+---------------------------\nA     | This is a very long string\nBC    | Short\n",
+		},
+		{
+			name:    "ANSI-colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[31mAlice\x1b[0m", "\x1b[32mActive\x1b[0m"},
+				{"Bob", "Inactive"},
+			},
+			want: "Name  | Status\n------+---------\n\x1b[31mAlice\x1b[0m | \x1b[32mActive\x1b[0m\nBob   | Inactive\n",
+		},
+		{
+			name:    "ANSI in header",
+			headers: []string{"\x1b[1;33mName\x1b[0m", "Status"},
+			rows: [][]string{
+				{"Alice", "Active"},
+			},
+			want: "\x1b[1;33mName\x1b[0m  | Status\n------+-------\nAlice | Active\n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"x", "y"},
+				{"1", "2", "3"},
+			},
+			want: "A | B | C\n--+---+--\nx | y | \n1 | 2 | 3\n",
+		},
+		{
+			name:    "three columns",
+			headers: []string{"ID", "Name", "Score"},
+			rows: [][]string{
+				{"1", "Alice", "95"},
+				{"2", "Bob", "87"},
+				{"3", "Charlie", "92"},
+			},
+			want: "ID | Name    | Score\n---+---------+------\n1  | Alice   | 95\n2  | Bob     | 87\n3  | Charlie | 92\n",
+		},
+		{
+			name:    "empty cells",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"", "value"},
+				{"value", ""},
+			},
+			want: "A     | B\n------+------\n      | value\nvalue | \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() output mismatch\ngot:\n%q\nwant:\n%q", got, tt.want)
+				t.Logf("got formatted:\n%s", got)
+				t.Logf("want formatted:\n%s", tt.want)
+			}
+		})
+	}
