+// FormatTable renders headers and rows as an aligned text table. Each column
+// is auto-sized based on the widest content in that column (including header).
+// ANSI-colored content is handled correctly using StripANSI for width
+// calculations. Returns a formatted table string with header/data separation.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths: max of header or any row content
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			w := len(StripANSI(row[i]))
+			if w > colWidths[i] {
+				colWidths[i] = w
+			}
+		}
+	}
+
+	// Build the table
+	var buf strings.Builder
+
+	// Format header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		// Pad header to column width
+		stripped := StripANSI(h)
+		padding := colWidths[i] - len(stripped)
+		buf.WriteString(h)
+		buf.WriteString(strings.Repeat(" ", padding))
+	}
+	buf.WriteString("\n")
+
+	// Format separator line
+	for i, w := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", w))
+	}
+	buf.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			// Pad cell to column width
+			stripped := StripANSI(cell)
+			padding := colWidths[i] - len(stripped)
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", padding))
+		}
+		buf.WriteString("\n")
+	}
+
+	return strings.TrimSuffix(buf.String(), "\n")
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 ",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"ID", "Status"},
+			rows:    [][]string{},
+			want:    "ID | Status\n---+-------",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"first"},
+				{"second"},
+				{"longer"},
+			},
+			want: "Item  \n------\nfirst \nsecond\nlonger",
+		},
+		{
+			name:    "wide content with multiple rows",
+			headers: []string{"Title", "Description"},
+			rows: [][]string{
+				{"A", "short"},
+				{"LongTitle", "A much longer description here"},
+			},
+			want: "Title     | Description                   \n----------+-------------------------------\nA         | short                         \nLongTitle | A much longer description here",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m", "\x1b[32mGreen\x1b[0m"},
+				{"Plain", "Text"},
+			},
+			want: "Name  | Status\n------+-------\n\x1b[31mRed\x1b[0m   | \x1b[32mGreen\x1b[0m \nPlain | Text  ",
+		},
+		{
+			name:    "mismatched columns - row shorter than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \nx | y | z",
+		},
+		{
+			name:    "mismatched columns - row longer than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+				{"x", "y"},
+			},
+			want: "A | B\n--+--\n1 | 2\nx | y",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "nil rows with headers",
+			headers: []string{"Col"},
+			rows:    nil,
+			want:    "Col\n---",
+		},
+		{
+			name:    "three columns varied widths",
+			headers: []string{"Short", "MediumCol", "VeryLongColumnName"},
+			rows: [][]string{
+				{"a", "b", "c"},
+				{"longer", "mid", "xyz"},
+			},
+			want: "Short  | MediumCol | VeryLongColumnName\n-------+-----------+-------------------\na      | b         | c                 \nlonger | mid       | xyz               ",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch\ngot:\n%q\n\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
+}
