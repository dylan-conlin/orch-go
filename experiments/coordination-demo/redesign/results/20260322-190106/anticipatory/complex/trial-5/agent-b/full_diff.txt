+// FormatTable formats headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including header).
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Handles edge cases: empty rows (headers only), rows with fewer columns than headers.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and rows
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	for _, row := range rows {
+		for i, cell := range row {
+			if i >= len(headers) {
+				break // Ignore columns beyond headers
+			}
+			width := len(StripANSI(cell))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Header row
+	for i, header := range headers {
+		if i > 0 {
+			buf.WriteString("|")
+		}
+		buf.WriteString(" ")
+		buf.WriteString(header)
+		buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(header))))
+		buf.WriteString(" ")
+	}
+	buf.WriteString("\n")
+
+	// Separator
+	for i, width := range colWidths {
+		if i > 0 {
+			buf.WriteString("+")
+		}
+		buf.WriteString(strings.Repeat("-", width+2))
+	}
+	buf.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString("|")
+			}
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(" ")
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(cell))))
+			buf.WriteString(" ")
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: " Name  | Age \n-------+-----\n Alice | 30  \n Bob   | 25  \n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want:    " Name | Age \n------+-----\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Name"},
+			rows: [][]string{
+				{"Alice"},
+				{"Bob"},
+			},
+			want: " Name  \n-------\n Alice \n Bob   \n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m"},
+				{"Normal"},
+			},
+			want: " Name   \n--------\n \x1b[31mRed\x1b[0m    \n Normal \n",
+		},
+		{
+			name:    "row with fewer columns",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows: [][]string{
+				{"A", "B"},
+				{"X", "Y", "Z"},
+			},
+			want: " Col1 | Col2 | Col3 \n------+------+------\n A    | B    |      \n X    | Y    | Z    \n",
+		},
+		{
+			name:    "row with more columns",
+			headers: []string{"Col1", "Col2"},
+			rows: [][]string{
+				{"A", "B", "C", "D"},
+			},
+			want: " Col1 | Col2 \n------+------\n A    | B    \n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongerContent"},
+			rows: [][]string{
+				{"X", "Y"},
+			},
+			want: " Short | LongerContent \n-------+---------------\n X     | Y             \n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"A"},
+			rows:    nil,
+			want:    " A \n---\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() =\n%q\nwant\n%q", got, tt.want)
+			}
+		})
+	}
