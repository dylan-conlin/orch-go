+// VisualWidth returns the visual display width of a string, excluding ANSI escape codes.
+// Unicode characters are counted as single runes. This is useful for calculating display width
+// when formatting strings with ANSI color codes or other escape sequences.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the width.
+// If the string is already as wide or wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"hello world", 11},
+		{"", 0},
+		{"a", 1},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[31mred\x1b[32m green\x1b[0m", 9},
+
+		// Unicode strings (CJK characters)
+		{"你好", 2},
+		{"こんにちは", 5},
+		{"한글", 2},
+
+		// Emoji (single runes, even though they may display wider in terminals)
+		{"😀", 1},
+		{"👋 hello", 7},
+
+		// Mixed: ANSI codes with Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1;32m😀 test\x1b[0m", 6},
+
+		// Edge cases
+		{"\x1b[31m\x1b[0m", 0},
+		{"   ", 3},
+		{"hello\nworld", 11},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+
+		// Mixed: ANSI codes with Unicode
+		{"\x1b[31m你好\x1b[0m", 4, "\x1b[31m你好\x1b[0m  "},

+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {

+		}
+	}
