+// FormatTable renders headers and rows as an aligned text table.
+// Columns are auto-sized based on the widest content in each column (including headers).
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Edge cases: empty rows (headers only), rows with fewer columns than headers, and nil rows are handled gracefully.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths (visual width without padding)
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	// Expand column widths based on row content
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Helper to format a cell with padding
+	formatCell := func(content string, width int) string {
+		visualWidth := len(StripANSI(content))
+		rightPadding := width - visualWidth
+		return " " + content + strings.Repeat(" ", rightPadding) + " "
+	}
+
+	// Format header row
+	headerParts := make([]string, len(headers))
+	for i, header := range headers {
+		headerParts[i] = formatCell(header, colWidths[i])
+	}
+	headerRow := "|" + strings.Join(headerParts, "|") + "|"
+
+	// Format separator row with dashes
+	separatorParts := make([]string, len(headers))
+	for i := range headers {
+		separatorParts[i] = strings.Repeat("-", colWidths[i]+2)
+	}
+	separatorRow := "+" + strings.Join(separatorParts, "+") + "+"
+
+	// Format data rows
+	var result []string
+	result = append(result, headerRow)
+	result = append(result, separatorRow)
+
+	for _, row := range rows {
+		rowParts := make([]string, len(headers))
+		for i := 0; i < len(headers); i++ {
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			rowParts[i] = formatCell(content, colWidths[i])
+		}
+		dataRow := "|" + strings.Join(rowParts, "|") + "|"
+		result = append(result, dataRow)
+	}
+
+	return strings.Join(result, "\n")
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table with headers and rows",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{{"Alice", "30"}, {"Bob", "25"}},
+			want: "| Name  | Age |\n" +
+				"+-------+-----+\n" +
+				"| Alice | 30  |\n" +
+				"| Bob   | 25  |",
+		},
+		{
+			name:    "headers only (no rows)",
+			headers: []string{"Name", "Email"},
+			rows:    [][]string{},
+			want: "| Name | Email |\n" +
+				"+------+-------+",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Status"},
+			rows:    [][]string{{"Active"}, {"Inactive"}},
+			want: "| Status   |\n" +
+				"+----------+\n" +
+				"| Active   |\n" +
+				"| Inactive |",
+		},
+		{
+			name:    "wide content in cells",
+			headers: []string{"Short", "VeryLongHeaderName"},
+			rows:    [][]string{{"A", "Short"}, {"B", "ExtremelyLongContentHere"}},
+			want: "| Short | VeryLongHeaderName       |\n" +
+				"+-------+--------------------------+\n" +
+				"| A     | Short                    |\n" +
+				"| B     | ExtremelyLongContentHere |",
+		},
+		{
+			name:    "ANSI-colored content alignment",
+			headers: []string{"Status"},
+			rows:    [][]string{{"\x1b[32mActive\x1b[0m"}, {"\x1b[31mError\x1b[0m"}},
+			want: "| Status |\n" +
+				"+--------+\n" +
+				"| \x1b[32mActive\x1b[0m |\n" +
+				"| \x1b[31mError\x1b[0m  |",
+		},
+		{
+			name:    "rows with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows:    [][]string{{"1", "2"}, {"3", "4", "5"}},
+			want: "| A | B | C |\n" +
+				"+---+---+---+\n" +
+				"| 1 | 2 |   |\n" +
+				"| 3 | 4 | 5 |",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{{"value"}},
+			want:    "",
+		},
+		{
+			name:    "single cell table",
+			headers: []string{"Data"},
+			rows:    [][]string{{"Value"}},
+			want: "| Data  |\n" +
+				"+-------+\n" +
+				"| Value |",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch\nGot:\n%q\n\nWant:\n%q", got, tt.want)
+			}
+		})
+	}
+}
