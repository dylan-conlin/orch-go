+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. Unicode characters are counted as single runes. This is useful
+// for padding and alignment in terminal output where ANSI codes should not
+// affect visible width.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved in the output but do not count toward the
+// width calculation. If the string's visual width is already >= width, it is
+// returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	paddingNeeded := width - visualWidth
+	return s + strings.Repeat(" ", paddingNeeded)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// ANSI color codes should not count toward width
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m test", 11},
+
+		// Unicode (CJK characters, emoji)
+		// CJK characters and emoji are counted as single runes
+		{"你好", 2},
+		{"こんにちは", 5},
+		{"😀😁😂", 3},
+		{"hello😀world", 11},
+
+		// Mixed: ANSI codes + Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1m😀\x1b[0m", 1},
+		{"\x1b[32mhello 世界\x1b[0m", 8},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input    string
+		width    int
+		want     string
+		wantLen  int // visual width of result
+	}{
+		// Plain ASCII, needs padding
+		{"hello", 10, "hello     ", 10},
+		{"hi", 5, "hi   ", 5},
+		{"", 3, "   ", 3},
+
+		// Plain ASCII, already at width
+		{"hello", 5, "hello", 5},
+		{"exact", 5, "exact", 5},
+
+		// Plain ASCII, wider than target (no padding)
+		{"hello", 3, "hello", 5},
+		{"longer", 2, "longer", 6},
+
+		// ANSI codes should be preserved but not count toward width
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  ", 5},
+		{"\x1b[1;32mbold\x1b[0m", 8, "\x1b[1;32mbold\x1b[0m    ", 8},
+
+		// Unicode (visual width 2, needs padding to 5)
+		{"你好", 5, "你好   ", 5},
+		{"😀", 3, "😀  ", 3},
+
+		// Mixed: ANSI + Unicode
+		{"\x1b[33m你好\x1b[0m", 6, "\x1b[33m你好\x1b[0m    ", 6},
+		{"\x1b[1m😀\x1b[0m test", 10, "\x1b[1m😀\x1b[0m test    ", 10},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		gotWidth := VisualWidth(got)
+		if gotWidth != tt.wantLen {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotWidth, tt.wantLen)
+		}
+	}
+}
