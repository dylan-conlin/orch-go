+// FormatTable renders headers and rows as an aligned text table with auto-sized columns.
+// Each column width is determined by the widest content (including headers), accounting
+// for ANSI color codes using visual width calculation. Rows with fewer columns than
+// headers are padded with empty strings; rows with more columns are truncated.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate max visual width for each column
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			val := ""
+			if i < len(row) {
+				val = row[i]
+			}
+			width := len(StripANSI(val))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	const padding = 1
+
+	// Helper to format a cell with padding
+	formatCell := func(content string, colWidth int) string {
+		visualWidth := len(StripANSI(content))
+		trailingSpaces := colWidth - visualWidth
+		return " " + content + strings.Repeat(" ", trailingSpaces) + " "
+	}
+
+	var buf strings.Builder
+
+	// Header row
+	buf.WriteString("|")
+	for i, h := range headers {
+		buf.WriteString(formatCell(h, colWidths[i]))
+		buf.WriteString("|")
+	}
+	buf.WriteString("\n")
+
+	// Separator row
+	buf.WriteString("|")
+	for _, w := range colWidths {
+		buf.WriteString(strings.Repeat("-", w+2*padding))
+		buf.WriteString("|")
+	}
+	buf.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		buf.WriteString("|")
+		for i := 0; i < len(headers); i++ {
+			val := ""
+			if i < len(row) {
+				val = row[i]
+			}
+			buf.WriteString(formatCell(val, colWidths[i]))
+			buf.WriteString("|")
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}
