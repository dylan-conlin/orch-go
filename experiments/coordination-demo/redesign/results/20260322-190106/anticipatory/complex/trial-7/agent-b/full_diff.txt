+// FormatTable renders headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including headers).
+// Uses StripANSI to calculate correct widths for ANSI-colored content.
+// Separates the header row from data rows with a line of dashes.
+// Handles edge cases: rows shorter than headers are padded with empty strings,
+// extra columns in rows are ignored.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on visual width (stripping ANSI codes)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			width := len(StripANSI(cell))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Format headers
+	buf.WriteString("| ")
+	for i, h := range headers {
+		stripped := StripANSI(h)
+		padding := colWidths[i] - len(stripped)
+		buf.WriteString(h)
+		buf.WriteString(strings.Repeat(" ", padding))
+		if i < len(headers)-1 {
+			buf.WriteString(" | ")
+		}
+	}
+	buf.WriteString(" |\n")
+
+	// Format separator
+	buf.WriteString("|")
+	for i := 0; i < len(headers); i++ {
+		buf.WriteString(strings.Repeat("-", colWidths[i]+2))
+		buf.WriteString("|")
+	}
+	buf.WriteString("\n")
+
+	// Format rows
+	for _, row := range rows {
+		buf.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			stripped := StripANSI(cell)
+			padding := colWidths[i] - len(stripped)
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", padding))
+			if i < len(headers)-1 {
+				buf.WriteString(" | ")
+			}
+		}
+		buf.WriteString(" |\n")
+	}
+
+	return strings.TrimSuffix(buf.String(), "\n")
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name  | Age |\n|-------|-----|\n| Alice | 30  |\n| Bob   | 25  |",
+		},
+		{
+			name:    "empty table (headers only)",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "| Col1 | Col2 |\n|------|------|",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"Active"},
+				{"Inactive"},
+			},
+			want: "| Status   |\n|----------|\n| Active   |\n| Inactive |",
+		},
+		{
+			name:    "ANSI-colored content",
+			headers: []string{"Name", "Color"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m", "\x1b[32mGreen\x1b[0m"},
+				{"Normal", "\x1b[1;35mMagenta\x1b[0m"},
+			},
+			want: "| Name   | Color   |\n|--------|---------|\n| \x1b[31mRed\x1b[0m    | \x1b[32mGreen\x1b[0m   |\n| Normal | \x1b[1;35mMagenta\x1b[0m |",
+		},
+		{
+			name:    "rows shorter than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3"},
+			},
+			want: "| A | B | C |\n|---|---|---|\n| 1 | 2 |   |\n| 3 |   |   |",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeaderName"},
+			rows: [][]string{
+				{"X", "Y"},
+				{"Content", "VeryLongContentString"},
+			},
+			want: "| Short   | VeryLongHeaderName    |\n|---------|-----------------------|\n| X       | Y                     |\n| Content | VeryLongContentString |",
+		},
+		{
+			name:    "three column table",
+			headers: []string{"ID", "Status", "Message"},
+			rows: [][]string{
+				{"1", "OK", "Success"},
+				{"2", "Error", "Failed"},
+			},
+			want: "| ID | Status | Message |\n|----|--------|---------|\n| 1  | OK     | Success |\n| 2  | Error  | Failed  |",
+		},
+		{
+			name:    "empty cell values",
+			headers: []string{"First", "Second"},
+			rows: [][]string{
+				{"Value", ""},
+				{"", "Other"},
+			},
+			want: "| First | Second |\n|-------|--------|\n| Value |        |\n|       | Other  |",
+		},
+	}
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
