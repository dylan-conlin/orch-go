+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. Unicode characters (including emoji and CJK) are counted
+// correctly by rune count.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the width calculation.
+// If the string is already at or wider than the target width, it is returned
+// unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	paddingNeeded := width - currentWidth
+	return s + strings.Repeat(" ", paddingNeeded)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"12345", 5},
+		// ANSI color codes (should be stripped)
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m", 6},
+		// Unicode: emoji and CJK
+		{"你好", 2},           // 2 Chinese characters
+		{"hello世界", 7},      // 5 ASCII + 2 CJK
+		{"👍", 1},             // emoji (single rune)
+		{"👍👎", 2},           // 2 emoji
+		{"café", 4},           // accented character
+		// Mixed ANSI and Unicode
+		{"\x1b[32m你好\x1b[0m", 2},  // ANSI + CJK
+		{"\x1b[31m👍\x1b[0m", 1},    // ANSI + emoji
+		// Edge cases
+		{" ", 1},              // space
+		{"   ", 3},            // multiple spaces
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		minLen int // minimum length (includes ANSI codes)
+	}{
+		// Plain ASCII, needs padding
+		{"hello", 10, "hello     ", 10},
+		{"a", 5, "a    ", 5},
+		{"", 5, "     ", 5},
+		// ASCII already at target width
+		{"hello", 5, "hello", 5},
+		// ASCII wider than target (no padding)
+		{"hello world", 5, "hello world", 11},
+		// ANSI codes preserved, padding added
+		{"\x1b[31mred\x1b[0m", 6, "\x1b[31mred\x1b[0m   ", 12},
+		// ANSI codes don't count but are preserved
+		{"\x1b[33myellow\x1b[0m", 8, "\x1b[33myellow\x1b[0m  ", 16},
+		// Unicode characters
+		{"你好", 5, "你好   ", 7},      // 2 CJK + 3 spaces
+		{"👍", 3, "👍  ", 3},          // emoji + 2 spaces (emoji is 1 visual width)
+		{"café", 6, "café  ", 6},     // accented + 2 spaces
+		// Unicode already at width
+		{"你好", 2, "你好", 2},
+		// Unicode wider than target
+		{"你好世", 2, "你好世", 3},
+		// Mixed ANSI and Unicode
+		{"\x1b[32m你好\x1b[0m", 4, "\x1b[32m你好\x1b[0m  ", 12},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		if len(got) < tt.minLen {
+			t.Errorf("PadToWidth(%q, %d) length = %d, want >= %d", tt.input, tt.width, len(got), tt.minLen)
+		}
+	}
+}
