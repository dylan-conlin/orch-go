+// FormatTable renders headers and rows as an aligned text table.
+// Columns are auto-sized based on the widest content in each column (including headers).
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Edge cases (empty rows, rows with fewer columns than headers, nil rows) are handled gracefully.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	numCols := len(headers)
+
+	// Calculate column widths based on headers and row content
+	colWidths := make([]int, numCols)
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	// Update column widths based on row content
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < numCols {
+				width := len(StripANSI(cell))
+				if width > colWidths[i] {
+					colWidths[i] = width
+				}
+			}
+		}
+	}
+
+	var result strings.Builder
+
+	// Header row
+	result.WriteString("|")
+	for i, header := range headers {
+		result.WriteString(" ")
+		padded := header + strings.Repeat(" ", colWidths[i]-len(StripANSI(header)))
+		result.WriteString(padded)
+		result.WriteString(" |")
+	}
+	result.WriteString("\n")
+
+	// Separator line
+	result.WriteString("|")
+	for i := 0; i < numCols; i++ {
+		result.WriteString(" " + strings.Repeat("-", colWidths[i]) + " |")
+	}
+	result.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		result.WriteString("|")
+		for i := 0; i < numCols; i++ {
+			result.WriteString(" ")
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			padded := cell + strings.Repeat(" ", colWidths[i]-len(StripANSI(cell)))
+			result.WriteString(padded)
+			result.WriteString(" |")
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}

+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{

+				{"Bob", "25"},

+			want: "| Name  | Age |\n| ----- | --- |\n| Alice | 30  |\n| Bob   | 25  |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want:    "| Name | Age |\n| ---- | --- |\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"\x1b[32mok\x1b[0m"},
+			},
+			want: "| Status |\n| ------ |\n| \x1b[32mok\x1b[0m     |\n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+			},
+			want: "| A | B | C |\n| - | - | - |\n| 1 | 2 |   |\n",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Value"},
+			rows: [][]string{
+				{"hello"},
+				{"world"},
+			},
+			want: "| Value |\n| ----- |\n| hello |\n| world |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongerHeader"},
+			rows: [][]string{
