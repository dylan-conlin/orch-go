+// FormatBytes formats a byte count into a human-readable string.
+// Output style: "512 B", "1.5 KiB", "2.3 MiB", etc.
+// Uses binary units (KiB, MiB, GiB, TiB based on powers of 1024).
+// Displays 1 decimal place for non-byte units. Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	const (
+		KiB = 1024
+		MiB = KiB * 1024
+		GiB = MiB * 1024
+		TiB = GiB * 1024
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	abs := bytes
+	if negative {
+		abs = -bytes
+	}
+
+	var result string
+
+	if abs >= TiB {
+		result = fmt.Sprintf("%.1f TiB", float64(abs)/float64(TiB))
+	} else if abs >= GiB {
+		result = fmt.Sprintf("%.1f GiB", float64(abs)/float64(GiB))
+	} else if abs >= MiB {
+		result = fmt.Sprintf("%.1f MiB", float64(abs)/float64(MiB))
+	} else if abs >= KiB {
+		result = fmt.Sprintf("%.1f KiB", float64(abs)/float64(KiB))
+	} else {
+		result = fmt.Sprintf("%d B", abs)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+		{5120, "5.0 KiB"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{2621440, "2.5 MiB"},
+		{10485760, "10.0 MiB"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{2684354560, "2.5 GiB"},
+		{10737418240, "10.0 GiB"},
+		// TiB boundary and values
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		{2748779069440, "2.5 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-512, "-512 B"},
+		{-1, "-1 B"},
+	}
