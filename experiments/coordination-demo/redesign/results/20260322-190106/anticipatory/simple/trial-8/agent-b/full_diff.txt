+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Output style: "512 B/s", "1.5 MiB/s", "2.3 GiB/s".
+// Uses binary units (KiB, MiB, GiB) and shows 1 decimal place for non-byte units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	value := bytesPerSec
+	if negative {
+		value = -value
+	}
+
+	// Convert to appropriate unit
+	units := []struct {
+		threshold float64
+		name      string
+	}{
+		{1024 * 1024 * 1024, "GiB/s"},
+		{1024 * 1024, "MiB/s"},
+		{1024, "KiB/s"},
+	}
+
+	for _, unit := range units {
+		if value >= unit.threshold {
+			result := fmt.Sprintf("%.1f %s", value/unit.threshold, unit.name)
+			if negative {
+				return "-" + result
+			}
+			return result
+		}
+	}
+
+	// For bytes per second, use integer formatting
+	result := fmt.Sprintf("%d B/s", int(value))
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// KiB/s boundary
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+
+		// MiB/s boundary
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+
+		// GiB/s boundary
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
