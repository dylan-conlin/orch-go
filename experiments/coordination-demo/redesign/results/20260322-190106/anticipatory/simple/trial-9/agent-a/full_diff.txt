+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", "2.0 GiB", "10.0 TiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	units := []struct {
+		name      string
+		threshold int64
+	}{
+		{"TiB", 1024 * 1024 * 1024 * 1024},
+		{"GiB", 1024 * 1024 * 1024},
+		{"MiB", 1024 * 1024},
+		{"KiB", 1024},
+		{"B", 1},
+	}
+
+	for _, unit := range units {
+		if bytes >= unit.threshold {
+			if unit.name == "B" {
+				// For bytes, no decimal places
+				if negative {
+					return fmt.Sprintf("-%d B", bytes)
+				}
+				return fmt.Sprintf("%d B", bytes)
+			}
+			// For other units, show 1 decimal place
+			value := float64(bytes) / float64(unit.threshold)
+			if negative {
+				return fmt.Sprintf("-%.1f %s", value, unit.name)
+			}
+			return fmt.Sprintf("%.1f %s", value, unit.name)
+		}
+	}
+
+	// Fallback for edge cases (should not be reached)
+	if negative {
+		return fmt.Sprintf("-%d B", bytes)
+	}
+	return fmt.Sprintf("%d B", bytes)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{5120, "5.0 KiB"},
+		// MiB boundaries
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{10485760, "10.0 MiB"},
+		// GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{5368709120, "5.0 GiB"},
+		// TiB boundaries
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
