+// FormatRate formats a transfer rate (bytes per second) as a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units show 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB"}
+	unitSizes := []float64{1, 1024, 1024 * 1024, 1024 * 1024 * 1024}
+
+	// Find the appropriate unit
+	for i := len(unitSizes) - 1; i >= 0; i-- {
+		if bytesPerSec >= unitSizes[i] {
+			value := bytesPerSec / unitSizes[i]
+			if i == 0 {
+				// B/s: show as integer
+				result := fmt.Sprintf("%d B/s", int(value))
+				if negative {
+					result = "-" + result
+				}
+				return result
+			}
+			// Other units: show 1 decimal place
+			result := fmt.Sprintf("%.1f %s/s", value, units[i])
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	// Fallback (should not reach here for non-zero values)
+	result := fmt.Sprintf("%.1f B/s", bytesPerSec)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero case
+		{0, "0 B/s"},
+		// Small values in B/s
+		{512, "512 B/s"},
+		{1000, "1000 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		// Large values
+		{10737418240, "10.0 GiB/s"},
+		// Fractional values in B/s (less than 1024)
+		{512.5, "512 B/s"},
+		{1023.9, "1023 B/s"},
+		// Fractional values in other units
+		{1024.5, "1.0 KiB/s"},
+		{2048.5, "2.0 KiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Very small positive values
+		{1, "1 B/s"},
+		{100, "100 B/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
