+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "512 B", "1.5 KiB", "1.0 MiB", "-2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	sign := ""
+	absBytes := bytes
+	if bytes < 0 {
+		sign = "-"
+		absBytes = -bytes
+	}
+
+	const (
+		B  = 1
+		KB = B * 1024
+		MB = KB * 1024
+		GB = MB * 1024
+		TB = GB * 1024
+	)
+
+	switch {
+	case absBytes < KB:
+		return fmt.Sprintf("%s%d B", sign, absBytes)
+	case absBytes < MB:
+		return fmt.Sprintf("%s%.1f KiB", sign, float64(absBytes)/float64(KB))
+	case absBytes < GB:
+		return fmt.Sprintf("%s%.1f MiB", sign, float64(absBytes)/float64(MB))
+	case absBytes < TB:
+		return fmt.Sprintf("%s%.1f GiB", sign, float64(absBytes)/float64(GB))
+	default:
+		return fmt.Sprintf("%s%.1f TiB", sign, float64(absBytes)/float64(TB))
+	}
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048575, "1024.0 KiB"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1073741823, "1024.0 MiB"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB boundary and values
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
