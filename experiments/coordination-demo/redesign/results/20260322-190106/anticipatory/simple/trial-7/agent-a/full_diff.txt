+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Shows 1 decimal place for non-byte units.
+func FormatBytes(bytes int64) string {
+	const (
+		B   = 1
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes < KiB:
+		result := fmt.Sprintf("%d B", bytes)
+		if negative {
+			result = "-" + result
+		}
+		return result
+	case bytes < MiB:
+		value = float64(bytes) / float64(KiB)
+		unit = "KiB"
+	case bytes < GiB:
+		value = float64(bytes) / float64(MiB)
+		unit = "MiB"
+	case bytes < TiB:
+		value = float64(bytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(bytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	result := fmt.Sprintf("%.1f %s", value, unit)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Bytes (below 1 KiB)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+		// MiB boundary and values
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// GiB boundary and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		// TiB boundary and values
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
