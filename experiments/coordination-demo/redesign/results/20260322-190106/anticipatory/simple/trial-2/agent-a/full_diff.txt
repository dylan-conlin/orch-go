+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", etc.
+// Handles negative values by prefixing with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const unitSize = 1024
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+
+	index := 0
+	value := float64(bytes)
+
+	for index < len(units)-1 && value >= unitSize {
+		value /= unitSize
+		index++
+	}
+
+	var formatted string
+	if index == 0 {
+		// For bytes, show no decimal places
+		formatted = fmt.Sprintf("%d %s", int64(value), units[index])
+	} else {
+		// For other units, show 1 decimal place
+		formatted = fmt.Sprintf("%.1f %s", value, units[index])
+	}
+
+	if negative {
+		return "-" + formatted
+	}
+	return formatted
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (under 1 KiB)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		// Large values
+		{1099511627776, "1.0 TiB"},
+		{2199023255552, "2.0 TiB"},
+		{3298534883328, "3.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
