+// FormatRate formats a transfer rate (bytes per second) into a human-readable string.
+// Output style: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s".
+// Uses binary units (1 KiB = 1024 bytes) with 1 decimal place for non-byte-per-sec units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	divisor := 1024.0
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for unitIndex < len(units)-1 && bytesPerSec >= divisor {
+		bytesPerSec /= divisor
+		unitIndex++
+	}
+
+	// Format based on unit
+	var formatted string
+	if unitIndex == 0 {
+		// B/s - show as integer
+		formatted = fmt.Sprintf("%d %s", int(bytesPerSec+0.5), units[unitIndex])
+	} else {
+		// KiB/s, MiB/s, GiB/s - show with 1 decimal place
+		formatted = fmt.Sprintf("%.1f %s", bytesPerSec, units[unitIndex])
+	}
+
+	if isNegative {
+		return "-" + formatted
+	}
+	return formatted
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero case
+		{0, "0 B/s"},
+
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries (KiB/s)
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{5120, "5.0 KiB/s"},
+
+		// MiB/s boundary
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+
+		// GiB/s boundary
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+
+		// Fractional values
+		{1024 + 512, "1.5 KiB/s"},
+		{1024*10 + 512, "10.5 KiB/s"},
+		{1048576*5 + 524288, "5.5 MiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+
+		// Large values
+		{1024 * 1024 * 1024 * 10, "10.0 GiB/s"},
+		{1024 * 1024 * 1024 * 100, "100.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
