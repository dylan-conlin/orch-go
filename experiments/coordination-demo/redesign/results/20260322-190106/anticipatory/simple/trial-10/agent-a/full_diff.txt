+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte values are shown with 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	units := []struct {
+		name  string
+		size  int64
+	}{
+		{"TiB", 1024 * 1024 * 1024 * 1024},
+		{"GiB", 1024 * 1024 * 1024},
+		{"MiB", 1024 * 1024},
+		{"KiB", 1024},
+		{"B", 1},
+	}
+
+	for _, unit := range units {
+		if absBytes >= unit.size {
+			if unit.name == "B" {
+				result := fmt.Sprintf("%d B", absBytes)
+				if negative {
+					result = "-" + result
+				}
+				return result
+			}
+			value := float64(absBytes) / float64(unit.size)
+			result := fmt.Sprintf("%.1f %s", value, unit.name)
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	return "0 B"
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Large values
+		{10737418240, "10.0 GiB"},
+		{1099511627776 * 5, "5.0 TiB"},
+		// Values between boundaries
+		{2560, "2.5 KiB"},
+		{5242880, "5.0 MiB"},
+		{10485760, "10.0 MiB"},
+		{1073741824 * 2, "2.0 GiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
