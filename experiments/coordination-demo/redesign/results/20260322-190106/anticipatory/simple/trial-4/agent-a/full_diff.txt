+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	// Binary units: B, KiB, MiB, GiB, TiB
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	const unit = 1024
+
+	var size float64 = float64(bytes)
+	index := 0
+
+	for size >= unit && index < len(units)-1 {
+		size /= unit
+		index++
+	}
+
+	var result string
+	if index == 0 {
+		// For bytes, show as integer
+		result = fmt.Sprintf("%d B", int64(size))
+	} else {
+		// For other units, show with 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[index])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+		{1649267441664, "1.5 TiB"},
+		// Larger values
+		{5242880, "5.0 MiB"},
+		{10737418240, "10.0 GiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-512, "-512 B"},
+		// Non-round values
+		{2048, "2.0 KiB"},
+		{2560, "2.5 KiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
