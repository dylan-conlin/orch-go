+// FormatRate formats a data transfer rate as a human-readable string.
+// Output style: "512 B/s", "1.5 KiB/s", "2.0 MiB/s", "3.5 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	rate := bytesPerSec
+	if rate < 0 {
+		negative = true
+		rate = -rate
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	unitIdx := 0
+
+	for unitIdx < len(units)-1 && rate >= 1024 {
+		rate /= 1024
+		unitIdx++
+	}
+
+	var result string
+	if unitIdx == 0 {
+		// For bytes per second, show as integer
+		result = fmt.Sprintf("%.0f %s", rate, units[unitIdx])
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", rate, units[unitIdx])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1048576 * 1.5, "1.5 MiB/s"},
+		{1048576 * 1024, "1.0 GiB/s"},
+		{1048576 * 1024 * 2.5, "2.5 GiB/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
