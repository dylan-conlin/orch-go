+// FormatBytes formats a byte count into a human-readable string.
+// Uses binary units: B, KiB, MiB, GiB, TiB.
+// Shows 1 decimal place for non-byte units (e.g., "1.5 MiB").
+// Handles negative values by prefixing with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case absBytes < KiB:
+		return fmt.Sprintf("%d B", bytes)
+	case absBytes < MiB:
+		value = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	case absBytes < GiB:
+		value = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes < TiB:
+		value = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	if negative {
+		value = -value
+	}
+
+	return fmt.Sprintf("%.1f %s", value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 * 1024, "1.0 MiB"},
+		{1024*1024 + 512*1024, "1.5 MiB"},
+		{10 * 1024 * 1024, "10.0 MiB"},
+		{1024 * 1024 * 1024, "1.0 GiB"},
+		{2*1024*1024*1024 + 512*1024*1024, "2.5 GiB"},
+		{1024 * 1024 * 1024 * 1024, "1.0 TiB"},
+		{5 * 1024 * 1024 * 1024 * 1024, "5.0 TiB"},
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1024 * 1024, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
