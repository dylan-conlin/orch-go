+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte units show 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kiB = 1024
+		miB = 1024 * 1024
+		giB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	switch {
+	case bytesPerSec < kiB:
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	case bytesPerSec < miB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kiB)
+	case bytesPerSec < giB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/miB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/giB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{512 * 1024, "512.0 KiB/s"},
+		{1024 * 1024, "1.0 MiB/s"},
+		{1.5 * 1024 * 1024, "1.5 MiB/s"},
+		{100 * 1024 * 1024, "100.0 MiB/s"},

+		{2.5 * 1024 * 1024 * 1024, "2.5 GiB/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},

+		{-1024 * 1024, "-1.0 MiB/s"},
