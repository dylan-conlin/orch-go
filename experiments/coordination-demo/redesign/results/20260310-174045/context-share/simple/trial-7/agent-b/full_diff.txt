+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Uses binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place for non-byte units.
+// Examples: "512 B/s", "1.5 MiB/s", "-1.0 KiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	sign := ""
+	rate := bytesPerSec
+	if rate < 0 {
+		sign = "-"
+		rate = -rate
+	}
+
+	const kib = 1024.0
+	const mib = 1024.0 * 1024.0
+	const gib = 1024.0 * 1024.0 * 1024.0
+
+	if rate < kib {
+		return fmt.Sprintf("%s%.0f B/s", sign, rate)
+	}
+	if rate < mib {
+		return fmt.Sprintf("%s%.1f KiB/s", sign, rate/kib)
+	}
+	if rate < gib {
+		return fmt.Sprintf("%s%.1f MiB/s", sign, rate/mib)
+	}
+	return fmt.Sprintf("%s%.1f GiB/s", sign, rate/gib)
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Large values
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-512, "-512 B/s"},
+		{-1048576, "-1.0 MiB/s"},
+		// Fractional values
+		{2560, "2.5 KiB/s"},
+		{2621440, "2.5 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
