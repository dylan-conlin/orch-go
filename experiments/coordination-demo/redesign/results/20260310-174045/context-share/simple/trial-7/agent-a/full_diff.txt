+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.3 MiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = KiB * 1024
+		GiB = MiB * 1024
+		TiB = GiB * 1024
+	)
+
+	var result string
+	switch {
+	case bytes < KiB:
+		result = fmt.Sprintf("%d B", bytes)
+	case bytes < MiB:
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/float64(KiB))
+	case bytes < GiB:
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/float64(MiB))
+	case bytes < TiB:
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/float64(GiB))
+	default:
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/float64(TiB))
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and fractions
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		// Between KiB and MiB
+		{51200, "50.0 KiB"},
+		{524288, "512.0 KiB"},
+		// MiB boundary and fractions
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Between MiB and GiB
+		{52428800, "50.0 MiB"},
+		{536870912, "512.0 MiB"},
+		// GiB boundary and fractions
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Between GiB and TiB
+		{549755813888, "512.0 GiB"},
+		// TiB boundary and fractions
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
