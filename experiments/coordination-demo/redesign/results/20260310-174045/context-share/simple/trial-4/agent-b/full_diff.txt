+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec units are shown with 1 decimal place (e.g., "1.5 MiB/s").
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	const (
+		kiB = 1024.0
+		miB = kiB * 1024
+		giB = miB * 1024
+	)
+
+	absRate := bytesPerSec
+	negative := false
+	if absRate < 0 {
+		negative = true
+		absRate = -absRate
+	}
+
+	var result string
+	switch {
+	case absRate < kiB:
+		result = fmt.Sprintf("%.0f B/s", absRate)
+	case absRate < miB:
+		result = fmt.Sprintf("%.1f KiB/s", absRate/kiB)
+	case absRate < giB:
+		result = fmt.Sprintf("%.1f MiB/s", absRate/miB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", absRate/giB)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary (1 KiB)
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB
+		{1536, "1.5 KiB/s"},
+		{1024 * 2.5, "2.5 KiB/s"},
+		// Exact boundary (1 MiB)
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB
+		{1048576 * 2.5, "2.5 MiB/s"},
+		// Exact boundary (1 GiB)
+		{1073741824, "1.0 GiB/s"},
+		// Large values
+		{1073741824 * 5.5, "5.5 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-512, "-512 B/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
