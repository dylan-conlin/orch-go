+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", "1.0 GiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	const unit = 1024
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+
+	b := float64(bytes)
+	for i := 0; i < len(units)-1; i++ {
+		if b < unit {
+			if i == 0 {
+				// For bytes, show as integer
+				result := fmt.Sprintf("%d %s", int64(b), units[i])
+				if negative {
+					return "-" + result
+				}
+				return result
+			}
+			// For other units, show 1 decimal place
+			result := fmt.Sprintf("%.1f %s", b, units[i])
+			if negative {
+				return "-" + result
+			}
+			return result
+		}
+		b /= unit
+	}
+
+	// TiB (last unit)
+	result := fmt.Sprintf("%.1f %s", b, units[len(units)-1])
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundaries and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 * 512, "512.0 KiB"},
+		// MiB boundaries and values
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{10485760, "10.0 MiB"},
+		// GiB boundaries and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
