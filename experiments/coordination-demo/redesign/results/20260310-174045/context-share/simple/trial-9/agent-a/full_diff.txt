+// FormatBytes formats a byte count into a human-readable string.
+// Uses binary units (B, KiB, MiB, GiB, TiB) with 1 decimal place for
+// non-byte units. Handles negative values by prefixing with "-".
+func FormatBytes(bytes int64) string {
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	// Handle zero
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	divisor := int64(1024)
+	value := float64(bytes)
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for unitIndex < len(units)-1 && value >= float64(divisor) {
+		value /= float64(divisor)
+		unitIndex++
+	}
+
+	// Format the output
+	var result string
+	if unitIndex == 0 {
+		// For bytes, display as integer
+		result = fmt.Sprintf("%d B", int64(value))
+	} else {
+		// For other units, display with 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	// Add negative sign if needed
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Exact boundaries and transitions
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+
+		// Between boundaries
+		{2560, "2.5 KiB"},
+		{5242880, "5.0 MiB"},
+		{10485760, "10.0 MiB"},
+		{1073741824 * 10, "10.0 GiB"},
+
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
