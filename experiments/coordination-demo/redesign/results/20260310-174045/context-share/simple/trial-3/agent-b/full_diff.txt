+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	abs := bytesPerSec
+	if negative {
+		abs = -abs
+	}
+
+	const (
+		kibi = 1024.0
+		mebi = 1024 * kibi
+		gibi = 1024 * mebi
+	)
+
+	sign := ""
+	if negative {
+		sign = "-"
+	}
+
+	switch {
+	case abs < kibi:
+		return fmt.Sprintf("%s%.0f B/s", sign, abs)
+	case abs < mebi:
+		return fmt.Sprintf("%s%.1f KiB/s", sign, abs/kibi)
+	case abs < gibi:
+		return fmt.Sprintf("%s%.1f MiB/s", sign, abs/mebi)
+	default:
+		return fmt.Sprintf("%s%.1f GiB/s", sign, abs/gibi)
+	}
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second, no decimal)
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Kilobytes per second (1 decimal)
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 * 100, "100.0 KiB/s"},
+		// Megabytes per second (1 decimal)
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1048576 * 10, "10.0 MiB/s"},
+		// Gigabytes per second (1 decimal)
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{1073741824 * 5, "5.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
