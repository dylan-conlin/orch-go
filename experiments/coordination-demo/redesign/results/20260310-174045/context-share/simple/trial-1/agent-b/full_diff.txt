+// FormatRate formats a transfer rate in bytes/sec into a human-readable string.
+// Uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	negative := bytesPerSec < 0
+	absRate := bytesPerSec
+	if negative {
+		absRate = -bytesPerSec
+	}
+
+	var result string
+	if absRate < KiB {
+		result = fmt.Sprintf("%.0f B/s", absRate)
+	} else if absRate < MiB {
+		result = fmt.Sprintf("%.1f KiB/s", absRate/KiB)
+	} else if absRate < GiB {
+		result = fmt.Sprintf("%.1f MiB/s", absRate/MiB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", absRate/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
