+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Negative values are prefixed with "-".
+// Examples: "0 B", "512 B", "1.5 MiB", "-1.0 KiB".
+func FormatBytes(bytes int64) string {
+	const (
+		B  = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -absBytes
+	}
+
+	var val float64
+	var unit string
+
+	switch {
+	case absBytes < KiB:
+		return fmt.Sprintf("%d B", bytes)
+	case absBytes < MiB:
+		val = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	case absBytes < GiB:
+		val = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes < TiB:
+		val = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		val = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	if negative {
+		return fmt.Sprintf("-%.1f %s", val, unit)
+	}
+	return fmt.Sprintf("%.1f %s", val, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
