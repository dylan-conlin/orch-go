+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "512 B", "1.5 KiB", "1.0 MiB", "1.0 GiB", "1.0 TiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitIndex := 0
+	value := float64(bytes)
+
+	for unitIndex < len(units)-1 && value >= 1024 {
+		value /= 1024
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// Bytes unit doesn't need decimal places
+		result = fmt.Sprintf("%d B", int64(value))
+	} else {
+		// Other units show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1024 * 1024, "1.0 MiB"},
+		{1024 * 1024 * 1024, "1.0 GiB"},
+		{1024 * 1024 * 1024 * 1024, "1.0 TiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{1024*1024 + 512*1024, "1.5 MiB"},
+		{1024*1024*1024 + 512*1024*1024, "1.5 GiB"},
+		// Large values
+		{5 * 1024 * 1024 * 1024, "5.0 GiB"},
+		{10 * 1024 * 1024 * 1024 * 1024, "10.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1024 * 1024 * 1024, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
