+// FormatRate formats a transfer rate (bytes per second) into a human-readable string.
+// Output style: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	unitSize := 1024.0
+
+	value := bytesPerSec
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for unitIndex < len(units)-1 && value >= unitSize {
+		value /= unitSize
+		unitIndex++
+	}
+
+	// Format the output
+	var result string
+	if unitIndex == 0 {
+		// For B/s, show as integer
+		result = fmt.Sprintf("%.0f %s", value, units[unitIndex])
+	} else {
+		// For larger units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
