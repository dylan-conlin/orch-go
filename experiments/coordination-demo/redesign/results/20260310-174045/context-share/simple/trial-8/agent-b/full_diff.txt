+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units with "/s" suffix: "B/s", "KiB/s", "MiB/s", "GiB/s".
+// Non-byte units are shown with 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	const (
+		KiB = 1024
+		MiB = KiB * 1024
+		GiB = MiB * 1024
+	)
+
+	// Handle zero
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative by computing absolute value and prefixing later
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	// Format with appropriate unit
+	var formatted string
+	if rate < KiB {
+		formatted = fmt.Sprintf("%.0f B/s", rate)
+	} else if rate < MiB {
+		formatted = fmt.Sprintf("%.1f KiB/s", rate/KiB)
+	} else if rate < GiB {
+		formatted = fmt.Sprintf("%.1f MiB/s", rate/MiB)
+	} else {
+		formatted = fmt.Sprintf("%.1f GiB/s", rate/GiB)
+	}
+
+	// Add negative prefix if needed
+	if negative {
+		return "-" + formatted
+	}
+	return formatted
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+		// Large KiB values near MiB boundary
+		{1024 * 1023, "1023.0 KiB/s"},
+		{1024 * 1024 - 1, "1024.0 KiB/s"},
+		// Exact MiB boundary
+		{1024 * 1024, "1.0 MiB/s"},
+		// Fractional MiB
+		{1024 * 1024 * 1.5, "1.5 MiB/s"},
+		{1024 * 1024 * 2, "2.0 MiB/s"},
+		{1024 * 1024 * 10, "10.0 MiB/s"},
+		// Large MiB values near GiB boundary
+		{1024 * 1024 * 1023, "1023.0 MiB/s"},
+		{1024 * 1024 * 1024 - 1, "1024.0 MiB/s"},
+		// GiB and above
+		{1024 * 1024 * 1024, "1.0 GiB/s"},
+		{1024 * 1024 * 1024 * 1.5, "1.5 GiB/s"},
+		{1024 * 1024 * 1024 * 10, "10.0 GiB/s"},
+		{1024 * 1024 * 1024 * 1024, "1024.0 GiB/s"},
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1024 * 1024, "-1.0 MiB/s"},
+		{-1024 * 1024 * 1.5, "-1.5 MiB/s"},
+		{-1024 * 1024 * 1024, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%.0f) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
