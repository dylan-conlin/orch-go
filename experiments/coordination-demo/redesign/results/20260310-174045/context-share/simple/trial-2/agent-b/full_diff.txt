+// FormatRate formats a transfer rate in bytes per second into a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec values show 1 decimal place. Negative rates are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kib = 1024
+		mib = 1024 * 1024
+		gib = 1024 * 1024 * 1024
+	)
+
+	var result string
+	switch {
+	case bytesPerSec < kib:
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	case bytesPerSec < mib:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kib)
+	case bytesPerSec < gib:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/mib)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/gib)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
