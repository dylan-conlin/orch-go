+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units: B/s, KiB/s, MiB/s, GiB/s.
+// Output style: "512 B/s", "1.5 MiB/s", "-1.0 KiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+
+	switch {
+	case bytesPerSec < KiB:
+		result = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	case bytesPerSec < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	case bytesPerSec < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero value
+		{0, "0 B/s"},
+		// Small values (bytes per second)

+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary (1 KiB)
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB/s
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},

+		{10240, "10.0 KiB/s"},
+		// Exact boundary (1 MiB)
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB/s
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Larger MiB/s values
+		{10485760, "10.0 MiB/s"},
+		// Exact boundary (1 GiB)
+		{1073741824, "1.0 GiB/s"},
+		// Fractional GiB/s

+		// Large GiB/s values
