+// FormatBytes formats a byte count into a human-readable string.
+// Uses binary units (KiB, MiB, GiB, TiB) with 1 decimal place for non-byte units.
+// Output style: "512 B", "1.5 KiB", "1.0 MiB".
+func FormatBytes(bytes int64) string {
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	value := bytes
+	if bytes < 0 {
+		negative = true
+		value = -bytes
+	}
+
+	var result string
+	switch {
+	case value < KiB:
+		result = fmt.Sprintf("%d B", value)
+	case value < MiB:
+		result = fmt.Sprintf("%.1f KiB", float64(value)/float64(KiB))
+	case value < GiB:
+		result = fmt.Sprintf("%.1f MiB", float64(value)/float64(MiB))
+	case value < TiB:
+		result = fmt.Sprintf("%.1f GiB", float64(value)/float64(GiB))
+	default:
+		result = fmt.Sprintf("%.1f TiB", float64(value)/float64(TiB))
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
