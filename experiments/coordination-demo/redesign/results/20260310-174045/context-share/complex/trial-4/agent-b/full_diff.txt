+// FormatTable renders headers and rows as an aligned text table.
+// It auto-sizes each column based on the widest content in that column
+// (including headers), using visual width to account for ANSI escape codes.
+// Headers are separated from data rows with a line of dashes.
+// Handles edge cases: empty rows (headers only), rows with fewer columns than headers.
+// Rows with more columns than headers are truncated to match header count.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on visual width (ignoring ANSI codes)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Update widths based on row content
+	for _, row := range rows {
+		for i, cell := range row {
+			if i >= len(headers) {
+				break // Ignore columns beyond header count
+			}
+			visualWidth := len(StripANSI(cell))
+			if visualWidth > colWidths[i] {
+				colWidths[i] = visualWidth
+			}
+		}
+	}
+
+	// Helper function to pad a cell to a target visual width
+	padCell := func(cell string, width int) string {
+		visualWidth := len(StripANSI(cell))
+		if visualWidth >= width {
+			return cell
+		}
+		padding := width - visualWidth
+		return cell + strings.Repeat(" ", padding)
+	}
+
+	// Build header row
+	var sb strings.Builder
+	for i, h := range headers {
+		if i > 0 {
+			sb.WriteString(" | ")
+		}
+		sb.WriteString(padCell(h, colWidths[i]))
+	}
+	sb.WriteString("\n")
+
+	// Build separator line
+	for i := 0; i < len(headers); i++ {
+		if i > 0 {
+			sb.WriteString("-+-")
+		}
+		sb.WriteString(strings.Repeat("-", colWidths[i]))
+	}
+	sb.WriteString("\n")
+
+	// Build data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				sb.WriteString(" | ")
+			}
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			sb.WriteString(padCell(cell, colWidths[i]))
+		}
+		sb.WriteString("\n")
+	}
+
+	return strings.TrimSuffix(sb.String(), "\n")
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 ",
+		},
+		{
+			name:    "empty table (headers only)",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "Col1 | Col2\n-----+-----",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"Apple"},
+				{"Banana"},
+			},
+			want: "Item  \n------\nApple \nBanana",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongContent"},
+			rows: [][]string{
+				{"A", "B"},
+			},
+			want: "Short | VeryLongContent\n------+----------------\nA     | B              ",
+		},
+		{
+			name:    "mismatched columns - row shorter than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z", "extra"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \nx | y | z",
+		},
+		{
+			name:    "ANSI-colored content alignment",
+			headers: []string{"Color", "Value"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m", "123"},
+				{"Plain", "\x1b[32mGreen\x1b[0m"},
+			},
+			want: "Color | Value\n------+------\n\x1b[31mRed\x1b[0m   | 123  \nPlain | \x1b[32mGreen\x1b[0m",
+		},
+		{
+			name:    "all empty content",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"", ""},
+			},
+			want: "A | B\n--+--\n  |  ",
+		},
+		{
+			name:    "header with spaces",
+			headers: []string{"Long Header", "Short"},
+			rows: [][]string{
+				{"x", "y"},
+			},
+			want: "Long Header | Short\n------------+------\nx           | y    ",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() = %q, want %q", got, tt.want)
+			}
+		})
+	}

+func TestFormatTableEdgeCases(t *testing.T) {
+	// Test nil inputs
+	result := FormatTable(nil, nil)
+	if result != "" {
+		t.Errorf("FormatTable(nil, nil) = %q, want empty string", result)
+	}
+
+	// Test empty headers
+	result = FormatTable([]string{}, [][]string{})
+	if result != "" {
+		t.Errorf("FormatTable(empty headers, empty rows) = %q, want empty string", result)
+	}
+
+	// Test nil rows
+	result = FormatTable([]string{"A", "B"}, nil)
+	expected := "A | B\n--+--"
+	if result != expected {
+		t.Errorf("FormatTable(headers, nil rows) = %q, want %q", result, expected)
+	}
+
+	// Test single header, multiple rows
+	result = FormatTable([]string{"Name"}, [][]string{
+		{"Alice"},
+		{"Bob"},
+		{"Charlie"},
+	})
+	expected = "Name   \n-------\nAlice  \nBob    \nCharlie"
+	if result != expected {
+		t.Errorf("FormatTable single column = %q, want %q", result, expected)
+	}
