+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. Unicode characters are counted as single runes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is
+// already wider than or equal to width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+		// ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[0mno color\x1b[0m", 8},
+		// Unicode and emoji
+		{"你好", 2},                           // Chinese characters (CJK)
+		{"こんにちは", 5},                        // Japanese hiragana (CJK)
+		{"🎉", 1},                              // Emoji (single rune)
+		{"hello🎉world", 11},                   // Mixed ASCII and emoji
+		// Mixed: ANSI + Unicode
+		{"\x1b[33m你好\x1b[0m", 2},             // ANSI + CJK
+		{"\x1b[1;36m🎉 celebrate\x1b[0m", 11}, // ANSI + emoji + ASCII
+		// Edge cases
+		{"  ", 2},           // Just spaces
+		{"\x1b[0m", 0},      // Just ANSI code
+		{"\x1b[31m\x1b[0m", 0}, // Only ANSI codes
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII - needs padding
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		{"", 3, "   "},
+		// Plain ASCII - already at width
+		{"hello", 5, "hello"},
+		{"a", 1, "a"},
+		// Plain ASCII - wider than target (unchanged)
+		{"hello", 3, "hello"},
+		{"hello world", 5, "hello world"},
+		// ANSI color codes
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[1;32mbold\x1b[0m", 6, "\x1b[1;32mbold\x1b[0m  "},
+		{"\x1b[31mred\x1b[0m", 3, "\x1b[31mred\x1b[0m"},
+		// Unicode strings
+		{"你好", 4, "你好  "},
+		{"🎉", 3, "🎉  "},
+		{"你好", 2, "你好"},
+		// Mixed: ANSI + Unicode
+		{"\x1b[33m你好\x1b[0m", 4, "\x1b[33m你好\x1b[0m  "},
+		{"\x1b[36m🎉\x1b[0m", 3, "\x1b[36m🎉\x1b[0m  "},
+		// Edge cases
+		{"", 0, ""},
+		{"  ", 5, "     "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width is correct (or wider if input was already wider)
+		visualWidth := VisualWidth(got)
+		if visualWidth < tt.width && VisualWidth(tt.input) < tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, visualWidth, tt.width)
+		}
+	}
+}
