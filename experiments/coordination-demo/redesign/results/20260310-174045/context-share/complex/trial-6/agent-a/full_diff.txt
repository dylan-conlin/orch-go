+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It correctly counts Unicode runes, not bytes.
+func VisualWidth(s string) int {
+	// Strip ANSI codes, then count runes to handle Unicode correctly
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved in the output but do not count toward the width.
+// If the string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// ANSI color codes (should be stripped)
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[38;5;196mcomplex\x1b[0m", 7},
+
+		// Unicode: CJK characters (typically 2 runes)
+		{"你好", 2},        // Chinese characters
+		{"こんにちは", 5},  // Japanese hiragana
+		{"안녕", 2},       // Korean Hangul
+
+		// Unicode: emoji
+		{"👋", 1},         // Emoji (1 rune)
+		{"hello👋world", 11}, // Mixed ASCII and emoji (5 + 1 + 5 = 11)
+
+		// Mixed: Unicode with ANSI codes
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[32m👋\x1b[0m", 1},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain strings: needs padding
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		{"", 3, "   "},
+
+		// Strings already at target width: no padding needed
+		{"exact", 5, "exact"},
+		{"abc", 3, "abc"},
+
+		// Strings already wider than target: return unchanged
+		{"hello world", 5, "hello world"},
+		{"longstring", 3, "longstring"},
+
+		// ANSI codes: preserved but don't count toward width
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[1;32mbold\x1b[0m", 10, "\x1b[1;32mbold\x1b[0m      "},
+
+		// Unicode: correctly padded
+		{"你好", 4, "你好  "},
+		{"👋", 3, "👋  "},
+
+		// Edge cases
+		{"a", 0, "a"},
+		{"test", 4, "test"},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify the visual width matches expectations
+		visualWidth := VisualWidth(got)
+		expectedWidth := VisualWidth(tt.want)
+		if visualWidth != expectedWidth {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, visualWidth, expectedWidth)
+		}
+	}
+}
