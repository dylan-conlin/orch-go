+// FormatTable renders headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including header).
+// A separator line of dashes is placed between the header and data rows.
+// ANSI escape codes are ignored when calculating column widths.
+// Rows with fewer columns than headers are padded with empty strings.
+// Rows with more columns than headers have extra columns ignored.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate max width for each column
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i, cell := range row {
+			if i >= len(headers) {
+				break // ignore extra columns
+			}
+			w := len(StripANSI(cell))
+			if w > colWidths[i] {
+				colWidths[i] = w
+			}
+		}
+	}
+
+	var sb strings.Builder
+
+	// Header row
+	for i, h := range headers {
+		if i > 0 {
+			sb.WriteString(" | ")
+		}
+		sb.WriteString(h)
+		// Pad to column width
+		padding := colWidths[i] - len(StripANSI(h))
+		sb.WriteString(strings.Repeat(" ", padding))
+	}
+	sb.WriteString("\n")
+
+	// Separator
+	for i, w := range colWidths {
+		if i > 0 {
+			sb.WriteString("-+-")
+		}
+		sb.WriteString(strings.Repeat("-", w))
+	}
+	sb.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				sb.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			sb.WriteString(cell)
+			// Pad to column width
+			padding := colWidths[i] - len(StripANSI(cell))
+			sb.WriteString(strings.Repeat(" ", padding))
+		}
+		sb.WriteString("\n")
+	}
+
+	return sb.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "Name  | Age | City\n------+-----+-----\nAlice | 30  | NYC \nBob   | 25  | LA  \n",
+		},
+		{
+			name:    "empty table (headers only)",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "Col1 | Col2\n-----+-----\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"Apple"},
+				{"Banana"},
+				{"Cherry"},
+			},
+			want: "Item  \n------\nApple \nBanana\nCherry\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status", "Value"},
+			rows: [][]string{
+				{"\x1b[32mOK\x1b[0m", "\x1b[31mFail\x1b[0m"},
+				{"Normal", "\x1b[1;33mWarn\x1b[0m"},
+			},
+			want: "Status | Value\n-------+------\n\x1b[32mOK\x1b[0m     | \x1b[31mFail\x1b[0m \nNormal | \x1b[1;33mWarn\x1b[0m \n",
+		},
+		{
+			name:    "mismatched columns (fewer columns in row)",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \n3 | 4 | 5\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"x", "short"},
+				{"abc", "this is a much longer string"},
+			},
+			want: "Short | VeryLongHeader              \n------+-----------------------------\nx     | short                       \nabc   | this is a much longer string\n",
+		},
+		{
+			name:    "single row",
+			headers: []string{"H1", "H2"},
+			rows: [][]string{
+				{"data1", "data2"},
+			},
+			want: "H1    | H2   \n------+------\ndata1 | data2\n",
+		},
+		{
+			name:    "empty cells",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"", "value"},
+				{"value", ""},
+			},
+			want: "A     | B    \n------+------\n      | value\nvalue |      \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch\ngot:\n%q\n\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
+}
