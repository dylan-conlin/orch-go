+// FormatTable renders headers and rows as an aligned text table.
+// It auto-sizes each column based on the widest content in that column
+// (including headers). The header is separated from data rows with a line
+// of dashes. ANSI-colored content is handled correctly using StripANSI for
+// width calculation. Handles edge cases: empty rows, rows with fewer columns
+// than headers, and nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Determine the number of columns
+	numCols := len(headers)
+	for _, row := range rows {
+		if len(row) > numCols {
+			numCols = len(row)
+		}
+	}
+
+	// Extend headers with empty strings if there are more columns
+	extendedHeaders := make([]string, numCols)
+	copy(extendedHeaders, headers)
+
+	// Calculate column widths (using visual width, ignoring ANSI codes)
+	colWidths := make([]int, numCols)
+	for i, header := range extendedHeaders {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < numCols; i++ {
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			width := len(StripANSI(content))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Build the table
+	var result strings.Builder
+
+	// Write header
+	for i, header := range extendedHeaders {
+		result.WriteString("| ")
+		visualWidth := len(StripANSI(header))
+		paddedHeader := header + strings.Repeat(" ", colWidths[i]-visualWidth)
+		result.WriteString(paddedHeader)
+		result.WriteString(" ")
+	}
+	result.WriteString("|\n")
+
+	// Write separator
+	for i := 0; i < numCols; i++ {
+		result.WriteString("+")
+		result.WriteString(strings.Repeat("-", colWidths[i]+2))
+	}
+	result.WriteString("+\n")
+
+	// Write rows
+	for _, row := range rows {
+		for i := 0; i < numCols; i++ {
+			result.WriteString("| ")
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			visualWidth := len(StripANSI(content))
+			paddedContent := content + strings.Repeat(" ", colWidths[i]-visualWidth)
+			result.WriteString(paddedContent)
+			result.WriteString(" ")
+		}
+		result.WriteString("|\n")
+	}
+
+	return result.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name  | Age |\n+-------+-----+\n| Alice | 30  |\n| Bob   | 25  |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Col1", "Col2"},
+			rows:    nil,
+			want:    "| Col1 | Col2 |\n+------+------+\n",
+		},
+		{
+			name:    "empty headers and rows",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"Ready"},
+				{"Running"},
+			},
+			want: "| Status  |\n+---------+\n| Ready   |\n| Running |\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Message"},
+			rows: [][]string{
+				{"\x1b[31mError\x1b[0m"},
+				{"Normal"},
+			},
+			want: "| Message |\n+---------+\n| \x1b[31mError\x1b[0m   |\n| Normal  |\n",
+		},
+		{
+			name:    "rows with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "| A | B | C |\n+---+---+---+\n| 1 | 2 |   |\n| x | y | z |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongerHeader"},
+			rows: [][]string{
+				{"a", "very long content"},
+				{"longer_value", "b"},
+			},
+			want: "| Short        | LongerHeader      |\n+--------------+-------------------+\n| a            | very long content |\n| longer_value | b                 |\n",
+		},
+		{
+			name:    "empty rows list",
+			headers: []string{"X", "Y"},
+			rows:    [][]string{},
+			want:    "| X | Y |\n+---+---+\n",
+		},
+		{
+			name:    "rows with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+				{"x", "y", "z"},
+			},
+			want: "| A | B |   |\n+---+---+---+\n| 1 | 2 | 3 |\n| x | y | z |\n",
+		},
+		{
+			name:    "mixed ANSI across columns",
+			headers: []string{"\x1b[1mBold\x1b[0m", "Normal"},
+			rows: [][]string{
+				{"data", "\x1b[32mgreen\x1b[0m"},
+			},
+			want: "| \x1b[1mBold\x1b[0m | Normal |\n+------+--------+\n| data | \x1b[32mgreen\x1b[0m  |\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() = %q, want %q", got, tt.want)
+			}
+		})
+	}
+}
