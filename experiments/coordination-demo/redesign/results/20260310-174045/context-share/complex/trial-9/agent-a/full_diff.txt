+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It counts runes (Unicode characters) in the stripped string.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is
+// already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"hello world", 11},
+		{"", 0},
+		{"a", 1},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[0m\x1b[1mtext\x1b[0m", 4},
+
+		// Unicode strings (CJK, emoji)
+		{"你好", 2},
+		{"你好世界", 4},
+		{"hello🌍world", 11}, // emoji counts as 1 rune
+
+		// Mixed: ANSI codes with Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1;32m🎉success\x1b[0m", 8},
+
+		// Multi-byte UTF-8 characters
+		{"café", 4},
+		{"naïve", 5},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		wantLen int // visual width of result
+	}{
+		// Plain ASCII strings - needs padding
+		{"hi", 5, "hi   ", 5},
+		{"hello", 10, "hello     ", 10},
+		{"a", 3, "a  ", 3},
+
+		// Strings already at target width
+		{"hello", 5, "hello", 5},
+		{"world", 5, "world", 5},
+		{"", 0, "", 0},
+
+		// Strings wider than target - returned unchanged
+		{"hello", 3, "hello", 5},
+		{"hello world", 5, "hello world", 11},
+
+		// ANSI codes with padding
+		{"\x1b[31mhi\x1b[0m", 5, "\x1b[31mhi\x1b[0m   ", 5},
+		{"\x1b[1;32mtest\x1b[0m", 8, "\x1b[1;32mtest\x1b[0m    ", 8},
+
+		// Unicode with padding
+		{"你好", 4, "你好  ", 4},
+		{"🎉", 3, "🎉  ", 3},
+
+		// Unicode + ANSI with padding
+		{"\x1b[31m你好\x1b[0m", 4, "\x1b[31m你好\x1b[0m  ", 4},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify the visual width of the result
+		gotWidth := VisualWidth(got)
+		if gotWidth != tt.wantLen {
+			t.Errorf("PadToWidth(%q, %d) resulted in visual width %d, want %d", tt.input, tt.width, gotWidth, tt.wantLen)
+		}
+	}
+}
