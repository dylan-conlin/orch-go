+// FormatTable renders headers and rows as an aligned text table. Each column
+// is sized to fit the widest content (including headers). ANSI color codes
+// are preserved but don't count toward column width. Rows with fewer columns
+// than headers are padded with empty cells. If rows have more columns than
+// headers, extra columns are omitted.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate max width for each column
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Write header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		paddedLen := colWidths[i] - len(StripANSI(h))
+		buf.WriteString(h)
+		buf.WriteString(strings.Repeat(" ", paddedLen))
+	}
+	buf.WriteString("\n")
+
+	// Write separator
+	for i, w := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", w))
+	}
+	buf.WriteString("\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			paddedLen := colWidths[i] - len(StripANSI(cell))
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", paddedLen))
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 \n",
+		},
+		{
+			name:    "headers only (no rows)",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows:    [][]string{},
+			want:    "Col1 | Col2 | Col3\n-----+------+-----\n",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Data"},
+			rows: [][]string{
+				{"Value1"},
+				{"Value2"},
+				{"LongerValue"},
+			},
+			want: "Data       \n-----------\nValue1     \nValue2     \nLongerValue\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"\x1b[32mSuccess\x1b[0m"},
+				{"\x1b[31mFailed\x1b[0m"},
+			},
+			want: "Status \n-------\n\x1b[32mSuccess\x1b[0m\n\x1b[31mFailed\x1b[0m \n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeaderName"},
+			rows: [][]string{
+				{"X", "Y"},
+				{"ABC", "This is a very long cell value"},
+			},
+			want: "Short | VeryLongHeaderName            \n------+-------------------------------\nX     | Y                             \nABC   | This is a very long cell value\n",
+		},
+		{
+			name:    "rows with fewer columns than headers",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows: [][]string{
+				{"A", "B"},
+				{"X", "Y", "Z"},
+				{"P"},
+			},
+			want: "Col1 | Col2 | Col3\n-----+------+-----\nA    | B    |     \nX    | Y    | Z   \nP    |      |     \n",
+		},
+		{
+			name:    "mixed content widths",
+			headers: []string{"ID", "Name", "Description"},
+			rows: [][]string{
+				{"1", "Alice", "Engineer"},
+				{"2", "Bob Johnson", "Manager"},
+				{"123", "A", "X"},
+			},
+			want: "ID  | Name        | Description\n----+-------------+------------\n1   | Alice       | Engineer   \n2   | Bob Johnson | Manager    \n123 | A           | X          \n",
+		},
+		{
+			name:    "empty headers and rows",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() got:\n%q\n\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
