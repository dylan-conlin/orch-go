+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It counts Unicode runes, not bytes, so emoji and CJK
+// characters are counted as 1 width unit each.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward the width calculation.
+// If the string is already at or wider than the target width, it is returned
+// unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// ANSI escape codes should not count
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m test", 11},
+
+		// Unicode and emoji
+		{"café", 4},
+		{"你好", 2},
+		{"😀", 1},
+		{"hello 😀 world", 13},
+
+		// Combined: ANSI + Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[32mCafé\x1b[0m", 4},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		wantLen int
+	}{
+		// Plain ASCII - padding needed
+		{"hello", 10, "hello     ", 10},
+		{"hi", 5, "hi   ", 5},
+
+		// Empty string
+		{"", 5, "     ", 5},
+		{"", 0, "", 0},
+
+		// Already at target width
+		{"hello", 5, "hello", 5},
+		{"test", 4, "test", 4},
+
+		// Already wider than target - no padding
+		{"hello", 3, "hello", 5},
+		{"hello world", 5, "hello world", 11},
+
+		// ANSI codes preserved but don't count toward width
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  ", 11}, // red is 3 chars, padded to 5, so 2 spaces added (invisible)
+		{"\x1b[1;32mbold\x1b[0m", 8, "\x1b[1;32mbold\x1b[0m    ", 20}, // bold is 4 chars, padded to 8, so 4 spaces added
+
+		// Unicode
+		{"café", 6, "café  ", 8},
+		{"你好", 4, "你好  ", 6},
+
+		// Unicode + ANSI
+		{"\x1b[36m你好\x1b[0m", 4, "\x1b[36m你好\x1b[0m  ", 10},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Also verify the visual width (ignoring ANSI)
+		gotVisualWidth := VisualWidth(got)
+		if gotVisualWidth < tt.width {
+			// Got visual width should be at least the target width (unless already wider)
+			originalWidth := VisualWidth(tt.input)
+			if originalWidth < tt.width && gotVisualWidth != tt.width {
+				t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotVisualWidth, tt.width)
+			}
+		}
+	}
+}
