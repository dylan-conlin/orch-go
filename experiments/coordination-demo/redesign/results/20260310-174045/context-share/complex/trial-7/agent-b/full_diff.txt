+// FormatTable renders headers and rows as an aligned text table. Columns are
+// auto-sized based on the widest content in each column, including headers.
+// ANSI escape codes in content are handled correctly for width calculation
+// using StripANSI. Rows with fewer columns than headers are padded with empty
+// cells. Empty tables (headers only) are handled gracefully.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on visual width (ignoring ANSI codes)
+	widths := make([]int, len(headers))
+	for i, h := range headers {
+		widths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(headers) {
+				w := len(StripANSI(cell))
+				if w > widths[i] {
+					widths[i] = w
+				}
+			}
+		}
+	}
+
+	// Helper to pad a string to target visual width while preserving ANSI codes
+	padToWidth := func(s string, width int) string {
+		visualWidth := len(StripANSI(s))
+		if visualWidth >= width {
+			return s
+		}
+		return s + strings.Repeat(" ", width-visualWidth)
+	}
+
+	var result strings.Builder
+
+	// Header row
+	for i, h := range headers {
+		if i > 0 {
+			result.WriteString(" | ")
+		}
+		result.WriteString(" ")
+		result.WriteString(padToWidth(h, widths[i]))
+		result.WriteString(" ")
+	}
+	result.WriteString("\n")
+
+	// Separator line
+	for i := range headers {
+		if i > 0 {
+			result.WriteString("-+-")
+		}
+		result.WriteString(strings.Repeat("-", widths[i]+2))
+	}
+	result.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		for i := range headers {
+			if i > 0 {
+				result.WriteString(" | ")
+			}
+			result.WriteString(" ")
+			if i < len(row) {
+				result.WriteString(padToWidth(row[i], widths[i]))
+			} else {
+				result.WriteString(strings.Repeat(" ", widths[i]))
+			}
+			result.WriteString(" ")
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: " Name   |  Age \n--------+------\n Alice  |  30  \n Bob    |  25  \n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows:    [][]string{},
+			want:    " Col1  |  Col2  |  Col3 \n-------+--------+-------\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+			},
+			want: " Item   \n--------\n apple  \n banana \n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"\x1b[32mgreen\x1b[0m"},
+				{"\x1b[31mred\x1b[0m"},
+			},
+			want: " Status \n--------\n \x1b[32mgreen\x1b[0m  \n \x1b[31mred\x1b[0m    \n",
+		},
+		{
+			name:    "mismatched column counts",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1"},
+				{"2", "3"},
+				{"4", "5", "6"},
+			},
+			want: " A  |  B  |  C \n----+-----+----\n 1  |     |    \n 2  |  3  |    \n 4  |  5  |  6 \n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeaderName"},
+			rows: [][]string{
+				{"x", "y"},
+				{"verylongcontent", "z"},
+			},
+			want: " Short            |  VeryLongHeaderName \n------------------+---------------------\n x                |  y                  \n verylongcontent  |  z                  \n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "single cell table",
+			headers: []string{"Only"},
+			rows: [][]string{
+				{"Data"},
+			},
+			want: " Only \n------\n Data \n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Header"},
+			rows:    nil,
+			want:    " Header \n--------\n",
+		},
+		{
+			name:    "ANSI multiple colors",
+			headers: []string{"Name", "Value"},
+			rows: [][]string{
+				{"\x1b[1;32mbold green\x1b[0m", "\x1b[1;31mbold red\x1b[0m"},
+			},
+			want: " Name        |  Value    \n-------------+-----------\n \x1b[1;32mbold green\x1b[0m  |  \x1b[1;31mbold red\x1b[0m \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+				// Print formatted output for visual inspection
+				t.Logf("Got output:\n%s", got)
+				t.Logf("Want output:\n%s", tt.want)

+		})
+	}
