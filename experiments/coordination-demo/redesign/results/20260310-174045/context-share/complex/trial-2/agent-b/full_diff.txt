+// FormatTable formats headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including header).
+// ANSI escape codes in content are stripped for width calculation but preserved in output.
+// Rows with fewer columns than headers are padded with empty strings.
+// If a row has more columns than headers, excess columns are ignored.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and data
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Write header
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(h)
+		// Pad all columns except the last
+		if i < len(headers)-1 {
+			visualWidth := len(StripANSI(h))
+			padding := colWidths[i] - visualWidth
+			if padding > 0 {
+				buf.WriteString(strings.Repeat(" ", padding))
+			}
+		}
+	}
+	buf.WriteString("\n")
+
+	// Write separator
+	for i, width := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", width))
+	}
+	buf.WriteString("\n")
+
+	// Write rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(cell)
+			// Pad all columns except the last
+			if i < len(headers)-1 {
+				visualWidth := len(StripANSI(cell))
+				padding := colWidths[i] - visualWidth
+				if padding > 0 {
+					buf.WriteString(strings.Repeat(" ", padding))
+				}
+			}
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table with headers and rows",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30\nBob   | 25\n",
+		},
+		{
+			name:    "headers only (no rows)",
+			headers: []string{"ID", "Status"},
+			rows:    nil,
+			want:    "ID | Status\n---+-------\n",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Label"},
+			rows: [][]string{
+				{"Value1"},
+				{"Value2"},
+			},
+			want: "Label\n------\nValue1\nValue2\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongHeader"},
+			rows: [][]string{
+				{"A", "VeryLongContent"},
+				{"X", "Y"},
+			},
+			want: "Short | LongHeader\n------+----------------\nA     | VeryLongContent\nX     | Y\n",
+		},
+		{
+			name:    "ANSI-colored content alignment",
+			headers: []string{"Color", "Value"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m", "text"},
+				{"Plain", "data"},
+			},
+			want: "Color | Value\n------+------\n\x1b[31mRed\x1b[0m   | text\nPlain | data\n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 | \nx | y | z\n",
+		},
+		{
+			name:    "row with more columns than headers (excess ignored)",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+				{"x", "y"},
+			},
+			want: "A | B\n--+--\n1 | 2\nx | y\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{{"data"}},
+			want:    "",
+		},
+		{
+			name:    "multiple ANSI codes and mixed content",
+			headers: []string{"Status", "Message"},
+			rows: [][]string{
+				{"\x1b[1;32m✓ OK\x1b[0m", "Success"},
+				{"\x1b[1;31m✗ Failed\x1b[0m", "Error"},
+			},
+			want: "Status     | Message\n-----------+--------\n\x1b[1;32m✓ OK\x1b[0m     | Success\n\x1b[1;31m✗ Failed\x1b[0m | Error\n",
+		},
+		{
+			name:    "all empty strings",
+			headers: []string{"H1", "H2"},
+			rows: [][]string{
+				{"", ""},
+				{"", ""},
+			},
+			want: "H1 | H2\n---+---\n   | \n   | \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
