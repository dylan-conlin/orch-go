+// FormatRate formats a transfer rate as a human-readable string using
+// binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	abs := bytesPerSec
+	if negative {
+		abs = -abs
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case abs >= GiB:
+		result = fmt.Sprintf("%.1f GiB/s", abs/GiB)
+	case abs >= MiB:
+		result = fmt.Sprintf("%.1f MiB/s", abs/MiB)
+	case abs >= KiB:
+		result = fmt.Sprintf("%.1f KiB/s", abs/KiB)
+	default:
+		result = fmt.Sprintf("%.0f B/s", abs)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+
+		// Fractional values
+		{1536.5, "1.5 KiB/s"},
+		{2560.5, "2.5 KiB/s"},
+		{1572864.5, "1.5 MiB/s"},
+
+		// Large values (GiB/s)
+		{1073741824 * 2, "2.0 GiB/s"},
+		{1073741824 * 5, "5.0 GiB/s"},
+		{1073741824 * 10.5, "10.5 GiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		{-1610612736, "-1.5 GiB/s"},

+		got := FormatRate(tt.input)
+		if got != tt.want {
