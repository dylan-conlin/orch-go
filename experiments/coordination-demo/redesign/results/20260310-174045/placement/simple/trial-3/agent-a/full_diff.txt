+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", etc.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	value := float64(bytes)
+	if negative {
+		value = -value
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	threshold := 1024.0
+
+	for idx := 0; idx < len(units)-1; idx++ {
+		if value < threshold {
+			if idx == 0 {
+				// Bytes: show as integer
+				if negative {
+					return fmt.Sprintf("-%d %s", int64(value), units[idx])
+				}
+				return fmt.Sprintf("%d %s", int64(value), units[idx])
+			}
+			// Other units: show with 1 decimal place
+			if negative {
+				return fmt.Sprintf("-%.1f %s", value, units[idx])
+			}
+			return fmt.Sprintf("%.1f %s", value, units[idx])
+		}
+		value /= threshold
+	}
+
+	// TiB (largest unit)
+	if negative {
+		return fmt.Sprintf("-%.1f %s", value, units[len(units)-1])
+	}
+	return fmt.Sprintf("%.1f %s", value, units[len(units)-1])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundaries and values
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		{1048575, "1024.0 KiB"},
+		// MiB boundaries and values
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1073741823, "1024.0 MiB"},
+		// GiB boundaries and values
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{10737418240, "10.0 GiB"},
+		// TiB boundaries and values
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
