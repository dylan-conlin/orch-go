+// FormatBytes formats byte counts into human-readable strings using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = KiB * 1024
+		GiB = MiB * 1024
+		TiB = GiB * 1024
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes < KiB:
+		return fmt.Sprintf("%d B", bytes)
+	case bytes < MiB:
+		value = float64(bytes) / KiB
+		unit = "KiB"
+	case bytes < GiB:
+		value = float64(bytes) / MiB
+		unit = "MiB"
+	case bytes < TiB:
+		value = float64(bytes) / GiB
+		unit = "GiB"
+	default:
+		value = float64(bytes) / TiB
+		unit = "TiB"
+	}
+
+	result := fmt.Sprintf("%.1f %s", value, unit)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{10737418240, "10.0 GiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		// Additional decimal cases
+		{2560, "2.5 KiB"},
+		{3584, "3.5 KiB"},
+		{1572864, "1.5 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
