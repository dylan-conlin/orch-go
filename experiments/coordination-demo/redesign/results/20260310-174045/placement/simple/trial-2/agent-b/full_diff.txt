+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units (B/s, KiB/s, MiB/s, GiB/s) with 1 decimal place
+// for non-byte-per-sec values.
+// Examples: "512 B/s", "1.5 MiB/s", "-1.0 KiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB"}
+	divisor := 1024.0
+	value := bytesPerSec
+
+	// Find the appropriate unit
+	unitIndex := 0
+	for i := 0; i < len(units)-1; i++ {
+		if value < divisor {
+			break
+		}
+		value /= divisor
+		unitIndex = i + 1
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// Bytes - no decimal
+		result = fmt.Sprintf("%.0f %s/s", value, units[unitIndex])
+	} else {
+		// KiB, MiB, GiB - 1 decimal
+		result = fmt.Sprintf("%.1f %s/s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}
