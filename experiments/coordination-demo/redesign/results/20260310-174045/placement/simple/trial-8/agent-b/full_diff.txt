+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-second units show 1 decimal place.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024.0
+		MiB = 1024.0 * 1024.0
+		GiB = 1024.0 * 1024.0 * 1024.0
+	)
+
+	var result string
+
+	switch {
+	case bytesPerSec < KiB:
+		result = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	case bytesPerSec < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	case bytesPerSec < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}
