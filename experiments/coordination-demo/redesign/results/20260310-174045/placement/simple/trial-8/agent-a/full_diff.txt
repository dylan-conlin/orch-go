+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.0 MiB".
+func FormatBytes(bytes int64) string {
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	absBytes := bytes
+	negative := false
+	if bytes < 0 {
+		negative = true
+		absBytes = -bytes
+	}
+
+	units := []struct {
+		size  int64
+		label string
+	}{
+		{TiB, "TiB"},
+		{GiB, "GiB"},
+		{MiB, "MiB"},
+		{KiB, "KiB"},
+	}
+
+	for _, unit := range units {
+		if absBytes >= unit.size {
+			value := float64(absBytes) / float64(unit.size)
+			result := fmt.Sprintf("%.1f %s", value, unit.label)
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	// Less than 1 KiB, show as bytes
+	result := fmt.Sprintf("%d B", absBytes)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
