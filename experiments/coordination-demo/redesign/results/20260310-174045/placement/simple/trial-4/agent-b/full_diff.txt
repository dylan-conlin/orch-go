+// FormatRate formats a data transfer rate into a human-readable string using
+// binary units and "/s" suffix.
+// Output style: "0 B/s", "512 B/s", "1.5 KiB/s", "1.0 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	// Handle zero
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	// Define binary units
+	const (
+		kibi = 1024
+		mebi = 1024 * 1024
+		gibi = 1024 * 1024 * 1024
+	)
+
+	var result string
+
+	// Format based on size
+	if bytesPerSec < kibi {
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else if bytesPerSec < mebi {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kibi)
+	} else if bytesPerSec < gibi {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/mebi)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/gibi)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},

+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundary and values in KiB/s
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{512 * 1024, "512.0 KiB/s"},
+		{1024 * 1024 - 1, "1024.0 KiB/s"},

+		{1048576 * 1.5, "1.5 MiB/s"},
+		{1048576 * 2, "2.0 MiB/s"},
+		{1048576 * 256, "256.0 MiB/s"},
+		{1073741824 - 1, "1024.0 MiB/s"},
+
+		// GiB/s
+		{1073741824, "1.0 GiB/s"},
+		{1073741824 * 1.5, "1.5 GiB/s"},
+		{1073741824 * 2.5, "2.5 GiB/s"},
+		{1073741824 * 10, "10.0 GiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
