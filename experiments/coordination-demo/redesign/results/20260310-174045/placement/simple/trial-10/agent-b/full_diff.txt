+// FormatRate formats a transfer rate in bytes per second as a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte values are shown with 1 decimal place (e.g., "1.5 MiB/s").
+// Examples: 0 → "0 B/s", 512 → "512 B/s", 1024 → "1.0 KiB/s", -1024 → "-1.0 KiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case rate < KiB:
+		result = fmt.Sprintf("%.0f B/s", rate)
+	case rate < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", rate/KiB)
+	case rate < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", rate/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", rate/GiB)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}
