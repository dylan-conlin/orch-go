+// FormatRate formats a transfer rate as a human-readable string.
+// Output uses binary units with "/s" suffix: "512 B/s", "1.5 MiB/s", "2.0 GiB/s".
+// Values < 1024 bytes/sec are shown without decimals; larger values use 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case bytesPerSec < KiB:
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	case bytesPerSec < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	case bytesPerSec < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
