+// VisualWidth returns the visual display width of a string, counting runes but
+// ignoring ANSI escape codes. Useful for measuring the displayed width of
+// strings that may contain color codes or other ANSI formatting.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward the width. If the string
+// is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myes\x1b[0m", 3},
+
+		// Unicode strings (CJK, emoji)
+		{"你好", 2},     // Two Chinese characters
+		{"こんにちは", 5}, // Five Japanese hiragana characters
+		{"😀", 1},       // Single emoji (counts as one rune)
+		{"hello😀world", 11}, // Mixed ASCII and emoji (5 + 1 + 5 = 11)
+
+		// Complex cases
+		{"\x1b[1;31mbold red with spaces\x1b[0m", 20}, // "bold red with spaces" = 20
+		{"\x1b[36mblue\x1b[0m and \x1b[35mmagenta\x1b[0m", 16}, // "blue and magenta" = 4 + 5 + 7 = 16
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		wantLen int // Visual width of result
+	}{
+		// Plain ASCII strings - should be padded
+		{"hello", 10, "hello     ", 10},
+		{"hi", 5, "hi   ", 5},
+		{"", 3, "   ", 3},
+
+		// Already at width - should not pad
+		{"exact", 5, "exact", 5},
+		{"hello", 5, "hello", 5},
+
+		// Already wider than width - should return unchanged
+		{"hello world", 5, "hello world", 11},
+		{"verylongstring", 3, "verylongstring", 14},
+
+		// ANSI color codes - width should ignore codes but preserve them
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  ", 5},
+		{"\x1b[1;32mbold\x1b[0m", 6, "\x1b[1;32mbold\x1b[0m  ", 6},
+
+		// Unicode strings
+		{"你好", 4, "你好  ", 4},
+		{"😀", 3, "😀  ", 3},
+
+		// Complex case with ANSI and already at width
+		{"\x1b[33mtest\x1b[0m", 4, "\x1b[33mtest\x1b[0m", 4},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		gotLen := VisualWidth(got)
+		if gotLen != tt.wantLen {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotLen, tt.wantLen)
+		}
+	}
+}
