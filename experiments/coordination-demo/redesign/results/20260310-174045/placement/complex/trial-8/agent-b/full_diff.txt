+// FormatTable renders headers and rows as an aligned text table with auto-sized
+// columns. Column widths are calculated based on the widest content (header or
+// data row), correctly handling ANSI-colored content by using StripANSI for
+// width calculation. The header is separated from data rows with a line of
+// dashes. Returns an empty string if headers is empty.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths using StripANSI for proper visual width
+	colWidths := make([]int, len(headers))
+
+	// Initialize with header widths
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Check all rows and update max widths
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			w := len(StripANSI(row[i]))
+			if w > colWidths[i] {
+				colWidths[i] = w
+			}
+		}
+	}
+
+	// Build table
+	var result strings.Builder
+	const border = "|"
+	const padding = " "
+
+	// Header row
+	result.WriteString(border)
+	for i, h := range headers {
+		result.WriteString(padding)
+		result.WriteString(padRight(h, colWidths[i]))
+		result.WriteString(padding)
+		result.WriteString(border)
+	}
+	result.WriteString("\n")
+
+	// Separator row
+	result.WriteString(border)
+	for _, w := range colWidths {
+		result.WriteString(strings.Repeat("-", w+2))
+		result.WriteString(border)
+	}
+	result.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		result.WriteString(border)
+		for i := 0; i < len(headers); i++ {
+			result.WriteString(padding)
+
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			result.WriteString(padRight(cell, colWidths[i]))
+			result.WriteString(padding)
+			result.WriteString(border)
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}

+// padRight pads a string (containing possible ANSI codes) to a given width
+// by adding spaces after the content, accounting for ANSI codes not taking space.
+func padRight(s string, width int) string {
+	visibleLen := len(StripANSI(s))
+	if visibleLen >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-visibleLen)
+}

+		headers []string

+		{

+			rows: [][]string{

+				{"3", "4", "5"},
+			},
+			want: "| A | B | C |\n|---|---|---|\n| 1 | 2 |   |\n| 3 | 4 | 5 |\n",

+			},
+			want: "| Color |\n|-------|\n| \x1b[31mred\x1b[0m   |\n| \x1b[32mgreen\x1b[0m |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongHeader"},
+			rows: [][]string{
+				{"x", "verylongvalue"},
+			},
+			want: "| Short | LongHeader    |\n|-------|---------------|\n| x     | verylongvalue |\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {

+			if got != tt.want {

+			}
+		})
+	}
