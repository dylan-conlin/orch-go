+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. Unicode characters are counted as single runes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the width. If the
+// string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-currentWidth)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"", 0},
+		{"hello", 5},
+		{"hello world", 11},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mred text\x1b[0m", 8},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m more", 11},
+
+		// Unicode strings (emoji and CJK characters)
+		{"emoji 😀 test", 12}, // emoji counts as 1 rune
+		{"你好", 2},           // CJK characters, each counts as 1 rune
+		{"こんにちは", 5},      // Japanese hiragana
+
+		// Mixed: ANSI codes with Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[32m😀\x1b[0m", 1},
+
+		// Edge cases
+		{"already exact", 13},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+	}{
+		// Plain ASCII - pad to width
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		{"", 5, "     "},
+
+		// Already at width - no padding
+		{"exact", 5, "exact"},
+
+		// Already wider - no padding, returned unchanged
+		{"longer", 4, "longer"},
+		{"verylongstring", 5, "verylongstring"},
+
+		// ANSI codes - preserve them, pad based on visual width
+		{"\x1b[31mhello\x1b[0m", 10, "\x1b[31mhello\x1b[0m     "},
+
+		// Unicode - pad correctly
+		{"你", 5, "你    "},
+		{"hello 😀", 10, "hello 😀   "},
+
+		// Mixed: ANSI + Unicode
+		{"\x1b[32m你\x1b[0m", 5, "\x1b[32m你\x1b[0m    "},
+
+		// Edge case: empty string to 0 width
+		{"", 0, ""},
+
+		// Already exact with ANSI codes
+		{"\x1b[31mhi\x1b[0m", 2, "\x1b[31mhi\x1b[0m"},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width is correct (should be >= target width)
+		gotVisualWidth := VisualWidth(got)
+		if gotVisualWidth < tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want >= %d", tt.input, tt.width, gotVisualWidth, tt.width)
+		}
+	}
+}
