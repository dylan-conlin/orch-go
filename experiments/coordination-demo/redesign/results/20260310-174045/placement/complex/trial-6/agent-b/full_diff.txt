+// FormatTable formats headers and rows as an aligned text table.
+// It auto-sizes each column based on the widest content (including headers),
+// and separates the header row from data rows with a line of dashes.
+// ANSI escape codes in content are handled correctly using StripANSI
+// for width calculation while preserving the colored text in output.
+// Handles edge cases: nil rows, rows with fewer columns than headers.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on header and row content
+	widths := make([]int, len(headers))
+	for i, h := range headers {
+		widths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			cellWidth := len(StripANSI(cell))
+			if cellWidth > widths[i] {
+				widths[i] = cellWidth
+			}
+		}
+	}
+
+	// Build the table
+	var sb strings.Builder
+
+	// Header row
+	sb.WriteString("| ")
+	for i, h := range headers {
+		sb.WriteString(h)
+		padding := widths[i] - len(StripANSI(h))
+		sb.WriteString(strings.Repeat(" ", padding))
+		if i < len(headers)-1 {
+			sb.WriteString(" | ")
+		}
+	}
+	sb.WriteString(" |\n")
+
+	// Separator row
+	sb.WriteString("+")
+	for i := 0; i < len(headers); i++ {
+		sb.WriteString(strings.Repeat("-", widths[i]+2))
+		sb.WriteString("+")
+	}
+	sb.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		sb.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			sb.WriteString(cell)
+			padding := widths[i] - len(StripANSI(cell))
+			sb.WriteString(strings.Repeat(" ", padding))
+			if i < len(headers)-1 {
+				sb.WriteString(" | ")
+			}
+		}
+		sb.WriteString(" |\n")
+	}
+
+	return sb.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name  | Age |\n+-------+-----+\n| Alice | 30  |\n| Bob   | 25  |\n",
+		},
+		{
+			name:    "headers only (empty rows)",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "| Col1 | Col2 |\n+------+------+\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},

+			want: "| Item   |\n+--------+\n| apple  |\n| banana |\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"\x1b[32mrunning\x1b[0m"},
+				{"\x1b[31mstopped\x1b[0m"},
+			},
+			want: "| Status  |\n+---------+\n| \x1b[32mrunning\x1b[0m |\n| \x1b[31mstopped\x1b[0m |\n",
+		},
+		{
+			name:    "mismatched column counts - fewer columns in row",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "| A | B | C |\n+---+---+---+\n| 1 | 2 |   |\n| x | y | z |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"a", "longer content here"},
+				{"much longer item", "b"},
+			},
+			want: "| Short            | VeryLongHeader      |\n+------------------+---------------------+\n| a                | longer content here |\n| much longer item | b                   |\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() output mismatch\ngot:\n%q\n\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
