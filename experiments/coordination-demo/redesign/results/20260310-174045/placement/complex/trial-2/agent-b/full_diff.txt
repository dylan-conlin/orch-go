+// FormatTable formats headers and rows as an aligned text table.
+// It auto-sizes each column based on the widest content in that column
+// (including header), and separates the header from data rows with a line of dashes.
+// Uses StripANSI for width calculation to handle ANSI-colored content correctly.
+// Returns an empty string if headers is empty.
+// Handles rows with fewer columns than headers by padding with empty cells.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on stripped content
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var width int
+			if i < len(row) {
+				width = len(StripANSI(row[i]))
+			}
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Helper to pad content considering ANSI codes
+	padContent := func(content string, width int) string {
+		stripped := StripANSI(content)
+		padding := width - len(stripped)
+		return content + strings.Repeat(" ", padding)
+	}
+
+	// Build separator line
+	var separatorParts []string
+	for _, width := range colWidths {
+		separatorParts = append(separatorParts, strings.Repeat("-", width))
+	}
+	separator := strings.Join(separatorParts, "-+-")
+
+	// Build header line
+	var headerParts []string
+	for i, header := range headers {
+		headerParts = append(headerParts, padContent(header, colWidths[i]))
+	}
+	headerLine := strings.Join(headerParts, " | ")
+
+	// Build data lines
+	var lines []string
+	for _, row := range rows {
+		var dataParts []string
+		for i := 0; i < len(headers); i++ {
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			dataParts = append(dataParts, padContent(content, colWidths[i]))
+		}
+		lines = append(lines, strings.Join(dataParts, " | "))
+	}
+
+	// Combine all parts
+	var result strings.Builder
+	result.WriteString(headerLine)
+	result.WriteString("\n")
+	result.WriteString(separator)
+	if len(lines) > 0 {
+		result.WriteString("\n")
+		result.WriteString(strings.Join(lines, "\n"))
+	}
+	return result.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 ",
+		},
+		{
+			name:    "empty table",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want:    "Name | Age\n-----+----",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Name"},
+			rows: [][]string{
+				{"Alice"},
+				{"Bob"},
+			},
+			want: "Name \n-----\nAlice\nBob  ",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"A", "This is a very long cell"},
+				{"B", "Short"},
+			},
+			want: "Short | VeryLongHeader          \n------+-------------------------\nA     | This is a very long cell\nB     | Short                   ",
+		},
+		{
+			name:    "mismatched columns",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},     // fewer columns
+				{"4", "5", "6"}, // matching columns
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \n4 | 5 | 6",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[32mAlice\x1b[0m", "\x1b[31mERROR\x1b[0m"},
+				{"Bob", "\x1b[32mOK\x1b[0m"},
+			},
+			want: "Name  | Status\n------+-------\n\x1b[32mAlice\x1b[0m | \x1b[31mERROR\x1b[0m \nBob   | \x1b[32mOK\x1b[0m    ",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "headers only no rows",
+			headers: []string{"Col1", "Col2", "Col3"},
+			rows:    nil,
+			want:    "Col1 | Col2 | Col3\n-----+------+-----",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable got:\n%q\n\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
+}
