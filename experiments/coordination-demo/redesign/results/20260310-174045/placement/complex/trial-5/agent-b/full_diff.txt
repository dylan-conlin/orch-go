+// FormatTable formats headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including header).
+// ANSI color codes are stripped when calculating column widths.
+// Rows with fewer columns are padded with empty strings; rows with more
+// columns are truncated to match the header count.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths (visual width, excluding ANSI codes)
+	widths := make([]int, len(headers))
+	for i, h := range headers {
+		widths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			visualWidth := len(StripANSI(content))
+			if visualWidth > widths[i] {
+				widths[i] = visualWidth
+			}
+		}
+	}
+
+	// Build table with pipe separators and 1-space padding on each side
+	var sb strings.Builder
+	padding := 1
+
+	// Header row
+	sb.WriteString("|")
+	for i, h := range headers {
+		sb.WriteString(strings.Repeat(" ", padding))
+		sb.WriteString(h)
+		visualWidth := len(StripANSI(h))
+		sb.WriteString(strings.Repeat(" ", widths[i]-visualWidth+padding))
+		sb.WriteString("|")
+	}
+	sb.WriteString("\n")
+
+	// Header separator
+	sb.WriteString("|")
+	for i := range headers {
+		sb.WriteString(strings.Repeat("-", widths[i]+2*padding))
+		sb.WriteString("|")
+	}
+	sb.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		sb.WriteString("|")
+		for i := 0; i < len(headers); i++ {
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			sb.WriteString(strings.Repeat(" ", padding))
+			sb.WriteString(content)
+			visualWidth := len(StripANSI(content))
+			sb.WriteString(strings.Repeat(" ", widths[i]-visualWidth+padding))
+			sb.WriteString("|")
+		}
+		sb.WriteString("\n")
+	}
+
+	return sb.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{

+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name  | Age |\n|-------|-----|\n| Alice | 30  |\n| Bob   | 25  |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"ID", "Status"},
+			rows:    [][]string{},
+			want:    "| ID | Status |\n|----|--------|\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Label"},
+			rows: [][]string{
+				{"Value1"},
+				{"Value2"},
+			},
+			want: "| Label  |\n|--------|\n| Value1 |\n| Value2 |\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name"},
+			rows: [][]string{

+				{"\x1b[32mGreen\x1b[0m"},

+			want: "| Name  |\n|-------|\n| \x1b[31mRed\x1b[0m   |\n| \x1b[32mGreen\x1b[0m |\n",
+		},
+		{
+			name:    "row with fewer columns",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "| A | B | C |\n|---|---|---|\n| 1 | 2 |   |\n| 3 | 4 | 5 |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"a", "this is a long cell"},
+				{"longer", "x"},
+			},
+			want: "| Short  | VeryLongHeader      |\n|--------|---------------------|\n| a      | this is a long cell |\n| longer | x                   |\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows: [][]string{
+				{"data"},
+			},
+			want: "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Col1"},
+			rows:    nil,
+			want:    "| Col1 |\n|------|\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch:\ngot:\n%q\nwant:\n%q", got, tt.want)
+				// Also print visual representation for debugging
+				t.Logf("got visual:\n%s", got)
+				t.Logf("want visual:\n%s", tt.want)
+			}
+		})
+	}
+}
