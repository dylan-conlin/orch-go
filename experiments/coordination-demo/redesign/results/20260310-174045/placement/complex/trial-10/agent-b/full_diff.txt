+// FormatTable renders headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (header or row).
+// ANSI-colored content widths are calculated correctly using StripANSI.
+// Handles edge cases: empty rows, mismatched column counts, nil rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths using visual width (ANSI-aware)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Update widths based on data rows
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(colWidths) {
+				w := len(StripANSI(cell))
+				if w > colWidths[i] {
+					colWidths[i] = w
+				}
+			}
+		}
+	}
+
+	// Build the table
+	var result strings.Builder
+
+	// Header row
+	result.WriteString("| ")
+	for i, h := range headers {
+		result.WriteString(h)
+		result.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(h))))
+		if i < len(headers)-1 {
+			result.WriteString(" | ")
+		} else {
+			result.WriteString(" |")
+		}
+	}
+	result.WriteString("\n")
+
+	// Separator line
+	result.WriteString("|")
+	for i, w := range colWidths {
+		result.WriteString(strings.Repeat("-", w+2))
+		if i < len(colWidths)-1 {
+			result.WriteString("|")
+		} else {
+			result.WriteString("|")
+		}
+	}
+	result.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		result.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			result.WriteString(cell)
+			result.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(cell))))
+			if i < len(headers)-1 {
+				result.WriteString(" | ")
+			} else {
+				result.WriteString(" |")
+			}
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "| Name  | Age |\n|-------|-----|\n| Alice | 30  |\n| Bob   | 25  |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "| Col1 | Col2 |\n|------|------|\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"ID"},
+			rows: [][]string{
+				{"1"},
+				{"2"},
+				{"3"},
+			},
+			want: "| ID |\n|----|\n| 1  |\n| 2  |\n| 3  |\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"\x1b[1mName\x1b[0m", "Status"},
+			rows: [][]string{
+				{"\x1b[32mRunning\x1b[0m", "OK"},
+				{"Stopped", "\x1b[31mError\x1b[0m"},
+			},
+			want: "| \x1b[1mName\x1b[0m    | Status |\n|---------|--------|\n| \x1b[32mRunning\x1b[0m | OK     |\n| Stopped | \x1b[31mError\x1b[0m  |\n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "| A | B | C |\n|---|---|---|\n| 1 | 2 |   |\n| x | y | z |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"x", "content here"},
+				{"verylongcontent", "y"},
+			},
+			want: "| Short           | VeryLongHeader |\n|-----------------|----------------|\n| x               | content here   |\n| verylongcontent | y              |\n",
+		},
+		{
+			name:    "empty string cells",
+			headers: []string{"Col1", "Col2"},
+			rows: [][]string{
+				{"", "value"},
+				{"key", ""},
+			},
+			want: "| Col1 | Col2  |\n|------|-------|\n|      | value |\n| key  |       |\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() output mismatch\ngot:\n%q\nwant:\n%q", got, tt.want)
+				// Also print visual output for debugging
+				t.Logf("Got visual output:\n%s", got)
+				t.Logf("Want visual output:\n%s", tt.want)
+			}
+		})
+	}
+
+	// Test edge cases
+	t.Run("nil rows", func(t *testing.T) {
+		got := FormatTable([]string{"A", "B"}, nil)
+		want := "| A | B |\n|---|---|\n"
+		if got != want {
+			t.Errorf("FormatTable with nil rows: got %q, want %q", got, want)
+		}
+	})
+
+	t.Run("empty headers", func(t *testing.T) {
+		got := FormatTable([]string{}, [][]string{{"a", "b"}})
+		want := ""
+		if got != want {
+			t.Errorf("FormatTable with empty headers: got %q, want %q", got, want)
+		}
+	})
+
+	t.Run("row with more columns than headers", func(t *testing.T) {
+		// Extra columns should be ignored
+		got := FormatTable([]string{"A"}, [][]string{{"1", "2", "3"}})
+		want := "| A |\n|---|\n| 1 |\n"
+		if got != want {
+			t.Errorf("FormatTable with extra columns: got %q, want %q", got, want)
+		}
+	})
+}
