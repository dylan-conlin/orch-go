+// FormatTable formats headers and rows as an aligned text table.
+// Headers and row content are auto-sized based on the widest content in each column.
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Each column is separated by " | " and a dash line separates headers from data rows.
+// Rows with fewer columns than headers are padded with empty cells.
+// Rows with more columns than headers have extra columns ignored.
+// Returns an empty string if headers is nil or empty.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on header and row content
+	colWidths := make([]int, len(headers))
+
+	// Start with header widths (using visual width, ignoring ANSI codes)
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	// Expand for row content
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Build table string
+	var sb strings.Builder
+
+	// Format headers
+	for i, h := range headers {
+		if i > 0 {
+			sb.WriteString(" | ")
+		}
+		if i < len(headers)-1 {
+			// Pad all columns except the last one
+			padded := padRight(h, colWidths[i])
+			sb.WriteString(padded)
+		} else {
+			// Last column: no padding
+			sb.WriteString(h)
+		}
+	}
+	sb.WriteString("\n")
+
+	// Format separator line
+	for i := 0; i < len(headers); i++ {
+		if i > 0 {
+			sb.WriteString("-+-")
+		}
+		sb.WriteString(strings.Repeat("-", colWidths[i]))
+	}
+	sb.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				sb.WriteString(" | ")
+			}
+			content := ""
+			if i < len(row) {
+				content = row[i]
+			}
+			if i < len(headers)-1 {
+				// Pad all columns except the last one
+				padded := padRight(content, colWidths[i])
+				sb.WriteString(padded)
+			} else {
+				// Last column: no padding
+				sb.WriteString(content)
+			}
+		}
+		sb.WriteString("\n")
+	}
+
+	// Remove trailing newline and trim trailing spaces from each line
+	result := strings.TrimSuffix(sb.String(), "\n")
+	lines := strings.Split(result, "\n")
+	for i := range lines {
+		lines[i] = strings.TrimRight(lines[i], " ")
+	}
+	return strings.Join(lines, "\n")
+}

+// padRight pads a string to reach targetWidth on the right side.
+// Uses StripANSI to calculate visual width, accounting for ANSI color codes.
+func padRight(s string, targetWidth int) string {
+	visualWidth := len(StripANSI(s))
+	if visualWidth >= targetWidth {
+		return s
+	}
+	padding := targetWidth - visualWidth
+	return s + strings.Repeat(" ", padding)
+}
