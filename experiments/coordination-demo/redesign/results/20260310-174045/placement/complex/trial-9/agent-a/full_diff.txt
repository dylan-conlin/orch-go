+// VisualWidth returns the visual display width of a string, ignoring ANSI escape codes.
+// It counts Unicode runes in the ANSI-stripped string.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward width. If the string is already
+// wider than or equal to width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	padCount := width - visualWidth
+	return s + strings.Repeat(" ", padCount)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"test string", 11},
+
+		// With ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[0m\x1b[1m\x1b[31mtest\x1b[0m", 4},
+
+		// Unicode
+		{"你好", 2},
+		{"😀", 1},
+		{"hello 世界", 8},
+		{"👋🌍", 2},
+
+		// Combined: ANSI codes with Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1;33m😀\x1b[0m", 1},
+		{"\x1b[32mhello 世\x1b[0m", 7},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII padding
+		{"hi", 5, "hi   "},
+		{"hello", 5, "hello"},
+		{"hello", 3, "hello"},
+		{"", 3, "   "},
+		{"a", 1, "a"},
+
+		// With ANSI codes
+		{"\x1b[31mhi\x1b[0m", 5, "\x1b[31mhi\x1b[0m   "},
+		{"\x1b[31mhello\x1b[0m", 5, "\x1b[31mhello\x1b[0m"},
+		{"\x1b[31mhello\x1b[0m", 3, "\x1b[31mhello\x1b[0m"},
+		{"\x1b[1;33mtest\x1b[0m", 6, "\x1b[1;33mtest\x1b[0m  "},
+
+		// Unicode padding
+		{"你", 3, "你  "},
+		{"😀", 3, "😀  "},
+		{"hello 世", 8, "hello 世 "},
+		{"你好", 4, "你好  "},
+
+		// Edge cases
+		{"", 0, ""},
+		{"x", 0, "x"},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+	}
+}
