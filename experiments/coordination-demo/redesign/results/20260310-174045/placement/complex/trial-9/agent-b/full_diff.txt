+// FormatTable renders headers and rows as an aligned text table.
+// Column widths are auto-sized based on the widest content (including header).
+// The header is separated from data rows with a line of dashes.
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Handles edge cases: empty rows (headers only), rows with fewer columns than headers.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on content
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(headers) {
+				width := len(StripANSI(cell))
+				if width > colWidths[i] {
+					colWidths[i] = width
+				}
+			}
+		}
+	}
+
+	var result strings.Builder
+
+	// Header row
+	result.WriteString(strings.TrimRight(formatTableRow(headers, colWidths), " "))
+	result.WriteString("\n")
+
+	// Separator line
+	var sep strings.Builder
+	for i, width := range colWidths {
+		if i > 0 {
+			sep.WriteString("-+-")
+		}
+		sep.WriteString(strings.Repeat("-", width+2)) // +2 for padding spaces
+	}
+	result.WriteString(sep.String())
+	result.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		// Pad row to match header length
+		paddedRow := make([]string, len(headers))
+		copy(paddedRow, row)
+		result.WriteString(strings.TrimRight(formatTableRow(paddedRow, colWidths), " "))
+		result.WriteString("\n")
+	}
+
+	return strings.TrimRight(result.String(), "\n")
+}
+
+// formatTableRow formats a single row with proper column alignment.
+func formatTableRow(cells []string, colWidths []int) string {
+	var result strings.Builder
+	for i, cell := range cells {
+		if i > 0 {
+			result.WriteString(" | ")
+		}
+		strippedLen := len(StripANSI(cell))
+		padding := colWidths[i] - strippedLen
+		result.WriteString(" " + cell + strings.Repeat(" ", padding))
+		if i < len(cells)-1 {
+			result.WriteString(" ")
+		}
+	}
+	return result.String()
+}

+----+-----+----
+ 1  |  2  |
+ 3  |     |`,
+		},
+		{
+			name:    "mismatched column counts - more columns in row",
+			headers: []string{"A", "B"},

+			want: ` A  |  B
+----+----
+ 1  |  2`,
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongColumnHeader"},

+			want: ` Short  |  LongColumnHeader
+--------+-------------------
+ a      |  content content`,
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{{"a", "b"}},
+			want:    "",

+--------+--------
+        |  value

+				t.Errorf("FormatTable() = %q, want %q", got, tt.want)
+			}
+		})
+	}
+}
