+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. Unicode characters (including emoji and CJK) are counted as
+// single runes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is
+// already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	paddingNeeded := width - currentWidth
+	return s + strings.Repeat(" ", paddingNeeded)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"test", 4},
+		// ANSI codes should be ignored
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		// Unicode (CJK, emoji)
+		{"你好", 2},
+		{"こんにちは", 5},
+		{"😀😁😂", 3},
+		// Mixed ASCII and Unicode
+		{"hello世界", 7},
+		// ANSI + Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		// Complex ANSI with multiple codes
+		{"\x1b[1;31;45mcomplex\x1b[0m", 7},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain strings - padding needed
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		// Empty string
+		{"", 3, "   "},
+		// Already at width
+		{"hello", 5, "hello"},
+		// Wider than target - unchanged
+		{"hello world", 5, "hello world"},
+		// ANSI codes - should pad based on visual width
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[1;32mbold green\x1b[0m", 15, "\x1b[1;32mbold green\x1b[0m     "},
+		// Unicode - pad correctly
+		{"你好", 4, "你好  "},
+		{"😀", 3, "😀  "},
+		// Mixed ASCII and Unicode
+		{"hello世", 8, "hello世  "},
+		// ANSI + Unicode
+		{"\x1b[33m你好\x1b[0m", 4, "\x1b[33m你好\x1b[0m  "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+	}
+}
