+// FormatBytes formats a byte count as a human-readable string.
+// Output style: "512 B", "1.5 KiB", "10.2 MiB", etc.
+// Uses binary units (1024-based): B, KiB, MiB, GiB, TiB.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	sign := ""
+	if bytes < 0 {
+		sign = "-"
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	value := float64(bytes)
+	unitIdx := 0
+
+	for unitIdx < len(units)-1 && value >= 1024 {
+		value /= 1024
+		unitIdx++
+	}
+
+	if unitIdx == 0 {
+		// Bytes: no decimal place
+		return fmt.Sprintf("%s%d %s", sign, int64(value), units[unitIdx])
+	}
+
+	// Other units: 1 decimal place
+	return fmt.Sprintf("%s%.1f %s", sign, value, units[unitIdx])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Boundary: 1024 bytes = 1 KiB
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		// Between KiB and MiB
+		{10240, "10.0 KiB"},
+		{102400, "100.0 KiB"},
+		{524288, "512.0 KiB"},
+		{1048575, "1024.0 KiB"},
+		// Boundary: 1048576 bytes = 1 MiB
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Between MiB and GiB
+		{10485760, "10.0 MiB"},
+		{104857600, "100.0 MiB"},
+		{536870912, "512.0 MiB"},
+		{1073741823, "1024.0 MiB"},
+		// Boundary: 1073741824 bytes = 1 GiB
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Between GiB and TiB
+		{10737418240, "10.0 GiB"},
+		{107374182400, "100.0 GiB"},
+		{549755813888, "512.0 GiB"},
+		{1099511627775, "1024.0 GiB"},
+		// Boundary: 1099511627776 bytes = 1 TiB
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		{-512, "-512 B"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
