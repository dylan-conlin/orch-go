+// FormatBytes formats a byte count as a human-readable string.
+// Output style: "512 B", "1.5 KiB", "2.3 MiB", "1.0 GiB".
+// Uses binary units (KiB, MiB, GiB, TiB) with 1 decimal place for non-byte units.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	b := bytes
+	if negative {
+		b = -b
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitIndex := 0
+
+	value := float64(b)
+	for unitIndex < len(units)-1 && value >= 1024 {
+		value /= 1024
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes, show as integer
+		result = fmt.Sprintf("%d B", int64(value))
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},

+		{1, "1 B"},
+		{100, "100 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Kibibyte boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		{1048575, "1024.0 KiB"},

+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{10485760, "10.0 MiB"},
+		// Gibibyte boundaries
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{10737418240, "10.0 GiB"},
+		// Tebibyte boundaries

+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
