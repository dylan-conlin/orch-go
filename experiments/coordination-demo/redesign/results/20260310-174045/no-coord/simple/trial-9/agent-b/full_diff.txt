+// FormatRate formats a transfer rate into a human-readable string using
+// binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	absRate := bytesPerSec
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		absRate = -bytesPerSec
+	}
+
+	var result string
+	if absRate >= GiB {
+		result = fmt.Sprintf("%.1f GiB/s", absRate/GiB)
+	} else if absRate >= MiB {
+		result = fmt.Sprintf("%.1f MiB/s", absRate/MiB)
+	} else if absRate >= KiB {
+		result = fmt.Sprintf("%.1f KiB/s", absRate/KiB)
+	} else {
+		result = fmt.Sprintf("%d B/s", int(absRate))
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// KiB/s boundary
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+
+		// MiB/s boundary
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+
+		// GiB/s boundary
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+
+		// Fractional values
+		{2560, "2.5 KiB/s"},
+		{3145728, "3.0 MiB/s"},
+		{5242880, "5.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
