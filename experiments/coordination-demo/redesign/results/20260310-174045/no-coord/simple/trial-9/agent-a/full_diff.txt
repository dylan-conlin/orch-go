+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Values are displayed with one decimal place
+// for non-byte units. Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	// Units in binary (1024-based)
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+
+	// Find appropriate unit
+	size := float64(bytes)
+	unitIndex := 0
+
+	for size >= 1024 && unitIndex < len(units)-1 {
+		size /= 1024
+		unitIndex++
+	}
+
+	// Format output
+	var result string
+	if unitIndex == 0 {
+		// Bytes - no decimal place
+		result = fmt.Sprintf("%d B", bytes)
+	} else {
+		// Other units - 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{5120, "5.0 KiB"},
+		{10240, "10.0 KiB"},
+		{1048575, "1024.0 KiB"},
+		// MiB boundary
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{5242880, "5.0 MiB"},
+		{104857600, "100.0 MiB"},
+		{1073741823, "1024.0 MiB"},
+		// GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{10737418240, "10.0 GiB"},
+		{1099511627775, "1024.0 GiB"},
+		// TiB boundary
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-512, "-512 B"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
