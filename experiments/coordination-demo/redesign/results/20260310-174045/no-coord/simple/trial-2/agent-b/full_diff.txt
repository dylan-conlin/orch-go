+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec values show 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	unitSize := 1024.0
+
+	value := bytesPerSec
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for unitIndex < len(units)-1 && value >= unitSize {
+		value /= unitSize
+		unitIndex++
+	}
+
+	// Format the result
+	var result string
+	if unitIndex == 0 {
+		// For bytes per second, show no decimal places
+		result = fmt.Sprintf("%d B/s", int(value))
+	} else {
+		// For larger units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s boundary
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 * 512, "512.0 KiB/s"},
+		// MiB/s boundary
+		{1048576, "1.0 MiB/s"},
+		{1048576 * 1.5, "1.5 MiB/s"},
+		{1048576 * 2, "2.0 MiB/s"},
+		{1048576 * 512, "512.0 MiB/s"},
+		// GiB/s boundary
+		{1073741824, "1.0 GiB/s"},
+		{1073741824 * 1.5, "1.5 GiB/s"},
+		{1073741824 * 2, "2.0 GiB/s"},
+		{1073741824 * 10, "10.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-512, "-512 B/s"},
+		// Fractional bytes per second (stays as B/s)
+		{0.5, "0 B/s"}, // Rounds down to 0
+		{100.5, "100 B/s"}, // Rounds down to 100
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
