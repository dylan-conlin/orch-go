+// FormatRate formats a transfer rate (bytes per second) as a human-readable
+// string using binary units. Output style: "512 B/s", "1.5 MiB/s", "2.0 GiB/s".
+// Shows 1 decimal place for units larger than bytes. Handles negative values
+// by prefixing with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	abs := bytesPerSec
+	if isNegative {
+		abs = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+
+	if abs >= GiB {
+		result = fmt.Sprintf("%.1f GiB/s", abs/GiB)
+	} else if abs >= MiB {
+		result = fmt.Sprintf("%.1f MiB/s", abs/MiB)
+	} else if abs >= KiB {
+		result = fmt.Sprintf("%.1f KiB/s", abs/KiB)
+	} else {
+		// For B/s, show as integer
+		result = fmt.Sprintf("%.0f B/s", abs)
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s - integers)
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s boundaries and fractional values
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+		{5120, "5.0 KiB/s"},
+		// MiB/s boundaries and fractional values
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		// GiB/s boundaries and fractional values
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{5368709120, "5.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		// Edge cases
+		{0.5, "0 B/s"},   // Below 1 B/s, rounds to 0
+		{512.5, "512 B/s"}, // Fractional B/s
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
