+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Examples: "512 B", "1.5 KiB", "1.0 MiB", "-1.0 KiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var result string
+	switch {
+	case bytes < KiB:
+		result = fmt.Sprintf("%d B", bytes)
+	case bytes < MiB:
+		result = fmt.Sprintf("%.1f KiB", float64(bytes)/KiB)
+	case bytes < GiB:
+		result = fmt.Sprintf("%.1f MiB", float64(bytes)/MiB)
+	case bytes < TiB:
+		result = fmt.Sprintf("%.1f GiB", float64(bytes)/GiB)
+	default:
+		result = fmt.Sprintf("%.1f TiB", float64(bytes)/TiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		{1046528, "1022.0 KiB"},
+		// Exact MiB boundaries
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{10485760, "10.0 MiB"},
+		{1071644672, "1022.0 MiB"},
+		// Exact GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{10737418240, "10.0 GiB"},
+		{1097364144128, "1022.0 GiB"},
+		// TiB range
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-512, "-512 B"},
+		{-1, "-1 B"},
+		// Values between boundaries
+		{2560, "2.5 KiB"},
+		{5242880, "5.0 MiB"},
+		{536870912, "512.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
