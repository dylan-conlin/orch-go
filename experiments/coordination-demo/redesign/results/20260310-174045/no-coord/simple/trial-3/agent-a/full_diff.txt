+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "512 B", "1.5 KiB", "256.0 MiB", "1.2 GiB", "2.0 TiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	size := float64(bytes)
+	unitIdx := 0
+
+	for size >= 1024 && unitIdx < len(units)-1 {
+		size /= 1024
+		unitIdx++
+	}
+
+	var result string
+	if unitIdx == 0 {
+		result = fmt.Sprintf("%d B", int64(size))
+	} else {
+		result = fmt.Sprintf("%.1f %s", size, units[unitIdx])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+
+		// Bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{5120, "5.0 KiB"},
+
+		// MiB boundaries
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{10485760, "10.0 MiB"},
+		{104857600, "100.0 MiB"},
+
+		// GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{10737418240, "10.0 GiB"},
+
+		// TiB boundaries
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{10995116277760, "10.0 TiB"},
+
+		// Large values
+		{109951162777600, "100.0 TiB"},
+		{1099511627776000, "1000.0 TiB"},
+
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
