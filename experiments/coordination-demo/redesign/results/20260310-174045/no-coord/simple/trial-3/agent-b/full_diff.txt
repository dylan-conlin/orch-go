+// FormatRate formats a transfer rate into a human-readable string using
+// binary units. Output style: "0 B/s", "512 B/s", "1.5 KiB/s", "1.0 MiB/s".
+// Non-byte-per-sec units display 1 decimal place. Negative rates are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	if bytesPerSec < KiB {
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else if bytesPerSec < MiB {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	} else if bytesPerSec < GiB {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+
+		// Fractional values
+		{1536, "1.5 KiB/s"},
+		{1024 * 1.5, "1.5 KiB/s"},
+		{1048576 * 2.5, "2.5 MiB/s"},
+		{1073741824 * 3.7, "3.7 GiB/s"},
+
+		// Larger values
+		{2048, "2.0 KiB/s"},
+		{5120, "5.0 KiB/s"},
+		{1048576 * 10, "10.0 MiB/s"},
+		{1073741824 * 100, "100.0 GiB/s"},
+
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+
+		// Additional tests for fractional B/s values
+		{256, "256 B/s"},
+		{768, "768 B/s"},
+		{512.5, "512 B/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
