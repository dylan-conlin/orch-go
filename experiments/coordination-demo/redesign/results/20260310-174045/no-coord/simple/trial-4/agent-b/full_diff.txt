+// FormatRate formats transfer rates into human-readable strings using binary
+// units with "/s" suffix. Returns "0 B/s" for zero, handles negative values
+// by prefixing with "-", and shows 1 decimal place for non-byte-per-sec units.
+// Examples: "512 B/s", "1.5 MiB/s", "2.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	// Handle zero and negative
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	rate := bytesPerSec
+	if negative {
+		rate = -rate
+	}
+
+	// Binary units: 1024 bytes = 1 KiB, 1048576 = 1 MiB, etc.
+	const (
+		kib = 1024
+		mib = 1024 * kib
+		gib = 1024 * mib
+	)
+
+	var result string
+	if rate >= gib {
+		result = fmt.Sprintf("%.1f GiB/s", rate/gib)
+	} else if rate >= mib {
+		result = fmt.Sprintf("%.1f MiB/s", rate/mib)
+	} else if rate >= kib {
+		result = fmt.Sprintf("%.1f KiB/s", rate/kib)
+	} else {
+		result = fmt.Sprintf("%.0f B/s", rate)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		bytesPerSec float64
+		want        string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Byte-level values (no decimal)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB values
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		// Just below MiB boundary
+		{1048575, "1024.0 KiB/s"},
+		// Exact MiB boundary
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB values
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		// Just below GiB boundary
+		{1073741823, "1024.0 MiB/s"},
+		// Exact GiB boundary
+		{1073741824, "1.0 GiB/s"},
+		// Fractional GiB values
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Large GiB values
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.bytesPerSec)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.bytesPerSec, got, tt.want)
+		}
+	}
+}
