+// FormatRate formats a transfer rate in bytes per second into a human-readable string.
+// Output uses binary units with "/s" suffix: "B/s", "KiB/s", "MiB/s", "GiB/s".
+// Non-byte units show 1 decimal place (e.g., "1.5 MiB/s").
+// Zero returns "0 B/s". Negative values are prefixed with "-".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		GiB = 1073741824 // 1024^3
+		MiB = 1048576    // 1024^2
+		KiB = 1024
+	)
+
+	var result string
+	switch {
+	case bytesPerSec >= GiB:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	case bytesPerSec >= MiB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	case bytesPerSec >= KiB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	default:
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values (B/s)
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// Exact and fractional KiB boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+		{512 * 1024, "512.0 KiB/s"},
+
+		// Exact and fractional MiB boundaries
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},                              // 1.5 MiB
+		{2 * 1048576, "2.0 MiB/s"},                          // 2 MiB
+		{1048576 + 512*1024, "1.5 MiB/s"},                   // 1.5 MiB
+		{512 * 1048576, "512.0 MiB/s"},                      // 512 MiB
+
+		// GiB boundaries
+		{1073741824, "1.0 GiB/s"},                           // 1 GiB
+		{1073741824 + 536870912, "1.5 GiB/s"},               // 1.5 GiB
+		{2 * 1073741824, "2.0 GiB/s"},                       // 2 GiB
+		{10 * 1073741824, "10.0 GiB/s"},                     // 10 GiB
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
