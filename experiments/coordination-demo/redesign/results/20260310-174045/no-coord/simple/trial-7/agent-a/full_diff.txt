+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units are displayed with 1 decimal
+// place. Negative values are prefixed with "-".
+// Examples: "512 B", "1.5 MiB", "-1.0 KiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	const unitSize = 1024.0
+
+	size := float64(bytes)
+	unitIndex := 0
+
+	for size >= unitSize && unitIndex < len(units)-1 {
+		size /= unitSize
+		unitIndex++
+	}
+
+	prefix := ""
+	if negative {
+		prefix = "-"
+	}
+
+	if unitIndex == 0 {
+		// Bytes: show as integer
+		return fmt.Sprintf("%s%d B", prefix, int64(size))
+	}
+
+	// Larger units: show 1 decimal place
+	return fmt.Sprintf("%s%.1f %s", prefix, size, units[unitIndex])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+
+		// Small bytes
+		{1, "1 B"},
+		{100, "100 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+
+		// Between boundaries (fractional values)
+		{1536, "1.5 KiB"},
+		{1792, "1.8 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{1887436, "1.8 MiB"},
+		{2097152, "2.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{1932735283, "1.8 GiB"},
+
+		// Large values
+		{5368709120, "5.0 GiB"},
+		{10995116277760, "10.0 TiB"},
+
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
