+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. Unicode characters (including emoji and CJK) are counted
+// correctly as single runes.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads s with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward the width. If the string
+// is already at or wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	paddingNeeded := width - currentWidth
+	return s + strings.Repeat(" ", paddingNeeded)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"", 0},
+		{"a", 1},
+		{"hello", 5},
+		{"hello world", 11},
+
+		// ANSI color codes (should be stripped)
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[36mcyan text\x1b[0m", 9},
+
+		// Unicode: emoji and CJK characters
+		{"😀", 1},
+		{"你好", 2},
+		{"Hello😀World", 11},
+		{"你好world", 7},
+
+		// Mix: ANSI + Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[32m😀😀\x1b[0m", 2},
+		{"\x1b[1;33mhello世界\x1b[0m", 7},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input   string
+		width   int
+		want    string
+		checkFn func(string, int) bool // Optional: check the result
+	}{
+		// Plain ASCII - needs padding
+		{"", 5, "     ", nil},
+		{"a", 5, "a    ", nil},
+		{"hi", 5, "hi   ", nil},
+		{"hello", 5, "hello", nil},
+
+		// Plain ASCII - already at or wider than width
+		{"hello", 3, "hello", nil},
+		{"hello world", 5, "hello world", nil},
+
+		// ANSI codes - padding should not count ANSI codes
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  ", nil},
+		{"\x1b[1;32mbold\x1b[0m", 6, "\x1b[1;32mbold\x1b[0m  ", nil},
+
+		// Unicode - padding should work correctly
+		{"你", 5, "你    ", nil},
+		{"你好", 5, "你好   ", nil},
+		{"😀", 3, "😀  ", nil},
+
+		// Mix: ANSI + Unicode
+		{"\x1b[31m你好\x1b[0m", 5, "\x1b[31m你好\x1b[0m   ", nil},
+		{"\x1b[32m😀\x1b[0m", 4, "\x1b[32m😀\x1b[0m   ", nil},
+
+		// Already at exact width
+		{"test", 4, "test", nil},
+		{"\x1b[33mhi\x1b[0m", 2, "\x1b[33mhi\x1b[0m", nil},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width is correct after padding
+		gotWidth := VisualWidth(got)
+		inputWidth := VisualWidth(tt.input)
+		expectedWidth := tt.width
+		if inputWidth < tt.width {
+			// If input was smaller, output should be exactly at width
+			if gotWidth != expectedWidth {
+				t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotWidth, expectedWidth)
+			}
+		} else {
+			// If input was already at or larger, width should stay same
+			if gotWidth != inputWidth {
+				t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotWidth, inputWidth)
+			}
+		}
+	}
+}
