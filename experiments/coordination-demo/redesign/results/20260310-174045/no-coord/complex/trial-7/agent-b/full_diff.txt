+// FormatTable formats headers and rows as an aligned text table. Each column is
+// auto-sized based on the widest content in that column (including headers).
+// ANSI-colored content is handled correctly by using StripANSI for width
+// calculation. The header is separated from data rows with a line of dashes.
+// Edge cases are handled: empty rows (headers only), rows with fewer columns
+// than headers (padded with empty strings), and nil rows (treated as empty).
+func FormatTable(headers []string, rows [][]string) string {
+	if headers == nil || len(headers) == 0 {
+		return ""
+	}
+
+	// Ensure rows is not nil
+	if rows == nil {
+		rows = [][]string{}
+	}
+
+	// Calculate max width for each column (based on headers and data)
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			width := len(StripANSI(cell))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var table strings.Builder
+
+	// Write header row: " content | content | content "
+	for i, header := range headers {
+		if i > 0 {
+			table.WriteString("|")
+		}
+		table.WriteString(" ")
+		table.WriteString(header)
+		stripWidth := len(StripANSI(header))
+		table.WriteString(strings.Repeat(" ", colWidths[i]-stripWidth))
+		table.WriteString(" ")
+	}
+	table.WriteString("\n")
+
+	// Write separator line: "-------+-------+-------"
+	for i, width := range colWidths {
+		if i > 0 {
+			table.WriteString("+")
+		}
+		table.WriteString(strings.Repeat("-", width+2))
+	}
+	table.WriteString("\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := 0; i < len(colWidths); i++ {
+			if i > 0 {
+				table.WriteString("|")
+			}
+			table.WriteString(" ")
+
+			// Get cell content or empty string
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+
+			table.WriteString(cell)
+			stripWidth := len(StripANSI(cell))
+			table.WriteString(strings.Repeat(" ", colWidths[i]-stripWidth))
+			table.WriteString(" ")
+		}
+		table.WriteString("\n")
+	}
+
+	return table.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table with headers and rows",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: " Name  | Age | City \n-------+-----+------\n Alice | 30  | NYC  \n Bob   | 25  | LA   \n",
+		},
+		{
+			name:    "empty table (headers only, no rows)",
+			headers: []string{"Header1", "Header2"},
+			rows:    [][]string{},
+			want:    " Header1 | Header2 \n---------+---------\n",
+		},
+		{
+			name:    "nil rows treated as empty",
+			headers: []string{"Col"},
+			rows:    nil,
+			want:    " Col \n-----\n",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Name"},
+			rows: [][]string{
+				{"Alice"},
+				{"Bob"},
+			},
+			want: " Name  \n-------\n Alice \n Bob   \n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"val1", "val2"},
+				{"val3", "val4", "val5"},
+			},
+			want: " A    | B    | C    \n------+------+------\n val1 | val2 |      \n val3 | val4 | val5 \n",
+		},
+		{
+			name:    "wide content requires wider columns",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"A", "B"},
+				{"LongContent", "X"},
+			},
+			want: " Short       | VeryLongHeader \n-------------+----------------\n A           | B              \n LongContent | X              \n",
+		},
+		{
+			name:    "ANSI-colored content alignment",
+			headers: []string{"Color", "Text"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m", "Plain"},
+				{"Plain", "\x1b[32mGreen\x1b[0m"},
+			},
+			want: " Color | Text  \n-------+-------\n \x1b[31mRed\x1b[0m   | Plain \n Plain | \x1b[32mGreen\x1b[0m \n",
+		},
+		{
+			name:    "row with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+				{"x", "y"},
+			},
+			want: " A | B \n---+---\n 1 | 2 \n x | y \n",
+		},
+		{
+			name:    "empty strings in cells",
+			headers: []string{"Col1", "Col2"},
+			rows: [][]string{
+				{"", "data"},
+				{"data", ""},
+			},
+			want: " Col1 | Col2 \n------+------\n      | data \n data |      \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
