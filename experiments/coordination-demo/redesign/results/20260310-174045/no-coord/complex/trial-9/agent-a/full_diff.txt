+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. Unicode characters are counted as individual runes, making
+// this suitable for calculating display width across different character sets.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved in the output but do not count toward the
+// visual width. If the string is already at or wider than width, it is
+// returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m", 6},
+
+		// Unicode - CJK characters (each counts as 1 rune)
+		{"你好", 2},
+		{"こんにちは", 5},
+		{"안녕", 2},
+
+		// Emoji
+		{"🎉", 1},
+		{"🎉🎊", 2},
+		{"hello🎉world", 11},
+
+		// Mixed cases
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1;32m🎉test\x1b[0m", 5},
+		{"plain text with no codes", 24},
+
+		// Multiple ANSI codes in one string
+		{"\x1b[1m\x1b[31mbold red\x1b[0m", 8},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		wantLen int // The visual width of the result
+	}{
+		// Plain ASCII - pad to larger width
+		{"hello", 10, "hello     ", 10},
+		{"hi", 5, "hi   ", 5},
+		{"", 5, "     ", 5},
+
+		// Already at target width - no padding
+		{"hello", 5, "hello", 5},
+		{"abc", 3, "abc", 3},
+
+		// Wider than target - return unchanged
+		{"hello world", 5, "hello world", 11},
+		{"toolong", 3, "toolong", 7},
+
+		// With ANSI codes - codes don't count toward width
+		{"\x1b[31mhi\x1b[0m", 5, "\x1b[31mhi\x1b[0m   ", 5},
+		{"\x1b[1;32mtest\x1b[0m", 6, "\x1b[1;32mtest\x1b[0m  ", 6},
+
+		// Unicode characters
+		{"你好", 4, "你好  ", 4},
+		{"🎉", 3, "🎉  ", 3},
+		{"🎉test", 7, "🎉test  ", 7},
+
+		// Mixed Unicode and ANSI
+		{"\x1b[31m你好\x1b[0m", 4, "\x1b[31m你好\x1b[0m  ", 4},
+		{"\x1b[33m🎉\x1b[0m", 5, "\x1b[33m🎉\x1b[0m    ", 5},
+
+		// Edge case: width of 0
+		{"hello", 0, "hello", 5},
+		{"", 0, "", 0},
+
+		// Width 1
+		{"a", 1, "a", 1},
+		{"a", 2, "a ", 2},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width of result matches expected
+		gotWidth := VisualWidth(got)
+		if gotWidth != tt.wantLen {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, gotWidth, tt.wantLen)
+		}
+	}
+}
