+// FormatTable formats headers and rows as an aligned text table with auto-sized columns.
+// Each column width is calculated based on the widest content in that column (including headers).
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Rows with fewer columns than headers are padded with empty strings.
+// Extra columns in rows beyond the header count are ignored.
+// If rows is empty, only headers and a separator line are displayed.
+// If headers is empty, an empty string is returned.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and all rows
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			width := len(StripANSI(cell))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Format header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(h)
+		buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(h))))
+	}
+	buf.WriteString("\n")
+
+	// Format separator line (dashes with + at junctions)
+	for i, width := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", width))
+	}
+	buf.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(cell)
+			buf.WriteString(strings.Repeat(" ", colWidths[i]-len(StripANSI(cell))))
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 \n",
+		},
+		{
+			name:    "empty table (headers only)",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "Col1 | Col2\n-----+-----\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Single"},
+			rows: [][]string{
+				{"value"},
+				{"another"},
+			},
+			want: "Single \n-------\nvalue  \nanother\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Header"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m"},
+				{"Plain"},
+			},
+			want: "Header\n------\n\x1b[31mRed\x1b[0m   \nPlain \n",
+		},
+		{
+			name:    "rows with fewer columns",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \nx |   |  \n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongerHeader"},
+			rows: [][]string{
+				{"x", "y"},
+				{"verylongvalue", "z"},
+			},
+			want: "Short         | LongerHeader\n--------------+-------------\nx             | y           \nverylongvalue | z           \n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "headers only with single column",
+			headers: []string{"OnlyHeader"},
+			rows:    [][]string{},
+			want:    "OnlyHeader\n----------\n",
+		},
+		{
+			name:    "multiple ANSI colors",
+			headers: []string{"Status", "Message"},
+			rows: [][]string{
+				{"\x1b[32mOK\x1b[0m", "\x1b[1mBold\x1b[0m"},
+				{"FAIL", "Normal"},
+			},
+			want: "Status | Message\n-------+--------\n\x1b[32mOK\x1b[0m     | \x1b[1mBold\x1b[0m   \nFAIL   | Normal \n",
+		},
+		{
+			name:    "rows with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "extra"},
+				{"x", "y", "ignored"},
+			},
+			want: "A | B\n--+--\n1 | 2\nx | y\n",
+		},
+		{
+			name:    "empty strings in data",
+			headers: []string{"Col1", "Col2"},
+			rows: [][]string{
+				{"value", ""},
+				{"", "text"},
+			},
+			want: "Col1  | Col2\n------+-----\nvalue |     \n      | text\n",
+		},
+		{
+			name:    "single cell table",
+			headers: []string{"Header"},
+			rows: [][]string{
+				{"Cell"},
+			},
+			want: "Header\n------\nCell  \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
