+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It counts runes correctly to handle Unicode characters.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is
+// already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-currentWidth)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// Strings with ANSI codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m text", 11},
+
+		// Unicode: CJK characters (2 runes each in typical representation)
+		{"你好", 2},
+		{"こんにちは", 5},
+		{"안녕하세요", 5},
+
+		// Emoji (typically 1 rune)
+		{"😀", 1},
+		{"👍", 1},
+		{"hello😀world", 11},
+
+		// Mixed: ANSI codes with Unicode
+		{"\x1b[36m你好\x1b[0m", 2},
+		{"\x1b[31m😀\x1b[0m", 1},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input       string
+		width       int
+		want        string
+		wantVWidth  int
+	}{
+		// Plain ASCII: shorter than target
+		{"hello", 10, "hello     ", 10},
+		{"hi", 5, "hi   ", 5},
+		{"", 3, "   ", 3},
+
+		// Plain ASCII: already at target width
+		{"hello", 5, "hello", 5},
+		{"abc", 3, "abc", 3},
+
+		// Plain ASCII: wider than target (unchanged)
+		{"hello world", 5, "hello world", 11},
+		{"toolong", 3, "toolong", 7},
+
+		// With ANSI codes (codes don't count toward width)
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  ", 5},
+		{"\x1b[1;32mbold\x1b[0m", 6, "\x1b[1;32mbold\x1b[0m  ", 6},
+
+		// Unicode strings: shorter than target
+		{"你好", 4, "你好  ", 4},
+		{"😀", 3, "😀  ", 3},
+
+		// Unicode strings: already at target width
+		{"你好", 2, "你好", 2},
+		{"😀", 1, "😀", 1},
+
+		// Unicode strings: wider than target (unchanged)
+		{"你好世界", 2, "你好世界", 4},
+
+		// Mixed: Unicode with ANSI codes
+		{"\x1b[36m你好\x1b[0m", 4, "\x1b[36m你好\x1b[0m  ", 4},
+		{"\x1b[31m😀\x1b[0m", 2, "\x1b[31m😀\x1b[0m ", 2},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width matches expected
+		gotVWidth := VisualWidth(got)
+		if gotVWidth != tt.wantVWidth {
+			t.Errorf("PadToWidth(%q, %d) resulted in visual width %d, want %d", tt.input, tt.width, gotVWidth, tt.wantVWidth)
+		}
+	}
+}
