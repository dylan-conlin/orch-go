+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It correctly handles Unicode characters by counting runes
+// rather than bytes.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward the width calculation.
+// If the string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualLen := VisualWidth(s)
+	if visualLen >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-visualLen)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"", 0},
+		{"hello", 5},
+		{"a", 1},
+		{"hello world", 11},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[36mcyan text\x1b[0m", 9},
+
+		// Unicode strings (CJK characters count as 1 rune)
+		{"你好", 2},
+		{"こんにちは", 5},
+		{"안녕하세요", 5},
+
+		// Emoji
+		{"😀", 1},
+		{"hello😀world", 11},
+
+		// Mixed: Unicode with ANSI codes
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1;32m😀\x1b[0m", 1},
+
+		// Multiple ANSI codes
+		{"\x1b[31mred\x1b[0m \x1b[32mgreen\x1b[0m", 9},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Empty string, pad to various widths
+		{"", 0, ""},
+		{"", 5, "     "},
+		{"", 1, " "},
+
+		// Plain ASCII strings
+		{"hello", 10, "hello     "},
+		{"hello", 5, "hello"},
+		{"hello", 3, "hello"},
+		{"hi", 10, "hi        "},
+
+		// Strings with ANSI codes (codes preserved, spaces added after)
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  "},
+		{"\x1b[31mred\x1b[0m", 3, "\x1b[31mred\x1b[0m"},
+		{"\x1b[31mred\x1b[0m", 2, "\x1b[31mred\x1b[0m"},
+
+		// Unicode strings
+		{"你好", 5, "你好   "},
+		{"你好", 2, "你好"},
+		{"你好", 1, "你好"},
+
+		// Emoji
+		{"😀", 5, "😀    "},
+		{"😀", 1, "😀"},
+
+		// Mixed: Unicode with ANSI codes
+		{"\x1b[31m你好\x1b[0m", 5, "\x1b[31m你好\x1b[0m   "},
+		{"\x1b[31m你好\x1b[0m", 2, "\x1b[31m你好\x1b[0m"},
+
+		// Already at exact width
+		{"hello", 5, "hello"},
+		{"test", 4, "test"},
+
+		// Width of 0 or negative (string wider than target)
+		{"hello", 0, "hello"},
+		{"hello", -1, "hello"},
+
+		// Long string not padded
+		{"hello world", 5, "hello world"},
+
+		// Multiple ANSI codes
+		{"\x1b[31mred\x1b[0m \x1b[32mgreen\x1b[0m", 15, "\x1b[31mred\x1b[0m \x1b[32mgreen\x1b[0m      "},
+		{"\x1b[31mred\x1b[0m \x1b[32mgreen\x1b[0m", 10, "\x1b[31mred\x1b[0m \x1b[32mgreen\x1b[0m "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify visual width
+		gotWidth := VisualWidth(got)
+		expectedWidth := tt.width
+		if gotWidth < expectedWidth && tt.width > 0 {
+			t.Errorf("PadToWidth(%q, %d) resulted in visual width %d, want %d", tt.input, tt.width, gotWidth, expectedWidth)
+		}
+	}
+}
