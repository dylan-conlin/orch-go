+// FormatTable renders headers and rows as an aligned text table.
+// Column widths are auto-sized based on the widest content in each column.
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Returns a formatted table with a header separator line.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and all rows
+	colWidths := make([]int, len(headers))
+	for i, header := range headers {
+		colWidths[i] = len(StripANSI(header))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var sb strings.Builder
+
+	// Write header row
+	for i, header := range headers {
+		if i > 0 {
+			sb.WriteString(" | ")
+		}
+		sb.WriteString(header)
+		// Pad to column width (accounting for ANSI codes)
+		visualWidth := len(StripANSI(header))
+		if visualWidth < colWidths[i] {
+			sb.WriteString(strings.Repeat(" ", colWidths[i]-visualWidth))
+		}
+	}
+	sb.WriteString("\n")
+
+	// Write separator line
+	for i, width := range colWidths {
+		if i > 0 {
+			sb.WriteString("-+-")
+		}
+		sb.WriteString(strings.Repeat("-", width))
+	}
+	sb.WriteString("\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				sb.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			sb.WriteString(cell)
+			// Pad to column width (accounting for ANSI codes)
+			visualWidth := len(StripANSI(cell))
+			if visualWidth < colWidths[i] {
+				sb.WriteString(strings.Repeat(" ", colWidths[i]-visualWidth))
+			}
+		}
+		sb.WriteString("\n")
+	}
+
+	return sb.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 \n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Col1", "Col2"},
+			rows:    [][]string{},
+			want:    "Col1 | Col2\n-----+-----\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Value"},
+			rows: [][]string{
+				{"test"},
+				{"hello"},
+			},
+			want: "Value\n-----\ntest \nhello\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"a", "b"},
+				{"verylongvalue", "x"},
+			},
+			want: "Short         | VeryLongHeader\n--------------+---------------\na             | b             \nverylongvalue | x             \n",
+		},
+		{
+			name:    "ansi colored content",
+			headers: []string{"\x1b[1;32mName\x1b[0m", "Age"},
+			rows: [][]string{
+				{"\x1b[31mAlice\x1b[0m", "30"},
+			},
+			want: "\x1b[1;32mName\x1b[0m  | Age\n------+----\n\x1b[31mAlice\x1b[0m | 30 \n",
+		},
+		{
+			name:    "row with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \nx | y | z\n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"H1", "H2"},
+			rows:    nil,
+			want:    "H1 | H2\n---+---\n",
+		},
+		{
+			name:    "mixed content lengths",
+			headers: []string{"ID", "Name", "Status"},
+			rows: [][]string{
+				{"1", "Alice", "active"},
+				{"2", "B", "pending"},
+				{"999", "Christopher", "inactive"},
+			},
+			want: "ID  | Name        | Status  \n----+-------------+---------\n1   | Alice       | active  \n2   | B           | pending \n999 | Christopher | inactive\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v)\ngot:\n%q\n\nwant:\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
