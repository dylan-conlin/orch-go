+// VisualWidth returns the visual display width of a string, counting runes
+// and ignoring ANSI escape codes. This is useful for formatting aligned output
+// when strings may contain color codes or other styling.
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the width. If the
+// string is already at or wider than the target width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// ANSI color codes (should be ignored)
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[91merror\x1b[0m", 5},
+
+		// Unicode: CJK characters (usually 2 columns wide, but counted as 1 rune)
+		{"你好", 2},
+		{"你好世界", 4},
+
+		// Unicode: combining characters
+		{"e\u0301", 2}, // e + combining acute accent
+
+		// Emoji (counted as 1 rune each)
+		{"😀", 1},
+		{"😀😀😀", 3},
+
+		// Mixed: ANSI + Unicode
+		{"\x1b[35m你好\x1b[0m", 2},
+		{"\x1b[33m😀\x1b[0m", 1},
+
+		// ANSI + ASCII + Unicode
+		{"\x1b[1m Hello \x1b[0m 世界", 10}, // " Hello  世界" = 10 runes (1+5+2+2)
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+		wantLen int // length of returned string (with padding, including ANSI codes)
+	}{
+		// Plain ASCII
+		{"hello", 10, "hello     ", 10},
+		{"hi", 5, "hi   ", 5},
+		{"", 3, "   ", 3},
+
+		// Already at width
+		{"hello", 5, "hello", 5},
+
+		// Already wider than target (should not pad)
+		{"hello world", 5, "hello world", 11},
+
+		// With ANSI codes (should preserve codes, pad based on visual width)
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  ", 14}, // "red" + 2 spaces + codes
+		{"\x1b[1;32mok\x1b[0m", 4, "\x1b[1;32mok\x1b[0m  ", 15}, // "ok" + 2 spaces + codes
+
+		// With Unicode
+		{"你好", 4, "你好  ", 8}, // 2 runes + 2 spaces
+		{"😀", 3, "😀  ", 6}, // 1 rune + 2 spaces
+
+		// ANSI + Unicode
+		{"\x1b[35m你\x1b[0m", 3, "\x1b[35m你\x1b[0m  ", 14}, // "你" (1 rune) + 2 spaces + codes
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		if len(got) != tt.wantLen {
+			t.Errorf("PadToWidth(%q, %d) length = %d, want %d", tt.input, tt.width, len(got), tt.wantLen)
+		}
+		// Verify visual width is at least the target (unless input already wider)
+		visualWidth := VisualWidth(got)
+		if visualWidth < tt.width && VisualWidth(tt.input) < tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want >= %d", tt.input, tt.width, visualWidth, tt.width)
+		}
+	}
+}
