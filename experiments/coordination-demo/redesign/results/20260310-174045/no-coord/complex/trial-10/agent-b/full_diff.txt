+// FormatTable renders headers and rows as an aligned text table with
+// auto-sized columns. Column widths are calculated using StripANSI to handle
+// ANSI-colored content correctly. Headers are separated from data rows by a
+// line of dashes. Rows with fewer columns than headers are padded with empty
+// strings. If rows have more columns than headers, extra columns are ignored.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Handle headers-only case
+	if len(rows) == 0 {
+		return formatTableRow(headers, nil)
+	}
+
+	// Calculate column widths (based on visual width without ANSI codes)
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var content string
+			if i < len(row) {
+				content = row[i]
+			}
+			width := len(StripANSI(content))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var sb strings.Builder
+
+	// Headers
+	sb.WriteString(formatTableRow(headers, colWidths))
+	sb.WriteString("\n")
+
+	// Separator
+	for i, w := range colWidths {
+		sb.WriteString(strings.Repeat("-", w))
+		if i < len(colWidths)-1 {
+			sb.WriteString("|")
+		}
+	}
+	sb.WriteString("\n")
+
+	// Data rows
+	for _, row := range rows {
+		paddedRow := make([]string, len(headers))
+		for i := 0; i < len(headers); i++ {
+			if i < len(row) {
+				paddedRow[i] = row[i]
+			}
+		}
+		sb.WriteString(formatTableRow(paddedRow, colWidths))
+		sb.WriteString("\n")
+	}
+
+	return strings.TrimSuffix(sb.String(), "\n")
+}
+
+// formatTableRow formats a single row with optional column widths for padding.
+// If widths is provided, columns are right-padded to the specified width
+// (except the last column).
+func formatTableRow(cells []string, widths []int) string {
+	var sb strings.Builder
+	for i, cell := range cells {
+		sb.WriteString(cell)
+
+		// Only pad non-last columns
+		if widths != nil && len(widths) > i && i < len(cells)-1 {
+			visualWidth := len(StripANSI(cell))
+			padding := widths[i] - visualWidth
+			if padding > 0 {
+				sb.WriteString(strings.Repeat(" ", padding))
+			}
+		}
+
+		if i < len(cells)-1 {
+			sb.WriteString(" | ")
+		}
+	}
+	return sb.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"ID", "Name", "Status"},
+			rows: [][]string{
+				{"1", "task1", "ready"},
+				{"2", "task2", "done"},
+			},
+			want: "ID | Name  | Status\n--|-----|------\n1  | task1 | ready\n2  | task2 | done",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"ID", "Name"},
+			rows:    [][]string{},
+			want:    "ID | Name",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"ready"},
+				{"done"},
+			},
+			want: "Status\n------\nready\ndone",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongHeaderName"},
+			rows: [][]string{
+				{"a", "b"},
+				{"longer", "verylongcontenthere"},
+			},
+			want: "Short  | LongHeaderName\n------|-------------------\na      | b\nlonger | verylongcontenthere",
+		},
+		{
+			name:    "rows with fewer columns than headers",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+				{"6"},
+			},
+			want: "A | B | C\n-|-|-\n1 | 2 | \n3 | 4 | 5\n6 |   | ",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"ID", "Status"},
+			rows: [][]string{
+				{"123", "\x1b[32mgreen\x1b[0m"},
+				{"456", "\x1b[31mred\x1b[0m"},
+			},
+			want: "ID  | Status\n---|------\n123 | \x1b[32mgreen\x1b[0m\n456 | \x1b[31mred\x1b[0m",
+		},
+		{
+			name:    "ANSI colored headers",
+			headers: []string{"\x1b[1mID\x1b[0m", "\x1b[1mName\x1b[0m"},
+			rows: [][]string{
+				{"1", "task"},
+			},
+			want: "\x1b[1mID\x1b[0m | \x1b[1mName\x1b[0m\n--|----\n1  | task",
+		},
+		{
+			name:    "nil rows slice",
+			headers: []string{"A", "B"},
+			rows:    nil,
+			want:    "A | B",
+		},
+		{
+			name:    "empty row slice",
+			headers: []string{"X", "Y", "Z"},
+			rows:    [][]string{},
+			want:    "X | Y | Z",
+		},
+		{
+			name:    "rows with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3"},
+				{"4", "5", "6", "7"},
+			},
+			want: "A | B\n-|-\n1 | 2\n4 | 5",
+		},
+		{
+			name:    "mixed content lengths",
+			headers: []string{"H1", "H2", "H3"},
+			rows: [][]string{

+				{"a", "", "c"},
+				{"aa", "bb", ""},
