+// FormatRate formats a transfer rate in bytes per second into a human-readable
+// string using binary units: B/s, KiB/s, MiB/s, GiB/s. Non-byte units show
+// 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+
+	switch {
+	case bytesPerSec < KiB:
+		result = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	case bytesPerSec < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	case bytesPerSec < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+
+		// KiB/s boundary and values
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		{1048575, "1024.0 KiB/s"},
+
+		// MiB/s boundary and values
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		{1073741823, "1024.0 MiB/s"},
+
+		// GiB/s values
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+
