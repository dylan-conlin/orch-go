+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.0 MiB", "1.0 GiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	const (
+		B  = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	var value float64
+	var unit string
+
+	switch {
+	case absBytes < KiB:
+		return fmt.Sprintf("%d B", bytes)
+	case absBytes < MiB:
+		value = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	case absBytes < GiB:
+		value = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes < TiB:
+		value = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	if negative {
+		return fmt.Sprintf("-%.1f %s", value, unit)
+	}
+	return fmt.Sprintf("%.1f %s", value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		// MiB boundary
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{10485760, "10.0 MiB"},
+		// GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{5368709120, "5.0 GiB"},
+		// TiB boundary
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{5497558138880, "5.0 TiB"},
+		// Large values
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+		// Between boundaries
+		{2560, "2.5 KiB"},
+		{3072000, "2.9 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
