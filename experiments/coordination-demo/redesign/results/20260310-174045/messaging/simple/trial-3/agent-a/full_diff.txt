+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "512 B", "1.0 KiB", "1.5 MiB", "2.3 GiB", etc.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	value := float64(bytes)
+
+	for i, unit := range units {
+		if value < 1024 {
+			if i == 0 {
+				// For bytes, show as integer without decimal
+				result := fmt.Sprintf("%d %s", int64(value), unit)
+				if negative {
+					result = "-" + result
+				}
+				return result
+			}
+			// For other units, show one decimal place
+			result := fmt.Sprintf("%.1f %s", value, unit)
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+		value /= 1024
+	}
+
+	// For very large values, use TiB with one decimal place
+	result := fmt.Sprintf("%.1f %s", value, units[len(units)-1])
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (no decimal)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 * 512, "512.0 KiB"},
+		// MiB boundaries
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1024 * 1024 * 512, "512.0 MiB"},
+		// GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 536870912, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{1024 * 1024 * 1024 * 512, "512.0 GiB"},
+		// TiB boundaries
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
