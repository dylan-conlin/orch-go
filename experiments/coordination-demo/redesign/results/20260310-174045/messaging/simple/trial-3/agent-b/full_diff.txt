+// FormatRate formats a transfer rate in bytes per second as a human-readable string.
+// Output uses binary units with "/s" suffix: "512 B/s", "1.5 MiB/s", "2.3 GiB/s".
+// Non-byte-per-second units are shown with 1 decimal place.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	sign := ""
+	value := bytesPerSec
+	if value < 0 {
+		sign = "-"
+		value = -value
+	}
+
+	const (
+		kib = 1024
+		mib = 1024 * 1024
+		gib = 1024 * 1024 * 1024
+	)
+
+	if value < kib {
+		return fmt.Sprintf("%s%.0f B/s", sign, value)
+	}
+	if value < mib {
+		return fmt.Sprintf("%s%.1f KiB/s", sign, value/kib)
+	}
+	if value < gib {
+		return fmt.Sprintf("%s%.1f MiB/s", sign, value/mib)
+	}
+	return fmt.Sprintf("%s%.1f GiB/s", sign, value/gib)
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B/s"},
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		// Fractional values
+		{1536, "1.5 KiB/s"},
+		{1.5 * 1024 * 1024, "1.5 MiB/s"},
+		{2.3 * 1024 * 1024 * 1024, "2.3 GiB/s"},
+		// Between boundaries
+		{2048, "2.0 KiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1.5 * 1024 * 1024, "-1.5 MiB/s"},
+		// Large values
+		{10 * 1024 * 1024 * 1024, "10.0 GiB/s"},
+		{5.75 * 1024 * 1024 * 1024, "5.8 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
