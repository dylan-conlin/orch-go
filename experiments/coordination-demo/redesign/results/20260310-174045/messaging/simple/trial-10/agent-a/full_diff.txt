+// FormatBytes formats a byte count as a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Shows 1 decimal place for non-byte units.
+// Handles negative values by prefixing with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -absBytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case absBytes < KiB:
+		return fmt.Sprintf("%d B", bytes)
+	case absBytes < MiB:
+		value = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	case absBytes < GiB:
+		value = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes < TiB:
+		value = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	sign := ""
+	if negative {
+		sign = "-"
+	}
+
+	return fmt.Sprintf("%s%.1f %s", sign, value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+		{1649267441664, "1.5 TiB"},
+
+		// Other fractional values
+		{2048, "2.0 KiB"},
+		{2097152, "2.0 MiB"},
+		{2147483648, "2.0 GiB"},
+
+		// Large values
+		{10737418240, "10.0 GiB"},
+		{11019960852480, "10.0 TiB"},
+
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1572864, "-1.5 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
