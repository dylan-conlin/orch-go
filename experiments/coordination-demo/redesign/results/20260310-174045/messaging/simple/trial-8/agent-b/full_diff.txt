+// FormatRate formats a transfer rate in bytes per second as a human-readable
+// string using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-B/s units (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	threshold := 1024.0
+	unitIndex := 0
+
+	for bytesPerSec >= threshold && unitIndex < len(units)-1 {
+		bytesPerSec /= threshold
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	} else {
+		result = fmt.Sprintf("%.1f %s", bytesPerSec, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		{0, "0 B/s"},
+		{512, "512 B/s"},
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{100, "100 B/s"},
+		{2048, "2.0 KiB/s"},
+		{5242880, "5.0 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
