+// FormatRate formats a transfer rate (bytes per second) into a human-readable
+// string using binary units with "/s" suffix: "B/s", "KiB/s", "MiB/s", "GiB/s".
+// Shows 1 decimal place for non-byte-per-sec units.
+// Handles zero and negative values: zero returns "0 B/s", negative values get "-" prefix.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kilobyte = 1024
+		megabyte = kilobyte * 1024
+		gigabyte = megabyte * 1024
+	)
+
+	var result string
+	if bytesPerSec < kilobyte {
+		result = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	} else if bytesPerSec < megabyte {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/kilobyte)
+	} else if bytesPerSec < gigabyte {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/megabyte)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/gigabyte)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s boundary and values
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{512 * 1024, "512.0 KiB/s"},
+		{1023.5 * 1024, "1023.5 KiB/s"},
+		// MiB/s boundary and values
+		{1048576, "1.0 MiB/s"},
+		{1.5 * 1048576, "1.5 MiB/s"},
+		{10 * 1048576, "10.0 MiB/s"},
+		{512 * 1048576, "512.0 MiB/s"},
+		// GiB/s values
+		{1024 * 1048576, "1.0 GiB/s"},
+		{1.5 * 1024 * 1048576, "1.5 GiB/s"},
+		{10 * 1024 * 1048576, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1024 * 1048576, "-1.0 GiB/s"},
+		// Fractional small values
+		{100.5, "100 B/s"},
+		{500.7, "500 B/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
