+// FormatRate formats a transfer rate (bytes per second) into human-readable
+// format using binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case bytesPerSec < KiB:
+		result = fmt.Sprintf("%d B/s", int(bytesPerSec))
+	case bytesPerSec < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	case bytesPerSec < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s boundary
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		// MiB/s boundary
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{5242880, "5.0 MiB/s"},
+		// GiB/s boundary
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{5368709120, "5.0 GiB/s"},
+		// Negative values
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-512, "-512 B/s"},
+		// Fractional values
+		{2560.5, "2.5 KiB/s"},
+		{1048576.0 * 2.25, "2.2 MiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
