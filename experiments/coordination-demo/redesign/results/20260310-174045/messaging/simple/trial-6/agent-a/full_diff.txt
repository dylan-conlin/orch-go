+// FormatBytes formats a byte count into a human-readable string using binary
+// units. Output style: "0 B", "512 B", "1.5 KiB", "1.0 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	thresholds := []int64{1024, 1024 * 1024, 1024 * 1024 * 1024, 1024 * 1024 * 1024 * 1024}
+
+	unitIndex := 0
+	for i, threshold := range thresholds {
+		if bytes < threshold {
+			unitIndex = i
+			break
+		}
+		unitIndex = i + 1
+	}
+
+	var result string
+	if unitIndex == 0 {
+		result = fmt.Sprintf("%d %s", bytes, units[0])
+	} else {
+		threshold := int64(1)
+		for i := 0; i < unitIndex; i++ {
+			threshold *= 1024
+		}
+		value := float64(bytes) / float64(threshold)
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Boundary: 1 KiB
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 + 512, "1.5 KiB"},
+		// Mid-range KiB
+		{10 * 1024, "10.0 KiB"},
+		{1023 * 1024, "1023.0 KiB"},
+		// Boundary: 1 MiB
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		{2 * 1048576, "2.0 MiB"},
+		// Mid-range MiB
+		{10 * 1048576, "10.0 MiB"},
+		{512 * 1048576, "512.0 MiB"},
+		// Boundary: 1 GiB
+		{1073741824, "1.0 GiB"},
+		{1073741824 + 537870912, "1.5 GiB"},
+		{2 * 1073741824, "2.0 GiB"},
+		// Mid-range GiB
+		{10 * 1073741824, "10.0 GiB"},
+		// Boundary: 1 TiB
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 + 549755813888, "1.5 TiB"},
+		// Large TiB values
+		{10 * 1099511627776, "10.0 TiB"},
+		{100 * 1099511627776, "100.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		// Fractional bytes (should round correctly)
+		{1536, "1.5 KiB"},
+		{1229, "1.2 KiB"},
+		{1331, "1.3 KiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
