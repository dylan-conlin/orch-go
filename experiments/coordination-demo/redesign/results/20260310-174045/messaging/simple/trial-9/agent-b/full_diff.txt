+// FormatRate formats a transfer rate (bytes per second) as a human-readable string.
+// Output style: "512 B/s", "1.0 KiB/s", "1.5 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	neg := false
+	if bytesPerSec < 0 {
+		neg = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	const divider = 1024.0
+	index := 0
+
+	for bytesPerSec >= divider && index < len(units)-1 {
+		bytesPerSec /= divider
+		index++
+	}
+
+	var result string
+	if index == 0 {
+		// For bytes per second, show as integer
+		result = fmt.Sprintf("%d %s", int(bytesPerSec), units[index])
+	} else {
+		// For larger units, show with 1 decimal place
+		result = fmt.Sprintf("%.1f %s", bytesPerSec, units[index])
+	}
+
+	if neg {
+		result = "-" + result
+	}
+	return result
+}
