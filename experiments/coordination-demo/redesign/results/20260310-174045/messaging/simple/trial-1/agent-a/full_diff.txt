+// FormatBytes formats a byte count as a human-readable string.
+// Uses binary units: B, KiB, MiB, GiB, TiB.
+// Shows 1 decimal place for non-byte units (e.g., "1.5 MiB").
+// Handles negative values by prefixing with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	isNegative := bytes < 0
+	absBytes := bytes
+	if isNegative {
+		absBytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+		TiB = 1024 * 1024 * 1024 * 1024
+	)
+
+	var value float64
+	var unit string
+
+	switch {
+	case absBytes >= TiB:
+		value = float64(absBytes) / float64(TiB)
+		unit = "TiB"
+	case absBytes >= GiB:
+		value = float64(absBytes) / float64(GiB)
+		unit = "GiB"
+	case absBytes >= MiB:
+		value = float64(absBytes) / float64(MiB)
+		unit = "MiB"
+	case absBytes >= KiB:
+		value = float64(absBytes) / float64(KiB)
+		unit = "KiB"
+	default:
+		// For bytes, return as integer
+		if isNegative {
+			return fmt.Sprintf("-%d B", absBytes)
+		}
+		return fmt.Sprintf("%d B", absBytes)
+	}
+
+	// Format with 1 decimal place
+	formatted := fmt.Sprintf("%.1f %s", value, unit)
+	if isNegative {
+		return "-" + formatted
+	}
+	return formatted
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		// Between KiB and MiB
+		{512 * 1024, "512.0 KiB"},
+		// Exact MiB boundary
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Exact GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		// Large values
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-512, "-512 B"},
+		{-1, "-1 B"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
