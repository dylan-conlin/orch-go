+// FormatRate formats a transfer rate in bytes per second as a human-readable
+// string using binary units: B/s, KiB/s, MiB/s, GiB/s. Shows 1 decimal place
+// for non-byte units (e.g., "1.5 MiB/s"). Handles zero and negative values.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []struct {
+		threshold float64
+		unit      string
+	}{
+		{1024 * 1024 * 1024, "GiB/s"},
+		{1024 * 1024, "MiB/s"},
+		{1024, "KiB/s"},
+		{0, "B/s"},
+	}
+
+	for _, u := range units {
+		if bytesPerSec >= u.threshold {
+			if u.threshold == 0 {
+				// B/s: no decimal
+				val := int(bytesPerSec)
+				if negative {
+					return fmt.Sprintf("-%d B/s", val)
+				}
+				return fmt.Sprintf("%d B/s", val)
+			}
+			// Other units: 1 decimal
+			val := bytesPerSec / u.threshold
+			if negative {
+				return fmt.Sprintf("-%.1f %s", val, u.unit)
+			}
+			return fmt.Sprintf("%.1f %s", val, u.unit)
+		}
+	}
+
+	// Fallback (shouldn't reach here)
+	if negative {
+		return fmt.Sprintf("-%d B/s", int(bytesPerSec))
+	}
+	return fmt.Sprintf("%d B/s", int(bytesPerSec))
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (B/s)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundary (1 KiB/s)
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB/s
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1024 + 512, "1.5 KiB/s"},
+		// Exact boundary (1 MiB/s)
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB/s
+		{1048576 + 524288, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Exact boundary (1 GiB/s)
+		{1073741824, "1.0 GiB/s"},
+		// Fractional GiB/s
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-1, "-1 B/s"},
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1536, "-1.5 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
