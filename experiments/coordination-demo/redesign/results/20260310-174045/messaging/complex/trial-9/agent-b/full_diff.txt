+// FormatTable formats headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (header or row).
+// ANSI escape codes are stripped for width calculation using StripANSI.
+// The header is separated from rows with a line of dashes.
+// Rows with fewer columns than headers are padded with empty cells.
+// Rows with more columns than headers are included with their extra columns.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and all rows
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(headers) {
+				w := len(StripANSI(cell))
+				if w > colWidths[i] {
+					colWidths[i] = w
+				}
+			}
+		}
+	}
+
+	// Build header row
+	var result strings.Builder
+	for i, h := range headers {
+		if i > 0 {
+			result.WriteString(" | ")
+		}
+		// Pad header to column width
+		padding := colWidths[i] - len(StripANSI(h))
+		result.WriteString(h)
+		result.WriteString(strings.Repeat(" ", padding))
+	}
+	result.WriteString("\n")
+
+	// Build separator row
+	for i, w := range colWidths {
+		if i > 0 {
+			result.WriteString("-+-")
+		}
+		result.WriteString(strings.Repeat("-", w))
+	}
+	result.WriteString("\n")
+
+	// Build data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				result.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			// Pad cell to column width
+			padding := colWidths[i] - len(StripANSI(cell))
+			result.WriteString(cell)
+			result.WriteString(strings.Repeat(" ", padding))
+		}
+		result.WriteString("\n")
+	}
+
+	return strings.TrimSuffix(result.String(), "\n")
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+----\nAlice | 30 \nBob   | 25 ",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want:    "Name | Age\n-----+----",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Name"},
+			rows: [][]string{
+				{"Alice"},
+				{"Bob"},
+			},
+			want: "Name \n-----\nAlice\nBob  ",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[32mAlice\x1b[0m", "\x1b[31merror\x1b[0m"},
+				{"Bob", "ok"},
+			},
+			want: "Name  | Status\n------+-------\n\x1b[32mAlice\x1b[0m | \x1b[31merror\x1b[0m \nBob   | ok    ",
+		},
+		{
+			name:    "mismatched columns - fewer in row",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 |  \n3 | 4 | 5",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "Long"},
+			rows: [][]string{
+				{"a", "this is a much longer string"},
+			},
+			want: "Short | Long                        \n------+-----------------------------\na     | this is a much longer string",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "empty cells",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"", "val"},
+				{"val", ""},
+			},
+			want: "A   | B  \n----+----\n    | val\nval |    ",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() mismatch:\ngot:\n%q\n\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
+}
