+// FormatTable renders headers and rows as an aligned text table.
+// Each column is auto-sized based on the widest content (including header).
+// Headers are separated from data rows by a line of dashes.
+// ANSI-colored content is handled correctly using StripANSI for width calculation.
+// Handles edge cases: empty rows (headers only), rows with fewer columns than headers,
+// and rows with more columns than headers (extra columns ignored).
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers) && i < len(row); i++ {
+			width := len(StripANSI(row[i]))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Render header row
+	buf.WriteString("│")
+	for i, h := range headers {
+		buf.WriteString(" ")
+		buf.WriteString(padToWidth(h, colWidths[i]))
+		buf.WriteString(" │")
+	}
+	buf.WriteString("\n")
+
+	// Render separator
+	buf.WriteString("├")
+	for i, width := range colWidths {
+		buf.WriteString(strings.Repeat("─", width+2))
+		if i < len(colWidths)-1 {
+			buf.WriteString("┼")
+		} else {
+			buf.WriteString("┤")
+		}
+	}
+	buf.WriteString("\n")
+
+	// Render data rows
+	for _, row := range rows {
+		buf.WriteString("│")
+		for i := 0; i < len(headers); i++ {
+			buf.WriteString(" ")
+			var cell string
+			if i < len(row) {
+				cell = padToWidth(row[i], colWidths[i])
+			} else {
+				cell = strings.Repeat(" ", colWidths[i])
+			}
+			buf.WriteString(cell)
+			buf.WriteString(" │")
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}
+
+// padToWidth right-pads a string with spaces to reach the target visual width.
+// Uses StripANSI to calculate true width, preserving ANSI codes in the output.
+func padToWidth(s string, width int) string {
+	visualWidth := len(StripANSI(s))
+	if visualWidth >= width {
+		return s
+	}
+	return s + strings.Repeat(" ", width-visualWidth)
+}

+func TestFormatTableEdgeCases(t *testing.T) {
+	// Test with nil rows
+	got := FormatTable([]string{"Header"}, nil)
+	if !strings.Contains(got, "Header") {
+		t.Errorf("FormatTable with nil rows should still have header")
+	}
+
+	// Test with empty row
+	got = FormatTable([]string{"A", "B"}, [][]string{
+		{},
+	})
+	if !strings.Contains(got, "│") {
+		t.Errorf("FormatTable with empty row should still render table")
+	}
+
+	// Test with very wide ANSI content
+	wideLine := "\x1b[1;32m" + strings.Repeat("X", 50) + "\x1b[0m"
+	got = FormatTable([]string{"Wide"}, [][]string{
+		{wideLine},
+	})
+	if !strings.Contains(got, "Wide") {
+		t.Errorf("FormatTable with wide ANSI content should render header")
+	}
