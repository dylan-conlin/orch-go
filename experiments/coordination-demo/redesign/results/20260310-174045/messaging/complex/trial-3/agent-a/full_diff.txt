+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It counts runes to handle Unicode characters correctly.
+func VisualWidth(s string) int {
+	return utf8.RuneCountInString(StripANSI(s))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward the width. If the string
+// is already at or wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mhello\x1b[0m", 5},                        // red text
+		{"\x1b[1;32mbold green\x1b[0m", 10},                // bold green
+		{"\x1b[38;5;196mext\x1b[0m", 3},                    // extended 256 color
+		{"\x1b[31mred\x1b[0m \x1b[32mgreen\x1b[0m", 9},    // multiple colors
+
+		// Unicode strings (CJK characters)
+		{"日本", 2},
+		{"你好世界", 4},
+
+		// Unicode with emoji
+		{"hello 😀", 7},
+		{"🎉🎊", 2},
+
+		// Unicode with ANSI codes
+		{"\x1b[32m日本\x1b[0m", 2},
+
+		// Edge cases
+		{"  ", 2},
+		{"\t", 1}, // tab is one rune
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII strings
+		{"hello", 10, "hello     "},
+		{"hello", 5, "hello"},
+		{"hi", 5, "hi   "},
+		{"", 5, "     "},
+
+		// Already at width
+		{"hello", 5, "hello"},
+		{"abc", 3, "abc"},
+
+		// Wider than target (should return unchanged)
+		{"hello world", 5, "hello world"},
+		{"test", 2, "test"},
+
+		// Strings with ANSI codes (should preserve codes and pad correctly)
+		{"\x1b[31mhello\x1b[0m", 10, "\x1b[31mhello\x1b[0m     "},
+		{"\x1b[1;32mbold\x1b[0m", 6, "\x1b[1;32mbold\x1b[0m  "},
+
+		// Unicode strings
+		{"日本", 4, "日本  "},
+		{"你好", 3, "你好 "},
+
+		// Unicode with ANSI codes
+		{"\x1b[32m日本\x1b[0m", 4, "\x1b[32m日本\x1b[0m  "},
+
+		// Empty string
+		{"", 0, ""},
+
+		// Single character padding
+		{"x", 2, "x "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify that the visual width is correct
+		visualWidth := VisualWidth(got)
+		if visualWidth < tt.width && tt.width > 0 {
+			// If original was already at or wider than width, visual width can be >= width
+			// Otherwise, visual width should equal width
+			if VisualWidth(tt.input) < tt.width && visualWidth != tt.width {
+				t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, visualWidth, tt.width)
+			}
+		}
+	}
+}
