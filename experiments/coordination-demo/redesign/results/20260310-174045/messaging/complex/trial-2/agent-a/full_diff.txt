+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It correctly handles Unicode by counting runes instead of bytes.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but do not count toward width. If the string is
+// already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+
+		// Strings with ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[31mred\x1b[0m \x1b[32mgreen\x1b[0m", 9}, // "red green"
+
+		// Unicode strings (CJK, emoji)
+		{"你好", 2},
+		{"你好世界", 4},
+		{"😀", 1},
+		{"Hello😀World", 11}, // "Hello" + "😀" + "World"
+
+		// Mixed: Unicode with ANSI
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1;33m😀test\x1b[0m", 5},
+
+		// Edge cases
+		{"   ", 3},
+		{"\x1b[0m", 0},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input   string
+		width   int
+		want    string
+		checkVW bool // check that visual width is <= width (false for strings wider than target)
+	}{
+		// Basic padding
+		{"hello", 10, "hello     ", true},
+		{"hello", 5, "hello", true},
+		{"hi", 5, "hi   ", true},
+
+		// Empty string
+		{"", 5, "     ", true},
+		{"", 0, "", true},
+
+		// Already at width
+		{"hello", 5, "hello", true},
+		{"hi", 2, "hi", true},
+
+		// Wider than target (should return unchanged) - don't check visual width
+		{"hello", 3, "hello", false},
+		{"hello world", 5, "hello world", false},
+
+		// Unicode strings
+		{"你好", 5, "你好   ", true}, // 2 runes + 3 spaces
+		{"😀", 3, "😀  ", true},      // 1 rune + 2 spaces
+
+		// Strings with ANSI codes
+		{"\x1b[31mred\x1b[0m", 5, "\x1b[31mred\x1b[0m  ", true}, // visual width 3, padded to 5
+		{"\x1b[32mhi\x1b[0m", 5, "\x1b[32mhi\x1b[0m   ", true},  // visual width 2, padded to 5
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// For padded cases, check visual width is correct
+		if tt.checkVW && VisualWidth(got) != tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want %d", tt.input, tt.width, VisualWidth(got), tt.width)
+		}
+	}
+}
