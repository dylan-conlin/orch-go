+// FormatTable renders headers and rows as an aligned text table with
+// auto-sized columns. ANSI escape codes are preserved but ignored for width
+// calculations. The header is separated from data rows with a line of dashes.
+// Handles edge cases: empty rows (headers only), rows with fewer columns than
+// headers (padded with empty strings), and rows with more columns than headers
+// (extra columns appended).
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths: max of header width and all row content widths
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i, cell := range row {
+			if i < len(headers) {
+				width := len(StripANSI(cell))
+				if width > colWidths[i] {
+					colWidths[i] = width
+				}
+			}
+		}
+	}
+
+	var sb strings.Builder
+
+	// Build header row
+	for i, h := range headers {
+		if i > 0 {
+			sb.WriteString(" | ")
+		}
+		padded := padRight(h, colWidths[i])
+		sb.WriteString(padded)
+	}
+	sb.WriteString("\n")
+
+	// Build separator row
+	for i, width := range colWidths {
+		if i > 0 {
+			sb.WriteString("-+-")
+		}
+		sb.WriteString(strings.Repeat("-", width))
+	}
+	sb.WriteString("\n")
+
+	// Build data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				sb.WriteString(" | ")
+			}
+			cell := ""
+			if i < len(row) {
+				cell = row[i]
+			}
+			padded := padRight(cell, colWidths[i])
+			sb.WriteString(padded)
+		}
+		sb.WriteString("\n")
+	}
+
+	return sb.String()
+}
+
+// padRight right-pads a string with spaces to reach the target width.
+// Uses StripANSI to calculate the visual width correctly.
+func padRight(s string, width int) string {
+	visualWidth := len(StripANSI(s))
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "Name  | Age | City\n------+-----+-----\nAlice | 30  | NYC \nBob   | 25  | LA  \n",
+		},
+		{
+			name:    "headers only (empty rows)",
+			headers: []string{"ID", "Status"},
+			rows:    [][]string{},
+			want:    "ID | Status\n---+-------\n",
+		},
+		{
+			name:    "single column table",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+			},
+			want: "Item  \n------\napple \nbanana\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"a", "x"},
+				{"abcdefgh", "y"},
+			},
+			want: "Short    | VeryLongHeader\n---------+---------------\na        | x             \nabcdefgh | y             \n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Name", "Status"},
+			rows: [][]string{
+				{"\x1b[32mAlice\x1b[0m", "\x1b[31mError\x1b[0m"},
+				{"Bob", "\x1b[32mOK\x1b[0m"},
+			},
+			want: "Name  | Status\n------+-------\n\x1b[32mAlice\x1b[0m | \x1b[31mError\x1b[0m \nBob   | \x1b[32mOK\x1b[0m    \n",
+		},
+		{
+			name:    "mismatched columns (fewer in row)",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"x", "y"},
+				{"p", "q", "r"},
+			},
+			want: "A | B | C\n--+---+--\nx | y |  \np | q | r\n",
+		},
+		{
+			name:    "all empty strings",
+			headers: []string{"", "", ""},
+			rows: [][]string{
+				{"", "", ""},
+			},
+			want: " |  | \n-+--+-\n |  | \n",
+		},
+		{
+			name:    "numeric alignment",
+			headers: []string{"Count", "Value"},
+			rows: [][]string{
+				{"1", "100"},
+				{"2000", "5"},
+			},
+			want: "Count | Value\n------+------\n1     | 100  \n2000  | 5    \n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable() = %q, want %q", got, tt.want)
+			}
+		})
+	}
+}

+func TestFormatTableEdgeCases(t *testing.T) {
+	// Empty headers should return empty string
+	got := FormatTable([]string{}, [][]string{})
+	if got != "" {
+		t.Errorf("FormatTable with empty headers = %q, want empty string", got)
+	}
+
+	// Single header, no rows
+	got = FormatTable([]string{"Header"}, [][]string{})
+	want := "Header\n------\n"
+	if got != want {
+		t.Errorf("FormatTable single header no rows = %q, want %q", got, want)
+	}
+
+	// Nil rows slice
+	got = FormatTable([]string{"A", "B"}, nil)
+	want = "A | B\n--+--\n"
+	if got != want {
+		t.Errorf("FormatTable with nil rows = %q, want %q", got, want)
+	}
+}
