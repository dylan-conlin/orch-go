+// FormatTable renders headers and rows as an aligned text table. Columns are
+// auto-sized based on the widest content in each column (including headers).
+// Headers are separated from data rows with a line of dashes. ANSI escape
+// codes are preserved in output but don't count toward visual width.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Determine the maximum number of columns (at least as many as headers)
+	maxCols := len(headers)
+	for _, row := range rows {
+		if len(row) > maxCols {
+			maxCols = len(row)
+		}
+	}
+
+	// Calculate column widths based on widest content in each column
+	colWidths := make([]int, maxCols)
+
+	// Consider header widths
+	for i, header := range headers {
+		width := len([]rune(StripANSI(header)))
+		if width > colWidths[i] {
+			colWidths[i] = width
+		}
+	}
+
+	// Consider row widths
+	for _, row := range rows {
+		for i, cell := range row {
+			width := len([]rune(StripANSI(cell)))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var result strings.Builder
+
+	// Format header row
+	for i := 0; i < maxCols; i++ {
+		if i > 0 {
+			result.WriteString(" | ")
+		}
+		var header string
+		if i < len(headers) {
+			header = headers[i]
+		}
+		result.WriteString(padCellToWidth(header, colWidths[i]))
+	}
+	result.WriteString("\n")
+
+	// Format separator row
+	for i := 0; i < maxCols; i++ {
+		if i > 0 {
+			result.WriteString("-+-")
+		}
+		result.WriteString(strings.Repeat("-", colWidths[i]))
+	}
+	result.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		for i := 0; i < maxCols; i++ {
+			if i > 0 {
+				result.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			result.WriteString(padCellToWidth(cell, colWidths[i]))
+		}
+		result.WriteString("\n")
+	}
+
+	return strings.TrimRight(result.String(), "\n")
+}
+
+// padCellToWidth right-pads a cell to the target visual width, preserving
+// ANSI codes but accounting for them in the width calculation.
+func padCellToWidth(cell string, width int) string {
+	stripped := StripANSI(cell)
+	currentWidth := len([]rune(stripped))
+	if currentWidth >= width {
+		return cell
+	}
+	padding := width - currentWidth
+	return cell + strings.Repeat(" ", padding)
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "Name  | Age | City\n" +
+				"------+-----+-----\n" +
+				"Alice | 30  | NYC \n" +
+				"Bob   | 25  | LA  ",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Header1", "Header2"},
+			rows:    [][]string{},
+			want: "Header1 | Header2\n" +
+				"--------+--------",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"Apple"},
+				{"Banana"},
+			},
+			want: "Item  \n" +
+				"------\n" +
+				"Apple \n" +
+				"Banana",
+		},
+		{
+			name:    "rows with fewer columns",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"3", "4", "5"},
+			},
+			want: "A | B | C\n" +
+				"--+---+--\n" +
+				"1 | 2 |  \n" +
+				"3 | 4 | 5",
+		},
+		{
+			name:    "rows with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3", "4"},
+			},
+			want: "A | B |   |  \n" +
+				"--+---+---+--\n" +
+				"1 | 2 | 3 | 4",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"X", "Y"},
+				{"ABC", "LongContent"},
+			},
+			want: "Short | VeryLongHeader\n" +
+				"------+---------------\n" +
+				"X     | Y             \n" +
+				"ABC   | LongContent   ",
+		},
+		{
+			name:    "ansi colored content",
+			headers: []string{"\x1b[31mName\x1b[0m", "\x1b[32mAge\x1b[0m"},
+			rows: [][]string{
+				{"\x1b[31mAlice\x1b[0m", "\x1b[32m30\x1b[0m"},
+				{"Bob", "25"},
+			},
+			want: "\x1b[31mName\x1b[0m  | \x1b[32mAge\x1b[0m\n" +
+				"------+----\n" +
+				"\x1b[31mAlice\x1b[0m | \x1b[32m30\x1b[0m \n" +
+				"Bob   | 25 ",
+		},
+		{
+			name:    "mixed width with ansi",
+			headers: []string{"ID", "Status"},
+			rows: [][]string{
+				{"123", "\x1b[32mActive\x1b[0m"},
+				{"4", "\x1b[31mInactive\x1b[0m"},
+			},
+			want: "ID  | Status  \n" +
+				"----+---------\n" +
+				"123 | \x1b[32mActive\x1b[0m  \n" +
+				"4   | \x1b[31mInactive\x1b[0m",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "single header single row",
+			headers: []string{"X"},
+			rows: [][]string{
+				{"Y"},
+			},
+			want: "X\n" +
+				"-\n" +
+				"Y",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable got:\n%q\n\nwant:\n%q", got, tt.want)
+			}
+		})
+	}
