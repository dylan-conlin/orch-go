+// FormatTable renders headers and rows as an aligned text table with
+// auto-sized columns. Column width is determined by the widest content
+// in that column (including header). ANSI color codes are handled
+// correctly using StripANSI for width calculation. Headers are separated
+// from data rows with a line of dashes. Rows with fewer columns than
+// headers are padded with empty cells; rows with more columns are
+// truncated. Returns a multi-line string.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len(StripANSI(h))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cellContent string
+			if i < len(row) {
+				cellContent = row[i]
+			}
+			visualWidth := len(StripANSI(cellContent))
+			if visualWidth > colWidths[i] {
+				colWidths[i] = visualWidth
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Write header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		isLastCol := i == len(headers)-1
+		if isLastCol {
+			buf.WriteString(h)
+		} else {
+			paddedHeader := padRight(h, colWidths[i])
+			buf.WriteString(paddedHeader)
+		}
+	}
+	buf.WriteString("\n")
+
+	// Write separator row
+	for i, width := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+-")
+		}
+		buf.WriteString(strings.Repeat("-", width))
+	}
+	buf.WriteString("\n")
+
+	// Write data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cellContent string
+			if i < len(row) {
+				cellContent = row[i]
+			}
+			isLastCol := i == len(headers)-1
+			if isLastCol {
+				buf.WriteString(cellContent)
+			} else {
+				paddedCell := padRight(cellContent, colWidths[i])
+				buf.WriteString(paddedCell)
+			}
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}
+
+// padRight pads a string (which may contain ANSI codes) with spaces on the
+// right to reach the target visual width. ANSI codes don't count toward
+// the width calculation.
+func padRight(s string, width int) string {
+	visualWidth := len(StripANSI(s))
+	if visualWidth >= width {
+		return s
+	}
+	paddingNeeded := width - visualWidth
+	return s + strings.Repeat(" ", paddingNeeded)
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "Name  | Age | City\n------+-----+-----\nAlice | 30  | NYC\nBob   | 25  | LA\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"ID", "Status"},
+			rows:    [][]string{},
+			want:    "ID | Status\n---+-------\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Items"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+			},
+			want: "Items\n------\napple\nbanana\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Task", "Status"},
+			rows: [][]string{
+				{"\x1b[32mSetup\x1b[0m", "\x1b[32mDone\x1b[0m"},
+				{"\x1b[31mFix\x1b[0m", "\x1b[31mFailed\x1b[0m"},
+			},
+			want: "Task  | Status\n------+-------\n\x1b[32mSetup\x1b[0m | \x1b[32mDone\x1b[0m\n\x1b[31mFix\x1b[0m   | \x1b[31mFailed\x1b[0m\n",
+		},
+		{
+			name:    "row with fewer columns",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"4", "5", "6"},
+			},
+			want: "A | B | C\n--+---+--\n1 | 2 | \n4 | 5 | 6\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"ID", "Description"},
+			rows: [][]string{
+				{"1", "short"},
+				{"2", "this is a much longer description"},
+			},
+			want: "ID | Description\n---+----------------------------------\n1  | short\n2  | this is a much longer description\n",
+		},
+		{
+			name:    "empty headers",
+			headers: []string{},
+			rows:    [][]string{},
+			want:    "",
+		},
+		{
+			name:    "row with more columns than headers",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"1", "2", "3", "4"},
+			},
+			want: "A | B\n--+--\n1 | 2\n",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"X", "Y"},
+			rows:    nil,
+			want:    "X | Y\n--+--\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v)\ngot:\n%q\nwant:\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
