+// VisualWidth returns the visual display width of a string, accounting for
+// ANSI escape codes and Unicode characters. ANSI codes are ignored, and
+// runes are counted (not bytes).
+func VisualWidth(s string) int {
+	stripped := StripANSI(s)
+	return len([]rune(stripped))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI escape codes are preserved but do not count toward the visual width.
+// If the string is already wider than or equal to width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	currentWidth := VisualWidth(s)
+	if currentWidth >= width {
+		return s
+	}
+	padding := width - currentWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+		// ANSI color codes
+		{"\x1b[31mred\x1b[0m", 3},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33m yellow \x1b[0m", 8},
+		// Unicode strings
+		{"café", 4},
+		{"你好", 2},
+		{"😀😁😂", 3},
+		// Mixed: ANSI + Unicode
+		{"\x1b[31m你好\x1b[0m", 2},
+		{"\x1b[1;35m🎉\x1b[0m", 1},
+		// Edge cases
+		{"   ", 3},
+		{"a\x1b[31mb\x1b[0mc", 3},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		width int
+		want  string
+	}{
+		// Plain ASCII padding
+		{"hello", 10, "hello     "},
+		{"hi", 5, "hi   "},
+		{"", 5, "     "},
+		// Already at width
+		{"hello", 5, "hello"},
+		// Already wider than target width
+		{"hello", 3, "hello"},
+		{"hello world", 5, "hello world"},
+		// ANSI codes (preserved, not counted in width)
+		{"\x1b[31mhi\x1b[0m", 5, "\x1b[31mhi\x1b[0m   "},
+		{"\x1b[1;32mbold\x1b[0m", 10, "\x1b[1;32mbold\x1b[0m      "},
+		// Unicode
+		{"café", 6, "café  "},
+		{"你好", 4, "你好  "},
+		// Mixed ANSI + Unicode (visual width counts runes, not terminal columns)
+		{"\x1b[31m你\x1b[0m", 3, "\x1b[31m你\x1b[0m  "},
+		{"\x1b[1;35m🎉\x1b[0m", 3, "\x1b[1;35m🎉\x1b[0m  "},
+		// No padding needed
+		{"test", 4, "test"},
+		{"test", 3, "test"},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify that visual width is at least the target (or exceeds if already wide)
+		visualWidth := VisualWidth(got)
+		if visualWidth < tt.width && VisualWidth(tt.input) < tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want >= %d", tt.input, tt.width, visualWidth, tt.width)
+		}
+	}
+}
