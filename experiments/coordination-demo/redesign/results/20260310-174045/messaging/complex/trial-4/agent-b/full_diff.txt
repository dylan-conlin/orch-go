+// FormatTable renders headers and rows as an aligned text table with
+// auto-sized columns. Column widths are determined by the widest content
+// (including headers) in each column. ANSI-colored content is handled
+// correctly by using StripANSI for width calculation. Rows with fewer
+// columns than headers are padded with empty strings. Returns a string
+// with newlines separating rows.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths from headers and all rows
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len([]rune(StripANSI(h)))
+	}
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			width := len([]rune(StripANSI(cell)))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	// Helper function to pad a cell to a target visual width
+	padCell := func(cell string, width int) string {
+		visualWidth := len([]rune(StripANSI(cell)))
+		padding := width - visualWidth
+		if padding < 0 {
+			padding = 0
+		}
+		return cell + strings.Repeat(" ", padding)
+	}
+
+	var result strings.Builder
+
+	// Format header row
+	result.WriteString("| ")
+	for i, h := range headers {
+		result.WriteString(padCell(h, colWidths[i]))
+		if i < len(headers)-1 {
+			result.WriteString(" | ")
+		} else {
+			result.WriteString(" |")
+		}
+	}
+	result.WriteString("\n")
+
+	// Format separator line
+	result.WriteString("+")
+	for _, width := range colWidths {
+		result.WriteString("-")
+		result.WriteString(strings.Repeat("-", width))
+		result.WriteString("-+")
+	}
+	result.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		result.WriteString("| ")
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			result.WriteString(padCell(cell, colWidths[i]))
+			if i < len(headers)-1 {
+				result.WriteString(" | ")
+			} else {
+				result.WriteString(" |")
+			}
+		}
+		result.WriteString("\n")
+	}
+
+	return result.String()
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age", "City"},
+			rows: [][]string{
+				{"Alice", "30", "NYC"},
+				{"Bob", "25", "LA"},
+			},
+			want: "| Name  | Age | City |\n+-------+-----+------+\n| Alice | 30  | NYC  |\n| Bob   | 25  | LA   |\n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"ID", "Status"},
+			rows:    [][]string{},
+			want:    "| ID | Status |\n+----+--------+\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"apple"},
+				{"banana"},
+			},
+			want: "| Item   |\n+--------+\n| apple  |\n| banana |\n",
+		},
+		{
+			name:    "ANSI colored content",
+			headers: []string{"Color", "Value"},
+			rows: [][]string{
+				{"\x1b[31mRed\x1b[0m", "100"},
+				{"\x1b[32mGreen\x1b[0m", "200"},
+			},
+			want: "| Color | Value |\n+-------+-------+\n| \x1b[31mRed\x1b[0m   | 100   |\n| \x1b[32mGreen\x1b[0m | 200   |\n",
+		},
+		{
+			name:    "mismatched columns (fewer columns in row)",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"x", "y"},
+				{"1", "2", "3"},
+			},
+			want: "| A | B | C |\n+---+---+---+\n| x | y |   |\n| 1 | 2 | 3 |\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "VeryLongHeader"},
+			rows: [][]string{
+				{"a", "b"},
+				{"exactly", "this is a very long string that should determine column width"},
+			},
+			want: "| Short   | VeryLongHeader                                                |\n+---------+---------------------------------------------------------------+\n| a       | b                                                             |\n| exactly | this is a very long string that should determine column width |\n",
+		},
+		{
+			name:    "empty strings in cells",
+			headers: []string{"A", "B"},
+			rows: [][]string{
+				{"", "val"},
+				{"val", ""},
+			},
+			want: "| A   | B   |\n+-----+-----+\n|     | val |\n| val |     |\n",
+		},
+		{
+			name:    "all empty headers",
+			headers: []string{"", "", ""},
+			rows: [][]string{
+				{"x", "y", "z"},
+			},
+			want: "|   |   |   |\n+---+---+---+\n| x | y | z |\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
