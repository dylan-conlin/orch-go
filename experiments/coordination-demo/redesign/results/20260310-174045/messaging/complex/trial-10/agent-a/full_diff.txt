+// VisualWidth returns the visual display width of a string, ignoring ANSI
+// escape codes. It counts runes (not bytes) to correctly handle Unicode.
+func VisualWidth(s string) int {
+	return len([]rune(StripANSI(s)))
+}
+
+// PadToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is
+// already wider than width, it is returned unchanged.
+func PadToWidth(s string, width int) string {
+	visualWidth := VisualWidth(s)
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestVisualWidth(t *testing.T) {
+	tests := []struct {
+		input string
+		want  int
+	}{
+		// Plain ASCII strings
+		{"hello", 5},
+		{"", 0},
+		{"a", 1},
+		{"hello world", 11},
+		// Strings with ANSI color codes
+		{"\x1b[31mred text\x1b[0m", 8},
+		{"\x1b[1;32mbold green\x1b[0m", 10},
+		{"\x1b[33myellow\x1b[0m", 6},
+		// Unicode strings (CJK characters)
+		{"你好", 2},
+		{"日本語", 3},
+		// Emoji (typically counted as 1 rune)
+		{"😀hello", 6},
+		{"test😀test", 9},
+		// Mixed: ASCII + ANSI + Unicode
+		{"\x1b[31m日本\x1b[0m", 2},
+		{"\x1b[32mhello世界\x1b[0m", 7},
+	}
+	for _, tt := range tests {
+		got := VisualWidth(tt.input)
+		if got != tt.want {
+			t.Errorf("VisualWidth(%q) = %d, want %d", tt.input, got, tt.want)
+		}
+	}
+}

+func TestPadToWidth(t *testing.T) {
+	tests := []struct {
+		input  string
+		width  int
+		want   string
+	}{
+		// Plain ASCII strings
+		{"hello", 10, "hello     "},
+		{"hello", 5, "hello"},
+		{"hello", 3, "hello"},
+		{"", 5, "     "},
+		{"", 0, ""},
+		// Strings with ANSI codes
+		{"\x1b[31mred\x1b[0m", 10, "\x1b[31mred\x1b[0m       "},
+		{"\x1b[31mred\x1b[0m", 3, "\x1b[31mred\x1b[0m"},
+		{"\x1b[1;32mbold\x1b[0m", 10, "\x1b[1;32mbold\x1b[0m      "},
+		// Unicode strings
+		{"你好", 5, "你好   "},
+		{"你好", 2, "你好"},
+		{"你好", 1, "你好"},
+		// Mixed: ASCII + ANSI + Unicode
+		{"\x1b[31m世界\x1b[0m", 5, "\x1b[31m世界\x1b[0m   "},
+	}
+	for _, tt := range tests {
+		got := PadToWidth(tt.input, tt.width)
+		if got != tt.want {
+			t.Errorf("PadToWidth(%q, %d) = %q, want %q", tt.input, tt.width, got, tt.want)
+		}
+		// Verify that visual width matches target (if input is narrower)
+		visualWidth := VisualWidth(got)
+		if visualWidth < tt.width {
+			t.Errorf("PadToWidth(%q, %d) visual width = %d, want >= %d", tt.input, tt.width, visualWidth, tt.width)
+		}
+	}
+}
