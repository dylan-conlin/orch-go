+// FormatTable renders headers and rows as an aligned text table. Each column
+// is auto-sized based on the widest content in that column (including headers).
+// The header is separated from data rows with a line of dashes. ANSI color codes
+// are preserved in output but ignored for width calculations using StripANSI.
+// Handles edge cases: empty rows, rows with fewer columns than headers (treated
+// as empty strings), and nil/empty row slices.
+func FormatTable(headers []string, rows [][]string) string {
+	if len(headers) == 0 {
+		return ""
+	}
+
+	// Calculate column widths based on headers and all row content
+	colWidths := make([]int, len(headers))
+	for i, h := range headers {
+		colWidths[i] = len([]rune(StripANSI(h)))
+	}
+
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			width := len([]rune(StripANSI(cell)))
+			if width > colWidths[i] {
+				colWidths[i] = width
+			}
+		}
+	}
+
+	var buf strings.Builder
+
+	// Format header row
+	for i, h := range headers {
+		if i > 0 {
+			buf.WriteString(" | ")
+		}
+		buf.WriteString(padToWidth(h, colWidths[i]))
+	}
+	buf.WriteString("\n")
+
+	// Separator line
+	for i, w := range colWidths {
+		if i > 0 {
+			buf.WriteString("-+")
+		}
+		buf.WriteString(strings.Repeat("-", w))
+	}
+	buf.WriteString("\n")
+
+	// Format data rows
+	for _, row := range rows {
+		for i := 0; i < len(headers); i++ {
+			if i > 0 {
+				buf.WriteString(" | ")
+			}
+			var cell string
+			if i < len(row) {
+				cell = row[i]
+			}
+			buf.WriteString(padToWidth(cell, colWidths[i]))
+		}
+		buf.WriteString("\n")
+	}
+
+	return buf.String()
+}
+
+// padToWidth right-pads a string with spaces to reach the target visual width.
+// ANSI codes are preserved but don't count toward width. If the string is
+// already wider than width, it is returned unchanged.
+func padToWidth(s string, width int) string {
+	visualWidth := len([]rune(StripANSI(s)))
+	if visualWidth >= width {
+		return s
+	}
+	padding := width - visualWidth
+	return s + strings.Repeat(" ", padding)
+}

+func TestFormatTable(t *testing.T) {
+	tests := []struct {
+		name    string
+		headers []string
+		rows    [][]string
+		want    string
+	}{
+		{
+			name:    "basic table",
+			headers: []string{"Name", "Age"},
+			rows: [][]string{
+				{"Alice", "30"},
+				{"Bob", "25"},
+			},
+			want: "Name  | Age\n------+---\nAlice | 30 \nBob   | 25 \n",
+		},
+		{
+			name:    "headers only",
+			headers: []string{"Name", "Age"},
+			rows:    [][]string{},
+			want:    "Name | Age\n-----+---\n",
+		},
+		{
+			name:    "single column",
+			headers: []string{"Item"},
+			rows: [][]string{
+				{"Apple"},
+				{"Banana"},
+			},
+			want: "Item  \n------\nApple \nBanana\n",
+		},
+		{
+			name:    "ansi colored content",
+			headers: []string{"Status"},
+			rows: [][]string{
+				{"\x1b[32mGreen\x1b[0m"},
+				{"\x1b[31mRed\x1b[0m"},
+			},
+			want: "Status\n------\n\x1b[32mGreen\x1b[0m \n\x1b[31mRed\x1b[0m   \n",
+		},
+		{
+			name:    "mismatched columns - fewer cells in row",
+			headers: []string{"A", "B", "C"},
+			rows: [][]string{
+				{"1", "2"},
+				{"x", "y", "z"},
+			},
+			want: "A | B | C\n--+--+-\n1 | 2 |  \nx | y | z\n",
+		},
+		{
+			name:    "wide content",
+			headers: []string{"Short", "LongColumn"},
+			rows: [][]string{
+				{"X", "This is a long string"},
+			},
+			want: "Short | LongColumn           \n------+---------------------\nX     | This is a long string\n",
+		},
+		{
+			name:    "empty headers returns empty",
+			headers: []string{},
+			rows: [][]string{
+				{"1", "2"},
+			},
+			want: "",
+		},
+		{
+			name:    "nil rows",
+			headers: []string{"Col"},
+			rows:    nil,
+			want:    "Col\n---\n",
+		},
+		{
+			name:    "three column table with varying widths",
+			headers: []string{"ID", "Name", "Status"},
+			rows: [][]string{
+				{"1", "Alice", "Active"},
+				{"999", "Bob", "Inactive"},
+			},
+			want: "ID  | Name  | Status  \n----+------+--------\n1   | Alice | Active  \n999 | Bob   | Inactive\n",
+		},
+	}
+
+	for _, tt := range tests {
+		t.Run(tt.name, func(t *testing.T) {
+			got := FormatTable(tt.headers, tt.rows)
+			if got != tt.want {
+				t.Errorf("FormatTable(%v, %v) =\n%q\nwant\n%q", tt.headers, tt.rows, got, tt.want)
+			}
+		})
+	}
+}
