+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output uses binary units (1024 basis): B, KiB, MiB, GiB, TiB.
+// Non-byte values show 1 decimal place (e.g., "1.5 MiB").
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	prefix := ""
+	if negative {
+		prefix = "-"
+	}
+
+	if bytes < KiB {
+		return fmt.Sprintf("%s%dB", prefix, bytes)
+	}
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytes < MiB:
+		value = float64(bytes) / float64(KiB)
+		unit = "KiB"
+	case bytes < GiB:
+		value = float64(bytes) / float64(MiB)
+		unit = "MiB"
+	case bytes < TiB:
+		value = float64(bytes) / float64(GiB)
+		unit = "GiB"
+	default:
+		value = float64(bytes) / float64(TiB)
+		unit = "TiB"
+	}
+
+	return fmt.Sprintf("%s%.1f %s", prefix, value, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1B"},
+		{512, "512B"},
+		{1023, "1023B"},
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		// MiB boundary
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{10485760, "10.0 MiB"},
+		// GiB boundary
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB boundary
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-512, "-512B"},
+		// Large values
+		{5497558138880, "5.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
