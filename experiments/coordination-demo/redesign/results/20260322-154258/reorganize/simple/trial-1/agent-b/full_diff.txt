+// FormatRate formats a transfer rate in bytes per second into a human-readable string.
+// Output style: "512 B/s", "1.5 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	sign := ""
+	if bytesPerSec < 0 {
+		sign = "-"
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		kiB = 1024
+		miB = 1024 * 1024
+		giB = 1024 * 1024 * 1024
+	)
+
+	if bytesPerSec < kiB {
+		return fmt.Sprintf("%s%.0f B/s", sign, bytesPerSec)
+	}
+	if bytesPerSec < miB {
+		return fmt.Sprintf("%s%.1f KiB/s", sign, bytesPerSec/kiB)
+	}
+	if bytesPerSec < giB {
+		return fmt.Sprintf("%s%.1f MiB/s", sign, bytesPerSec/miB)
+	}
+	return fmt.Sprintf("%s%.1f GiB/s", sign, bytesPerSec/giB)
+}
