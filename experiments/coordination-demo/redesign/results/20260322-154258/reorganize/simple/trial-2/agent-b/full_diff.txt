+// FormatRate formats a transfer rate into a human-readable string using binary units.
+// Output style: "512 B/s", "1.0 KiB/s", "1.5 MiB/s", "2.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	if negative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	value := bytesPerSec
+	unitIndex := 0
+
+	for value >= 1024 && unitIndex < len(units)-1 {
+		value /= 1024
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// B/s: show as integer
+		result = fmt.Sprintf("%d %s", int(value), units[unitIndex])
+	} else {
+		// Other units: show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
