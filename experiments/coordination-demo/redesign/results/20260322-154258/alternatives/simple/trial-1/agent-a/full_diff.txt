+// FormatBytes formats a byte count into a human-readable string.
+// Uses binary units (B, KiB, MiB, GiB, TiB) with 1 decimal place for non-byte units.
+// Examples: "0 B", "512 B", "1.0 KiB", "1.5 MiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	neg := bytes < 0
+	if neg {
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	size := float64(bytes)
+	unitIndex := 0
+
+	for size >= 1024 && unitIndex < len(units)-1 {
+		size /= 1024
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes, show as integer
+		result = fmt.Sprintf("%d %s", bytes, units[unitIndex])
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[unitIndex])
+	}
+
+	if neg {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact KiB boundary
+		{1024, "1.0 KiB"},
+		// Between KiB boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{5120, "5.0 KiB"},
+		// Exact MiB boundary
+		{1048576, "1.0 MiB"},
+		// Between MiB boundaries
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// Exact GiB boundary
+		{1073741824, "1.0 GiB"},
+		// Between GiB boundaries
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// Exact TiB boundary
+		{1099511627776, "1.0 TiB"},
+		// Between TiB boundaries
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Large TiB values
+		{10995116277760, "10.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
