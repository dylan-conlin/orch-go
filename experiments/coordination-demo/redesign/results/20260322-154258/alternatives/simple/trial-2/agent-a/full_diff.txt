+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output uses binary units (B, KiB, MiB, GiB, TiB) with one decimal place for
+// non-byte units. Examples: "0 B", "512 B", "1.5 MiB", "-1.0 KiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var result string
+
+	switch {
+	case absBytes < KiB:
+		result = fmt.Sprintf("%d B", absBytes)
+	case absBytes < MiB:
+		value := float64(absBytes) / float64(KiB)
+		result = fmt.Sprintf("%.1f KiB", value)
+	case absBytes < GiB:
+		value := float64(absBytes) / float64(MiB)
+		result = fmt.Sprintf("%.1f MiB", value)
+	case absBytes < TiB:
+		value := float64(absBytes) / float64(GiB)
+		result = fmt.Sprintf("%.1f GiB", value)
+	default:
+		value := float64(absBytes) / float64(TiB)
+		result = fmt.Sprintf("%.1f TiB", value)
+	}
+
+	if negative {
+		return "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Bytes
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary and range
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		// MiB boundary and range
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		// GiB boundary and range
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		// TiB boundary and range
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1572864, "-1.5 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1610612736, "-1.5 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
