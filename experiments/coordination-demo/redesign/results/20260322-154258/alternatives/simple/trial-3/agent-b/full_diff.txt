+// FormatRate formats a transfer rate into human-readable string using binary units.
+// Output style: "512 B/s", "1.5 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	divisor := 1024.0
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for unitIndex < len(units)-1 && bytesPerSec >= divisor {
+		bytesPerSec /= divisor
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes per second, use integer format
+		result = fmt.Sprintf("%.0f %s", bytesPerSec, units[unitIndex])
+	} else {
+		// For other units, use 1 decimal place
+		result = fmt.Sprintf("%.1f %s", bytesPerSec, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
