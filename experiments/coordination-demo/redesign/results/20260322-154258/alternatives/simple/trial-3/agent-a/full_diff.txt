+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+
+	// Handle negative values
+	negative := bytes < 0
+	abytes := bytes
+	if negative {
+		abytes = -bytes
+	}
+
+	// Find appropriate unit
+	size := float64(abytes)
+	unitIndex := 0
+	for unitIndex < len(units)-1 && size >= 1024 {
+		size /= 1024
+		unitIndex++
+	}
+
+	// Format based on unit
+	var result string
+	if unitIndex == 0 {
+		// Bytes - no decimal
+		result = fmt.Sprintf("%d B", abytes)
+	} else {
+		// Other units - 1 decimal place
+		result = fmt.Sprintf("%.1f %s", size, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{1048576, "1.0 MiB"},
+		{1048576 + 524288, "1.5 MiB"},
+		// Larger values
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		{1099511627776 * 2, "2.0 TiB"},
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+		// Decimal precision
+		{2560, "2.5 KiB"},
+		{3072, "3.0 KiB"},
+		{3584, "3.5 KiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
