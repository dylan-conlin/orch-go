+// FormatRate formats transfer rates into human-readable binary unit strings.
+// Uses units B/s, KiB/s, MiB/s, GiB/s with 1 decimal place for non-byte rates.
+// Examples: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+	)
+
+	var result string
+	switch {
+	case bytesPerSec < KiB:
+		result = fmt.Sprintf("%.0f B/s", bytesPerSec)
+	case bytesPerSec < MiB:
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	case bytesPerSec < GiB:
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	default:
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+	return result
+}
