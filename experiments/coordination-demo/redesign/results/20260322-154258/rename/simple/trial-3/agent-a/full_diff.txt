+// FormatBytes formats a byte count into a human-readable string using
+// binary units (B, KiB, MiB, GiB, TiB) with 1 decimal place for non-byte units.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var (
+		value float64
+		unit  string
+	)
+
+	switch {
+	case bytes >= TiB:
+		value = float64(bytes) / float64(TiB)
+		unit = "TiB"
+	case bytes >= GiB:
+		value = float64(bytes) / float64(GiB)
+		unit = "GiB"
+	case bytes >= MiB:
+		value = float64(bytes) / float64(MiB)
+		unit = "MiB"
+	case bytes >= KiB:
+		value = float64(bytes) / float64(KiB)
+		unit = "KiB"
+	default:
+		result := fmt.Sprintf("%d B", bytes)
+		if negative {
+			result = "-" + result
+		}
+		return result
+	}
+
+	result := fmt.Sprintf("%.1f %s", value, unit)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}
