+// FormatRate formats a transfer rate into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte-per-sec values are shown with 1 decimal place.
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := bytesPerSec < 0
+	abs := bytesPerSec
+	if negative {
+		abs = -abs
+	}
+
+	const (
+		kb = 1024
+		mb = 1024 * 1024
+		gb = 1024 * 1024 * 1024
+	)
+
+	var result string
+	if abs < kb {
+		result = fmt.Sprintf("%.0f B/s", abs)
+	} else if abs < mb {
+		result = fmt.Sprintf("%.1f KiB/s", abs/kb)
+	} else if abs < gb {
+		result = fmt.Sprintf("%.1f MiB/s", abs/mb)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", abs/gb)
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
