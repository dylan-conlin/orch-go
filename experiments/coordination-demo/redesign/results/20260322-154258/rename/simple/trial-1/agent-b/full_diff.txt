+// FormatRate formats a transfer rate as a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Non-byte units show 1 decimal place (e.g., "1.5 MiB/s").
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []struct {
+		name  string
+		bytes float64
+	}{
+		{"GiB/s", 1024 * 1024 * 1024},
+		{"MiB/s", 1024 * 1024},
+		{"KiB/s", 1024},
+		{"B/s", 1},
+	}
+
+	for _, unit := range units {
+		if bytesPerSec >= unit.bytes {
+			value := bytesPerSec / unit.bytes
+			sign := ""
+			if isNegative {
+				sign = "-"
+			}
+			if unit.bytes == 1 {
+				// B/s: no decimal places
+				return fmt.Sprintf("%s%d %s", sign, int(value), unit.name)
+			}
+			// Other units: 1 decimal place
+			return fmt.Sprintf("%s%.1f %s", sign, value, unit.name)
+		}
+	}
+
+	// Fallback for values < 1 B/s
+	sign := ""
+	if isNegative {
+		sign = "-"
+	}
+	return fmt.Sprintf("%s%d B/s", sign, int(bytesPerSec))
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{256, "256 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// Exact boundaries
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		// KiB range
+		{5120, "5.0 KiB/s"},
+		{10240, "10.0 KiB/s"},
+		// MiB boundary
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Large values in MiB
+		{10485760, "10.0 MiB/s"},
+		// GiB boundary
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Large GiB values
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
