+// FormatBytes formats a byte count into a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte units display 1 decimal place.
+// Negative values are formatted with a "-" prefix.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []struct {
+		divisor int64
+		suffix  string
+	}{
+		{1024 * 1024 * 1024 * 1024, "TiB"},
+		{1024 * 1024 * 1024, "GiB"},
+		{1024 * 1024, "MiB"},
+		{1024, "KiB"},
+	}
+
+	for _, unit := range units {
+		if bytes >= unit.divisor {
+			value := float64(bytes) / float64(unit.divisor)
+			result := fmt.Sprintf("%.1f %s", value, unit.suffix)
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	result := fmt.Sprintf("%d B", bytes)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+		{1024, "1.0 KiB"},

+		{1649267441664, "1.5 TiB"},
