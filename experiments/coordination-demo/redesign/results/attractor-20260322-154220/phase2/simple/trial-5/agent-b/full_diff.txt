+// FormatRate formats a transfer rate as a human-readable string.
+// Output style: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s", "1.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	prefix := ""
+	if negative {
+		prefix = "-"
+	}
+
+	if bytesPerSec < KiB {
+		return fmt.Sprintf("%s%d B/s", prefix, int(bytesPerSec))
+	}
+
+	var value float64
+	var unit string
+
+	switch {
+	case bytesPerSec < MiB:
+		value = bytesPerSec / KiB
+		unit = "KiB/s"
+	case bytesPerSec < GiB:
+		value = bytesPerSec / MiB
+		unit = "MiB/s"
+	default:
+		value = bytesPerSec / GiB
+		unit = "GiB/s"
+	}
+
+	return fmt.Sprintf("%s%.1f %s", prefix, value, unit)
+}
