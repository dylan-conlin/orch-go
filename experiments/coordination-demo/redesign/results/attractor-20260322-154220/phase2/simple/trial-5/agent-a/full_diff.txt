+// FormatBytes formats a byte count into human-readable binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", etc.
+func FormatBytes(bytes int64) string {
+	const (
+		B   = 1
+		KiB = 1024 * B
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	var prefix string
+	if negative {
+		prefix = "-"
+	}
+
+	if bytes < KiB {
+		return fmt.Sprintf("%s%d B", prefix, bytes)
+	}
+	if bytes < MiB {
+		return fmt.Sprintf("%s%.1f KiB", prefix, float64(bytes)/float64(KiB))
+	}
+	if bytes < GiB {
+		return fmt.Sprintf("%s%.1f MiB", prefix, float64(bytes)/float64(MiB))
+	}
+	if bytes < TiB {
+		return fmt.Sprintf("%s%.1f GiB", prefix, float64(bytes)/float64(GiB))
+	}
+
+	return fmt.Sprintf("%s%.1f TiB", prefix, float64(bytes)/float64(TiB))
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+		// Large values
+		{1099511627776 * 2, "2.0 TiB"},
+		{1099511627776 * 10, "10.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1572864, "-1.5 MiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
