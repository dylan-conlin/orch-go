+// FormatBytes formats a byte count as a human-readable string.
+// Uses binary units: B, KiB, MiB, GiB, TiB.
+// Output examples: "512 B", "1.5 MiB", "-1.0 KiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	const unitSize = 1024
+
+	// Find the appropriate unit
+	unitIndex := 0
+	value := float64(bytes)
+	for unitIndex < len(units)-1 && value >= unitSize {
+		value /= unitSize
+		unitIndex++
+	}
+
+	// Format the result
+	var result string
+	if unitIndex == 0 {
+		// Bytes - no decimal
+		result = fmt.Sprintf("%d B", int64(value))
+	} else {
+		// Other units - 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}
