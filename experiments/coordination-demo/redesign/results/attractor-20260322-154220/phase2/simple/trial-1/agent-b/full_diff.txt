+// FormatRate formats a transfer rate in bytes per second into a human-readable string.
+// Output uses binary units with "/s" suffix: B/s, KiB/s, MiB/s, GiB/s.
+// Shows 1 decimal place for non-byte-per-sec units.
+// Examples: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s", "-1.0 KiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var value float64
+	var unit string
+
+	prefix := ""
+	if negative {
+		prefix = "-"
+	}
+
+	switch {
+	case bytesPerSec < KiB:
+		return fmt.Sprintf("%s%d B/s", prefix, int(bytesPerSec))
+	case bytesPerSec < MiB:
+		value = bytesPerSec / KiB
+		unit = "KiB/s"
+	case bytesPerSec < GiB:
+		value = bytesPerSec / MiB
+		unit = "MiB/s"
+	default:
+		value = bytesPerSec / GiB
+		unit = "GiB/s"
+	}
+
+	return fmt.Sprintf("%s%.1f %s", prefix, value, unit)
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		// KiB/s (1024 = 1 KiB)
+		{1024, "1.0 KiB/s"},
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{2560, "2.5 KiB/s"},
+		// MiB/s (1048576 = 1 MiB)
+		{1048576, "1.0 MiB/s"},
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		{10485760, "10.0 MiB/s"},
+		// GiB/s (1073741824 = 1 GiB)
+		{1073741824, "1.0 GiB/s"},
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		// Negative values
+		{-1, "-1 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+		// Boundary values and fractional bytes
+		{1023.9, "1023 B/s"},
+		{1024.1, "1.0 KiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%v) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
