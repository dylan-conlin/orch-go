+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "512 B", "1.0 KiB", "1.5 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	absBytes := bytes
+	negative := false
+	if bytes < 0 {
+		negative = true
+		absBytes = -bytes
+	}
+
+	value := float64(absBytes)
+	unitIndex := 0
+
+	// Find the appropriate unit
+	for unitIndex = 0; unitIndex < len(units)-1; unitIndex++ {
+		if value < 1024 {
+			break
+		}
+		value /= 1024
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes, show as integer
+		result = fmt.Sprintf("%d B", absBytes)
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10240, "10.0 KiB"},
+		{1048575, "1024.0 KiB"},
+		// MiB boundaries
+		{1048576, "1.0 MiB"},
+		{1572864, "1.5 MiB"},
+		{5242880, "5.0 MiB"},
+		{1073741823, "1024.0 MiB"},
+		// GiB boundaries
+		{1073741824, "1.0 GiB"},
+		{1610612736, "1.5 GiB"},
+		{5368709120, "5.0 GiB"},
+		{1099511627775, "1024.0 GiB"},
+		// TiB
+		{1099511627776, "1.0 TiB"},
+		{1649267441664, "1.5 TiB"},
+		{5497558138880, "5.0 TiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
