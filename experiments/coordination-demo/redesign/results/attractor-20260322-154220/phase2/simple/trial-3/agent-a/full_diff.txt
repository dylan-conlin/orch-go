+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.0 KiB", "1.5 MiB", "1.0 GiB", "1.0 TiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	isNegative := bytes < 0
+	absBytes := bytes
+	if isNegative {
+		absBytes = -bytes
+	}
+
+	units := []struct {
+		name  string
+		bytes int64
+	}{
+		{"TiB", 1024 * 1024 * 1024 * 1024},
+		{"GiB", 1024 * 1024 * 1024},
+		{"MiB", 1024 * 1024},
+		{"KiB", 1024},
+		{"B", 1},
+	}
+
+	for _, unit := range units {
+		if absBytes >= unit.bytes {
+			if unit.bytes == 1 {
+				// For bytes, don't show decimal places
+				if isNegative {
+					return fmt.Sprintf("-%d %s", absBytes, unit.name)
+				}
+				return fmt.Sprintf("%d %s", absBytes, unit.name)
+			}
+			value := float64(absBytes) / float64(unit.bytes)
+			if isNegative {
+				return fmt.Sprintf("%.1f %s", -value, unit.name)
+			}
+			return fmt.Sprintf("%.1f %s", value, unit.name)
+		}
+	}
+
+	// Should not reach here
+	if isNegative {
+		return fmt.Sprintf("-%d B", absBytes)
+	}
+	return fmt.Sprintf("%d B", absBytes)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundaries
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1024 * 10, "10.0 KiB"},
+		// MiB boundaries
+		{1024 * 1024, "1.0 MiB"},
+		{1024*1024 + 512*1024, "1.5 MiB"},
+		{2 * 1024 * 1024, "2.0 MiB"},
+		// GiB boundaries
+		{1024 * 1024 * 1024, "1.0 GiB"},
+		{1024*1024*1024 + 512*1024*1024, "1.5 GiB"},
+		// TiB boundaries
+		{1024 * 1024 * 1024 * 1024, "1.0 TiB"},
+		{1024*1024*1024*1024 + 512*1024*1024*1024, "1.5 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1024 * 1024, "-1.0 MiB"},
+		{-1024 * 1024 * 1024, "-1.0 GiB"},
+		{-1024 * 1024 * 1024 * 1024, "-1.0 TiB"},
+		// Large values
+		{999 * 1024 * 1024 * 1024, "999.0 GiB"},
+		{999 * 1024 * 1024 * 1024 * 1024, "999.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
