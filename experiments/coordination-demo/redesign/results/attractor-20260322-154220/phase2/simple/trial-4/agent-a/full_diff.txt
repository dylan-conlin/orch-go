+// FormatBytes formats a byte count as a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.3 MiB", "1.0 GiB", "1.5 TiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	divisor := int64(1)
+	unitIdx := 0
+
+	// Find the appropriate unit
+	for i := 0; i < len(units)-1; i++ {
+		if bytes < divisor*1024 {
+			break
+		}
+		divisor *= 1024
+		unitIdx++
+	}
+
+	var result string
+	if unitIdx == 0 {
+		// For bytes, show as integer
+		result = fmt.Sprintf("%d B", bytes)
+	} else {
+		// For other units, show with 1 decimal place
+		value := float64(bytes) / float64(divisor)
+		result = fmt.Sprintf("%.1f %s", value, units[unitIdx])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{1024 * 10, "10.0 KiB"},
+		{1024 * 512, "512.0 KiB"},
+		{1024*1024 - 1, "1024.0 KiB"},
+
+		// MiB boundary
+		{1024 * 1024, "1.0 MiB"},
+		{1024*1024 + 512*1024, "1.5 MiB"},
+		{1024 * 1024 * 10, "10.0 MiB"},
+		{1024*1024*1024 - 1, "1024.0 MiB"},
+
+		// GiB boundary
+		{1024 * 1024 * 1024, "1.0 GiB"},
+		{1024*1024*1024 + 512*1024*1024, "1.5 GiB"},
+		{1024 * 1024 * 1024 * 10, "10.0 GiB"},
+
+		// TiB boundary
+		{1024 * 1024 * 1024 * 1024, "1.0 TiB"},
+		{1024*1024*1024*1024 + 512*1024*1024*1024, "1.5 TiB"},
+
+		// Negative values
+		{-1024, "-1.0 KiB"},
+		{-512, "-512 B"},
+		{-1048576, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
