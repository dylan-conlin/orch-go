+// FormatRate formats a transfer rate into a human-readable string using binary
+// units. Output style: "512 B/s", "1.0 KiB/s", "1.5 MiB/s", "1.0 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytesPerSec < 0 {
+		negative = true
+		bytesPerSec = -bytesPerSec
+	}
+
+	units := []string{"B/s", "KiB/s", "MiB/s", "GiB/s"}
+	const threshold = 1024.0
+
+	i := 0
+	for i < len(units)-1 && bytesPerSec >= threshold {
+		bytesPerSec /= threshold
+		i++
+	}
+
+	// Format the result
+	var result string
+	if i == 0 {
+		// For bytes per second, no decimal place
+		result = fmt.Sprintf("%d %s", int(bytesPerSec), units[i])
+	} else {
+		// For larger units, show one decimal place
+		result = fmt.Sprintf("%.1f %s", bytesPerSec, units[i])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+
+	return result
+}
