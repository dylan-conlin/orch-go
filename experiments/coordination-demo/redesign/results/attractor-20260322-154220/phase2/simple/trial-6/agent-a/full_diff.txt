+// FormatBytes formats a byte count as a human-readable string using binary
+// units (B, KiB, MiB, GiB, TiB). Non-byte values show 1 decimal place.
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	absBytes := bytes
+	if negative {
+		absBytes = -bytes
+	}
+
+	// Binary units: 1 KiB = 1024 B
+	const (
+		KiB = 1024
+		MiB = 1024 * KiB
+		GiB = 1024 * MiB
+		TiB = 1024 * GiB
+	)
+
+	var size float64
+	var unit string
+
+	switch {
+	case absBytes < KiB:
+		// For bytes, return as integer without decimal
+		if negative {
+			return fmt.Sprintf("-%d B", absBytes)
+		}
+		return fmt.Sprintf("%d B", absBytes)
+	case absBytes < MiB:
+		size = float64(absBytes) / KiB
+		unit = "KiB"
+	case absBytes < GiB:
+		size = float64(absBytes) / MiB
+		unit = "MiB"
+	case absBytes < TiB:
+		size = float64(absBytes) / GiB
+		unit = "GiB"
+	default:
+		size = float64(absBytes) / TiB
+		unit = "TiB"
+	}
+
+	if negative {
+		return fmt.Sprintf("-%.1f %s", size, unit)
+	}
+	return fmt.Sprintf("%.1f %s", size, unit)
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small bytes
+		{0, "0 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Between boundaries
+		{1536, "1.5 KiB"},
+		{1572864, "1.5 MiB"},
+		{1610612736, "1.5 GiB"},
+		// Larger values
+		{5242880, "5.0 MiB"},
+		{10737418240, "10.0 GiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
