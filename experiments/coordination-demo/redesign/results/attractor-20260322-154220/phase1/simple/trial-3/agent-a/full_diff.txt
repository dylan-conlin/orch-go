+// FormatBytes formats a byte count as a human-readable string.
+// Uses binary units (B, KiB, MiB, GiB, TiB) with one decimal place
+// for non-byte units. Negative values are prefixed with "-".
+// Examples: "512 B", "1.5 KiB", "2.0 MiB", "-1.0 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	abs := bytes
+	if negative {
+		abs = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	unitIndex := 0
+	value := float64(abs)
+
+	for unitIndex < len(units)-1 && value >= 1024 {
+		value /= 1024
+		unitIndex++
+	}
+
+	// For bytes (unitIndex == 0), show as integer
+	if unitIndex == 0 {
+		if negative {
+			return fmt.Sprintf("-%d B", int64(value))
+		}
+		return fmt.Sprintf("%d B", int64(value))
+	}
+
+	// For other units, show with 1 decimal place
+	if negative {
+		return fmt.Sprintf("-%.1f %s", value, units[unitIndex])
+	}
+	return fmt.Sprintf("%.1f %s", value, units[unitIndex])
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero
+		{0, "0 B"},
+		// Small bytes (no unit conversion)
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// Exact boundaries
+		{1024, "1.0 KiB"},
+		{1048576, "1.0 MiB"},
+		{1073741824, "1.0 GiB"},
+		{1099511627776, "1.0 TiB"},
+		// Values between boundaries
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{1572864, "1.5 MiB"},
+		{2097152, "2.0 MiB"},
+		{1610612736, "1.5 GiB"},
+		{2147483648, "2.0 GiB"},
+		{1649267441664, "1.5 TiB"},
+		{2199023255552, "2.0 TiB"},
+		// Large values
+		{107374182400, "100.0 GiB"},
+		{10995116277760, "10.0 TiB"},
+		// Values that need rounding
+		{1234567, "1.2 MiB"},
+		// Negative values
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1048576, "-1.0 MiB"},
+		{-1073741824, "-1.0 GiB"},
+		{-1099511627776, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
