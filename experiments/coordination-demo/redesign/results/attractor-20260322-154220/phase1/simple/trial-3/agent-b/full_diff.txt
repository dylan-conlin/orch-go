+// FormatRate formats a transfer rate in bytes per second into a human-readable
+// string using binary units with "/s" suffix.
+// Output style: "0 B/s", "512 B/s", "1.0 KiB/s", "1.5 MiB/s", "2.3 GiB/s".
+func FormatRate(bytesPerSec float64) string {
+	if bytesPerSec == 0 {
+		return "0 B/s"
+	}
+
+	isNegative := bytesPerSec < 0
+	if isNegative {
+		bytesPerSec = -bytesPerSec
+	}
+
+	const (
+		KiB = 1024
+		MiB = 1024 * 1024
+		GiB = 1024 * 1024 * 1024
+	)
+
+	var result string
+	if bytesPerSec < KiB {
+		result = fmt.Sprintf("%g B/s", bytesPerSec)
+	} else if bytesPerSec < MiB {
+		result = fmt.Sprintf("%.1f KiB/s", bytesPerSec/KiB)
+	} else if bytesPerSec < GiB {
+		result = fmt.Sprintf("%.1f MiB/s", bytesPerSec/MiB)
+	} else {
+		result = fmt.Sprintf("%.1f GiB/s", bytesPerSec/GiB)
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+
+	return result
+}

+func TestFormatRate(t *testing.T) {
+	tests := []struct {
+		input float64
+		want  string
+	}{
+		// Zero
+		{0, "0 B/s"},
+		// Small values (bytes per second)
+		{1, "1 B/s"},
+		{512, "512 B/s"},
+		{1023, "1023 B/s"},
+		{512.5, "512.5 B/s"},
+		// Boundary: 1024 bytes = 1.0 KiB
+		{1024, "1.0 KiB/s"},
+		// Fractional KiB
+		{1536, "1.5 KiB/s"},
+		{2048, "2.0 KiB/s"},
+		{1331, "1.3 KiB/s"},
+		// Large KiB values
+		{524288, "512.0 KiB/s"},
+		// Boundary: 1048576 bytes = 1.0 MiB
+		{1048576, "1.0 MiB/s"},
+		// Fractional MiB
+		{1572864, "1.5 MiB/s"},
+		{2097152, "2.0 MiB/s"},
+		// Large MiB values
+		{536870912, "512.0 MiB/s"},
+		// Boundary: 1073741824 bytes = 1.0 GiB
+		{1073741824, "1.0 GiB/s"},
+		// Large GiB values
+		{1610612736, "1.5 GiB/s"},
+		{2147483648, "2.0 GiB/s"},
+		{10737418240, "10.0 GiB/s"},
+		// Negative values
+		{-512, "-512 B/s"},
+		{-1024, "-1.0 KiB/s"},
+		{-1048576, "-1.0 MiB/s"},
+		{-1073741824, "-1.0 GiB/s"},
+	}
+	for _, tt := range tests {
+		got := FormatRate(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatRate(%g) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
