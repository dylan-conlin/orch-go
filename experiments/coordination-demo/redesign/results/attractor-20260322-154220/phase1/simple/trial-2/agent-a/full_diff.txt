+// FormatBytes formats a byte count as a human-readable string using binary units
+// (B, KiB, MiB, GiB, TiB). Non-byte units show 1 decimal place.
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := false
+	if bytes < 0 {
+		negative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	threshold := int64(1024)
+	unitIndex := 0
+
+	value := float64(bytes)
+	for unitIndex < len(units)-1 && value >= float64(threshold) {
+		value /= float64(threshold)
+		unitIndex++
+	}
+
+	var result string
+	if unitIndex == 0 {
+		// For bytes, show no decimal places
+		result = fmt.Sprintf("%d %s", int64(value), units[unitIndex])
+	} else {
+		// For other units, show 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if negative {
+		result = "-" + result
+	}
+	return result
+}

+func TestFormatBytes(t *testing.T) {
+	tests := []struct {
+		input int64
+		want  string
+	}{
+		// Zero and small values
+		{0, "0 B"},
+		{1, "1 B"},
+		{512, "512 B"},
+		{1023, "1023 B"},
+		// KiB boundary
+		{1024, "1.0 KiB"},
+		{1536, "1.5 KiB"},
+		{2048, "2.0 KiB"},
+		{10 * 1024, "10.0 KiB"},
+		{1024*1024 - 1, "1024.0 KiB"},
+		// MiB boundary
+		{1024 * 1024, "1.0 MiB"},
+		{1024*1024 + 512*1024, "1.5 MiB"},
+		{2 * 1024 * 1024, "2.0 MiB"},
+		{100 * 1024 * 1024, "100.0 MiB"},
+		{1024*1024*1024 - 1, "1024.0 MiB"},
+		// GiB boundary
+		{1024 * 1024 * 1024, "1.0 GiB"},
+		{1024*1024*1024 + 512*1024*1024, "1.5 GiB"},
+		{5 * 1024 * 1024 * 1024, "5.0 GiB"},
+		{1024*1024*1024*1024 - 1, "1024.0 GiB"},
+		// TiB boundary
+		{1024 * 1024 * 1024 * 1024, "1.0 TiB"},
+		{1024*1024*1024*1024 + 512*1024*1024*1024, "1.5 TiB"},
+		{10 * 1024 * 1024 * 1024 * 1024, "10.0 TiB"},
+		// Negative values
+		{-1, "-1 B"},
+		{-512, "-512 B"},
+		{-1024, "-1.0 KiB"},
+		{-1536, "-1.5 KiB"},
+		{-1024 * 1024, "-1.0 MiB"},
+		{-1024 * 1024 * 1024, "-1.0 GiB"},
+		{-1024 * 1024 * 1024 * 1024, "-1.0 TiB"},
+	}
+	for _, tt := range tests {
+		got := FormatBytes(tt.input)
+		if got != tt.want {
+			t.Errorf("FormatBytes(%d) = %q, want %q", tt.input, got, tt.want)
+		}
+	}
+}
