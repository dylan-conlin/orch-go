+// FormatBytes formats byte counts into human-readable strings using binary units.
+// Output style: "0 B", "512 B", "1.5 MiB", "2.3 GiB".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	negative := bytes < 0
+	if negative {
+		bytes = -bytes
+	}
+
+	// Define units: threshold (in bytes), unit name
+	type unit struct {
+		threshold int64
+		name      string
+	}
+
+	units := []unit{
+		{1024 * 1024 * 1024 * 1024, "TiB"},
+		{1024 * 1024 * 1024, "GiB"},
+		{1024 * 1024, "MiB"},
+		{1024, "KiB"},
+	}
+
+	for _, u := range units {
+		if bytes >= u.threshold {
+			value := float64(bytes) / float64(u.threshold)
+			result := fmt.Sprintf("%.1f %s", value, u.name)
+			if negative {
+				result = "-" + result
+			}
+			return result
+		}
+	}
+
+	// Less than 1 KiB
+	result := fmt.Sprintf("%d B", bytes)
+	if negative {
+		result = "-" + result
+	}
+	return result
+}
