+// FormatBytes formats a byte count into a human-readable string using binary units.
+// Output style: "0 B", "512 B", "1.5 KiB", "2.0 MiB", "1.5 GiB", "3.2 TiB".
+// Negative values are prefixed with "-".
+func FormatBytes(bytes int64) string {
+	if bytes == 0 {
+		return "0 B"
+	}
+
+	// Handle negative values
+	isNegative := false
+	if bytes < 0 {
+		isNegative = true
+		bytes = -bytes
+	}
+
+	units := []string{"B", "KiB", "MiB", "GiB", "TiB"}
+	value := float64(bytes)
+
+	unitIndex := 0
+	for unitIndex < len(units)-1 && value >= 1024 {
+		value /= 1024
+		unitIndex++
+	}
+
+	// Format the string
+	var result string
+	if unitIndex == 0 {
+		// For bytes, no decimal places
+		result = fmt.Sprintf("%d B", int64(value))
+	} else {
+		// For other units, 1 decimal place
+		result = fmt.Sprintf("%.1f %s", value, units[unitIndex])
+	}
+
+	if isNegative {
+		result = "-" + result
+	}
+
+	return result
+}
